  padding: 10px 20px;

CSS Modules ​

基本用法 ​

jsx
// Button.module.css
.button {
  background-color: #007bff;
  color: white;
  padding: 10px 20px;
  border: none;
  border-radius: 4px;
  cursor: pointer;
}

.button:hover {
  background-color: #0056b3;
}

.primary {
  background-color: #28a745;
}

.primary:hover {
  background-color: #218838;
}

// Button.jsx

function Button({ primary, children }) {
  return (
    <button className={`${styles.button} ${primary ? styles.primary : ''}`}>
      {children}
    </button>
  )
}

样式面试题 ​

1. React 中有哪些样式处理方式？ ​

答案：

React 中的样式处理方式：

  1. 内联样式:

    jsx
    const style = { color: 'red', fontSize: '16px' }
<div style={style}>内容</div>

  2. CSS 类名:

    jsx
    import './styles.css'
<div className="container">内容</div>

  3. CSS Modules:

    jsx
    import styles from './styles.module.css'
<div className={styles.container}>内容</div>

  4. CSS-in-JS:

    jsx
    import styled from 'styled-components'
const Button = styled.button`color: red;`

  5. Tailwind CSS:

    jsx
    <div className="bg-blue-500 text-white">内容</div>

2. CSS-in-JS 的优缺点是什么？ ​

答案：

优点：

  1. 样式隔离: 自动生成唯一的类名，避免样式冲突
  2. 动态样式: 可以使用 JavaScript 动态生成样式
  3. 组件化: 样式与组件绑定，提高可维护性
  4. 主题支持: 轻松实现主题切换
  5. 类型安全: 与 TypeScript 集成良好

缺点：

  1. 性能: 运行时生成样式，可能影响性能
  2. 包大小: 增加打包体积
  3. 学习成本: 需要学习新的语法和 API
  4. 调试: 生成的类名不易调试

3. CSS Modules 的工作原理是什么？ ​

答案：

CSS Modules 的工作原理：

  1. 编译时转换:

    • 构建工具将 CSS 类名转换为唯一的标识符
    • 例如: .button → .button__hash123

  2. 导入样式:

    jsx
    import styles from './Button.module.css'

  3. 使用样式:

    jsx
    <button className={styles.button}>按钮</button>

  4. 生成 HTML:

    html
    <button class="Button_button__hash123">按钮</button>

  5. 优点:

    • 样式隔离
    • 避免全局污染
    • 支持组合样式

4. styled-components 和 Emotion 的区别是什么？ ​

答案：

主要区别：

  1. 性能:

    • Emotion: 更轻量，性能更好
    • styled-components: 功能更丰富，但性能稍差

  2. API 设计:

    • Emotion: 支持 css prop 和 styled API
    • styled-components: 只支持 styled API

  3. 包大小:

    • Emotion: 约 12KB
    • styled-components: 约 14KB

  4. TypeScript 支持:

    • Emotion: 更好的 TypeScript 支持
    • styled-components: 也支持，但稍弱

  5. 使用示例:

    jsx
    // styled-components
const Button = styled.button`color: red;`
<Button>按钮</Button>

// Emotion
const buttonStyle = css`color: red;`
<button css={buttonStyle}>按钮</button>

5. Tailwind CSS 的优缺点是什么？ ​

答案：

优点：

  1. 开发效率: 不需要切换文件，直接在 JSX 中写样式
  2. 一致性: 使用预设的设计系统，保持样式一致
  3. 响应式: 内置响应式工具类
  4. 可定制: 可以自定义配置
  5. 性能: 生产环境自动移除未使用的样式

缺点：

  1. HTML 冗余: 类名较多，HTML 可读性差
  2. 学习成本: 需要记忆大量的工具类
  3. 样式复用: 需要创建自定义组件来复用样式
  4. 设计限制: 受限于预设的设计系统

6. 如何在 React 中实现主题切换？ ​

答案：

实现主题切换：

  1. 使用 CSS Variables:

    jsx
    // styles.css
:root {
  --primary-color: #007bff;
  --background-color: white;
}

[data-theme="dark"] {
  --primary-color: #0056b3;
  --background-color: #1a1a1a;
}

// ThemeProvider.jsx
function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light')

  useEffect(() => {
    document.documentElement.setAttribute('data-theme', theme)
  }, [theme])

  return (
    <ThemeContext.Provider value={{ theme, setTheme }}>
      {children}
    </ThemeContext.Provider>
  )
}

// 使用
function App() {
  const { theme, setTheme } = useContext(ThemeContext)

  return (
    <ThemeProvider>
      <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
        切换主题
      </button>
    </ThemeProvider>
  )
}

  2. 使用 styled-components:

    jsx
    import { ThemeProvider } from 'styled-components'

const lightTheme = {
  primary: '#007bff',
  background: 'white'
}

const darkTheme = {
  primary: '#0056b3',
  background: '#1a1a1a'
}

function App() {
  const [theme, setTheme] = useState(lightTheme)

  return (
    <ThemeProvider theme={theme}>
      <button onClick={() => setTheme(theme === lightTheme ? darkTheme : lightTheme)}>
        切换主题
      </button>
    </ThemeProvider>
  )
}

7. 如何优化 React 应用的样式性能？ ​

答案：

优化样式性能：

  1. 减少样式计算:

    jsx
    // 使用 CSS Modules 或 CSS-in-JS 避免全局样式
import styles from './styles.module.css'

  2. 使用 CSS Variables:

    jsx
    :root {
  --primary-color: #007bff;
}

.button {
  background-color: var(--primary-color);
}

  3. 避免内联样式:

    jsx
    // 不推荐
<div style={{ color: 'red' }}>内容</div>

// 推荐
<div className="text-red">内容</div>

  4. 使用 PurgeCSS:

    js
    // tailwind.config.js
module.exports = {
  purge: ['./src/**/*.{js,jsx,ts,tsx}'],
  // ...
}

  5. 代码分割:

    jsx
    import { lazy, Suspense } from 'react'

const HeavyComponent = lazy(() => import('./HeavyComponent'))

function App() {
  return (
    <Suspense fallback={<div>Loading...</div>}>
      <HeavyComponent />
    </Suspense>
  )
}

8. 如何在 React 中实现响应式设计？ ​

答案：

实现响应式设计：

  1. 使用 CSS Media Queries:

    css
    /* styles.css */
.container {
  max-width: 1200px;
  margin: 0 auto;
  padding: 20px;
}

@media (max-width: 768px) {
  .container {
    padding: 10px;
  }
}

  2. 使用 Tailwind CSS:

    jsx
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
  <div>卡片 1</div>
  <div>卡片 2</div>
  <div>卡片 3</div>
</div>

  3. 使用 React Hooks:

    jsx
    import { useState, useEffect } from 'react'

function useWindowSize() {
  const [size, setSize] = useState({
    width: window.innerWidth,
    height: window.innerHeight
  })

  useEffect(() => {
    const handleResize = () => {
      setSize({
        width: window.innerWidth,
        height: window.innerHeight
      })
    }

    window.addEventListener('resize', handleResize)
    return () => window.removeEventListener('resize', handleResize)
  }, [])

  return size
}

function ResponsiveComponent() {
  const { width } = useWindowSize()

  return (
    <div>
      {width < 768 ? <MobileView /> : <DesktopView />}
    </div>
  )
}

  4. 使用 CSS Grid:

    jsx
    function ResponsiveGrid() {
  return (
    <div style={{
      display: 'grid',
      gridTemplateColumns: 'repeat(auto-fit, minmax(300px, 1fr))',
      gap: '20px'
    }}>
      <div>卡片 1</div>
      <div>卡片 2</div>
      <div>卡片 3</div>
    </div>
  )
}
