1. 小程序框架概述 ​

框架特点：

  • MVVM架构
  • 数据驱动视图
  • 组件化开发
  • 响应式数据绑定

  • WXSS：页面样式
  • JS：页面逻辑
  • JSON：页面配置

2. 数据绑定 ​

数据绑定语法：

<!-- 文本绑定 -->
<view>{{message}}</view>

<!-- 属性绑定 -->
<view id="{{id}}">内容</view>

<!-- 表单绑定 -->
<input value="{{value}}" bindinput="onInput" />

<!-- 条件渲染 -->
<view wx:if="{{condition}}">显示</view>
<view wx:elif="{{condition2}}">显示2</view>
<view wx:else>显示3</view>

<!-- 列表渲染 -->
<view wx:for="{{list}}" wx:key="id">
  {{item.name}}
</view>

<!-- 模板引用 -->
<template is="templateName" data="{{...item}}" />

数据更新：

javascript
Page({
  data: {
    message: 'Hello',
    list: [
      { id: 1, name: '张三' },
      { id: 2, name: '李四' }
    ]
  },

  onLoad() {
    // 直接修改数据
    this.setData({
      message: 'World'
    })

    // 修改数组元素
    this.setData({
      'list[0].name': '王五'
    })

    // 添加数组元素
    this.setData({
      'list': [...this.data.list, { id: 3, name: '赵六' }]
    })
  }
})

3. 事件处理 ​

事件绑定：

<!-- bind事件绑定，不会阻止冒泡 -->
<view bindtap="onTap">点击</view>

<!-- catch事件绑定，会阻止冒泡 -->
<view catchtap="onTap">点击</view>

<!-- capture-bind事件绑定，捕获阶段触发 -->
<view capture-bindtap="onTap">点击</view>

<!-- capture-catch事件绑定，捕获阶段触发并阻止冒泡 -->
<view capture-catchtap="onTap">点击</view>

<!-- 事件传参 -->
<view bindtap="onTap" data-id="{{id}}">点击</view>

事件处理：

javascript
Page({
  onTap(e) {
    // 获取事件对象
    console.log(e)

    // 获取自定义数据
    const id = e.currentTarget.dataset.id

    // 获取触摸点
    const touch = e.touches[0]
  },

  onInput(e) {
    // 获取输入值
    const value = e.detail.value

    // 更新数据
    this.setData({
      value: value
    })
  }
})

小程序组件 ​

4. 组件定义 ​

组件定义：

javascript
// components/my-component/my-component.js
Component({
  properties: {
    // 组件属性
    title: {
      type: String,
      value: '默认标题'
    },
    count: {
      type: Number,
      value: 0
    }
  },

  data: {
    // 组件内部数据
    internalData: '内部数据'
  },

  methods: {
    // 组件方法
    handleClick() {
      this.setData({
        count: this.data.count + 1
      })

      // 触发自定义事件
      this.triggerEvent('customEvent', {
        count: this.data.count
      })
    }
  },

  lifetimes: {
    // 组件生命周期
    attached() {
      console.log('组件挂载')
    },
    detached() {
      console.log('组件卸载')
    }
  },

  pageLifetimes: {
    // 组件所在页面的生命周期
    show() {
      console.log('页面显示')
    },
    hide() {
      console.log('页面隐藏')
    }
  }
})

组件模板：

<!-- components/my-component/my-component.wxml -->
<view class="my-component">
  <view class="title">{{title}}</view>
  <view class="count">{{count}}</view>
  <button bindtap="handleClick">点击</button>
</view>

组件样式：

css
/* components/my-component/my-component.wxss */
.my-component {
  padding: 20rpx;
  background: #fff;
  border-radius: 10rpx;
}

.title {
  font-size: 32rpx;
  font-weight: bold;
}

.count {
  font-size: 28rpx;
  color: #666;
}

组件配置：

json
{
  "component": true,

}

5. 组件使用 ​

页面引入组件：

json
{
  "usingComponents": {
    "my-component": "/components/my-component/my-component"
  }
}

页面使用组件：

<my-component
  title="自定义标题"
  count="{{count}}"
  bind:customEvent="onCustomEvent"
/>

页面处理组件事件：

javascript
Page({
  data: {
    count: 0
  },

  onCustomEvent(e) {
    const { count } = e.detail
    this.setData({
      count: count
    })
  }
})

小程序生命周期 ​

7. 页面生命周期 ​

页面生命周期：

javascript
Page({
  onLoad(options) {
    // 页面加载
    // options为页面参数
    console.log('onLoad', options)
  },

  onShow() {
    // 页面显示
    console.log('onShow')
  },

  onReady() {
    // 页面初次渲染完成
    console.log('onReady')
  },

  onHide() {
    // 页面隐藏
    console.log('onHide')
  },

  onUnload() {
    // 页面卸载
    console.log('onUnload')
  },

  onPullDownRefresh() {
    // 下拉刷新
    console.log('onPullDownRefresh')
  },

  onReachBottom() {
    // 上拉触底
    console.log('onReachBottom')
  },

  onShareAppMessage() {
    // 分享
    return {
      title: '分享标题',
      path: '/pages/index/index'
    }
  }
})

生命周期执行顺序：

  1. onLoad：页面加载
  2. onShow：页面显示
  3. onReady：页面初次渲染完成
  4. onHide：页面隐藏
  5. onUnload：页面卸载

8. 组件生命周期 ​

组件生命周期：

javascript
Component({
  lifetimes: {
    created() {
      // 组件实例化
      console.log('created')
    },

    attached() {
      // 组件挂载
      console.log('attached')
    },

    ready() {
      // 组件布局完成
      console.log('ready')
    },

    moved() {
      // 组件移动
      console.log('moved')
    },

    detached() {
      // 组件卸载
      console.log('detached')
    }
  },

  pageLifetimes: {
    show() {
      // 组件所在页面显示
      console.log('page show')
    },

    hide() {
      // 组件所在页面隐藏
      console.log('page hide')
    },

    resize(size) {
      // 组件所在页面尺寸变化
      console.log('page resize', size)
    }
  }
})

生命周期执行顺序：

  1. created：组件实例化
  2. attached：组件挂载
  3. ready：组件布局完成
  4. moved：组件移动
  5. detached：组件卸载

9. 小程序生命周期对比 ​

页面生命周期 vs 组件生命周期：

页面生命周期组件生命周期说明
onLoadcreated实例化
onShowattached挂载
onReadyready布局完成
onHide-隐藏
onUnloaddetached卸载

使用场景：

  • onLoad：初始化数据，获取页面参数
  • onShow：刷新数据，更新状态
  • onReady：操作DOM，获取节点信息
  • onHide：暂停定时器，保存状态
  • onUnload：清理资源，取消监听

10. 小程序路由 ​

路由方法：

javascript
// 保留当前页面，跳转到新页面
wx.navigateTo({
  url: '/pages/detail/detail?id=1',
  success(res) {
    console.log('跳转成功')
  },
  fail(err) {
    console.log('跳转失败', err)
  }
})

// 关闭当前页面，跳转到新页面
wx.redirectTo({
  url: '/pages/detail/detail?id=1'
})

// 跳转到tabBar页面
wx.switchTab({
  url: '/pages/index/index'
})

// 关闭所有页面，跳转到新页面
wx.reLaunch({
  url: '/pages/index/index'
})

// 返回上一页
wx.navigateBack({
  delta: 1
})

路由参数：

javascript
// 传递参数
wx.navigateTo({
  url: '/pages/detail/detail?id=1&name=张三'
})

// 接收参数
Page({
  onLoad(options) {
    const { id, name } = options
    console.log(id, name)
  }
})

路由事件：

javascript
// 监听路由变化
Page({
  onShow() {
    // 页面显示
    const pages = getCurrentPages()
    const currentPage = pages[pages.length - 1]
    console.log(currentPage.route)
  }
})

9. 小程序的自定义组件和页面有什么区别？ ​

参考答案：

小程序的自定义组件和页面的区别：

  1. 定义方式

    • 页面：使用Page构造器
    • 组件：使用Component构造器

  2. 生命周期

    • 页面：onLoad、onShow、onReady、onHide、onUnload
    • 组件：created、attached、ready、moved、detached

  3. 数据

    • 页面：data、setData
    • 组件：properties、data、setData

  4. 事件

    • 页面：onLoad、onShow等
    • 组件：triggerEvent触发事件

  5. 使用方式

    • 页面：直接访问
    • 组件：需要引入和使用

10. 小程序的条件渲染和列表渲染是如何实现的？ ​

参考答案：

小程序的条件渲染和列表渲染实现：

  1. 条件渲染

    <view wx:if="{{condition}}">显示</view>
<view wx:elif="{{condition2}}">显示2</view>
<view wx:else>显示3</view>

  2. 列表渲染

    <view wx:for="{{list}}" wx:key="id">
  {{item.name}}
</view>

  3. block

    <block wx:if="{{condition}}">
  <view>内容1</view>
  <view>内容2</view>
</block>

  4. 注意事项

    • wx:key提高渲染性能
    • wx:for支持嵌套
    • wx:if和hidden的区别
