JavaScript 对象与数组 ​

JavaScript 对象与数组相关面试题，包括对象方法、数组方法、深浅拷贝、数组去重、数组扁平化等。

对象方法 ​

Object.keys ​

javascript
const obj = { a: 1, b: 2, c: 3 };
const keys = Object.keys(obj);
console.log(keys);  // ['a', 'b', 'c']

Object.values ​

javascript
const obj = { a: 1, b: 2, c: 3 };
const values = Object.values(obj);
console.log(values);  // [1, 2, 3]

Object.entries ​

javascript
const obj = { a: 1, b: 2, c: 3 };
const entries = Object.entries(obj);
console.log(entries);  // [['a', 1], ['b', 2], ['c', 3]]

// 转 Map
const map = new Map(entries);
console.log(map.get('a'));  // 1

Object.assign ​

javascript
const target = { a: 1 };
const source = { b: 2, c: 3 };
const result = Object.assign(target, source);
console.log(result);  // { a: 1, b: 2, c: 3 }
console.log(target);  // { a: 1, b: 2, c: 3 }

Object.freeze ​

javascript
const obj = { a: 1 };
Object.freeze(obj);

obj.a = 2;  // 严格模式下报错
obj.b = 3;  // 严格模式下报错

常见面试问题 ​

Q: Object.assign 和展开运算符的区别？

javascript
// Object.assign 会触发 setter
const obj = {
  _name: 'Alice',
  get name() {
    return this._name;
  },
  set name(value) {
    console.log('Setting name to', value);
    this._name = value;
  }
};

Object.assign({}, obj);  // 触发 setter
{ ...obj };  // 不触发 setter

常用方法示例 ​

javascript
// map
[1, 2, 3].map(x => x * 2);  // [2, 4, 6]

// filter
[1, 2, 3, 4].filter(x => x % 2 === 0);  // [2, 4]

// reduce
[1, 2, 3].reduce((sum, x) => sum + x, 0);  // 6

// find
[1, 2, 3, 4].find(x => x > 2);  // 3

// some
[1, 2, 3].some(x => x > 2);  // true

// every
[1, 2, 3].every(x => x > 0);  // true

// includes
[1, 2, 3].includes(2);  // true

// flat
[1, [2, 3], 4].flat();  // [1, 2, 3, 4]
[1, [2, 3], 4].flat(2);  // [1, 2, 3, 4]

手动实现 map ​

javascript
Array.prototype.myMap = function(fn) {
  const result = [];
  for (let i = 0; i < this.length; i++) {
    result.push(fn(this[i], i, this));
  }
  return result;
};

手动实现 filter ​

javascript
Array.prototype.myFilter = function(fn) {
  const result = [];
  for (let i = 0; i < this.length; i++) {
    if (fn(this[i], i, this)) {
      result.push(this[i]);
    }
  }
  return result;
};

手动实现 reduce ​

javascript
Array.prototype.myReduce = function(fn, initialValue) {
  let accumulator = initialValue;
  for (let i = 0; i < this.length; i++) {
    accumulator = fn(accumulator, this[i], i, this);
  }
  return accumulator;
};

常见面试问题 ​

Q: map 和 forEach 的区别？

  • map：返回新数组
  • forEach：不返回值，用于副作用

Q: find 和 filter 的区别？

  • find：返回第一个满足条件的元素
  • filter：返回所有满足条件的元素

深浅拷贝 ​

浅拷贝 ​

javascript
// 方法 1：Object.assign
const obj = { a: 1, b: { c: 2 } };
const copy = Object.assign({}, obj);

// 方法 2：展开运算符
const copy = { ...obj };

// 方法 3：Array.slice（数组）
const arr = [1, 2, 3];
const copy = arr.slice();

深拷贝 ​

javascript
// 方法 1：JSON 方法
const obj = { a: 1, b: { c: 2 } };
const copy = JSON.parse(JSON.stringify(obj));

// 方法 2：递归
function deepClone(obj) {
  if (obj === null || typeof obj !== 'object') return obj;
  const clone = Array.isArray(obj) ? [] : {};
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key]);
    }
  }
  return clone;
}

// 方法 3：structuredClone（现代浏览器）
const copy = structuredClone(obj);

常见面试问题 ​

Q: 深拷贝和浅拷贝的区别？

  • 浅拷贝：只复制第一层属性，嵌套对象仍然是引用
  • 深拷贝：递归复制所有层级，完全独立

数组去重 ​

方法 1：Set ​

javascript
const arr = [1, 2, 2, 3, 3, 4];
const uniqueArr = [...new Set(arr)];
console.log(uniqueArr);  // [1, 2, 3, 4]

方法 2：filter + indexOf ​

javascript
const arr = [1, 2, 2, 3, 3, 4];
const uniqueArr = arr.filter((item, index) => arr.indexOf(item) === index);
console.log(uniqueArr);  // [1, 2, 3, 4]

方法 3：reduce ​

javascript
const arr = [1, 2, 2, 3, 3, 4];
const uniqueArr = arr.reduce((acc, item) => {
  if (!acc.includes(item)) {
    acc.push(item);
  }
  return acc;
}, []);
console.log(uniqueArr);  // [1, 2, 3, 4]

对象数组去重 ​

javascript
const arr = [
  { id: 1, name: 'Alice' },
  { id: 2, name: 'Bob' },
  { id: 1, name: 'Alice' }
];

const uniqueArr = arr.filter((item, index) => {
  return arr.findIndex(i => i.id === item.id) === index;
});
console.log(uniqueArr);  // [{ id: 1, name: 'Alice' }, { id: 2, name: 'Bob' }]

常见面试问题 ​

Q: 对象数组如何去重？

使用 filter + findIndex 或 Map。

数组扁平化 ​

方法 1：flat ​

javascript
const arr = [1, [2, 3], [4, [5, 6]]];
const flatArr = arr.flat();
console.log(flatArr);  // [1, 2, 3, 4, [5, 6]]

const flatArr2 = arr.flat(2);
console.log(flatArr2);  // [1, 2, 3, 4, 5, 6]

方法 2：flat + Infinity ​

javascript
const arr = [1, [2, [3, [4, [5]]]];
const flatArr = arr.flat(Infinity);
console.log(flatArr);  // [1, 2, 3, 4, 5]

方法 3：递归 ​

javascript
function flatten(arr) {
  return arr.reduce((acc, item) => {
    if (Array.isArray(item)) {
      return acc.concat(flatten(item));
    } else {
      acc.push(item);
      return acc;
    }
  }, []);
}

const arr = [1, [2, [3, [4, [5]]]];
const flatArr = flatten(arr);
console.log(flatArr);  // [1, 2, 3, 4, 5]

方法 4：toString + split ​

javascript
const arr = [1, [2, [3, [4, [5]]]];
const flatArr = arr.toString().split(',').map(Number);
console.log(flatArr);  // [1, 2, 3, 4, 5]

flat 是 ES2019 的特性，旧浏览器需要 polyfill。

数组排序 ​

基本排序 ​

javascript
[3, 1, 4, 2].sort();  // [1, 2, 3, 4]

// 数字排序
[3, 1, 4, 2].sort((a, b) => a - b);  // [1, 2, 3, 4]

// 对象排序
const users = [
  { name: 'Alice', age: 20 },
  { name: 'Bob', age: 18 },
  { name: 'Charlie', age: 25 }
];

users.sort((a, b) => a.age - b.age);
console.log(users);  // 按年龄升序排列

冒泡排序 ​

javascript
function bubbleSort(arr) {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = 0; j < arr.length - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
      }
    }
  }
  return arr;
}

快速排序 ​

javascript
function quickSort(arr) {
  if (arr.length <= 1) return arr;
  
  const pivot = arr[0];
  const left = [];
  const right = [];
  
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] < pivot) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }
  
  return [...quickSort(left), pivot, ...quickSort(right)];
}

默认按字符串排序，数字需要传入比较函数。

javascript
const arr = [1, 2, 3];
const arrayLike = { 0: 1, 1: 2, 2: 3, length: 3 };

Array.isArray(arr);      // true
Array.isArray(arrayLike); // false

// 转为数组
const realArr = Array.from(arrayLike);
console.log(Array.isArray(realArr));  // true
