WebSocket ​

WebSocket 基本概念 ​

WebSocket 原理 ​

javascript
// WebSocket
// 全双工通信协议
// 基于 TCP
// 可以在单个连接上进行全双工通信

// WebSocket 特点
// 1. 全双工：客户端和服务器可以同时发送和接收消息
// 2. 实时通信：消息实时传输
// 3. 低延迟：比 HTTP 轮询延迟更低
// 4. 节省带宽：不需要发送头部信息
// 5. 服务器推送：服务器可以主动推送消息

WebSocket 与 HTTP 的区别 ​

javascript
// WebSocket 与 HTTP 的区别
// 1. 通信方式
//    - HTTP: 半双工（请求-响应）
//    - WebSocket: 全双工（双向通信）

// 2. 连接方式
//    - HTTP: 每次请求都需要建立连接
//    - WebSocket: 建立长连接

// 3. 状态
//    - HTTP: 无状态
//    - WebSocket: 有状态

// 4. 性能
//    - HTTP: 性能较差（每次请求都需要建立连接）
//    - WebSocket: 性能较好（长连接）

WebSocket 连接 ​

客户端连接 ​

javascript
// 客户端连接
const socket = new WebSocket('ws://localhost:8080')

// 连接打开
socket.onopen = function(event) {
  console.log('WebSocket 连接已打开')
  socket.send('Hello Server')
}

// 接收消息
socket.onmessage = function(event) {
  console.log('收到消息:', event.data)
}

// 连接关闭
socket.onclose = function(event) {
  console.log('WebSocket 连接已关闭')
}

// 连接错误
socket.onerror = function(error) {
  console.log('WebSocket 错误:', error)
}

服务器连接（Node.js） ​

javascript
// 服务器连接（Node.js）
const WebSocket = require('ws')
const wss = new WebSocket.Server({ port: 8080 })

wss.on('connection', function connection(ws) {
  console.log('客户端已连接')

  // 接收消息
  ws.on('message', function incoming(message) {
    console.log('收到消息:', message)
    ws.send('Hello Client')
  })

  // 连接关闭
  ws.on('close', function() {
    console.log('客户端已断开')
  })
})

WebSocket 消息 ​

发送消息 ​

javascript
// 发送消息
const socket = new WebSocket('ws://localhost:8080')

socket.onopen = function(event) {
  // 发送文本消息
  socket.send('Hello Server')

  // 发送 JSON 消息
  socket.send(JSON.stringify({ type: 'chat', message: 'Hello' }))

  // 发送二进制消息
  const arrayBuffer = new ArrayBuffer(10)
  socket.send(arrayBuffer)

  // 发送 Blob
  const blob = new Blob(['Hello'], { type: 'text/plain' })
  socket.send(blob)
}

接收消息 ​

javascript
// 接收消息
const socket = new WebSocket('ws://localhost:8080')

socket.onmessage = function(event) {
  // 文本消息
  if (typeof event.data === 'string') {
    console.log('文本消息:', event.data)
  }

  // JSON 消息
  if (typeof event.data === 'string') {
    const data = JSON.parse(event.data)
    console.log('JSON 消息:', data)
  }

  // 二进制消息
  if (event.data instanceof ArrayBuffer) {
    console.log('二进制消息:', event.data)
  }

  // Blob 消息
  if (event.data instanceof Blob) {
    console.log('Blob 消息:', event.data)
  }
}

WebSocket 状态 ​

WebSocket 状态 ​

javascript
// WebSocket 状态
const socket = new WebSocket('ws://localhost:8080')

// 0: CONNECTING - 正在连接
console.log(socket.readyState) // 0

// 1: OPEN - 连接已打开
socket.onopen = function(event) {
  console.log(socket.readyState) // 1
}

// 2: CLOSING - 正在关闭
socket.onclose = function(event) {
  console.log(socket.readyState) // 2
}

// 3: CLOSED - 连接已关闭
setTimeout(() => {
  console.log(socket.readyState) // 3
}, 1000)

关闭连接 ​

javascript
// 关闭连接
const socket = new WebSocket('ws://localhost:8080')

socket.onopen = function(event) {
  // 关闭连接
  socket.close()

  // 关闭连接并指定状态码和原因
  socket.close(1000, 'Normal Closure')
}

// 连接关闭
socket.onclose = function(event) {
  console.log('连接已关闭')
  console.log('状态码:', event.code)
  console.log('原因:', event.reason)
  console.log('是否由服务器关闭:', event.wasClean)
}

面试题 ​

1. WebSocket 和 HTTP 的区别是什么？ ​

答案：

主要区别：

  1. 通信方式:

    • HTTP: 半双工（请求-响应）
    • WebSocket: 全双工（双向通信）

  2. 连接方式:

    • HTTP: 每次请求都需要建立连接
    • WebSocket: 建立长连接

  3. 状态:

    • HTTP: 无状态
    • WebSocket: 有状态

  4. 性能:

    • HTTP: 性能较差（每次请求都需要建立连接）
    • WebSocket: 性能较好（长连接）

2. WebSocket 的使用场景有哪些？ ​

答案：

WebSocket 的使用场景：

  1. 聊天应用: 实时聊天
  2. 实时数据: 实时数据更新
  3. 在线游戏: 在线多人游戏
  4. 协同编辑: 实时协同编辑
  5. 股票行情: 实时股票行情

3. 如何实现 WebSocket 连接？ ​

答案：

实现 WebSocket 连接：

  1. 客户端连接:

    javascript
    const socket = new WebSocket('ws://localhost:8080')

socket.onopen = function(event) {
  console.log('WebSocket 连接已打开')
  socket.send('Hello Server')
}

socket.onmessage = function(event) {
  console.log('收到消息:', event.data)
}

  2. 服务器连接（Node.js）:

    javascript
    const WebSocket = require('ws')
const wss = new WebSocket.Server({ port: 8080 })

wss.on('connection', function connection(ws) {
  ws.on('message', function incoming(message) {
    console.log('收到消息:', message)
    ws.send('Hello Client')
  })
})

4. WebSocket 的状态有哪些？ ​

答案：

WebSocket 的状态：

  1. CONNECTING (0): 正在连接
  2. OPEN (1): 连接已打开
  3. CLOSING (2): 正在关闭
  4. CLOSED (3): 连接已关闭

5. 如何关闭 WebSocket 连接？ ​

答案：

关闭 WebSocket 连接：

  1. 关闭连接:

    javascript
    socket.close()

  2. 关闭连接并指定状态码和原因:

    javascript
    socket.close(1000, 'Normal Closure')

  3. 监听关闭事件:

    javascript
    socket.onclose = function(event) {
  console.log('连接已关闭')
  console.log('状态码:', event.code)
  console.log('原因:', event.reason)
}
