前端监控架构 ​

知识点介绍 ​

前端监控的概念 ​

前端监控是指监控前端应用的运行状态，包括错误监控、性能监控、用户行为监控等，及时发现和解决问题。

前端监控的分类 ​

  1. 错误监控：监控 JavaScript 错误、Promise 错误等
  2. 性能监控：监控页面加载性能、运行性能等
  3. 用户行为监控：监控用户点击、滚动等行为
  4. 业务监控：监控业务指标，如转化率、留存率等

前端监控工具 ​

常见的前端监控工具：

  1. Sentry：错误监控平台
  2. Google Analytics：用户行为分析
  3. Hotjar：用户行为分析
  4. LogRocket：用户行为回放
  5. Datadog：全栈监控平台

2. 如何实现错误监控？ ​

答案：

实现错误监控：

javascript
// 监控 JavaScript 错误
window.addEventListener('error', (event) => {
  console.error('JavaScript Error:', event.error)
  
  // 发送到监控平台
  sendError({
    type: 'javascript',
    message: event.message,
    filename: event.filename,
    lineno: event.lineno,
    colno: event.colno,
    stack: event.error?.stack
  })
})

// 监控 Promise 错误
window.addEventListener('unhandledrejection', (event) => {
  console.error('Promise Error:', event.reason)
  
  // 发送到监控平台
  sendError({
    type: 'promise',
    reason: event.reason,
    stack: event.reason?.stack
  })
})

// 监控资源加载错误
window.addEventListener('error', (event) => {
  if (event.target !== window) {
    console.error('Resource Error:', event.target)
    
    // 发送到监控平台
    sendError({
      type: 'resource',
      target: event.target.src || event.target.href
    })
  }
}, true)

// 使用 try-catch 捕获错误
try {
  // 可能出错的代码
} catch (error) {
  console.error('Caught Error:', error)
  
  // 发送到监控平台
  sendError({
    type: 'caught',
    message: error.message,
    stack: error.stack
  })
}

// 使用 React Error Boundary
class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props)
    this.state = { hasError: false }
  }
  
  static getDerivedStateFromError(error) {
    return { hasError: true }
  }
  
  componentDidCatch(error, errorInfo) {
    console.error('React Error:', error, errorInfo)
    
    // 发送到监控平台
    sendError({
      type: 'react',
      message: error.message,
      stack: error.stack,
      componentStack: errorInfo.componentStack
    })
  }
  
  render() {
    if (this.state.hasError) {
      return <h1>Something went wrong.</h1>
    }
    return this.props.children
  }
}

// 使用 Performance API
// 监控页面加载性能
window.addEventListener('load', () => {
  const perfData = performance.getEntriesByType('navigation')[0]
  
  const metrics = {
    // DNS 查询时间
    dns: perfData.domainLookupEnd - perfData.domainLookupStart,
    // TCP 连接时间
    tcp: perfData.connectEnd - perfData.connectStart,
    // 请求时间
    request: perfData.responseStart - perfData.requestStart,
    // 响应时间
    response: perfData.responseEnd - perfData.responseStart,
    // DOM 解析时间
    dom: perfData.domComplete - perfData.domInteractive,
    // 首次内容绘制时间
    fcp: perfData.responseStart - perfData.fetchStart,
    // 首次有意义绘制时间
    fmp: perfData.domContentLoadedEventEnd - perfData.fetchStart,
    // 完全加载时间
    load: perfData.loadEventEnd - perfData.fetchStart
  }
  
  // 发送到监控平台
  sendMetrics(metrics)
})

// 使用 Performance Observer
const observer = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    console.log(entry.name, entry.duration)
    
    // 发送到监控平台
    sendMetrics({
      name: entry.name,
      duration: entry.duration,
      startTime: entry.startTime
    })
  }
})

observer.observe({ entryTypes: ['measure', 'paint', 'navigation'] })

// 使用 Web Vitals
import { getCLS, getFID, getLCP } from 'web-vitals'

getCLS(console.log)
getFID(console.log)
getLCP(console.log)

// 自定义性能监控
function measurePerformance(name, fn) {
  const start = performance.now()
  const result = fn()
  const end = performance.now()
  
  const duration = end - start
  
  console.log(`${name} took ${duration}ms`)
  
  // 发送到监控平台
  sendMetrics({
    name,
    duration
  })
  
  return result
}

// 使用
measurePerformance('fetchData', () => {
  return fetchData()
})

// 监控点击事件

    },
    position: {

// 监控滚动事件
let scrollTimeout
document.addEventListener('scroll', () => {
  clearTimeout(scrollTimeout)
  
  scrollTimeout = setTimeout(() => {
    const scrollDepth = window.scrollY / (document.body.scrollHeight - window.innerHeight)
    
    console.log('Scroll Depth:', scrollDepth)
    
    // 发送到监控平台
    sendUserAction({
      type: 'scroll',
      depth: scrollDepth
    })
  }, 100)
})

// 监控表单提交
document.addEventListener('submit', (event) => {
  const form = event.target
  
  console.log('Form Submit:', {
    id: form.id,
    className: form.className,
    action: form.action,
    method: form.method
  })
  
  // 发送到监控平台
  sendUserAction({
    type: 'form-submit',
    form: {
      id: form.id,
      className: form.className,
      action: form.action,
      method: form.method
    }
  })
})

// 监控路由变化
let lastPath = window.location.pathname

function checkRouteChange() {
  const currentPath = window.location.pathname
  
  if (currentPath !== lastPath) {
    console.log('Route Change:', {
      from: lastPath,
      to: currentPath
    })
    
    // 发送到监控平台
    sendUserAction({
      type: 'route-change',
      from: lastPath,
      to: currentPath
    })
    
    lastPath = currentPath
  }
}

// 使用 History API 监听路由变化
const originalPushState = history.pushState
const originalReplaceState = history.replaceState

history.pushState = function(...args) {
  originalPushState.apply(this, args)
  checkRouteChange()
}

history.replaceState = function(...args) {
  originalReplaceState.apply(this, args)
  checkRouteChange()
}

window.addEventListener('popstate', checkRouteChange)

7. 如何实现错误告警？ ​

答案：

实现错误告警：

javascript
// 错误告警配置
const errorAlertConfig = {
  // 错误阈值
  threshold: 10,
  // 时间窗口（毫秒）
  timeWindow: 60000,
  // 告警方式
  alertMethods: ['email', 'slack', 'webhook']
}

// 错误计数器
const errorCounts = new Map()

// 发送告警
function sendAlert(error) {
  const alertData = {
    type: 'error',
    message: error.message,
    stack: error.stack,
    timestamp: new Date().toISOString()
  }
  
  // 发送邮件
  if (errorAlertConfig.alertMethods.includes('email')) {
    sendEmailAlert(alertData)
  }
  
  // 发送 Slack
  if (errorAlertConfig.alertMethods.includes('slack')) {
    sendSlackAlert(alertData)
  }
  
  // 发送 Webhook
  if (errorAlertConfig.alertMethods.includes('webhook')) {
    sendWebhookAlert(alertData)
  }
}

// 检查是否需要告警
function checkAlert(error) {
  const key = error.message
  const now = Date.now()
  
  // 获取或创建错误计数
  let errorCount = errorCounts.get(key)
  
  if (!errorCount || now - errorCount.timestamp > errorAlertConfig.timeWindow) {
    errorCount = {
      count: 0,
      timestamp: now
    }
  }
  
  // 增加计数
  errorCount.count++
  errorCounts.set(key, errorCount)
  
  // 检查是否超过阈值
  if (errorCount.count >= errorAlertConfig.threshold) {
    sendAlert(error)
    
    // 重置计数
    errorCount.count = 0
    errorCount.timestamp = now
  }
}

// 监听错误
window.addEventListener('error', (event) => {
  checkAlert(event.error)
})

// 性能告警配置
const performanceAlertConfig = {
  // FCP 阈值（毫秒）
  fcpThreshold: 1800,
  // LCP 阈值（毫秒）
  lcpThreshold: 2500,
  // FID 阈值（毫秒）
  fidThreshold: 100,
  // CLS 阈值
  clsThreshold: 0.1
}

// 发送告警
function sendPerformanceAlert(metric, value, threshold) {
  const alertData = {
    type: 'performance',
    metric,
    value,
    threshold,
    timestamp: new Date().toISOString()
  }
  
  // 发送到监控平台
  sendAlert(alertData)
}

// 监控性能指标
import { getCLS, getFID, getLCP } from 'web-vitals'

getCLS((metric) => {
  if (metric.value > performanceAlertConfig.clsThreshold) {
    sendPerformanceAlert('CLS', metric.value, performanceAlertConfig.clsThreshold)
  }
})

getFID((metric) => {
  if (metric.value > performanceAlertConfig.fidThreshold) {
    sendPerformanceAlert('FID', metric.value, performanceAlertConfig.fidThreshold)
  }
})

getLCP((metric) => {
  if (metric.value > performanceAlertConfig.lcpThreshold) {
    sendPerformanceAlert('LCP', metric.value, performanceAlertConfig.lcpThreshold)
  }
})

10. 前端监控的最佳实践是什么？ ​

答案：

前端监控的最佳实践：

  1. 全面监控：

    • 监控错误、性能、用户行为
    • 覆盖所有关键页面和功能

  2. 合理采样：

    • 对性能数据进行采样
    • 对用户行为数据进行采样

  3. 隐私保护：

    • 不收集敏感信息
    • 遵守隐私法规

  4. 性能优化：

    • 使用异步发送
    • 批量发送数据

  5. 错误处理：

    • 监控代码本身不能影响应用
    • 使用 try-catch 保护监控代码

  6. 告警策略：

    • 设置合理的告警阈值
    • 避免告警疲劳

  7. 数据分析：

    • 定期分析监控数据
    • 发现和解决问题

  8. 持续改进：

    • 根据监控数据优化应用
    • 持续改进监控策略

示例：

javascript
// 异步发送监控数据
async function sendMonitoringData(data) {
  try {
    // 使用 sendBeacon API
    if (navigator.sendBeacon) {
      const blob = new Blob([JSON.stringify(data)], { type: 'application/json' })
      navigator.sendBeacon('/api/monitoring', blob)
    } else {
      // 使用 fetch API
      await fetch('/api/monitoring', {
        method: 'POST',
        body: JSON.stringify(data),
        keepalive: true
      })
    }
  } catch (error) {
    // 监控代码本身不能影响应用
    console.error('Failed to send monitoring data:', error)
  }
}

// 批量发送数据
const dataQueue = []
let sendTimer = null

function queueData(data) {
  dataQueue.push(data)
  
  if (!sendTimer) {
    sendTimer = setTimeout(() => {
      sendMonitoringData(dataQueue)
      dataQueue.length = 0
      sendTimer = null
    }, 5000)
  }
}

// 使用
queueData({ type: 'error', message: 'Something went wrong' })
queueData({ type: 'performance', metric: 'FCP', value: 1500 })
