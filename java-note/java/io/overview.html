Java IO流详解 ​

📚 概述 ​

Java IO（Input/Output）流是Java中处理输入输出操作的核心机制，用于在程序与外部数据源（如文件、网络、内存等）之间传输数据。Java IO系统设计基于流（Stream）的概念，提供了统一的接口来处理不同类型的数据源。从Java 1.0的传统IO到Java 1.4引入的NIO（New IO），再到Java 1.7的NIO.2，Java的IO系统不断演进，为开发者提供了丰富而强大的IO处理能力。理解Java IO系统对于文件操作、网络编程、数据序列化等应用场景至关重要。

🎯 学习目标 ​

  • 深入理解Java IO流的基本概念和分类
  • 掌握字节流和字符流的使用方法和区别
  • 熟练使用缓冲流、转换流等包装流
  • 理解对象序列化和反序列化机制
  • 掌握文件和目录操作的最佳实践
  • 了解NIO和传统IO的区别和应用场景
  • 学会处理IO异常和资源管理
  • 掌握IO性能优化技巧

    // IO流分类演示
    public static void demonstrateIOStreamClassification() {
        System.out.println("=== Java IO流分类演示 ===");
        
        /*
         * Java IO流分类：
         * 
         * 按数据类型分类：
         * 1. 字节流（Byte Streams）- 处理8位字节数据
         *    ├── InputStream（输入字节流抽象类）
         *    │   ├── FileInputStream - 文件输入流
         *    │   ├── ByteArrayInputStream - 字节数组输入流
         *    │   ├── BufferedInputStream - 缓冲输入流
         *    │   └── ObjectInputStream - 对象输入流
         *    └── OutputStream（输出字节流抽象类）
         *        ├── FileOutputStream - 文件输出流
         *        ├── ByteArrayOutputStream - 字节数组输出流
         *        ├── BufferedOutputStream - 缓冲输出流
         *        └── ObjectOutputStream - 对象输出流
         * 
         * 2. 字符流（Character Streams）- 处理16位Unicode字符
         *    ├── Reader（字符输入流抽象类）
         *    │   ├── FileReader - 文件字符输入流
         *    │   ├── StringReader - 字符串输入流
         *    │   ├── BufferedReader - 缓冲字符输入流
         *    │   └── InputStreamReader - 字节流到字符流转换
         *    └── Writer（字符输出流抽象类）
         *        ├── FileWriter - 文件字符输出流
         *        ├── StringWriter - 字符串输出流
         *        ├── BufferedWriter - 缓冲字符输出流
         *        └── OutputStreamWriter - 字符流到字节流转换
         * 
         * 按功能分类：
         * 1. 节点流（Node Streams）- 直接连接数据源
         * 2. 包装流（Wrapper Streams）- 包装其他流，提供额外功能
         */
        
        System.out.println("IO流分类特点:");
        System.out.println("1. 字节流 - 处理二进制数据，适用于所有类型文件");
        System.out.println("2. 字符流 - 处理文本数据，自动处理字符编码");
        System.out.println("3. 节点流 - 直接操作数据源，如文件、数组");
        System.out.println("4. 包装流 - 增强功能，如缓冲、转换、压缩");
        
        // 演示流的层次结构
        demonstrateStreamHierarchy();
        
        System.out.println();
    }

    // 文件字节流演示
    public static void demonstrateFileByteStreams() {
        System.out.println("\n--- 文件字节流操作 ---");
        
        String sourceFile = "source.txt";
        String targetFile = "target.txt";
        
        try {
            // 创建源文件
            String content = "Hello, Java IO Streams!\n这是字节流测试内容。\n包含中文字符。";
            Files.write(Paths.get(sourceFile), content.getBytes(StandardCharsets.UTF_8));
            
            // 使用字节流复制文件
            long startTime = System.currentTimeMillis();
            
            try (FileInputStream fis = new FileInputStream(sourceFile);
                 FileOutputStream fos = new FileOutputStream(targetFile)) {
                
                byte[] buffer = new byte[1024];
                int bytesRead;
                int totalBytes = 0;
                
                while ((bytesRead = fis.read(buffer)) != -1) {
                    fos.write(buffer, 0, bytesRead);
                    totalBytes += bytesRead;
                }
                
                long endTime = System.currentTimeMillis();
                System.out.println("文件复制完成:");
                System.out.println("  复制字节数: " + totalBytes);
                System.out.println("  耗时: " + (endTime - startTime) + "ms");
            }
            
            // 验证复制结果
            byte[] sourceBytes = Files.readAllBytes(Paths.get(sourceFile));
            byte[] targetBytes = Files.readAllBytes(Paths.get(targetFile));
            
            System.out.println("  复制验证: " + Arrays.equals(sourceBytes, targetBytes));
            
            // 清理文件
            Files.deleteIfExists(Paths.get(sourceFile));
            Files.deleteIfExists(Paths.get(targetFile));
            
        } catch (IOException e) {
            System.out.println("文件操作异常: " + e.getMessage());
        }
    }
    
    // 字节数组流演示
    public static void demonstrateByteArrayStreams() {
        System.out.println("\n--- 字节数组流操作 ---");
        
        try {
            // ByteArrayOutputStream - 写入字节数组
            ByteArrayOutputStream baos = new ByteArrayOutputStream();
            
            String[] messages = {"Hello", " ", "Java", " ", "IO", "!"};
            for (String msg : messages) {
                baos.write(msg.getBytes(StandardCharsets.UTF_8));
            }
            
            byte[] data = baos.toByteArray();
            System.out.println("写入数据: " + new String(data, StandardCharsets.UTF_8));
            System.out.println("数据长度: " + data.length + " 字节");
            
            // ByteArrayInputStream - 读取字节数组
            ByteArrayInputStream bais = new ByteArrayInputStream(data);
            
            byte[] buffer = new byte[5];
            int bytesRead;
            System.out.println("分块读取:");
            
            while ((bytesRead = bais.read(buffer)) != -1) {
                String chunk = new String(buffer, 0, bytesRead, StandardCharsets.UTF_8);
                System.out.println("  读取: '" + chunk + "' (" + bytesRead + " 字节)");
            }
            
            // 关闭流
            baos.close();
            bais.close();
            
        } catch (IOException e) {
            System.out.println("字节数组流操作异常: " + e.getMessage());
        }
    }
    
    // 数据流演示
    public static void demonstrateDataStreams() {
        System.out.println("\n--- 数据流操作 ---");

            // 写入不同类型的数据
            try (DataOutputStream dos = new DataOutputStream(
                    new FileOutputStream(dataFile))) {

                dos.writeUTF("Hello, DataStream!");
                dos.writeLong(System.currentTimeMillis());
                
                System.out.println("数据写入完成");
            }
            
            // 读取数据（必须按写入顺序读取）
            try (DataInputStream dis = new DataInputStream(
                    new FileInputStream(dataFile))) {
                
                int intValue = dis.readInt();
                double doubleValue = dis.readDouble();
                boolean boolValue = dis.readBoolean();
                String stringValue = dis.readUTF();
                long longValue = dis.readLong();
                
                System.out.println("读取的数据:");
                System.out.println("  整数: " + intValue);
                System.out.println("  浮点数: " + doubleValue);
                System.out.println("  布尔值: " + boolValue);
                System.out.println("  字符串: " + stringValue);
                System.out.println("  长整数: " + longValue);
            }
            
            // 显示文件大小
            long fileSize = Files.size(Paths.get(dataFile));
            System.out.println("数据文件大小: " + fileSize + " 字节");
            
            // 清理文件
            Files.deleteIfExists(Paths.get(dataFile));

    // 缓冲流性能对比
    public static void demonstrateBufferedPerformance() {
        System.out.println("\n--- 缓冲流性能对比 ---");
        
        String sourceFile = "performance_test.txt";
        String targetFile1 = "target_unbuffered.txt";
        String targetFile2 = "target_buffered.txt";
        
        try {
            // 创建测试文件
            createLargeTestFile(sourceFile, 10000);
            
            // 无缓冲复制
            long startTime = System.currentTimeMillis();
            try (FileInputStream fis = new FileInputStream(sourceFile);
                 FileOutputStream fos = new FileOutputStream(targetFile1)) {
                
                int byteData;
                while ((byteData = fis.read()) != -1) {
                    fos.write(byteData);
                }
            }
            long unbufferedTime = System.currentTimeMillis() - startTime;
            
            // 缓冲流复制
            startTime = System.currentTimeMillis();
            try (BufferedInputStream bis = new BufferedInputStream(new FileInputStream(sourceFile));
                 BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(targetFile2))) {
                
                int byteData;
                while ((byteData = bis.read()) != -1) {
                    bos.write(byteData);
                }
            }
            long bufferedTime = System.currentTimeMillis() - startTime;
            
            System.out.println("性能对比结果:");
            System.out.println("  无缓冲流耗时: " + unbufferedTime + "ms");
            System.out.println("  缓冲流耗时: " + bufferedTime + "ms");
            System.out.println("  性能提升: " + String.format("%.1f", (double) unbufferedTime / bufferedTime) + "倍");
            
            // 清理文件
            Files.deleteIfExists(Paths.get(sourceFile));
            Files.deleteIfExists(Paths.get(targetFile1));
            Files.deleteIfExists(Paths.get(targetFile2));
            
        } catch (IOException e) {
            System.out.println("缓冲流性能测试异常: " + e.getMessage());
        }
    }

        try {
            // 创建多行文本文件
            List<String> lines = Arrays.asList(

                "第四行：mark()和reset()功能",
                "第五行：文件结束"
            );
            Files.write(Paths.get(textFile), lines, StandardCharsets.UTF_8);
            
            try (BufferedReader reader = new BufferedReader(
                    new FileReader(textFile, StandardCharsets.UTF_8))) {
                
                // readLine()方法
                System.out.println("使用readLine()读取:");
                String line;
                int lineNum = 1;
                while ((line = reader.readLine()) != null) {
                    System.out.println("  " + lineNum + ": " + line);
                    lineNum++;
                    
                    // 演示mark和reset（如果支持）
                    if (lineNum == 3 && reader.markSupported()) {
                        System.out.println("  [标记当前位置]");
                        reader.mark(1000); // 标记，参数是预读缓冲区大小
                    }
                    
                    if (lineNum == 4 && reader.markSupported()) {
                        System.out.println("  [重置到标记位置]");
                        reader.reset();
                        line = reader.readLine(); // 重新读取第3行
                        System.out.println("  重读: " + line);
                    }
                }
            }
            
            // 清理文件
            Files.deleteIfExists(Paths.get(textFile));
            
        } catch (IOException e) {
            System.out.println("BufferedReader功能演示异常: " + e.getMessage());
        }
    }

    // 安全的文件复制
    public static void demonstrateSafeFileCopy() {
        System.out.println("\n--- 安全的文件复制 ---");

        String sourceFile = "source_safe.txt";
        String targetFile = "target_safe.txt";

        try {
            // 创建源文件
            String content = "这是安全文件复制的测试内容。\n包含多行文本数据。";
            Files.write(Paths.get(sourceFile), content.getBytes(StandardCharsets.UTF_8));

            // 安全的文件复制方法
            boolean success = safeFileCopy(sourceFile, targetFile);

            if (success) {
                System.out.println("文件复制成功");

                // 验证复制结果
                String originalContent = Files.readString(Paths.get(sourceFile), StandardCharsets.UTF_8);
                String copiedContent = Files.readString(Paths.get(targetFile), StandardCharsets.UTF_8);

                System.out.println("内容验证: " + originalContent.equals(copiedContent));
            } else {
                System.out.println("文件复制失败");
            }

            // 清理文件
            Files.deleteIfExists(Paths.get(sourceFile));
            Files.deleteIfExists(Paths.get(targetFile));

        } catch (IOException e) {
            System.out.println("安全文件复制异常: " + e.getMessage());
        }
    }

    // 安全文件复制方法
    public static boolean safeFileCopy(String sourcePath, String targetPath) {
        try (InputStream input = Files.newInputStream(Paths.get(sourcePath));
             OutputStream output = Files.newOutputStream(Paths.get(targetPath))) {

            byte[] buffer = new byte[8192]; // 8KB缓冲区
            int bytesRead;

            while ((bytesRead = input.read(buffer)) != -1) {
                output.write(buffer, 0, bytesRead);
            }

            return true;

        } catch (IOException e) {
            System.out.println("文件复制过程中发生错误: " + e.getMessage());

            // 清理可能创建的不完整目标文件
            try {
                Files.deleteIfExists(Paths.get(targetPath));
            } catch (IOException cleanupException) {
                System.out.println("清理失败的目标文件时发生错误: " + cleanupException.getMessage());
            }

            return false;
        }
    }

    // 目录操作
    public static void demonstrateDirectoryOperations() {
        System.out.println("\n--- 目录操作 ---");

        String testDir = "test_directory";
        String subDir = "test_directory/sub_dir";

        try {
            // 创建目录
            Path dirPath = Paths.get(testDir);
            Path subDirPath = Paths.get(subDir);

            Files.createDirectories(subDirPath);
            System.out.println("目录创建成功: " + subDir);

            // 在目录中创建文件
            for (int i = 1; i <= 3; i++) {
                String fileName = testDir + "/file" + i + ".txt";
                Files.write(Paths.get(fileName), ("文件" + i + "的内容").getBytes(StandardCharsets.UTF_8));
            }

            // 列出目录内容
            System.out.println("目录内容:");
            Files.list(dirPath).forEach(path -> {
                try {
                    String type = Files.isDirectory(path) ? "[目录]" : "[文件]";
                    long size = Files.isRegularFile(path) ? Files.size(path) : 0;
                    System.out.println("  " + type + " " + path.getFileName() +
                                     (size > 0 ? " (" + size + " 字节)" : ""));
                } catch (IOException e) {
                    System.out.println("  [错误] " + path.getFileName());
                }
            });

            // 递归删除目录
            deleteDirectoryRecursively(dirPath);
            System.out.println("目录删除完成");

        } catch (IOException e) {
            System.out.println("目录操作异常: " + e.getMessage());
        }
    }

    // 递归删除目录
    public static void deleteDirectoryRecursively(Path directory) throws IOException {
        if (Files.exists(directory)) {
            Files.walk(directory)
                 .sorted((path1, path2) -> path2.compareTo(path1)) // 先删除文件，后删除目录
                 .forEach(path -> {
                     try {
                         Files.delete(path);
                     } catch (IOException e) {
                         System.out.println("删除失败: " + path);
                     }
                 });
        }
    }

    // IO方法比较演示
    public static void demonstrateIOMethodComparison() {
        System.out.println("\n--- 不同IO方法性能比较 ---");

        String sourceFile = "io_comparison_source.txt";
        String targetFile1 = "target_traditional.txt";
        String targetFile2 = "target_nio.txt";
        String targetFile3 = "target_files.txt";

        try {
            // 创建测试文件
            createLargeTestFile(sourceFile, 10000);

            // 传统IO方式
            long startTime = System.currentTimeMillis();
            try (BufferedInputStream bis = new BufferedInputStream(new FileInputStream(sourceFile));
                 BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(targetFile1))) {

                byte[] buffer = new byte[8192];
                int bytesRead;
                while ((bytesRead = bis.read(buffer)) != -1) {
                    bos.write(buffer, 0, bytesRead);
                }
            }
            long traditionalTime = System.currentTimeMillis() - startTime;

            // NIO方式
            startTime = System.currentTimeMillis();
            try (var sourceChannel = FileChannel.open(Paths.get(sourceFile), StandardOpenOption.READ);
                 var targetChannel = FileChannel.open(Paths.get(targetFile2),
                     StandardOpenOption.CREATE, StandardOpenOption.WRITE)) {

                sourceChannel.transferTo(0, sourceChannel.size(), targetChannel);
            }
            long nioTime = System.currentTimeMillis() - startTime;

            // Files工具类方式
            startTime = System.currentTimeMillis();
            Files.copy(Paths.get(sourceFile), Paths.get(targetFile3), StandardCopyOption.REPLACE_EXISTING);
            long filesTime = System.currentTimeMillis() - startTime;

            System.out.println("IO方法性能比较:");
            System.out.println("  传统IO (BufferedStream): " + traditionalTime + "ms");
            System.out.println("  NIO (FileChannel): " + nioTime + "ms");
            System.out.println("  Files工具类: " + filesTime + "ms");

            // 清理文件
            Files.deleteIfExists(Paths.get(sourceFile));
            Files.deleteIfExists(Paths.get(targetFile1));
            Files.deleteIfExists(Paths.get(targetFile2));
            Files.deleteIfExists(Paths.get(targetFile3));

        } catch (IOException e) {
            System.out.println("IO方法比较异常: " + e.getMessage());
        }
    }

// 可序列化的Person类
class Person implements Serializable {
    private static final long serialVersionUID = 1L;

    private String name;
    private int age;
    private String profession;
    private List<String> skills;
    private transient String temporaryData; // 不会被序列化

    public Person(String name, int age, String profession) {
        this.name = name;
        this.age = age;
        this.profession = profession;
        this.skills = new ArrayList<>();
        this.temporaryData = "临时数据";
    }

    public void addSkill(String skill) {
        skills.add(skill);
    }

    @Override
    public String toString() {
        return String.format("Person{name='%s', age=%d, profession='%s', skills=%s, temporaryData='%s'}",
                           name, age, profession, skills, temporaryData);
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;

        Person person = (Person) obj;
        return age == person.age &&
               Objects.equals(name, person.name) &&
               Objects.equals(profession, person.profession) &&
               Objects.equals(skills, person.skills);
    }

    @Override
    public int hashCode() {
        return Objects.hash(name, age, profession, skills);
    }
}
