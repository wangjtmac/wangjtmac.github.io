    public static String formatMacAddress(byte[] mac) {

    private static void handleAccept(SelectionKey key, Selector selector) throws IOException {
        ServerSocketChannel serverChannel = (ServerSocketChannel) key.channel();
        SocketChannel clientChannel = serverChannel.accept();

        if (clientChannel != null) {
            clientChannel.configureBlocking(false);

    // å¤„ç†è¯»å–æ•°æ®
    private static void handleRead(SelectionKey key) throws IOException {
        SocketChannel clientChannel = (SocketChannel) key.channel();
        ByteBuffer buffer = ByteBuffer.allocate(1024);

        int bytesRead = clientChannel.read(buffer);

        if (bytesRead > 0) {
            buffer.flip();
            String message = new String(buffer.array(), 0, buffer.limit());
            System.out.println("NIOæœåŠ¡å™¨æ”¶åˆ°: " + message.trim());

            // å›å¤å®¢æˆ·ç«¯
            String response = "NIOå›å¤: " + message.toUpperCase();
            ByteBuffer responseBuffer = ByteBuffer.wrap(response.getBytes());
            clientChannel.write(responseBuffer);

        } else if (bytesRead == -1) {
            System.out.println("å®¢æˆ·ç«¯æ–­å¼€è¿æ¥");
            key.cancel();
            clientChannel.close();
        }
    }

class ConnectionLimiter {
    private final int maxConnections;
    private final Set<String> activeConnections;

    public ConnectionLimiter(int maxConnections) {
        this.maxConnections = maxConnections;
        this.activeConnections = new HashSet<>();
    }

    public synchronized boolean allowConnection(String clientAddress) {
        if (activeConnections.size() >= maxConnections) {
            return false;
        }

        activeConnections.add(clientAddress);
        return true;
    }

    public synchronized void releaseConnection(String clientAddress) {
        activeConnections.remove(clientAddress);
    }
}

class SimpleConnectionPool {
    private final Queue<Socket> availableConnections;
    private final Set<Socket> usedConnections;
    private final int maxSize;

    public SimpleConnectionPool(int maxSize) {
        this.maxSize = maxSize;
        this.availableConnections = new LinkedList<>();
        this.usedConnections = new HashSet<>();
    }

    public synchronized Socket getConnection(String host, int port) throws IOException {
        if (!availableConnections.isEmpty()) {
            Socket connection = availableConnections.poll();
            usedConnections.add(connection);
            return connection;
        }

        if (usedConnections.size() < maxSize) {
            Socket connection = new Socket(); // åˆ›å»ºæ–°è¿æ¥ï¼ˆç¤ºä¾‹ä¸­ä¸å®é™…è¿æ¥ï¼‰
            usedConnections.add(connection);
            return connection;
        }

        throw new IOException("è¿æ¥æ± å·²æ»¡");
    }

    public synchronized void releaseConnection(Socket connection) {
        if (usedConnections.remove(connection)) {
            if (!connection.isClosed()) {
                availableConnections.offer(connection);
            }
        }
    }

    public synchronized void close() {
        for (Socket connection : availableConnections) {
            try {
                connection.close();
            } catch (IOException e) {
                // å¿½ç•¥å…³é—­å¼‚å¸¸
            }
        }

        for (Socket connection : usedConnections) {
            try {
                connection.close();
            } catch (IOException e) {
                // å¿½ç•¥å…³é—­å¼‚å¸¸
            }
        }

        availableConnections.clear();
        usedConnections.clear();
    }
