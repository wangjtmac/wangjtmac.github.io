Java网络编程详解 ​

📚 概述 ​

Java网络编程是Java平台的重要组成部分，提供了丰富的API来支持网络通信。从最基础的Socket编程到高级的NIO和异步IO，Java为开发者提供了多层次的网络编程解决方案。网络编程涉及TCP/UDP协议、HTTP通信、WebSocket、RMI等多种技术，广泛应用于分布式系统、Web服务、即时通讯、游戏开发等领域。掌握Java网络编程不仅能够开发高性能的网络应用，更能深入理解现代互联网应用的底层通信机制。Java的网络编程API设计遵循了面向对象的原则，提供了简洁而强大的编程接口。

🎯 学习目标 ​

  • 深入理解网络编程的基本概念和协议
  • 掌握Socket编程的客户端和服务端开发
  • 熟练使用TCP和UDP进行网络通信
  • 理解HTTP协议和Web服务开发
  • 掌握NIO和异步网络编程技术
  • 学会处理网络异常和连接管理
  • 了解网络安全和性能优化
  • 掌握现代网络编程最佳实践

    // 格式化MAC地址
    public static String formatMacAddress(byte[] mac) {
        if (mac == null) return "未知";
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < mac.length; i++) {
            sb.append(String.format("%02X", mac[i]));
            if (i < mac.length - 1) {
                sb.append(":");
            }
        }
        return sb.toString();
    }

    // TCP服务器
    public static void runTCPServer(int port) {
        try (ServerSocket serverSocket = new ServerSocket(port)) {
            System.out.println("TCP服务器启动，监听端口: " + port);
            
            // 设置超时，避免无限等待
            serverSocket.setSoTimeout(10000);
            
            try (Socket clientSocket = serverSocket.accept()) {
                System.out.println("客户端连接: " + clientSocket.getRemoteSocketAddress());
                
                // 创建输入输出流
                BufferedReader reader = new BufferedReader(
                    new InputStreamReader(clientSocket.getInputStream()));
                PrintWriter writer = new PrintWriter(
                    clientSocket.getOutputStream(), true);
                
                // 读取客户端消息
                String message;
                while ((message = reader.readLine()) != null) {
                    System.out.println("服务器收到: " + message);
                    
                    // 回复客户端
                    String response = "服务器回复: " + message.toUpperCase();
                    writer.println(response);
                    
                    if ("bye".equalsIgnoreCase(message)) {
                        break;
                    }
                }
                
                System.out.println("客户端断开连接");
                
            } catch (SocketTimeoutException e) {
                System.out.println("服务器等待连接超时");
            }
            
        } catch (IOException e) {
            System.out.println("TCP服务器异常: " + e.getMessage());
        }
    }

    // UDP服务器
    public static void runUDPServer(int port) {
        try (DatagramSocket socket = new DatagramSocket(port)) {
            System.out.println("UDP服务器启动，监听端口: " + port);
            
            byte[] buffer = new byte[1024];
            
            for (int i = 0; i < 5; i++) { // 接收5个数据包
                DatagramPacket packet = new DatagramPacket(buffer, buffer.length);
                socket.receive(packet);
                
                String message = new String(packet.getData(), 0, packet.getLength());
                System.out.println("UDP服务器收到: " + message + 
                                 " (来自: " + packet.getAddress() + ":" + packet.getPort() + ")");
                
                // 回复客户端
                String response = "UDP回复: " + message.toUpperCase();
                byte[] responseData = response.getBytes();
                
                DatagramPacket responsePacket = new DatagramPacket(
                    responseData, responseData.length, 
                    packet.getAddress(), packet.getPort());
                
                socket.send(responsePacket);
            }
            
            System.out.println("UDP服务器结束");
            
        } catch (IOException e) {
            System.out.println("UDP服务器异常: " + e.getMessage());
        }
    }

    // 传统HTTP连接演示
    public static void demonstrateTraditionalHTTP() {
        System.out.println("\n--- 传统HTTP连接 ---");
        
        try {
            // 创建URL连接
            URL url = new URL("http://httpbin.org/get");
            HttpURLConnection connection = (HttpURLConnection) url.openConnection();
            
            // 设置请求属性
            connection.setRequestMethod("GET");
            connection.setRequestProperty("User-Agent", "Java-HTTP-Client");
            connection.setConnectTimeout(5000);
            connection.setReadTimeout(5000);
            
            // 获取响应
            int responseCode = connection.getResponseCode();
            System.out.println("HTTP响应码: " + responseCode);
            System.out.println("响应消息: " + connection.getResponseMessage());
            
            // 读取响应头
            System.out.println("响应头:");
            Map<String, List<String>> headers = connection.getHeaderFields();
            for (Map.Entry<String, List<String>> entry : headers.entrySet()) {
                if (entry.getKey() != null) {
                    System.out.println("  " + entry.getKey() + ": " + entry.getValue());
                }
            }
            
            // 读取响应体
            if (responseCode == HttpURLConnection.HTTP_OK) {
                try (BufferedReader reader = new BufferedReader(
                        new InputStreamReader(connection.getInputStream()))) {
                    
                    String line;
                    StringBuilder response = new StringBuilder();
                    
                    while ((line = reader.readLine()) != null) {
                        response.append(line).append("\n");
                    }
                    
                    System.out.println("响应体:");
                    System.out.println(response.toString());
                }
            }
            
            connection.disconnect();
            
        } catch (IOException e) {
            System.out.println("HTTP请求异常: " + e.getMessage());
        }
    }

    // NIO服务器演示
    public static void demonstrateNIOServer() {
        System.out.println("\n--- NIO服务器演示 ---");

        int port = 12347;

        try {
            // 创建选择器
            Selector selector = Selector.open();

            // 创建服务器套接字通道
            ServerSocketChannel serverChannel = ServerSocketChannel.open();
            serverChannel.configureBlocking(false);
            serverChannel.bind(new InetSocketAddress(port));

            // 注册接受连接事件
            serverChannel.register(selector, SelectionKey.OP_ACCEPT);

            System.out.println("NIO服务器启动，监听端口: " + port);

            // 启动客户端测试
            Thread clientThread = new Thread(() -> testNIOClient("localhost", port));
            clientThread.start();

            // 服务器事件循环
            long startTime = System.currentTimeMillis();
            while (System.currentTimeMillis() - startTime < 10000) { // 运行10秒

                if (selector.select(1000) > 0) {
                    Set<SelectionKey> selectedKeys = selector.selectedKeys();
                    Iterator<SelectionKey> iterator = selectedKeys.iterator();

                    while (iterator.hasNext()) {
                        SelectionKey key = iterator.next();
                        iterator.remove();

                        if (key.isAcceptable()) {
                            handleAccept(key, selector);
                        } else if (key.isReadable()) {
                            handleRead(key);
                        }
                    }
                }
            }

            // 清理资源
            selector.close();
            serverChannel.close();

            // 等待客户端线程结束
            clientThread.join(2000);

        } catch (IOException | InterruptedException e) {
            System.out.println("NIO服务器异常: " + e.getMessage());
        }
    }

    // 处理连接接受
    private static void handleAccept(SelectionKey key, Selector selector) throws IOException {
        ServerSocketChannel serverChannel = (ServerSocketChannel) key.channel();
        SocketChannel clientChannel = serverChannel.accept();

        if (clientChannel != null) {
            clientChannel.configureBlocking(false);
            clientChannel.register(selector, SelectionKey.OP_READ);

            System.out.println("NIO服务器接受连接: " + clientChannel.getRemoteAddress());
        }
    }

    // 处理读取数据
    private static void handleRead(SelectionKey key) throws IOException {
        SocketChannel clientChannel = (SocketChannel) key.channel();
        ByteBuffer buffer = ByteBuffer.allocate(1024);

        int bytesRead = clientChannel.read(buffer);

        if (bytesRead > 0) {
            buffer.flip();
            String message = new String(buffer.array(), 0, buffer.limit());
            System.out.println("NIO服务器收到: " + message.trim());

            // 回复客户端
            String response = "NIO回复: " + message.toUpperCase();
            ByteBuffer responseBuffer = ByteBuffer.wrap(response.getBytes());
            clientChannel.write(responseBuffer);

        } else if (bytesRead == -1) {
            System.out.println("客户端断开连接");
            key.cancel();
            clientChannel.close();
        }
    }

    public static void testNIOClient(String host, int port) {

        }
    }

    // SSL连接演示
    public static void demonstrateSSLConnection() {
        System.out.println("\n--- SSL连接演示 ---");

        try {
            // 创建SSL连接到HTTPS网站
            URL url = new URL("https://httpbin.org/get");
            HttpsURLConnection connection = (HttpsURLConnection) url.openConnection();

            // 设置SSL相关属性
            connection.setRequestMethod("GET");
            connection.setRequestProperty("User-Agent", "Java-HTTPS-Client");
            connection.setConnectTimeout(5000);
            connection.setReadTimeout(5000);

            // 获取SSL信息
            connection.connect();

            System.out.println("SSL连接信息:");
            System.out.println("  协议: " + connection.getCipherSuite());
            System.out.println("  响应码: " + connection.getResponseCode());

            // 获取证书信息
            Certificate[] certificates = connection.getServerCertificates();
            if (certificates.length > 0 && certificates[0] instanceof X509Certificate) {
                X509Certificate cert = (X509Certificate) certificates[0];
                System.out.println("  证书主题: " + cert.getSubjectDN());
                System.out.println("  证书颁发者: " + cert.getIssuerDN());
                System.out.println("  有效期: " + cert.getNotBefore() + " 到 " + cert.getNotAfter());
            }

            connection.disconnect();

        } catch (IOException e) {
            System.out.println("SSL连接异常: " + e.getMessage());
        }
    }

    // 输入验证方法
    public static boolean validateInput(String input) {
        if (input == null || input.trim().isEmpty()) {
            return false;
        }

        // 检查危险字符
        String[] dangerousPatterns = {"<script>", "../", "javascript:", "eval("};
        for (String pattern : dangerousPatterns) {
            if (input.toLowerCase().contains(pattern)) {
                return false;
            }
        }

        // 检查长度
        if (input.length() > 1000) {
            return false;
        }

        return true;
    }

            executor.submit(() -> {

    // 缓冲区优化演示
    public static void demonstrateBufferOptimization() {
        System.out.println("\n--- 缓冲区优化演示 ---");

        int[] bufferSizes = {1024, 4096, 8192, 16384};
        byte[] testData = new byte[100000]; // 100KB测试数据
        Arrays.fill(testData, (byte) 'A');

        System.out.println("缓冲区大小对性能的影响:");

        for (int bufferSize : bufferSizes) {
            long startTime = System.currentTimeMillis();

            try (ByteArrayInputStream input = new ByteArrayInputStream(testData);
                 ByteArrayOutputStream output = new ByteArrayOutputStream()) {

                byte[] buffer = new byte[bufferSize];
                int bytesRead;

                while ((bytesRead = input.read(buffer)) != -1) {
                    output.write(buffer, 0, bytesRead);
                }

                long endTime = System.currentTimeMillis();
                System.out.println("  缓冲区 " + bufferSize + " 字节: " + (endTime - startTime) + "ms");

            } catch (IOException e) {
                System.out.println("缓冲区测试异常: " + e.getMessage());
            }
        }
    }

// 连接限制器
class ConnectionLimiter {
    private final int maxConnections;
    private final Set<String> activeConnections;

    public ConnectionLimiter(int maxConnections) {
        this.maxConnections = maxConnections;
        this.activeConnections = new HashSet<>();
    }

    public synchronized boolean allowConnection(String clientAddress) {
        if (activeConnections.size() >= maxConnections) {
            return false;
        }

        activeConnections.add(clientAddress);
        return true;
    }

    public synchronized void releaseConnection(String clientAddress) {
        activeConnections.remove(clientAddress);
    }
}

// 简单连接池
class SimpleConnectionPool {
    private final Queue<Socket> availableConnections;
    private final Set<Socket> usedConnections;
    private final int maxSize;

    public SimpleConnectionPool(int maxSize) {
        this.maxSize = maxSize;
        this.availableConnections = new LinkedList<>();
        this.usedConnections = new HashSet<>();
    }

    public synchronized Socket getConnection(String host, int port) throws IOException {
        if (!availableConnections.isEmpty()) {
            Socket connection = availableConnections.poll();
            usedConnections.add(connection);
            return connection;
        }

        if (usedConnections.size() < maxSize) {
            Socket connection = new Socket(); // 创建新连接（示例中不实际连接）
            usedConnections.add(connection);
            return connection;
        }

        throw new IOException("连接池已满");
    }

    public synchronized void releaseConnection(Socket connection) {
        if (usedConnections.remove(connection)) {
            if (!connection.isClosed()) {
                availableConnections.offer(connection);
            }
        }
    }

    public synchronized void close() {
        for (Socket connection : availableConnections) {
            try {
                connection.close();
            } catch (IOException e) {
                // 忽略关闭异常
            }
        }

        for (Socket connection : usedConnections) {
            try {
                connection.close();
            } catch (IOException e) {
                // 忽略关闭异常
            }
        }

        availableConnections.clear();
        usedConnections.clear();
    }
}
