Java 并发编程基础详解 ​

📚 概述 ​

并发编程是现代 Java 开发的核心技能之一，它允许程序同时执行多个任务，充分利用多核处理器的性能。Java 提供了丰富的并发编程支持，从基础的线程机制到高级的并发工具类，从传统的 synchronized 关键字到现代的 java.util.concurrent 包。深入理解并发编程对于开发高性能、可扩展的 Java 应用程序至关重要。并发编程不仅涉及技术实现，更需要理解并发模型、线程安全、性能优化等深层概念。

🎯 学习目标 ​

  • 深入理解线程的概念、生命周期和状态转换
  • 掌握线程的创建方式和基本操作
  • 理解线程安全问题和同步机制
  • 熟练使用 synchronized 关键字和 volatile 关键字
  • 掌握 java.util.concurrent 包的核心工具类
  • 理解线程池的原理和使用方法
  • 学会使用并发集合和原子操作类
  • 了解并发编程的最佳实践和常见陷阱
  • 理解 Java 内存模型和 happens-before 关系

    // 线程状态详细演示
    public static void demonstrateThreadStates() {
        System.out.println("=== 线程状态详细演示 ===");
        
        Object lock = new Object();
        
        // 创建演示不同状态的线程
        Thread blockedThread = new Thread(() -> {
            synchronized (lock) {
                try {
                    Thread.sleep(2000);
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            }
        }, "BlockedThread");
        
        Thread waitingThread = new Thread(() -> {
            synchronized (lock) {
                try {
                    lock.wait();  // WAITING 状态
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            }
        }, "WaitingThread");
        
        Thread timedWaitingThread = new Thread(() -> {
            try {
                Thread.sleep(3000);  // TIMED_WAITING 状态
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        }, "TimedWaitingThread");
        
        // 启动线程并观察状态
        blockedThread.start();
        waitingThread.start();
        timedWaitingThread.start();
        
        try {
            Thread.sleep(100);  // 让线程开始执行
            
            System.out.println("线程状态观察:");
            System.out.println("BlockedThread 状态: " + blockedThread.getState());
            System.out.println("WaitingThread 状态: " + waitingThread.getState());
            System.out.println("TimedWaitingThread 状态: " + timedWaitingThread.getState());
            
            // 唤醒等待线程
            synchronized (lock) {
                lock.notify();
            }
            
            // 等待所有线程结束
            blockedThread.join();
            waitingThread.join();
            timedWaitingThread.join();
            
        } catch (InterruptedException e) {
            System.out.println("主线程被中断");
        }
        
        System.out.println();
    }
    
    // 辅助方法：获取状态描述
    private static String getStateDescription(Thread.State state) {
        switch (state) {
            case NEW: return "线程已创建但未启动";
            case RUNNABLE: return "线程正在运行或等待 CPU 调度";
            case BLOCKED: return "线程被阻塞等待监视器锁";
            case WAITING: return "线程无限期等待另一个线程执行特定操作";
            case TIMED_WAITING: return "线程等待另一个线程执行操作，但有时间限制";
            case TERMINATED: return "线程已退出";
            default: return "未知状态";
        }
    }
}

// 线程基本操作详细演示
public class ThreadOperationsDemo {
    
    public static void main(String[] args) {
        demonstrateThreadControl();
        demonstrateThreadInterruption();
        demonstrateThreadJoin();
        demonstrateThreadPriority();
    }
    
    // 线程控制演示
    public static void demonstrateThreadControl() {
        System.out.println("=== 线程控制演示 ===");
        
        Thread workerThread = new Thread(() -> {
            for (int i = 0; i < 5; i++) {
                System.out.println("工作线程执行: " + i);
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    System.out.println("工作线程被中断");
                    return;
                }
            }
            System.out.println("工作线程完成");
        }, "WorkerThread");
        
        System.out.println("启动工作线程");
        workerThread.start();
        
        try {
            Thread.sleep(2500);
            System.out.println("主线程继续执行其他任务");
            
            // 等待工作线程完成
            workerThread.join();
            System.out.println("工作线程已完成，主线程继续");
            
        } catch (InterruptedException e) {
            System.out.println("主线程被中断");
        }
        
        System.out.println();
    }
    
    // 线程中断演示
    public static void demonstrateThreadInterruption() {
        System.out.println("=== 线程中断演示 ===");
        
        Thread longRunningThread = new Thread(() -> {
            System.out.println("长时间运行的线程开始");
            
            while (!Thread.currentThread().isInterrupted()) {
                try {
                    System.out.println("执行任务...");
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    System.out.println("线程在睡眠时被中断");
                    // 重新设置中断状态
                    Thread.currentThread().interrupt();
                    break;
                }
            }
            
            System.out.println("长时间运行的线程结束");
        }, "LongRunningThread");
        
        longRunningThread.start();
        
        try {
            Thread.sleep(3500);
            System.out.println("主线程发送中断信号");
            longRunningThread.interrupt();
            
            longRunningThread.join();
            System.out.println("长时间运行的线程已被中断并结束");
            
        } catch (InterruptedException e) {
            System.out.println("主线程被中断");
        }
        
        System.out.println();
    }
    
    // 线程 join 演示
    public static void demonstrateThreadJoin() {
        System.out.println("=== 线程 join 演示 ===");
        
        Thread[] workers = new Thread[3];
        
        // 创建多个工作线程
        for (int i = 0; i < workers.length; i++) {
            final int workerId = i;
            workers[i] = new Thread(() -> {
                System.out.println("工作者 " + workerId + " 开始工作");
                try {
                    Thread.sleep((workerId + 1) * 1000);  // 不同的工作时间
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
                System.out.println("工作者 " + workerId + " 完成工作");
            }, "Worker-" + i);
        }
        
        // 启动所有工作线程
        System.out.println("启动所有工作线程");
        for (Thread worker : workers) {
            worker.start();
        }
        
        // 等待所有工作线程完成
        System.out.println("等待所有工作线程完成");
        try {
            for (Thread worker : workers) {
                worker.join();
            }
            System.out.println("所有工作线程已完成");
        } catch (InterruptedException e) {
            System.out.println("等待工作线程时被中断");
        }
        
        System.out.println();
    }
    
    // 线程优先级演示
    public static void demonstrateThreadPriority() {
        System.out.println("=== 线程优先级演示 ===");
        
        System.out.println("线程优先级范围: " + Thread.MIN_PRIORITY + " - " + Thread.MAX_PRIORITY);
        System.out.println("默认优先级: " + Thread.NORM_PRIORITY);
        
        Thread highPriorityThread = new Thread(() -> {
            for (int i = 0; i < 5; i++) {
                System.out.println("高优先级线程: " + i);
                Thread.yield();  // 让出 CPU 时间片
            }
        }, "HighPriorityThread");
        
        Thread lowPriorityThread = new Thread(() -> {
            for (int i = 0; i < 5; i++) {
                System.out.println("低优先级线程: " + i);
                Thread.yield();
            }
        }, "LowPriorityThread");
        
        // 设置线程优先级
        highPriorityThread.setPriority(Thread.MAX_PRIORITY);
        lowPriorityThread.setPriority(Thread.MIN_PRIORITY);
        
        System.out.println("高优先级线程优先级: " + highPriorityThread.getPriority());
        System.out.println("低优先级线程优先级: " + lowPriorityThread.getPriority());
        
        // 启动线程
        lowPriorityThread.start();
        highPriorityThread.start();
        
        try {
            highPriorityThread.join();
            lowPriorityThread.join();
        } catch (InterruptedException e) {
            System.out.println("等待线程时被中断");
        }
        
        System.out.println("注意：线程优先级只是建议，实际调度取决于操作系统");
        System.out.println();
    }
}

// 线程安全问题演示
public class ThreadSafetyDemo {

    public static void main(String[] args) {
        demonstrateRaceCondition();
        demonstrateAtomicityProblem();
        demonstrateVisibilityProblem();
        demonstrateOrderingProblem();
    }

    // 竞态条件演示
    public static void demonstrateRaceCondition() {
        System.out.println("=== 竞态条件演示 ===");

        UnsafeCounter unsafeCounter = new UnsafeCounter();
        int threadCount = 10;
        int incrementsPerThread = 1000;

        Thread[] threads = new Thread[threadCount];

        // 创建多个线程同时增加计数器
        for (int i = 0; i < threadCount; i++) {
            threads[i] = new Thread(() -> {
                for (int j = 0; j < incrementsPerThread; j++) {
                    unsafeCounter.increment();
                }
            }, "Thread-" + i);
        }

        // 启动所有线程
        long startTime = System.currentTimeMillis();
        for (Thread thread : threads) {
            thread.start();
        }

        // 等待所有线程完成
        try {
            for (Thread thread : threads) {
                thread.join();
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }

        long endTime = System.currentTimeMillis();

        System.out.println("期望结果: " + (threadCount * incrementsPerThread));
        System.out.println("实际结果: " + unsafeCounter.getCount());
        System.out.println("执行时间: " + (endTime - startTime) + "ms");
        System.out.println("数据丢失: " + (threadCount * incrementsPerThread - unsafeCounter.getCount()));

        System.out.println();
    }

    // 原子性问题演示
    public static void demonstrateAtomicityProblem() {
        System.out.println("=== 原子性问题演示 ===");

        /*
         * 原子性问题：
         * count++ 操作实际包含三个步骤：
         * 1. 读取 count 的值
         * 2. 将值加 1
         * 3. 将结果写回 count
         * 这三个步骤不是原子的，可能被其他线程中断
         */

        System.out.println("count++ 操作的字节码分析:");
        System.out.println("1. getfield    #2  // 读取字段值");
        System.out.println("2. iconst_1        // 加载常量 1");
        System.out.println("3. iadd            // 执行加法");
        System.out.println("4. putfield   #2  // 写回字段值");
        System.out.println();

        System.out.println("多线程执行时可能的交错:");
        System.out.println("线程A: 读取 count=0");
        System.out.println("线程B: 读取 count=0");
        System.out.println("线程A: 计算 0+1=1");
        System.out.println("线程B: 计算 0+1=1");
        System.out.println("线程A: 写回 count=1");
        System.out.println("线程B: 写回 count=1");
        System.out.println("结果: 两次增加操作，但 count 只增加了 1");

        System.out.println();
    }

    // 可见性问题演示
    public static void demonstrateVisibilityProblem() {
        System.out.println("=== 可见性问题演示 ===");

        VisibilityExample example = new VisibilityExample();

        // 启动读取线程
        Thread readerThread = new Thread(() -> {
            while (!example.isFlag()) {
                // 忙等待
            }
            System.out.println("读取线程检测到 flag 变为 true");
        }, "ReaderThread");

        readerThread.start();

        // 主线程修改 flag
        try {
            Thread.sleep(1000);
            System.out.println("主线程设置 flag 为 true");
            example.setFlag(true);

            // 等待读取线程结束（可能永远不会结束）
            readerThread.join(2000);  // 最多等待 2 秒

            if (readerThread.isAlive()) {
                System.out.println("读取线程仍在运行 - 可见性问题！");
                readerThread.interrupt();
            }

        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }

        System.out.println();
    }

    // 有序性问题演示
    public static void demonstrateOrderingProblem() {
        System.out.println("=== 有序性问题演示 ===");

        /*
         * 有序性问题：
         * 编译器和处理器可能会重排序指令以优化性能
         * 这可能导致程序的执行顺序与代码顺序不一致
         */

        System.out.println("指令重排序示例:");
        System.out.println("原始代码:");
        System.out.println("  int a = 1;  // 语句1");
        System.out.println("  int b = 2;  // 语句2");
        System.out.println("  int c = a + b;  // 语句3");

        System.out.println("\n可能的重排序:");
        System.out.println("  int b = 2;  // 语句2");
        System.out.println("  int a = 1;  // 语句1");
        System.out.println("  int c = a + b;  // 语句3");

        System.out.println("\n在单线程中，重排序不会影响结果");
        System.out.println("但在多线程中，重排序可能导致意外的结果");

        System.out.println();
    }
}

    }

        synchronized (lock) {
            balance += amount;
            transactionCount++;
            System.out.println(Thread.currentThread().getName() +
                             " 存款 " + amount + "，余额: " + balance);
        }
    }

    public void withdraw(double amount) {
        synchronized (lock) {
            if (balance >= amount) {
                balance -= amount;
                transactionCount++;
                System.out.println(Thread.currentThread().getName() +
                                 " 取款 " + amount + "，余额: " + balance);
            } else {
                System.out.println(Thread.currentThread().getName() +
                                 " 取款失败，余额不足");
            }
        }
    }

    public synchronized double getBalance() {
        return balance;
    }

    public synchronized int getTransactionCount() {
        return transactionCount;
    }

        }, "ReaderThread");

            final int taskId = i;
            try {
                executor.submit(() -> {
                    System.out.println("自定义池任务 " + taskId + " 开始 - " +
                                     Thread.currentThread().getName());
                    try {
                        Thread.sleep(2000);  // 模拟长时间任务
                    } catch (InterruptedException e) {
                        Thread.currentThread().interrupt();
                    }
                    System.out.println("自定义池任务 " + taskId + " 完成 - " +
                                     Thread.currentThread().getName());
                });
            } catch (RejectedExecutionException e) {
                System.out.println("任务 " + taskId + " 被拒绝执行");
            }

            // 控制提交速度
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        }

        // 关闭线程池
        executor.shutdown();
        try {
            if (!executor.awaitTermination(15, TimeUnit.SECONDS)) {
                executor.shutdownNow();
            }
        } catch (InterruptedException e) {
            executor.shutdownNow();
        }

        System.out.println("自定义线程池演示完成\n");
    }
}

// ThreadLocal 使用示例
class ThreadLocalExample {
    private static final ThreadLocal<String> userContext = new ThreadLocal<>();

    public void setUserId(String userId) {
        userContext.set(userId);
        System.out.println("设置用户上下文: " + userId + " - " +
                         Thread.currentThread().getName());
    }

    public void processRequest() {
        String userId = userContext.get();
        System.out.println("处理请求，用户: " + userId + " - " +
                         Thread.currentThread().getName());

        // 模拟业务处理
        try {
            Thread.sleep(100);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }

    public void cleanup() {
        userContext.remove();  // 重要：避免内存泄漏
        System.out.println("清理用户上下文 - " + Thread.currentThread().getName());
    }
}

// 正确的锁使用示例
class ProperLockingExample {
    private final ReadWriteLock lock = new ReentrantReadWriteLock();
    private final Lock readLock = lock.readLock();
    private final Lock writeLock = lock.writeLock();
    private int data = 0;

    public void readOperation() {
        readLock.lock();
        try {
            System.out.println("读取数据: " + data + " - " +
                             Thread.currentThread().getName());
            Thread.sleep(100);  // 模拟读取时间
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        } finally {
            readLock.unlock();  // 确保在 finally 中释放锁
        }
    }

    public void writeOperation(int newValue) {
        writeLock.lock();
        try {
            System.out.println("写入数据: " + newValue + " - " +
                             Thread.currentThread().getName());
            data = newValue;
            Thread.sleep(200);  // 模拟写入时间
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        } finally {
            writeLock.unlock();  // 确保在 finally 中释放锁
        }
    }
}

💡 最佳实践总结 ​

设计原则 ​

  1. 优先使用不可变对象：不可变对象天然线程安全
  2. 最小化共享状态：减少线程间的数据共享
  3. 使用线程安全的工具类：优先使用 java.util.concurrent 包
  4. 合理的锁粒度：避免过粗或过细的锁

同步策略 ​

  1. 优先级顺序：不可变 > ThreadLocal > 无锁 > 锁
  2. 锁的选择：读多写少用读写锁，简单场景用 synchronized
  3. 原子操作：简单的数值操作使用原子类
  4. 批量操作：减少锁的获取和释放次数

性能优化 ​

  1. 减少上下文切换：合理设置线程数量
  2. 避免锁竞争：使用分段锁、无锁算法
  3. 合理使用线程池：避免频繁创建销毁线程
  4. 监控和调优：使用工具监控线程状态

错误处理 ​

  1. 正确处理中断：响应 InterruptedException
  2. 资源清理：使用 try-finally 确保资源释放
  3. 避免死锁：按固定顺序获取锁，使用超时
  4. 异常传播：合理处理线程中的异常
