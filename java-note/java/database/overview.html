Java数据库编程详解 ​

📚 概述 ​

Java数据库编程是企业级应用开发的核心技能之一，涉及与各种数据库系统的交互和数据管理。从传统的JDBC API到现代的ORM框架，Java提供了丰富的数据库访问技术。数据库编程不仅包括基本的CRUD操作，还涉及事务管理、连接池优化、SQL注入防护、性能调优等高级主题。掌握Java数据库编程对于开发高性能、高可靠性的企业应用至关重要。本文将从JDBC基础开始，逐步深入到连接池、事务处理、ORM框架等高级主题，帮助开发者建立完整的数据库编程知识体系。

🎯 学习目标 ​

  • 深入理解JDBC API的核心概念和使用方法
  • 掌握数据库连接管理和连接池技术
  • 熟练进行CRUD操作和复杂查询
  • 理解事务管理和并发控制机制
  • 学会防范SQL注入等安全问题
  • 掌握数据库性能优化技巧
  • 了解ORM框架的基本原理和使用
  • 学会处理大数据量和批处理操作

                System.out.println("驱动版本: " + connection.getMetaData().getDriverVersion());

                System.out.println("事务隔离级别: " + getTransactionIsolationName(connection.getTransactionIsolation()));

            System.out.println("数据库驱动未找到: " + e.getMessage());

    // 获取事务隔离级别名称
    public static String getTransactionIsolationName(int level) {
        switch (level) {
            case Connection.TRANSACTION_NONE: return "NONE";
            case Connection.TRANSACTION_READ_UNCOMMITTED: return "READ_UNCOMMITTED";
            case Connection.TRANSACTION_READ_COMMITTED: return "READ_COMMITTED";
            case Connection.TRANSACTION_REPEATABLE_READ: return "REPEATABLE_READ";
            case Connection.TRANSACTION_SERIALIZABLE: return "SERIALIZABLE";
            default: return "UNKNOWN";
        }
    }

    // 创建用户表
    public static void createUserTable(Connection connection) throws SQLException {
        String createTableSQL = """
            CREATE TABLE IF NOT EXISTS users (
                id INT PRIMARY KEY AUTO_INCREMENT,
                username VARCHAR(50) NOT NULL UNIQUE,
                email VARCHAR(100) NOT NULL,
                age INT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """;
        
        try (Statement statement = connection.createStatement()) {
            statement.execute(createTableSQL);
            System.out.println("\n用户表创建成功");
        }
    }
    
    // 插入用户数据
    public static void insertUsers(Connection connection) throws SQLException {
        String insertSQL = "INSERT INTO users (username, email, age) VALUES (?, ?, ?)";
        
        try (PreparedStatement pstmt = connection.prepareStatement(insertSQL)) {
            
            // 插入多个用户
            String[][] users = {
                {"alice", "alice@example.com", "25"},
                {"bob", "bob@example.com", "30"},
                {"charlie", "charlie@example.com", "28"},
                {"diana", "diana@example.com", "22"}
            };
            
            for (String[] user : users) {
                pstmt.setString(1, user[0]);
                pstmt.setString(2, user[1]);
                pstmt.setInt(3, Integer.parseInt(user[2]));
                
                int rowsAffected = pstmt.executeUpdate();
                System.out.println("插入用户 " + user[0] + ": " + rowsAffected + " 行受影响");
            }
        }
    }

             ResultSet resultSet = statement.executeQuery(querySQL)) {

    public static void demonstrateTransactionManagement() {
        System.out.println("=== 事务管理演示 ===");
        
        /*
         * 事务特性（ACID）：
         * 1. Atomicity - 原子性：事务中的操作要么全部成功，要么全部失败

         * 3. Isolation - 隔离性：并发事务之间相互隔离
         * 4. Durability - 持久性：事务提交后数据永久保存
         */
        
        System.out.println("事务ACID特性:");
        System.out.println("1. Atomicity - 原子性");
        System.out.println("2. Consistency - 一致性");
        System.out.println("3. Isolation - 隔离性");
        System.out.println("4. Durability - 持久性");
        
        // 成功事务演示
        demonstrateSuccessfulTransaction();
        
        // 失败事务回滚演示
        demonstrateTransactionRollback();
        
        System.out.println();

    // 成功事务演示
    public static void demonstrateSuccessfulTransaction() {
        System.out.println("\n--- 成功事务演示 ---");
        
        try (Connection connection = DriverManager.getConnection(DB_URL, DB_USER, DB_PASSWORD)) {
            
            // 关闭自动提交
            connection.setAutoCommit(false);
            
            try {
                // 创建账户表
                createAccountTable(connection);
                
                // 初始化账户
                insertAccount(connection, "account1", 1000.00);
                insertAccount(connection, "account2", 500.00);
                
                System.out.println("转账前账户余额:");
                queryAccounts(connection);
                
                // 执行转账操作
                transferMoney(connection, "account1", "account2", 200.00);
                
                // 提交事务
                connection.commit();
                System.out.println("事务提交成功");
                
                System.out.println("转账后账户余额:");
                queryAccounts(connection);
                
            } catch (SQLException e) {
                // 回滚事务
                connection.rollback();
                System.out.println("事务回滚: " + e.getMessage());
            } finally {
                // 恢复自动提交
                connection.setAutoCommit(true);
            }
            
        } catch (SQLException e) {
            System.out.println("事务管理异常: " + e.getMessage());
        }
    }
    
    // 失败事务回滚演示
    public static void demonstrateTransactionRollback() {
        System.out.println("\n--- 失败事务回滚演示 ---");
        
        try (Connection connection = DriverManager.getConnection(DB_URL, DB_USER, DB_PASSWORD)) {
            
            connection.setAutoCommit(false);
            
            try {
                System.out.println("尝试无效转账前账户余额:");
                queryAccounts(connection);
                
                // 尝试转账超过余额的金额（会失败）
                transferMoney(connection, "account2", "account1", 1000.00);
                
                connection.commit();
                
            } catch (SQLException e) {
                connection.rollback();
                System.out.println("事务回滚: " + e.getMessage());
                
                System.out.println("回滚后账户余额:");
                queryAccounts(connection);
            } finally {
                connection.setAutoCommit(true);
            }
            
        } catch (SQLException e) {
            System.out.println("事务回滚演示异常: " + e.getMessage());
        }
    }

    // 转账操作
    public static void transferMoney(Connection connection, String fromAccount, String toAccount, double amount) throws SQLException {
        // 检查余额
        String checkBalanceSQL = "SELECT balance FROM accounts WHERE account_id = ?";
        try (PreparedStatement pstmt = connection.prepareStatement(checkBalanceSQL)) {
            pstmt.setString(1, fromAccount);
            try (ResultSet rs = pstmt.executeQuery()) {
                if (rs.next()) {
                    double balance = rs.getDouble("balance");
                    if (balance < amount) {
                        throw new SQLException("余额不足: 当前余额 " + balance + ", 转账金额 " + amount);
                    }
                } else {
                    throw new SQLException("账户不存在: " + fromAccount);
                }
            }
        }
        
        // 扣款
        String debitSQL = "UPDATE accounts SET balance = balance - ? WHERE account_id = ?";
        try (PreparedStatement pstmt = connection.prepareStatement(debitSQL)) {
            pstmt.setDouble(1, amount);
            pstmt.setString(2, fromAccount);
            pstmt.executeUpdate();
        }
        
        // 入账
        String creditSQL = "UPDATE accounts SET balance = balance + ? WHERE account_id = ?";
        try (PreparedStatement pstmt = connection.prepareStatement(creditSQL)) {
            pstmt.setDouble(1, amount);
            pstmt.setString(2, toAccount);
            pstmt.executeUpdate();
        }
        
        System.out.println("转账成功: " + fromAccount + " -> " + toAccount + ", 金额: " + amount);
    }

    // 简单连接池演示
    public static void demonstrateSimpleConnectionPool() {
        System.out.println("\n--- 简单连接池演示 ---");
        
        SimpleConnectionPool pool = new SimpleConnectionPool(DB_URL, DB_USER, DB_PASSWORD, 3);
        
        // 模拟多个线程使用连接池
        ExecutorService executor = Executors.newFixedThreadPool(5);
        
        for (int i = 1; i <= 5; i++) {
            final int taskId = i;
            executor.submit(() -> {
                try {
                    Connection connection = pool.getConnection();
                    System.out.println("任务" + taskId + " 获得连接: " + connection.hashCode());
                    
                    // 模拟数据库操作
                    try (Statement stmt = connection.createStatement();
                         ResultSet rs = stmt.executeQuery("SELECT 1")) {
                        if (rs.next()) {
                            System.out.println("任务" + taskId + " 执行查询成功");
                        }
                    }
                    
                    // 模拟操作时间
                    Thread.sleep(2000);
                    
                    pool.releaseConnection(connection);
                    System.out.println("任务" + taskId + " 释放连接");
                    
                } catch (Exception e) {
                    System.out.println("任务" + taskId + " 异常: " + e.getMessage());
                }
            });
        }
        
        executor.shutdown();
        try {
            executor.awaitTermination(15, TimeUnit.SECONDS);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        pool.close();
        System.out.println("连接池已关闭");
    }
}

// 简单连接池实现
class SimpleConnectionPool {
    private final String url;
    private final String username;
    private final String password;
    private final Queue<Connection> availableConnections;
    private final Set<Connection> usedConnections;
    private final int maxSize;
    
    public SimpleConnectionPool(String url, String username, String password, int maxSize) {
        this.url = url;
        this.username = username;
        this.password = password;
        this.maxSize = maxSize;
        this.availableConnections = new LinkedList<>();
        this.usedConnections = new HashSet<>();
        
        // 初始化连接池
        initializePool();
    }
    
    private void initializePool() {
        for (int i = 0; i < maxSize; i++) {
            try {
                Connection connection = DriverManager.getConnection(url, username, password);
                availableConnections.offer(connection);
            } catch (SQLException e) {
                System.out.println("初始化连接池失败: " + e.getMessage());
            }
        }
    }
    
    public synchronized Connection getConnection() throws SQLException {
        if (availableConnections.isEmpty()) {
            if (usedConnections.size() < maxSize) {
                Connection connection = DriverManager.getConnection(url, username, password);
                usedConnections.add(connection);
                return connection;
            } else {
                throw new SQLException("连接池已满，无可用连接");
            }
        }
        
        Connection connection = availableConnections.poll();
        usedConnections.add(connection);
        return connection;
    }
    
    public synchronized void releaseConnection(Connection connection) {
        if (usedConnections.remove(connection)) {
            if (!isConnectionValid(connection)) {
                try {
                    connection.close();
                    Connection newConnection = DriverManager.getConnection(url, username, password);
                    availableConnections.offer(newConnection);
                } catch (SQLException e) {
                    System.out.println("重新创建连接失败: " + e.getMessage());
                }
            } else {
                availableConnections.offer(connection);
            }
        }
    }
    
    private boolean isConnectionValid(Connection connection) {
        try {
            return connection != null && !connection.isClosed() && connection.isValid(1);
        } catch (SQLException e) {
            return false;
        }
    }
    
    public synchronized void close() {
        for (Connection connection : availableConnections) {
            try {
                connection.close();
            } catch (SQLException e) {
                System.out.println("关闭连接异常: " + e.getMessage());
            }
        }
        
        for (Connection connection : usedConnections) {
            try {
                connection.close();
            } catch (SQLException e) {
                System.out.println("关闭连接异常: " + e.getMessage());
            }
        }
        
        availableConnections.clear();
        usedConnections.clear();
    }
}
