Java 数组详解 ​

📚 概述 ​

数组是 Java 中最基本的数据结构之一，用于存储相同类型的多个元素。数组在内存中是连续存储的，可以通过索引快速访问元素。掌握数组的使用是 Java 编程的基础技能。

🎯 学习目标 ​

  • 理解数组的概念和特点
  • 掌握一维数组和多维数组的声明、创建和初始化
  • 熟练使用数组的各种操作方法
  • 了解数组的内存结构和性能特点
  • 掌握数组的常见应用场景和最佳实践

📊 数组基础概念 ​

数组的特点 ​

// 数组的主要特点：
// 1. 存储相同类型的元素
// 2. 长度固定，创建后不能改变
// 3. 元素在内存中连续存储
// 4. 通过索引访问元素（从0开始）

        // 数组是引用类型

        System.out.println(arr1[0]);  // 输出 100

        System.out.println("数组长度: " + arr1.length);  // 3
        // arr1.length = 5;  // 编译错误，length 是 final 的
    }
}

public class ArrayAccess {
    public static void main(String[] args) {
        int[] numbers = {10, 20, 30, 40, 50};
        
        // 访问元素
        System.out.println("第一个元素: " + numbers[0]);    // 10
        System.out.println("最后一个元素: " + numbers[4]);   // 50
        System.out.println("最后一个元素: " + numbers[numbers.length - 1]);  // 50
        
        // 修改元素
        numbers[0] = 100;
        numbers[numbers.length - 1] = 500;
        
        // 遍历数组
        System.out.println("修改后的数组:");
        for (int i = 0; i < numbers.length; i++) {
            System.out.println("numbers[" + i + "] = " + numbers[i]);
        }
        
        // 增强for循环（推荐用于只读遍历）
        System.out.println("使用增强for循环:");
        for (int num : numbers) {
            System.out.print(num + " ");
        }
        System.out.println();
        
        // 数组越界异常
        try {
            System.out.println(numbers[10]);  // ArrayIndexOutOfBoundsException
        } catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("数组越界: " + e.getMessage());
        }
        
        // 安全访问
        int index = 10;
        if (index >= 0 && index < numbers.length) {
            System.out.println(numbers[index]);
        } else {
            System.out.println("索引超出范围");
        }
    }
}

import java.util.Arrays;

public class ArrayCopy {
    public static void main(String[] args) {
        int[] original = {1, 2, 3, 4, 5};
        
        // 方式1：使用循环复制
        int[] copy1 = new int[original.length];
        for (int i = 0; i < original.length; i++) {
            copy1[i] = original[i];
        }
        
        // 方式2：使用 System.arraycopy()
        int[] copy2 = new int[original.length];
        System.arraycopy(original, 0, copy2, 0, original.length);
        
        // 方式3：使用 Arrays.copyOf()
        int[] copy3 = Arrays.copyOf(original, original.length);
        
        // 方式4：使用 Arrays.copyOfRange()
        int[] copy4 = Arrays.copyOfRange(original, 1, 4);  // 复制索引1到3的元素
        
        // 方式5：使用 clone() 方法
        int[] copy5 = original.clone();
        
        // 验证复制结果
        System.out.println("原数组: " + Arrays.toString(original));
        System.out.println("copy1: " + Arrays.toString(copy1));
        System.out.println("copy2: " + Arrays.toString(copy2));
        System.out.println("copy3: " + Arrays.toString(copy3));
        System.out.println("copy4: " + Arrays.toString(copy4));
        System.out.println("copy5: " + Arrays.toString(copy5));
        
        // 浅拷贝 vs 深拷贝
        String[] names = {"Alice", "Bob", "Charlie"};
        String[] namesCopy = names.clone();  // 浅拷贝
        
        namesCopy[0] = "David";  // 修改副本
        System.out.println("原数组: " + Arrays.toString(names));      // Alice, Bob, Charlie
        System.out.println("副本: " + Arrays.toString(namesCopy));    // David, Bob, Charlie
        
        // 对象数组的深拷贝需要特殊处理
        Person[] people = {new Person("Alice"), new Person("Bob")};
        Person[] peopleCopy = new Person[people.length];
        for (int i = 0; i < people.length; i++) {
            peopleCopy[i] = new Person(people[i].name);  // 创建新对象
        }
    }
}

2. 数组排序和搜索 ​

java
import java.util.Arrays;

public class ArraySortAndSearch {
    public static void main(String[] args) {
        int[] numbers = {64, 34, 25, 12, 22, 11, 90};
        
        // 使用 Arrays.sort() 排序
        int[] sortedNumbers = numbers.clone();
        Arrays.sort(sortedNumbers);
        System.out.println("排序后: " + Arrays.toString(sortedNumbers));
        
        // 二分查找（数组必须已排序）
        int target = 25;
        int index = Arrays.binarySearch(sortedNumbers, target);
        System.out.println(target + " 的索引: " + index);
        
        // 手动实现冒泡排序
        int[] bubbleSort = numbers.clone();
        bubbleSortArray(bubbleSort);
        System.out.println("冒泡排序: " + Arrays.toString(bubbleSort));
        
        // 手动实现线性查找
        int linearIndex = linearSearch(numbers, target);
        System.out.println("线性查找 " + target + " 的索引: " + linearIndex);
        
        // 字符串数组排序
        String[] names = {"Charlie", "Alice", "Bob", "David"};
        Arrays.sort(names);
        System.out.println("排序后的名字: " + Arrays.toString(names));
        
        // 自定义排序（降序）
        Integer[] nums = {64, 34, 25, 12, 22, 11, 90};
        Arrays.sort(nums, (a, b) -> b - a);  // 降序排序
        System.out.println("降序排序: " + Arrays.toString(nums));
    }
    
    // 冒泡排序实现
    public static void bubbleSortArray(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {
                    // 交换元素
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
    }
    
    // 线性查找实现
    public static int linearSearch(int[] arr, int target) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                return i;
            }
        }
        return -1;  // 未找到
    }
}

import java.util.Arrays;
import java.util.List;

public class ArraysUtility {
    public static void main(String[] args) {
        int[] numbers = {3, 1, 4, 1, 5, 9, 2, 6};
        
        // toString() - 转换为字符串
        System.out.println("数组内容: " + Arrays.toString(numbers));
        
        // sort() - 排序
        Arrays.sort(numbers);
        System.out.println("排序后: " + Arrays.toString(numbers));
        
        // binarySearch() - 二分查找
        int index = Arrays.binarySearch(numbers, 5);
        System.out.println("5的索引: " + index);
        
        // fill() - 填充数组
        int[] filled = new int[5];
        Arrays.fill(filled, 42);
        System.out.println("填充数组: " + Arrays.toString(filled));
        
        // equals() - 比较数组
        int[] arr1 = {1, 2, 3};
        int[] arr2 = {1, 2, 3};
        int[] arr3 = {1, 2, 4};
        System.out.println("arr1 equals arr2: " + Arrays.equals(arr1, arr2));  // true
        System.out.println("arr1 equals arr3: " + Arrays.equals(arr1, arr3));  // false
        
        // deepEquals() - 深度比较多维数组
        int[][] matrix1 = {{1, 2}, {3, 4}};
        int[][] matrix2 = {{1, 2}, {3, 4}};
        System.out.println("矩阵相等: " + Arrays.deepEquals(matrix1, matrix2));  // true
        
        // deepToString() - 多维数组转字符串
        System.out.println("矩阵内容: " + Arrays.deepToString(matrix1));
        
        // asList() - 数组转List
        String[] names = {"Alice", "Bob", "Charlie"};
        List<String> nameList = Arrays.asList(names);
        System.out.println("转换为List: " + nameList);
        
        // 注意：asList返回的List是固定大小的
        try {
            nameList.add("David");  // UnsupportedOperationException
        } catch (UnsupportedOperationException e) {
            System.out.println("不能向固定大小的List添加元素");
        }
        
        // copyOf() - 复制数组
        int[] original = {1, 2, 3, 4, 5};
        int[] copy = Arrays.copyOf(original, 3);  // 复制前3个元素
        System.out.println("复制数组: " + Arrays.toString(copy));
        
        // copyOfRange() - 复制指定范围
        int[] range = Arrays.copyOfRange(original, 1, 4);  // 复制索引1到3
        System.out.println("范围复制: " + Arrays.toString(range));
        
        // stream() - 转换为流（Java 8+）
        int sum = Arrays.stream(numbers).sum();
        System.out.println("数组元素和: " + sum);
        
        double average = Arrays.stream(numbers).average().orElse(0.0);
        System.out.println("平均值: " + average);
    }
}

示例1：学生成绩管理系统 ​

java
import java.util.Arrays;
import java.util.Scanner;

public class StudentGradeManager {
    private String[] studentNames;
    private double[] grades;
    private int studentCount;
    
    public StudentGradeManager(int capacity) {
        studentNames = new String[capacity];
        grades = new double[capacity];
        studentCount = 0;
    }
    
    // 添加学生成绩
    public boolean addStudent(String name, double grade) {
        if (studentCount >= studentNames.length) {
            System.out.println("学生数量已达上限");
            return false;
        }
        
        studentNames[studentCount] = name;
        grades[studentCount] = grade;
        studentCount++;
        return true;
    }
    
    // 计算平均分
    public double calculateAverage() {
        if (studentCount == 0) return 0.0;
        
        double sum = 0;
        for (int i = 0; i < studentCount; i++) {
            sum += grades[i];
        }
        return sum / studentCount;
    }
    
    // 找到最高分
    public double findMaxGrade() {
        if (studentCount == 0) return 0.0;
        
        double max = grades[0];
        for (int i = 1; i < studentCount; i++) {
            if (grades[i] > max) {
                max = grades[i];
            }
        }
        return max;
    }
    
    // 找到最低分
    public double findMinGrade() {
        if (studentCount == 0) return 0.0;
        
        double min = grades[0];
        for (int i = 1; i < studentCount; i++) {
            if (grades[i] < min) {
                min = grades[i];
            }
        }
        return min;
    }
    
    // 按成绩排序（冒泡排序）
    public void sortByGrade() {
        for (int i = 0; i < studentCount - 1; i++) {
            for (int j = 0; j < studentCount - i - 1; j++) {
                if (grades[j] > grades[j + 1]) {
                    // 交换成绩
                    double tempGrade = grades[j];
                    grades[j] = grades[j + 1];
                    grades[j + 1] = tempGrade;
                    
                    // 交换姓名
                    String tempName = studentNames[j];
                    studentNames[j] = studentNames[j + 1];
                    studentNames[j + 1] = tempName;
                }
            }
        }
    }
    
    // 查找学生
    public int findStudent(String name) {
        for (int i = 0; i < studentCount; i++) {
            if (studentNames[i].equals(name)) {
                return i;
            }
        }
        return -1;
    }
    
    // 显示所有学生信息
    public void displayAllStudents() {
        System.out.println("\n=== 学生成绩单 ===");
        for (int i = 0; i < studentCount; i++) {
            System.out.printf("%-10s: %.2f\n", studentNames[i], grades[i]);
        }
        System.out.printf("\n平均分: %.2f\n", calculateAverage());
        System.out.printf("最高分: %.2f\n", findMaxGrade());
        System.out.printf("最低分: %.2f\n", findMinGrade());
    }
    
    public static void main(String[] args) {
        StudentGradeManager manager = new StudentGradeManager(10);
        Scanner scanner = new Scanner(System.in);
        
        // 添加示例数据
        manager.addStudent("张三", 85.5);
        manager.addStudent("李四", 92.0);
        manager.addStudent("王五", 78.5);
        manager.addStudent("赵六", 88.0);
        
        while (true) {
            System.out.println("\n=== 学生成绩管理系统 ===");
            System.out.println("1. 添加学生");
            System.out.println("2. 查找学生");
            System.out.println("3. 显示所有学生");
            System.out.println("4. 按成绩排序");
            System.out.println("5. 统计信息");
            System.out.println("0. 退出");
            System.out.print("请选择: ");
            
            int choice = scanner.nextInt();
            scanner.nextLine(); // 消费换行符
            
            switch (choice) {
                case 1:
                    System.out.print("请输入学生姓名: ");
                    String name = scanner.nextLine();
                    System.out.print("请输入成绩: ");
                    double grade = scanner.nextDouble();
                    if (manager.addStudent(name, grade)) {
                        System.out.println("添加成功");
                    }
                    break;
                    
                case 2:
                    System.out.print("请输入要查找的学生姓名: ");
                    String searchName = scanner.nextLine();
                    int index = manager.findStudent(searchName);
                    if (index != -1) {
                        System.out.printf("找到学生: %s, 成绩: %.2f\n", 
                            manager.studentNames[index], manager.grades[index]);
                    } else {
                        System.out.println("未找到该学生");
                    }
                    break;
                    
                case 3:
                    manager.displayAllStudents();
                    break;
                    
                case 4:
                    manager.sortByGrade();
                    System.out.println("排序完成");
                    manager.displayAllStudents();
                    break;
                    
                case 5:
                    System.out.printf("学生总数: %d\n", manager.studentCount);
                    System.out.printf("平均分: %.2f\n", manager.calculateAverage());
                    System.out.printf("最高分: %.2f\n", manager.findMaxGrade());
                    System.out.printf("最低分: %.2f\n", manager.findMinGrade());
                    break;
                    
                case 0:
                    System.out.println("程序结束");
                    scanner.close();
                    return;
                    
                default:
                    System.out.println("无效选择");
            }
        }
    }
}

1. 数组使用注意事项 ​

java
public class ArrayBestPractices {
    public static void main(String[] args) {
        // 1. 避免数组越界
        int[] arr = {1, 2, 3, 4, 5};
        
        // 错误的做法
        // System.out.println(arr[5]);  // ArrayIndexOutOfBoundsException
        
        // 正确的做法
        int index = 5;
        if (index >= 0 && index < arr.length) {
            System.out.println(arr[index]);
        } else {
            System.out.println("索引超出范围");
        }
        
        // 2. 初始化时注意默认值
        int[] numbers = new int[3];     // {0, 0, 0}
        boolean[] flags = new boolean[3]; // {false, false, false}
        String[] names = new String[3];   // {null, null, null}
        
        // 避免空指针异常
        for (int i = 0; i < names.length; i++) {
            if (names[i] != null) {
                System.out.println(names[i].length());
            }
        }
        
        // 3. 数组复制要注意浅拷贝问题
        int[][] matrix = {{1, 2}, {3, 4}};
        int[][] copy = matrix.clone();  // 浅拷贝
        copy[0][0] = 100;  // 会影响原数组
        System.out.println("原数组第一个元素: " + matrix[0][0]);  // 100
        
        // 深拷贝的正确做法
        int[][] deepCopy = new int[matrix.length][];
        for (int i = 0; i < matrix.length; i++) {
            deepCopy[i] = matrix[i].clone();
        }
        
        // 4. 使用增强for循环的限制
        int[] nums = {1, 2, 3, 4, 5};
        
        // 不能修改数组元素
        for (int num : nums) {
            num = num * 2;  // 这不会改变原数组
        }
        
        // 不能获取索引
        // for (int num : nums) {
        //     System.out.println("索引: " + i);  // 编译错误
        // }
        
        // 5. 数组长度是final的
        // arr.length = 10;  // 编译错误
        
        // 6. 数组是对象，可以为null
        int[] nullArray = null;
        // System.out.println(nullArray.length);  // NullPointerException
        
        if (nullArray != null) {
            System.out.println(nullArray.length);
        }
    }
}

2. 性能考虑 ​

java
import java.util.ArrayList;
import java.util.List;

public class ArrayPerformance {
    public static void main(String[] args) {
        // 1. 数组 vs ArrayList 性能比较
        int size = 1000000;
        
        // 数组访问性能测试
        int[] array = new int[size];
        long startTime = System.nanoTime();
        for (int i = 0; i < size; i++) {
            array[i] = i;
        }
        long arrayTime = System.nanoTime() - startTime;
        
        // ArrayList访问性能测试
        List<Integer> list = new ArrayList<>();
        startTime = System.nanoTime();
        for (int i = 0; i < size; i++) {
            list.add(i);
        }
        long listTime = System.nanoTime() - startTime;
        
        System.out.println("数组赋值时间: " + arrayTime + " ns");
        System.out.println("ArrayList添加时间: " + listTime + " ns");
        
        // 2. 多维数组的内存布局
        // 连续的二维数组（更好的缓存性能）
        int rows = 1000, cols = 1000;
        int[][] matrix2D = new int[rows][cols];
        
        // 一维数组模拟二维（更好的性能）
        int[] matrix1D = new int[rows * cols];
        
        // 访问元素：matrix1D[i * cols + j] 等价于 matrix2D[i][j]
        
        // 3. 数组初始化的性能
        int[] arr1 = new int[1000];  // 快速，只分配内存
        int[] arr2 = {1, 2, 3, 4, 5};  // 编译时确定，快速
        
        int[] arr3 = new int[1000];
        Arrays.fill(arr3, 42);  // 需要遍历，较慢
    }
}

  3. 多维数组实际上是数组的数组
  4. 使用Arrays工具类可以简化数组操作
  5. 根据使用场景选择合适的遍历方式
  6. 注意数组复制时的浅拷贝和深拷贝问题
