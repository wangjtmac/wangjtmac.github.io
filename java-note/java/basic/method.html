Java 方法和函数详解 ​

📚 概述 ​

方法（Method）是 Java 中组织代码的基本单位，用于封装特定功能的代码块。方法可以接收参数、执行操作并返回结果。掌握方法的定义和使用是编写模块化、可重用代码的关键。

🎯 学习目标 ​

  • 理解方法的概念和作用
  • 掌握方法的定义语法和调用方式
  • 熟悉参数传递机制和返回值处理
  • 理解方法重载的概念和应用
  • 掌握递归方法的编写技巧
  • 了解方法的最佳实践和设计原则

📖 方法基础概念 ​

方法的作用 ​

java
// 方法的主要作用：
// 1. 代码重用：避免重复编写相同的代码
// 2. 模块化：将复杂问题分解为小的子问题
// 3. 抽象：隐藏实现细节，提供简洁的接口
// 4. 维护性：便于代码的修改和维护
// 5. 可读性：提高代码的可读性和理解性

public class MethodBasics {
    public static void main(String[] args) {
        // 不使用方法的代码（重复且难维护）
        System.out.println("计算圆的面积:");
        double radius1 = 5.0;
        double area1 = 3.14159 * radius1 * radius1;
        System.out.println("半径为 " + radius1 + " 的圆面积: " + area1);
        
        double radius2 = 3.0;
        double area2 = 3.14159 * radius2 * radius2;
        System.out.println("半径为 " + radius2 + " 的圆面积: " + area2);
        
        // 使用方法的代码（简洁且可重用）
        System.out.println("\n使用方法计算:");
        System.out.println("半径为 5.0 的圆面积: " + calculateCircleArea(5.0));
        System.out.println("半径为 3.0 的圆面积: " + calculateCircleArea(3.0));
    }
    
    // 计算圆面积的方法
    public static double calculateCircleArea(double radius) {
        return Math.PI * radius * radius;
    }
}

1. 方法定义语法 ​

java
public class MethodSyntax {
    // 方法定义的完整语法：
    // [访问修饰符] [其他修饰符] 返回类型 方法名(参数列表) [throws 异常列表] {
    //     方法体
    //     [return 返回值;]
    // }
    
    // 1. 无参数无返回值的方法
    public static void sayHello() {
        System.out.println("Hello, World!");
    }
    
    // 2. 有参数无返回值的方法
    public static void greetPerson(String name) {
        System.out.println("Hello, " + name + "!");
    }
    
    // 3. 无参数有返回值的方法
    public static int getCurrentYear() {
        return 2024;
    }
    
    // 4. 有参数有返回值的方法
    public static int add(int a, int b) {
        return a + b;
    }
    
    // 5. 多个参数的方法
    public static double calculateRectangleArea(double length, double width) {
        return length * width;
    }
    
    // 6. 不同访问修饰符的方法
    public static void publicMethod() {
        System.out.println("公共方法");
    }
    
    private static void privateMethod() {
        System.out.println("私有方法");
    }
    
    protected static void protectedMethod() {
        System.out.println("受保护方法");
    }
    
    static void packageMethod() {
        System.out.println("包级私有方法");
    }
    
    // 7. 其他修饰符的方法
    public final static void finalMethod() {
        System.out.println("final方法，不能被重写");
    }
    
    public static void main(String[] args) {
        // 调用各种方法
        sayHello();
        greetPerson("Alice");
        System.out.println("当前年份: " + getCurrentYear());
        System.out.println("5 + 3 = " + add(5, 3));
        System.out.println("矩形面积: " + calculateRectangleArea(5.0, 3.0));
        
        // 调用不同访问级别的方法
        publicMethod();
        privateMethod();    // 只能在同一个类中调用
        protectedMethod();
        packageMethod();
        finalMethod();
    }
}

public class ReturnTypes {
    // 基本数据类型返回值
    public static int getInteger() {
        return 42;
    }
    
    public static double getDouble() {
        return 3.14159;
    }
    
    public static boolean isEven(int number) {
        return number % 2 == 0;
    }
    
    public static char getFirstChar(String str) {
        if (str != null && str.length() > 0) {
            return str.charAt(0);
        }
        return '\0';  // 空字符
    }
    
    // 引用类型返回值
    public static String createGreeting(String name) {
        return "Hello, " + name + "!";
    }
    
    public static int[] createArray(int size) {
        return new int[size];
    }
    
    public static String[] splitString(String str, String delimiter) {
        return str.split(delimiter);
    }
    
    // void 返回类型（无返回值）
    public static void printMessage(String message) {
        System.out.println(message);
        // 可以使用 return; 提前结束方法
        if (message == null) {
            return;
        }
        System.out.println("消息长度: " + message.length());
    }
    
    // 多个返回点
    public static String getGrade(int score) {
        if (score >= 90) {
            return "A";
        } else if (score >= 80) {
            return "B";
        } else if (score >= 70) {
            return "C";
        } else if (score >= 60) {
            return "D";
        } else {
            return "F";
        }
    }
    
    // 条件返回
    public static int getAbsoluteValue(int number) {
        return number >= 0 ? number : -number;
    }
    
    public static void main(String[] args) {
        System.out.println("整数: " + getInteger());
        System.out.println("双精度: " + getDouble());
        System.out.println("10是偶数: " + isEven(10));
        System.out.println("首字符: " + getFirstChar("Hello"));
        System.out.println("问候语: " + createGreeting("Bob"));
        
        int[] arr = createArray(5);
        System.out.println("数组长度: " + arr.length);
        
        String[] parts = splitString("apple,banana,orange", ",");
        System.out.println("分割结果: " + java.util.Arrays.toString(parts));
        
        printMessage("测试消息");
        printMessage(null);
        
        System.out.println("成绩等级: " + getGrade(85));
        System.out.println("绝对值: " + getAbsoluteValue(-10));
    }
}

📥 参数传递机制 ​

1. 值传递 vs 引用传递 ​

java
public class ParameterPassing {
    // Java 中所有参数都是按值传递的
    // 对于基本类型：传递的是值的副本
    // 对于引用类型：传递的是引用的副本（指向同一个对象）
    
    // 基本类型参数传递
    public static void modifyPrimitive(int x) {
        x = 100;  // 只修改副本，不影响原变量
        System.out.println("方法内 x = " + x);
    }
    
    // 引用类型参数传递
    public static void modifyArray(int[] arr) {
        arr[0] = 999;  // 修改数组内容，会影响原数组
        System.out.println("方法内数组第一个元素: " + arr[0]);
    }
    
    public static void reassignArray(int[] arr) {
        arr = new int[]{100, 200, 300};  // 重新赋值，不影响原数组引用
        System.out.println("方法内重新赋值后: " + java.util.Arrays.toString(arr));
    }
    
    // 字符串参数传递
    public static void modifyString(String str) {
        str = "Modified";  // 字符串不可变，不影响原字符串
        System.out.println("方法内字符串: " + str);
    }
    
    // 对象参数传递
    public static void modifyPerson(Person person) {
        person.name = "Modified Name";  // 修改对象属性，会影响原对象
        person.age = 100;
    }
    
    public static void reassignPerson(Person person) {
        person = new Person("New Person", 25);  // 重新赋值，不影响原对象引用
    }
    
    public static void main(String[] args) {
        // 基本类型测试
        int number = 10;
        System.out.println("调用前 number = " + number);
        modifyPrimitive(number);
        System.out.println("调用后 number = " + number);  // 仍然是 10
        
        System.out.println();
        
        // 数组测试
        int[] array = {1, 2, 3, 4, 5};
        System.out.println("调用前数组: " + java.util.Arrays.toString(array));
        modifyArray(array);
        System.out.println("调用后数组: " + java.util.Arrays.toString(array));  // 第一个元素变为999
        
        reassignArray(array);
        System.out.println("重新赋值后原数组: " + java.util.Arrays.toString(array));  // 原数组不变
        
        System.out.println();
        
        // 字符串测试
        String text = "Original";
        System.out.println("调用前字符串: " + text);
        modifyString(text);
        System.out.println("调用后字符串: " + text);  // 仍然是 "Original"
        
        System.out.println();
        
        // 对象测试
        Person person = new Person("Alice", 20);
        System.out.println("调用前: " + person);
        modifyPerson(person);
        System.out.println("修改后: " + person);  // 属性被修改
        
        reassignPerson(person);
        System.out.println("重新赋值后: " + person);  // 对象引用不变
    }
}

class Person {
    String name;
    int age;

        this.age = age;
    }
    
    @Override

}

2. 可变参数（Varargs） ​

java
public class VarargsExample {
    // 可变参数方法
    public static int sum(int... numbers) {
        int total = 0;
        for (int num : numbers) {
            total += num;
        }
        return total;
    }
    
    // 可变参数与普通参数混合
    public static void printInfo(String prefix, int... numbers) {
        System.out.print(prefix + ": ");
        for (int i = 0; i < numbers.length; i++) {
            System.out.print(numbers[i]);
            if (i < numbers.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println();
    }
    
    // 字符串可变参数
    public static String concatenate(String... strings) {
        StringBuilder sb = new StringBuilder();
        for (String str : strings) {
            sb.append(str);
        }
        return sb.toString();
    }
    
    // 泛型可变参数
    @SafeVarargs
    public static <T> void printAll(T... items) {
        for (T item : items) {
            System.out.println(item);
        }
    }
    
    public static void main(String[] args) {
        // 调用可变参数方法
        System.out.println("无参数求和: " + sum());                    // 0
        System.out.println("单个参数求和: " + sum(5));                 // 5
        System.out.println("多个参数求和: " + sum(1, 2, 3, 4, 5));     // 15
        
        // 传递数组给可变参数方法
        int[] numbers = {10, 20, 30};
        System.out.println("数组参数求和: " + sum(numbers));           // 60
        
        // 混合参数
        printInfo("数字", 1, 2, 3, 4, 5);
        printInfo("偶数", 2, 4, 6, 8);
        
        // 字符串连接
        String result = concatenate("Hello", " ", "World", "!");
        System.out.println("连接结果: " + result);
        
        // 泛型可变参数
        printAll("Apple", "Banana", "Orange");
        printAll(1, 2, 3, 4, 5);
        printAll(true, false, true);
    }
}

🔄 方法重载（Overloading） ​

java
public class MethodOverloading {
    // 方法重载：同一个类中方法名相同，但参数列表不同
    
    // 1. 参数个数不同
    public static int add(int a, int b) {
        System.out.println("调用 add(int, int)");
        return a + b;
    }
    
    public static int add(int a, int b, int c) {
        System.out.println("调用 add(int, int, int)");
        return a + b + c;
    }
    
    // 2. 参数类型不同
    public static double add(double a, double b) {
        System.out.println("调用 add(double, double)");
        return a + b;
    }
    
    public static String add(String a, String b) {
        System.out.println("调用 add(String, String)");
        return a + b;
    }
    
    // 3. 参数顺序不同
    public static void printInfo(String name, int age) {
        System.out.println("姓名: " + name + ", 年龄: " + age);
    }
    
    public static void printInfo(int age, String name) {
        System.out.println("年龄: " + age + ", 姓名: " + name);
    }
    
    // 4. 可变参数的重载
    public static void print(String message) {
        System.out.println("单个消息: " + message);
    }
    
    public static void print(String... messages) {
        System.out.println("多个消息: " + java.util.Arrays.toString(messages));
    }
    
    // 5. 数组参数的重载
    public static void process(int[] array) {
        System.out.println("处理整数数组: " + java.util.Arrays.toString(array));
    }
    
    public static void process(String[] array) {
        System.out.println("处理字符串数组: " + java.util.Arrays.toString(array));
    }
    
    // 注意：仅返回类型不同不能构成重载
    // public static double add(int a, int b) {  // 编译错误
    //     return (double)(a + b);
    // }
    
    public static void main(String[] args) {
        // 编译器根据参数自动选择合适的重载方法
        
        System.out.println("整数相加: " + add(5, 3));           // 调用 add(int, int)
        System.out.println("三个整数相加: " + add(1, 2, 3));     // 调用 add(int, int, int)
        System.out.println("浮点数相加: " + add(2.5, 3.7));     // 调用 add(double, double)
        System.out.println("字符串相加: " + add("Hello", "World")); // 调用 add(String, String)
        
        // 参数顺序不同的重载
        printInfo("Alice", 25);    // 调用 printInfo(String, int)
        printInfo(30, "Bob");      // 调用 printInfo(int, String)
        
        // 可变参数重载
        print("单个消息");                           // 调用 print(String)
        print("消息1", "消息2", "消息3");            // 调用 print(String...)
        
        // 数组参数重载
        process(new int[]{1, 2, 3});
        process(new String[]{"A", "B", "C"});
        
        // 自动类型转换与重载
        byte b = 10;
        short s = 20;
        System.out.println("byte + short: " + add(b, s));  // 自动转换为int，调用add(int, int)
        
        // 装箱与重载
        Integer i1 = 10;
        Integer i2 = 20;
        System.out.println("Integer相加: " + add(i1, i2));  // 自动拆箱，调用add(int, int)
    }
}

🔁 递归方法 ​

java
public class RecursiveMethods {
    // 递归的基本要素：
    // 1. 基础情况（Base Case）：递归的终止条件
    // 2. 递归情况（Recursive Case）：方法调用自身
    // 3. 递归必须向基础情况收敛
    
    // 1. 计算阶乘
    public static long factorial(int n) {
        // 基础情况
        if (n <= 1) {
            return 1;
        }
        // 递归情况
        return n * factorial(n - 1);
    }
    
    // 2. 斐波那契数列
    public static int fibonacci(int n) {
        if (n <= 1) {
            return n;
        }
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
    
    // 3. 计算幂次
    public static double power(double base, int exponent) {
        if (exponent == 0) {
            return 1;
        }
        if (exponent < 0) {
            return 1.0 / power(base, -exponent);
        }
        return base * power(base, exponent - 1);
    }
    
    // 4. 二分查找（递归版本）
    public static int binarySearch(int[] arr, int target, int left, int right) {
        if (left > right) {
            return -1;  // 未找到
        }
        
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            return mid;
        } else if (arr[mid] > target) {
            return binarySearch(arr, target, left, mid - 1);
        } else {
            return binarySearch(arr, target, mid + 1, right);
        }
    }
    
    // 5. 汉诺塔问题
    public static void hanoi(int n, char from, char to, char aux) {
        if (n == 1) {
            System.out.println("移动盘子 1 从 " + from + " 到 " + to);
            return;
        }
        
        hanoi(n - 1, from, aux, to);
        System.out.println("移动盘子 " + n + " 从 " + from + " 到 " + to);
        hanoi(n - 1, aux, to, from);
    }
    
    // 6. 计算最大公约数（欧几里得算法）
    public static int gcd(int a, int b) {
        if (b == 0) {
            return a;
        }
        return gcd(b, a % b);
    }
    
    // 7. 字符串反转
    public static String reverse(String str) {
        if (str == null || str.length() <= 1) {
            return str;
        }
        return reverse(str.substring(1)) + str.charAt(0);
    }
    
    // 8. 判断回文字符串
    public static boolean isPalindrome(String str, int start, int end) {
        if (start >= end) {
            return true;
        }
        if (str.charAt(start) != str.charAt(end)) {
            return false;
        }
        return isPalindrome(str, start + 1, end - 1);
    }
    
    // 9. 树的遍历（模拟）
    public static void printNumbers(int n) {
        if (n <= 0) {
            return;
        }
        System.out.println("进入: " + n);
        printNumbers(n - 1);
        System.out.println("退出: " + n);
    }
    
    // 优化的斐波那契（记忆化）
    private static long[] fibCache = new long[100];
    
    public static long fibonacciOptimized(int n) {
        if (n <= 1) {
            return n;
        }
        if (fibCache[n] != 0) {
            return fibCache[n];
        }
        fibCache[n] = fibonacciOptimized(n - 1) + fibonacciOptimized(n - 2);
        return fibCache[n];
    }
    
    public static void main(String[] args) {
        // 阶乘测试
        System.out.println("5! = " + factorial(5));
        
        // 斐波那契测试
        System.out.println("斐波那契数列前10项:");
        for (int i = 0; i < 10; i++) {
            System.out.print(fibonacci(i) + " ");
        }
        System.out.println();
        
        // 幂次计算
        System.out.println("2^10 = " + power(2, 10));
        System.out.println("2^(-3) = " + power(2, -3));
        
        // 二分查找
        int[] sortedArray = {1, 3, 5, 7, 9, 11, 13, 15};
        int index = binarySearch(sortedArray, 7, 0, sortedArray.length - 1);
        System.out.println("7 在数组中的索引: " + index);
        
        // 汉诺塔
        System.out.println("\n汉诺塔问题（3个盘子）:");
        hanoi(3, 'A', 'C', 'B');
        
        // 最大公约数
        System.out.println("\ngcd(48, 18) = " + gcd(48, 18));
        
        // 字符串反转
        System.out.println("reverse('hello') = " + reverse("hello"));
        
        // 回文判断
        String palindrome = "racecar";
        boolean isPalin = isPalindrome(palindrome, 0, palindrome.length() - 1);
        System.out.println(palindrome + " 是回文: " + isPalin);
        
        // 递归调用栈演示
        System.out.println("\n递归调用栈演示:");
        printNumbers(3);
        
        // 优化的斐波那契
        System.out.println("\n优化的斐波那契:");
        long start = System.nanoTime();
        System.out.println("fibonacci(40) = " + fibonacci(40));
        long time1 = System.nanoTime() - start;
        
        start = System.nanoTime();
        System.out.println("fibonacciOptimized(40) = " + fibonacciOptimized(40));
        long time2 = System.nanoTime() - start;
        
        System.out.println("普通递归耗时: " + time1 + " ns");
        System.out.println("优化递归耗时: " + time2 + " ns");
    }
}

public class MathUtils {
    // 常量定义
    public static final double PI = 3.14159265359;
    public static final double E = 2.71828182846;
    
    // 1. 基本数学运算
    public static double add(double a, double b) {
        return a + b;
    }
    
    public static double subtract(double a, double b) {
        return a - b;
    }
    
    public static double multiply(double a, double b) {
        return a * b;
    }
    
    public static double divide(double a, double b) {
        if (b == 0) {
            throw new IllegalArgumentException("除数不能为零");
        }
        return a / b;
    }
    
    // 2. 高级数学函数
    public static double power(double base, int exponent) {
        if (exponent == 0) return 1;
        if (exponent < 0) return 1.0 / power(base, -exponent);
        
        double result = 1;
        for (int i = 0; i < exponent; i++) {
            result *= base;
        }
        return result;
    }
    
    public static double sqrt(double number) {
        if (number < 0) {
            throw new IllegalArgumentException("负数没有实数平方根");
        }
        if (number == 0) return 0;
        
        // 牛顿法求平方根
        double guess = number / 2;
        double epsilon = 1e-10;
        
        while (Math.abs(guess * guess - number) > epsilon) {
            guess = (guess + number / guess) / 2;
        }
        return guess;
    }
    
    // 3. 几何计算
    public static double circleArea(double radius) {
        if (radius < 0) {
            throw new IllegalArgumentException("半径不能为负数");
        }
        return PI * radius * radius;
    }
    
    public static double circleCircumference(double radius) {
        if (radius < 0) {
            throw new IllegalArgumentException("半径不能为负数");
        }
        return 2 * PI * radius;
    }
    
    public static double rectangleArea(double length, double width) {
        if (length < 0 || width < 0) {
            throw new IllegalArgumentException("长度和宽度不能为负数");
        }
        return length * width;
    }
    
    public static double triangleArea(double base, double height) {
        if (base < 0 || height < 0) {
            throw new IllegalArgumentException("底边和高不能为负数");
        }
        return 0.5 * base * height;
    }
    
    // 4. 统计函数
    public static double max(double... numbers) {
        if (numbers.length == 0) {
            throw new IllegalArgumentException("至少需要一个数字");
        }
        
        double maximum = numbers[0];
        for (int i = 1; i < numbers.length; i++) {
            if (numbers[i] > maximum) {
                maximum = numbers[i];
            }
        }
        return maximum;
    }
    
    public static double min(double... numbers) {
        if (numbers.length == 0) {
            throw new IllegalArgumentException("至少需要一个数字");
        }
        
        double minimum = numbers[0];
        for (int i = 1; i < numbers.length; i++) {
            if (numbers[i] < minimum) {
                minimum = numbers[i];
            }
        }
        return minimum;
    }
    
    public static double average(double... numbers) {
        if (numbers.length == 0) {
            throw new IllegalArgumentException("至少需要一个数字");
        }
        
        double sum = 0;
        for (double num : numbers) {
            sum += num;
        }
        return sum / numbers.length;
    }
    
    // 5. 数论函数
    public static boolean isPrime(int number) {
        if (number <= 1) return false;
        if (number <= 3) return true;
        if (number % 2 == 0 || number % 3 == 0) return false;
        
        for (int i = 5; i * i <= number; i += 6) {
            if (number % i == 0 || number % (i + 2) == 0) {
                return false;
            }
        }
        return true;
    }
    
    public static int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return Math.abs(a);
    }
    
    public static int lcm(int a, int b) {
        return Math.abs(a * b) / gcd(a, b);
    }
    
    // 6. 工具方法
    public static boolean isEven(int number) {
        return number % 2 == 0;
    }
    
    public static boolean isOdd(int number) {
        return number % 2 != 0;
    }
    
    public static int abs(int number) {
        return number >= 0 ? number : -number;
    }
    
    public static double abs(double number) {
        return number >= 0 ? number : -number;
    }
    
    // 测试方法
    public static void main(String[] args) {
        // 基本运算测试
        System.out.println("=== 基本运算 ===");
        System.out.println("10 + 5 = " + add(10, 5));
        System.out.println("10 - 5 = " + subtract(10, 5));
        System.out.println("10 * 5 = " + multiply(10, 5));
        System.out.println("10 / 5 = " + divide(10, 5));
        
        // 高级函数测试

        // 几何计算测试
        System.out.println("\n=== 几何计算 ===");
        System.out.println("半径为5的圆面积: " + circleArea(5));

        System.out.println("5x3矩形面积: " + rectangleArea(5, 3));

        System.out.println("最大值: " + max(numbers));
        System.out.println("最小值: " + min(numbers));

public class MethodDesignPrinciples {
    // 1. 单一职责原则：一个方法只做一件事
    
    // 不好的设计：方法做了太多事情
    public static void badProcessUser(String name, int age, String email) {
        // 验证输入
        if (name == null || name.trim().isEmpty()) {
            System.out.println("姓名不能为空");
            return;
        }
        if (age < 0 || age > 150) {
            System.out.println("年龄无效");
            return;
        }
        if (!email.contains("@")) {
            System.out.println("邮箱格式无效");
            return;
        }
        
        // 处理数据
        String processedName = name.trim().toLowerCase();
        
        // 保存到数据库（模拟）
        System.out.println("保存用户: " + processedName + ", " + age + ", " + email);
        
        // 发送邮件（模拟）
        System.out.println("发送欢迎邮件到: " + email);
    }
    
    // 好的设计：职责分离
    public static boolean validateName(String name) {
        return name != null && !name.trim().isEmpty();
    }
    
    public static boolean validateAge(int age) {
        return age >= 0 && age <= 150;
    }
    
    public static boolean validateEmail(String email) {
        return email != null && email.contains("@");
    }
    
    public static String processName(String name) {
        return name.trim().toLowerCase();
    }
    
    public static void saveUser(String name, int age, String email) {
        System.out.println("保存用户: " + name + ", " + age + ", " + email);
    }
    
    public static void sendWelcomeEmail(String email) {
        System.out.println("发送欢迎邮件到: " + email);
    }
    
    public static void goodProcessUser(String name, int age, String email) {
        if (!validateName(name)) {
            System.out.println("姓名不能为空");
            return;
        }
        if (!validateAge(age)) {
            System.out.println("年龄无效");
            return;
        }
        if (!validateEmail(email)) {
            System.out.println("邮箱格式无效");
            return;
        }
        
        String processedName = processName(name);
        saveUser(processedName, age, email);
        sendWelcomeEmail(email);
    }
    
    // 2. 方法命名规范
    
    // 动词开头，描述方法的行为
    public static void calculateTotal() { }      // 计算
    public static void printReport() { }         // 打印
    public static boolean isValid() { }          // 判断
    public static String getName() { }           // 获取
    public static void setAge(int age) { }       // 设置
    
    // 3. 参数数量控制
    
    // 不好：参数太多
    public static void badCreateUser(String firstName, String lastName, 
                                   int age, String email, String phone, 
                                   String address, String city, String country) {
        // 实现
    }
    
    // 好：使用对象封装参数
    public static void goodCreateUser(UserInfo userInfo) {
        // 实现
    }
    
    // 4. 返回值设计
    
    // 不好：使用特殊值表示错误
    public static int badDivide(int a, int b) {
        if (b == 0) {
            return -1;  // 特殊值表示错误，容易混淆
        }
        return a / b;
    }
    
    // 好：抛出异常或使用Optional
    public static int goodDivide(int a, int b) {
        if (b == 0) {
            throw new IllegalArgumentException("除数不能为零");
        }
        return a / b;
    }
    
    public static java.util.Optional<Integer> safeDivide(int a, int b) {
        if (b == 0) {
            return java.util.Optional.empty();
        }
        return java.util.Optional.of(a / b);
    }
    
    public static void main(String[] args) {
        // 测试好的设计
        goodProcessUser("  ALICE  ", 25, "alice@example.com");
        
        // 测试安全除法
        try {
            System.out.println("10 / 2 = " + goodDivide(10, 2));
            System.out.println("10 / 0 = " + goodDivide(10, 0));
        } catch (IllegalArgumentException e) {
            System.out.println("错误: " + e.getMessage());
        }
        
        // 使用Optional
        java.util.Optional<Integer> result = safeDivide(10, 0);
        if (result.isPresent()) {
            System.out.println("结果: " + result.get());
        } else {
            System.out.println("除法操作失败");
        }
    }
}
