Java 泛型编程详解 ​

📚 概述 ​

泛型（Generics）是 Java 5 引入的重要特性，它提供了编译时类型安全检查，消除了类型转换的需要，并使程序员能够实现通用算法。泛型让代码更加类型安全、可读性更强，并且能够在编译时捕获更多的错误。理解和掌握泛型是现代 Java 编程的基础技能。

🎯 学习目标 ​

  • 深入理解泛型的概念和作用
  • 掌握泛型类、泛型接口和泛型方法的定义和使用
  • 理解类型参数、类型边界和通配符
  • 熟练使用泛型集合和泛型算法
  • 掌握泛型的继承和多态特性
  • 理解类型擦除机制和泛型的限制
  • 学会设计和使用复杂的泛型结构
  • 掌握泛型编程的最佳实践

// 泛型出现前的问题演示
public class WithoutGenericsDemo {
    
    public static void main(String[] args) {
        demonstrateProblemsWithoutGenerics();
        demonstrateSolutionWithGenerics();
    }
    
    // 没有泛型时的问题
    public static void demonstrateProblemsWithoutGenerics() {
        System.out.println("=== 没有泛型的问题 ===");
        
        // 问题1：类型不安全
        java.util.List list = new java.util.ArrayList();
        list.add("Hello");
        list.add(123);        // 可以添加不同类型的对象
        list.add(new Object());
        
        // 问题2：需要强制类型转换
        for (int i = 0; i < list.size(); i++) {
            try {
                String str = (String) list.get(i);  // 强制转换可能失败
                System.out.println("字符串: " + str);
            } catch (ClassCastException e) {
                System.out.println("类型转换失败: " + list.get(i));
            }
        }
        
        // 问题3：运行时才能发现错误
        System.out.println("类型错误只能在运行时发现");
        System.out.println();
    }
    
    // 使用泛型解决问题
    public static void demonstrateSolutionWithGenerics() {
        System.out.println("=== 使用泛型的解决方案 ===");
        
        // 优势1：类型安全
        java.util.List<String> stringList = new java.util.ArrayList<>();
        stringList.add("Hello");
        stringList.add("World");
        // stringList.add(123);  // 编译错误：类型不匹配
        
        // 优势2：无需类型转换
        for (String str : stringList) {
            System.out.println("字符串: " + str);  // 无需强制转换
        }
        
        // 优势3：编译时类型检查
        System.out.println("类型错误在编译时就能发现");
        System.out.println();
    }
}

        System.out.println("交换前: " + java.util.Arrays.toString(stringArray));

    public static void demonstrateGenericInterfaces() {
        System.out.println("=== 泛型接口演示 ===");
        
        Processor<String> stringProcessor = new StringProcessor();
        Processor<Integer> intProcessor = new IntegerProcessor();
        
        System.out.println("处理字符串: " + stringProcessor.process("hello"));
        System.out.println("处理整数: " + intProcessor.process(42));
        
        System.out.println();
    }
    
    // 泛型方法：获取数组第一个元素
    public static <T> T getFirst(T[] array) {
        if (array == null || array.length == 0) {
            return null;
        }
        return array[0];
    }
    
    // 泛型方法：交换数组元素
    public static <T> void swap(T[] array, int i, int j) {
        if (array != null && i >= 0 && j >= 0 && i < array.length && j < array.length) {
            T temp = array[i];
            array[i] = array[j];
            array[j] = temp;
        }
    }
}

// 简单泛型类
class Box<T> {
    private T content;
    
    public void set(T content) {
        this.content = content;
    }
    
    public T get() {
        return content;
    }
    
    public boolean isEmpty() {
        return content == null;
    }
}

// 多个类型参数的泛型类
class Pair<T, U> {
    private T first;
    private U second;
    
    public Pair(T first, U second) {
        this.first = first;
        this.second = second;
    }
    
    public T getFirst() {
        return first;
    }
    
    public U getSecond() {
        return second;
    }
    
    public void setFirst(T first) {
        this.first = first;
    }
    
    public void setSecond(U second) {
        this.second = second;
    }
    
    @Override
    public String toString() {
        return "(" + first + ", " + second + ")";
    }
}

// 三个类型参数的泛型类
class Triple<T, U, V> {
    private T first;
    private U second;
    private V third;
    
    public Triple(T first, U second, V third) {
        this.first = first;
        this.second = second;
        this.third = third;
    }
    
    public T getFirst() { return first; }
    public U getSecond() { return second; }
    public V getThird() { return third; }
    
    @Override
    public String toString() {
        return "(" + first + ", " + second + ", " + third + ")";
    }
}

// 泛型类的继承
class NumberBox<T extends Number> extends Box<T> {
    
    public double getNumber() {
        T content = get();
        return content != null ? content.doubleValue() : 0.0;
    }
    
    public NumberBox(T content) {
        set(content);
    }
}

class UserRepository implements Repository<User> {

// 泛型方法详细演示
public class GenericMethodDemo {

    public static void main(String[] args) {
        demonstrateBasicGenericMethods();
        demonstrateBoundedGenericMethods();
        demonstrateMultipleTypeBounds();
        demonstrateGenericMethodInClass();
    }

    // 基本泛型方法演示
    public static void demonstrateBasicGenericMethods() {
        System.out.println("=== 基本泛型方法演示 ===");

        // 类型推断
        String[] strings = {"apple", "banana", "cherry"};
        Integer[] numbers = {1, 2, 3, 4, 5};

        printArray(strings);   // 自动推断为 String
        printArray(numbers);   // 自动推断为 Integer

        // 显式指定类型
        GenericMethodDemo.<String>printArray(strings);

        // 泛型方法返回值
        String middle = getMiddleElement(strings);
        Integer middleNum = getMiddleElement(numbers);
        System.out.println("中间字符串: " + middle);
        System.out.println("中间数字: " + middleNum);

        System.out.println();
    }

    // 有界泛型方法演示
    public static void demonstrateBoundedGenericMethods() {
        System.out.println("=== 有界泛型方法演示 ===");

        Integer[] intArray = {1, 5, 3, 9, 2};
        Double[] doubleArray = {1.1, 5.5, 3.3, 9.9, 2.2};

        System.out.println("整数数组最大值: " + findMax(intArray));
        System.out.println("浮点数组最大值: " + findMax(doubleArray));

        // String[] stringArray = {"a", "b", "c"};
        // findMax(stringArray);  // 编译错误：String 不是 Comparable 的子类型

        System.out.println();
    }

    // 多重类型边界演示
    public static void demonstrateMultipleTypeBounds() {
        System.out.println("=== 多重类型边界演示 ===");

        NumberContainer<Integer> intContainer = new NumberContainer<>(42);
        NumberContainer<Double> doubleContainer = new NumberContainer<>(3.14);

        processNumberContainer(intContainer);
        processNumberContainer(doubleContainer);

        System.out.println();
    }

    // 类中的泛型方法演示
    public static void demonstrateGenericMethodInClass() {
        System.out.println("=== 类中的泛型方法演示 ===");

        Utility utility = new Utility();

        String[] words = {"hello", "world", "java"};
        Integer[] nums = {10, 20, 30};

        utility.reverse(words);
        utility.reverse(nums);

        System.out.println("反转后的字符串数组: " + java.util.Arrays.toString(words));
        System.out.println("反转后的数字数组: " + java.util.Arrays.toString(nums));

        System.out.println();
    }

    // 泛型方法：打印数组
    public static <T> void printArray(T[] array) {
        System.out.print("数组内容: ");
        for (T element : array) {
            System.out.print(element + " ");
        }
        System.out.println();
    }

    // 泛型方法：获取中间元素
    public static <T> T getMiddleElement(T[] array) {
        if (array == null || array.length == 0) {
            return null;
        }
        return array[array.length / 2];
    }

    // 有界泛型方法：查找最大值
    public static <T extends Comparable<T>> T findMax(T[] array) {
        if (array == null || array.length == 0) {
            return null;
        }

        T max = array[0];
        for (int i = 1; i < array.length; i++) {
            if (array[i].compareTo(max) > 0) {
                max = array[i];
            }
        }
        return max;
    }

    // 多重边界的泛型方法
    public static <T extends Number & Comparable<T>> void processNumberContainer(NumberContainer<T> container) {
        T value = container.getValue();
        System.out.println("处理数值容器: " + value + " (类型: " + value.getClass().getSimpleName() + ")");
        System.out.println("  数值: " + value.doubleValue());
        System.out.println("  与0比较: " + value.compareTo((T) Integer.valueOf(0)));
    }
}

// 数值容器类
class NumberContainer<T extends Number> {
    private T value;

    public NumberContainer(T value) {
        this.value = value;
    }

    public T getValue() {
        return value;
    }
}

// 工具类
class Utility {

    // 泛型方法：反转数组
    public <T> void reverse(T[] array) {
        if (array == null) return;

        int left = 0;
        int right = array.length - 1;

        while (left < right) {
            T temp = array[left];
            array[left] = array[right];
            array[right] = temp;
            left++;
            right--;
        }
    }

    // 泛型方法：查找元素
    public <T> int indexOf(T[] array, T target) {
        if (array == null || target == null) {
            return -1;
        }

        for (int i = 0; i < array.length; i++) {
            if (target.equals(array[i])) {
                return i;
            }
        }
        return -1;
    }
}

        System.out.println("=== 下界通配符演示 ===");

        java.util.List<Number> numberList = new java.util.ArrayList<>();
        java.util.List<Object> objectList = new java.util.ArrayList<>();
        java.util.List<Integer> intList = new java.util.ArrayList<>();

        addNumbers(numberList);
        addNumbers(objectList);
        // addNumbers(intList);  // 编译错误：Integer 不是 Integer 的超类

        System.out.println("Number 列表: " + numberList);
        System.out.println("Object 列表: " + objectList);

        System.out.println();
    }

    // 通配符捕获演示
    public static void demonstrateWildcardCapture() {
        System.out.println("=== 通配符捕获演示 ===");

        java.util.List<String> stringList = new java.util.ArrayList<>(java.util.Arrays.asList("a", "b", "c"));
        java.util.List<Integer> intList = new java.util.ArrayList<>(java.util.Arrays.asList(1, 2, 3));

        System.out.println("交换前字符串列表: " + stringList);
        swapElements(stringList, 0, 2);
        System.out.println("交换后字符串列表: " + stringList);

        System.out.println("交换前整数列表: " + intList);
        swapElements(intList, 0, 2);
        System.out.println("交换后整数列表: " + intList);

        System.out.println();
    }

    // 无界通配符方法
    public static void printListSize(java.util.List<?> list) {
        System.out.println("列表大小: " + list.size());
        // list.add("something");  // 编译错误：不能添加元素到 ? 类型的列表
    }

    // 上界通配符方法
    public static double sumNumbers(java.util.List<? extends Number> numbers) {
        double sum = 0.0;
        for (Number num : numbers) {
            sum += num.doubleValue();
        }
        return sum;
    }

    // 下界通配符方法
    public static void addNumbers(java.util.List<? super Integer> list) {
        list.add(1);
        list.add(2);
        list.add(3);
        // Integer item = list.get(0);  // 编译错误：不能确定返回类型
    }

    // 通配符捕获辅助方法
    public static void swapElements(java.util.List<?> list, int i, int j) {
        swapHelper(list, i, j);
    }

    private static <T> void swapHelper(java.util.List<T> list, int i, int j) {
        T temp = list.get(i);
        list.set(i, list.get(j));
        list.set(j, temp);
    }
}

            }
        } catch (NoSuchFieldException e) {
            e.printStackTrace();
        }

        System.out.println();
    }

// 泛型最佳实践演示
public class GenericBestPracticesDemo {

    public static void main(String[] args) {
        demonstrateNamingConventions();
        demonstrateProducerExtendsConsumerSuper();
        demonstrateGenericSingleton();
        demonstrateTypeInference();
    }

    // 命名约定演示
    public static void demonstrateNamingConventions() {
        System.out.println("=== 泛型命名约定演示 ===");

        // 常见的类型参数命名
        Container<String> container = new Container<>();  // T - Type
        KeyValuePair<String, Integer> pair = new KeyValuePair<>("key", 42);  // K, V - Key, Value
        ResponseWrapper<User> response = new ResponseWrapper<>(new User(1, "Alice", "alice@example.com"));  // E - Element, R - Result

        System.out.println("容器: " + container);
        System.out.println("键值对: " + pair);
        System.out.println("响应包装: " + response);

        System.out.println();
    }

    // PECS 原则演示（Producer Extends, Consumer Super）
    public static void demonstrateProducerExtendsConsumerSuper() {
        System.out.println("=== PECS 原则演示 ===");

        java.util.List<Integer> intList = java.util.Arrays.asList(1, 2, 3);
        java.util.List<Double> doubleList = java.util.Arrays.asList(1.1, 2.2, 3.3);
        java.util.List<Number> numberList = new java.util.ArrayList<>();

        // Producer Extends: 从集合中读取数据
        copyNumbers(intList, numberList);
        copyNumbers(doubleList, numberList);

        System.out.println("复制后的数字列表: " + numberList);

        // Consumer Super: 向集合中写入数据
        java.util.List<Object> objectList = new java.util.ArrayList<>();
        addIntegers(numberList);
        addIntegers(objectList);

        System.out.println("添加整数后的 Number 列表: " + numberList);
        System.out.println("添加整数后的 Object 列表: " + objectList);

        System.out.println();
    }

    // 泛型单例演示
    public static void demonstrateGenericSingleton() {
        System.out.println("=== 泛型单例演示 ===");

        java.util.Comparator<String> stringComparator = GenericSingleton.reverseOrder();
        java.util.Comparator<Integer> intComparator = GenericSingleton.reverseOrder();

        java.util.List<String> strings = java.util.Arrays.asList("c", "a", "b");
        java.util.List<Integer> integers = java.util.Arrays.asList(3, 1, 2);

        strings.sort(stringComparator);
        integers.sort(intComparator);

        System.out.println("反向排序的字符串: " + strings);
        System.out.println("反向排序的整数: " + integers);

        System.out.println();
    }

    // 类型推断演示
    public static void demonstrateTypeInference() {
        System.out.println("=== 类型推断演示 ===");

        // Java 7+ 钻石操作符
        java.util.List<String> list1 = new java.util.ArrayList<>();  // 推断为 ArrayList<String>
        java.util.Map<String, Integer> map1 = new java.util.HashMap<>();  // 推断为 HashMap<String, Integer>

        // 方法类型推断
        java.util.List<String> list2 = createList("a", "b", "c");
        java.util.List<Integer> list3 = createList(1, 2, 3);

        System.out.println("推断的字符串列表: " + list2);
        System.out.println("推断的整数列表: " + list3);

        System.out.println();
    }

    // PECS - Producer Extends
    public static <T extends Number> void copyNumbers(java.util.List<? extends T> source, java.util.List<? super T> dest) {
        for (T number : source) {
            dest.add(number);
        }
    }

    // PECS - Consumer Super
    public static void addIntegers(java.util.List<? super Integer> list) {
        list.add(10);
        list.add(20);
        list.add(30);
    }

    // 泛型方法类型推断
    @SafeVarargs
    public static <T> java.util.List<T> createList(T... elements) {
        return new java.util.ArrayList<>(java.util.Arrays.asList(elements));
    }
}

class KeyValuePair<K, V> {
    private K key;
    private V value;

    public KeyValuePair(K key, V value) {
        this.key = key;
        this.value = value;
    }

    public K getKey() { return key; }
    public V getValue() { return value; }

    @Override
    public String toString() {

// 响应包装类
class ResponseWrapper<R> {
    private R data;
    private String message;
    private boolean success;

    public ResponseWrapper(R data) {
        this.data = data;
        this.success = true;
        this.message = "Success";
    }

    public R getData() { return data; }
    public String getMessage() { return message; }
    public boolean isSuccess() { return success; }

    @Override
    public String toString() {
        return "Response{data=" + data + ", success=" + success + "}";
    }
}

class GenericSingleton {
    private static final java.util.Comparator<Object> REVERSE_ORDER = new java.util.Comparator<Object>() {
        @Override
        public int compare(Object o1, Object o2) {
            return ((Comparable<Object>) o2).compareTo(o1);
        }
    };

    @SuppressWarnings("unchecked")
    public static <T> java.util.Comparator<T> reverseOrder() {
        return (java.util.Comparator<T>) REVERSE_ORDER;
    }
}

// 通配符常见错误
public class WildcardMistakes {

    // 错误：混淆 extends 和 super
    public void wrongWildcardUsage() {
        java.util.List<? extends Number> numbers = new java.util.ArrayList<Integer>();
        // numbers.add(1);  // 编译错误：不能添加元素到 extends 通配符

        java.util.List<? super Integer> integers = new java.util.ArrayList<Number>();

        // Integer value = integers.get(0);  // 编译错误：返回类型是 Object
    }

    public <T> void copyList(java.util.List<? extends T> source, java.util.List<? super T> dest) {
        for (T item : source) {
            dest.add(item);
        }
    }

设计原则 ​
