Java 访问权限控制详解 ​

📚 概述 ​

访问权限控制（Access Control）是 Java 面向对象编程中实现封装的核心机制。通过访问修饰符，Java 提供了对类、方法、变量和构造器的访问控制，确保代码的安全性、可维护性和模块化设计。理解和正确使用访问权限控制是编写高质量 Java 代码的基础。

🎯 学习目标 ​

  • 深入理解 Java 四种访问修饰符的作用和范围
  • 掌握访问权限在不同场景下的应用
  • 熟练使用访问控制实现封装和信息隐藏
  • 理解包（package）在访问控制中的作用
  • 掌握访问权限在继承中的规则和限制
  • 学会设计合理的访问权限策略
  • 了解访问权限与设计模式的关系
  • 掌握访问权限控制的最佳实践

🔐 访问修饰符详解 ​

1. 四种访问修饰符 ​

java
// 演示四种访问修饰符
public class AccessModifierDemo {

    // public：公共访问，任何地方都可以访问
    public String publicField = "公共字段";
    public static final String PUBLIC_CONSTANT = "公共常量";

    // protected：受保护访问，同包或子类可以访问
    protected String protectedField = "受保护字段";
    protected static int protectedStaticField = 100;

    // 默认（包级私有）：同包内可以访问
    String packageField = "包级私有字段";
    static String packageStaticField = "包级私有静态字段";

    // private：私有访问，只有本类内部可以访问
    private String privateField = "私有字段";
    private static final String PRIVATE_CONSTANT = "私有常量";

    // 构造器的访问修饰符
    public AccessModifierDemo() {
        System.out.println("公共构造器");
    }

    protected AccessModifierDemo(String name) {
        System.out.println("受保护构造器: " + name);
    }

    AccessModifierDemo(int value) {
        System.out.println("包级私有构造器: " + value);
    }

    private AccessModifierDemo(boolean flag) {
        System.out.println("私有构造器: " + flag);
    }

    // 方法的访问修饰符
    public void publicMethod() {
        System.out.println("公共方法 - 可以从任何地方调用");
        // 私有方法只能在类内部调用
        privateMethod();
    }

    protected void protectedMethod() {
        System.out.println("受保护方法 - 同包或子类可以调用");
    }

    void packageMethod() {
        System.out.println("包级私有方法 - 同包内可以调用");
    }

    private void privateMethod() {
        System.out.println("私有方法 - 只能在本类内部调用");
        // 可以访问所有字段，包括私有字段
        System.out.println("访问私有字段: " + privateField);
    }

    // 演示内部访问
    public void demonstrateInternalAccess() {
        System.out.println("=== 类内部访问演示 ===");
        System.out.println("公共字段: " + publicField);
        System.out.println("受保护字段: " + protectedField);
        System.out.println("包级私有字段: " + packageField);
        System.out.println("私有字段: " + privateField);

        publicMethod();
        protectedMethod();
        packageMethod();
        privateMethod();

        // 调用私有构造器
        AccessModifierDemo privateInstance = new AccessModifierDemo(true);
    }
}

2. protected 访问的特殊规则 ​

java
// 演示 protected 访问的细节规则
package com.example.base;

public class BaseClass {
    protected String protectedField = "受保护字段";
    protected static String protectedStaticField = "受保护静态字段";

    protected void protectedMethod() {
        System.out.println("受保护方法");
    }

    protected static void protectedStaticMethod() {
        System.out.println("受保护静态方法");
    }
}

// 不同包的子类
package com.example.derived;
import com.example.base.BaseClass;

public class DerivedClass extends BaseClass {

    public void demonstrateProtectedAccess() {
        // 1. 可以直接访问继承的 protected 成员
        System.out.println("直接访问: " + protectedField);
        protectedMethod();

        // 2. 可以访问静态 protected 成员
        System.out.println("静态字段: " + protectedStaticField);
        protectedStaticMethod();

        // 3. 可以通过 this 访问
        System.out.println("this 访问: " + this.protectedField);
        this.protectedMethod();

        // 4. 可以通过同类型实例访问
        DerivedClass other = new DerivedClass();
        System.out.println("同类型实例: " + other.protectedField);
        other.protectedMethod();

        // 5. 不能通过父类型实例访问（即使是同一个对象）
        BaseClass base = this;
        // System.out.println("父类型实例: " + base.protectedField); // ✗ 编译错误
        // base.protectedMethod(); // ✗ 编译错误

        // 6. 不能通过其他父类实例访问
        BaseClass anotherBase = new BaseClass();
        // System.out.println("其他父类实例: " + anotherBase.protectedField); // ✗ 编译错误
    }
}

2. 嵌套类的访问控制 ​

java
public class OuterClass {
    private String outerPrivateField = "外部类私有字段";
    protected String outerProtectedField = "外部类受保护字段";
    public String outerPublicField = "外部类公共字段";

    // 公共静态嵌套类
    public static class PublicStaticNestedClass {
        public void method() {
            System.out.println("公共静态嵌套类");
            // 静态嵌套类不能直接访问外部类的实例成员
            // System.out.println(outerPrivateField); // 编译错误
        }
    }

    // 私有静态嵌套类
    private static class PrivateStaticNestedClass {
        public void method() {
            System.out.println("私有静态嵌套类");
        }
    }

    // 公共内部类
    public class PublicInnerClass {
        public void method() {
            System.out.println("公共内部类");
            // 内部类可以访问外部类的所有成员，包括私有成员
            System.out.println("访问外部类私有字段: " + outerPrivateField);
        }
    }

    // 私有内部类
    private class PrivateInnerClass {
        public void method() {
            System.out.println("私有内部类");
            System.out.println("访问外部类字段: " + outerPrivateField);
        }
    }

    // 受保护内部类
    protected class ProtectedInnerClass {
        public void method() {
            System.out.println("受保护内部类");
        }
    }

    // 演示嵌套类的使用
    public void demonstrateNestedClasses() {
        // 创建静态嵌套类实例
        PublicStaticNestedClass staticNested = new PublicStaticNestedClass();
        staticNested.method();

        PrivateStaticNestedClass privateStatic = new PrivateStaticNestedClass();
        privateStatic.method();

        // 创建内部类实例
        PublicInnerClass publicInner = new PublicInnerClass();
        publicInner.method();

        PrivateInnerClass privateInner = new PrivateInnerClass();
        privateInner.method();

        ProtectedInnerClass protectedInner = new ProtectedInnerClass();
        protectedInner.method();
    }
}

// 外部类使用嵌套类
class ExternalClassUsingNested {
    public void useNestedClasses() {
        // 可以访问公共静态嵌套类
        OuterClass.PublicStaticNestedClass staticNested = new OuterClass.PublicStaticNestedClass();
        staticNested.method();

        // 不能访问私有静态嵌套类
        // OuterClass.PrivateStaticNestedClass privateStatic = ...; // 编译错误

        // 访问公共内部类需要外部类实例
        OuterClass outer = new OuterClass();
        OuterClass.PublicInnerClass publicInner = outer.new PublicInnerClass();
        publicInner.method();

        // 不能访问私有内部类
        // OuterClass.PrivateInnerClass privateInner = ...; // 编译错误
    }
}

🔄 继承中的访问权限 ​

1. 方法重写的访问权限规则 ​

java
// 父类
class Parent {
    public void publicMethod() {
        System.out.println("Parent public method");
    }

    protected void protectedMethod() {
        System.out.println("Parent protected method");
    }

    void packageMethod() {
        System.out.println("Parent package method");
    }

    // private 方法不能被重写，只能被隐藏
    private void privateMethod() {
        System.out.println("Parent private method");
    }
}

// 子类
class Child extends Parent {

    // 可以保持相同的访问级别
    @Override
    public void publicMethod() {
        System.out.println("Child public method");
    }

    // 可以扩大访问权限：protected -> public
    @Override
    public void protectedMethod() {
        System.out.println("Child protected method (now public)");
    }

    // 不能缩小访问权限
    // @Override
    // private void packageMethod() { } // 编译错误：不能将包级私有改为私有

    @Override
    void packageMethod() {
        System.out.println("Child package method");
    }

    // 这不是重写，而是定义新方法（因为父类的 privateMethod 是私有的）
    private void privateMethod() {
        System.out.println("Child private method (not override)");
    }

    // 演示访问权限规则
    public void demonstrateAccessRules() {
        publicMethod();     // 调用重写的方法
        protectedMethod();  // 调用重写的方法
        packageMethod();    // 调用重写的方法
        privateMethod();    // 调用子类自己的私有方法

        // 调用父类的方法
        super.publicMethod();
        super.protectedMethod();
        super.packageMethod();
        // super.privateMethod(); // 编译错误：无法访问父类私有方法
    }
}

💼 实际应用案例 ​

1. 银行账户系统设计 ​

java
// 银行账户系统：演示访问权限的实际应用
public abstract class BankAccount {
    // 私有字段：外部无法直接访问
    private String accountNumber;
    private String accountHolder;
    private double balance;
    private boolean isActive;

    // 受保护字段：子类可以访问
    protected String bankCode;
    protected java.time.LocalDateTime createdTime;

    // 公共构造器
    public BankAccount(String accountNumber, String accountHolder, double initialBalance, String bankCode) {
        this.accountNumber = accountNumber;
        this.accountHolder = accountHolder;
        this.balance = initialBalance >= 0 ? initialBalance : 0;
        this.isActive = true;
        this.bankCode = bankCode;
        this.createdTime = java.time.LocalDateTime.now();
    }

    // 公共方法：外部可以调用
    public String getAccountNumber() {
        return accountNumber;
    }

    public String getAccountHolder() {
        return accountHolder;
    }

    public double getBalance() {
        return isActive ? balance : 0;
    }

    public boolean isActive() {
        return isActive;
    }

    // 受保护方法：子类可以重写
    protected boolean validateAmount(double amount) {
        return amount > 0;
    }

    protected void logTransaction(String operation, double amount) {
        System.out.printf("[%s] %s: %s %.2f, 余额: %.2f%n",
                         createdTime.toLocalDate(), accountNumber, operation, amount, balance);
    }

    // 包级私有方法：同包的工具类可以访问
    void updateBalance(double newBalance) {
        this.balance = newBalance;
    }

    void setActive(boolean active) {
        this.isActive = active;
    }

    // 私有方法：只有本类可以访问
    private boolean checkAccountStatus() {
        return isActive && balance >= 0;
    }

    // 公共业务方法
    public boolean deposit(double amount) {
        if (!checkAccountStatus()) {
            System.out.println("账户状态异常，无法存款");
            return false;
        }

        if (!validateAmount(amount)) {
            System.out.println("存款金额无效");
            return false;
        }

        balance += amount;
        logTransaction("存款", amount);
        return true;
    }

    public boolean withdraw(double amount) {
        if (!checkAccountStatus()) {
            System.out.println("账户状态异常，无法取款");
            return false;
        }

        if (!validateAmount(amount)) {
            System.out.println("取款金额无效");
            return false;
        }

        if (!canWithdraw(amount)) {
            System.out.println("余额不足或超出限制");
            return false;
        }

        balance -= amount;
        logTransaction("取款", amount);
        return true;
    }

    // 抽象方法：子类必须实现
    protected abstract boolean canWithdraw(double amount);
    protected abstract double getWithdrawLimit();
}

// 储蓄账户
class SavingsAccount extends BankAccount {
    private static final double WITHDRAW_LIMIT = 10000; // 每日取款限制
    private double dailyWithdrawn = 0;

    public SavingsAccount(String accountNumber, String accountHolder, double initialBalance) {
        super(accountNumber, accountHolder, initialBalance, "SAV");
    }

    @Override
    protected boolean canWithdraw(double amount) {
        return getBalance() >= amount && (dailyWithdrawn + amount) <= WITHDRAW_LIMIT;
    }

    @Override
    protected double getWithdrawLimit() {
        return WITHDRAW_LIMIT - dailyWithdrawn;
    }

    @Override
    protected void logTransaction(String operation, double amount) {
        super.logTransaction(operation, amount);
        if ("取款".equals(operation)) {
            dailyWithdrawn += amount;
        }
    }

    // 储蓄账户特有方法
    public void calculateInterest() {
        double interest = getBalance() * 0.02; // 2% 年利率
        if (interest > 0) {
            updateBalance(getBalance() + interest); // 使用包级私有方法
            logTransaction("利息", interest);
        }
    }
}

// 信用账户
class CreditAccount extends BankAccount {
    private double creditLimit;

    public CreditAccount(String accountNumber, String accountHolder, double creditLimit) {
        super(accountNumber, accountHolder, 0, "CRD");
        this.creditLimit = creditLimit;
    }

    @Override
    protected boolean canWithdraw(double amount) {
        return (getBalance() + creditLimit) >= amount;
    }

    @Override
    protected double getWithdrawLimit() {
        return getBalance() + creditLimit;
    }

    @Override
    protected boolean validateAmount(double amount) {
        // 信用账户允许更大的金额
        return amount > 0 && amount <= 50000;
    }

    public double getAvailableCredit() {
        return creditLimit + Math.min(0, getBalance());
    }
}

class BankAccountManager {
    // 可以访问包级私有方法
    public void freezeAccount(BankAccount account) {
        account.setActive(false);
        System.out.println("账户已冻结: " + account.getAccountNumber());
    }

    public void adjustBalance(BankAccount account, double adjustment) {
        double newBalance = account.getBalance() + adjustment;
        account.updateBalance(newBalance);
        System.out.printf("账户 %s 余额调整: %.2f%n", account.getAccountNumber(), adjustment);
    }
}

public class GoodDesignExample {
    // 私有字段：外部不需要直接访问
    private String internalId;
    private List<String> items;

    // 受保护字段：子类可能需要访问
    protected String status;

    // 公共构造器：外部需要创建对象
    public GoodDesignExample(String id) {
        this.internalId = id;
        this.items = new ArrayList<>();
        this.status = "ACTIVE";
    }

    // 公共方法：提供必要的外部接口
    public String getId() {
        return internalId;
    }

    public void addItem(String item) {
        if (validateItem(item)) {
            items.add(item);
            updateStatus();
        }
    }

    public List<String> getItems() {
        // 返回不可变副本，保护内部状态
        return Collections.unmodifiableList(items);
    }

    // 受保护方法：子类可能需要重写
    protected boolean validateItem(String item) {
        return item != null && !item.trim().isEmpty();
    }

    protected void updateStatus() {
        status = items.isEmpty() ? "EMPTY" : "HAS_ITEMS";
    }

    // 私有方法：内部实现细节
    private void logOperation(String operation) {
        System.out.println("Operation: " + operation + " on " + internalId);
    }
}

// 接口设计：合理的访问权限
public interface PaymentProcessor {
    // 接口方法默认是 public abstract
    boolean processPayment(double amount);
    String getPaymentMethod();

    // 默认方法：提供通用实现
    default void logPayment(double amount) {
        System.out.printf("Payment processed: %.2f via %s%n", amount, getPaymentMethod());
    }

    // 静态方法：工具方法
    static boolean isValidAmount(double amount) {
        return amount > 0 && amount <= 1000000;
    }
}

// 抽象类设计：合理的访问权限层次
public abstract class AbstractPaymentProcessor implements PaymentProcessor {
    // 受保护字段：子类可以访问
    protected String processorId;
    protected boolean isEnabled;

    // 私有字段：内部状态
    private int transactionCount;

    protected AbstractPaymentProcessor(String processorId) {
        this.processorId = processorId;
        this.isEnabled = true;
        this.transactionCount = 0;
    }

    // 模板方法：定义处理流程
    @Override
    public final boolean processPayment(double amount) {
        if (!isEnabled) {
            return false;
        }

        if (!PaymentProcessor.isValidAmount(amount)) {
            return false;
        }

        boolean result = doProcessPayment(amount);
        if (result) {
            incrementTransactionCount();
            logPayment(amount);
        }
        return result;
    }

    // 抽象方法：子类实现具体逻辑
    protected abstract boolean doProcessPayment(double amount);

    // 受保护方法：子类可以重写
    protected void incrementTransactionCount() {
        transactionCount++;
    }

    // 包级私有方法：同包工具类可以访问
    int getTransactionCount() {
        return transactionCount;
    }

    // 公共方法：外部接口
    public boolean isEnabled() {
        return isEnabled;
    }

    public void setEnabled(boolean enabled) {
        this.isEnabled = enabled;
    }
}

public class CommonMistakes {

    public List<String> items;  // 应该是私有的，提供受控访问

    // 错误3：构造器访问权限不当

public class CorrectDesign {

    private String name;

    protected String status;

    // 公共构造器
    public CorrectDesign(String name) {

        this.items = new ArrayList<>();
        this.status = "INITIALIZED";

💡 最佳实践总结 ​

设计原则 ​

  1. 最小权限原则：给予最小必要的访问权限
  2. 封装原则：隐藏实现细节，暴露必要接口
  3. 一致性原则：在整个项目中保持一致的访问权限策略
  4. 可维护性原则：考虑未来的扩展和维护需求

实践建议 ​

  1. 字段默认私有：除非有特殊需求，字段应该是私有的
  2. 方法按需公开：只公开必要的方法，内部方法保持私有
  3. 构造器合理设计：根据对象创建需求设计构造器访问权限
  4. 继承谨慎设计：合理使用 protected，避免过度暴露
