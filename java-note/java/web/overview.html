Java Web开发详解 ​

📚 概述 ​

Java Web开发是企业级应用开发的重要组成部分，涵盖了从传统的Servlet/JSP技术到现代的Spring Boot微服务架构。Java Web技术栈丰富多样，包括Servlet容器、MVC框架、模板引擎、RESTful API、微服务等多个层面。随着云计算和微服务架构的兴起，Java Web开发也在不断演进，从单体应用向分布式、云原生应用转变。掌握Java Web开发不仅需要理解HTTP协议和Web应用架构，还需要熟悉各种框架和工具的使用。本文将从Servlet基础开始，逐步深入到现代Web开发的各个方面，帮助开发者建立完整的Web开发知识体系。

🎯 学习目标 ​

  • 深入理解HTTP协议和Web应用架构
  • 掌握Servlet和JSP的核心概念和使用
  • 熟练使用Spring框架进行Web开发
  • 理解MVC设计模式和RESTful API设计
  • 学会处理Web安全和会话管理
  • 掌握前后端分离开发模式
  • 了解微服务架构和云原生开发
  • 学会Web应用的部署和运维

// Cookie管理器
class CookieManager {
    private Map<String, CookieInfo> cookies = new HashMap<>();
    
    public void setCookie(String name, String value, int maxAge) {
        cookies.put(name, new CookieInfo(value, maxAge));
        System.out.println("设置Cookie: " + name + "=" + value + " (有效期: " + maxAge + "秒)");
    }
    
    public String getCookie(String name) {
        CookieInfo cookie = cookies.get(name);
        return cookie != null ? cookie.value : null;
    }
    
    public void deleteCookie(String name) {
        cookies.remove(name);
        System.out.println("删除Cookie: " + name);
    }
    
    public void printCookies() {
        for (Map.Entry<String, CookieInfo> entry : cookies.entrySet()) {
            System.out.println("  " + entry.getKey() + "=" + entry.getValue().value);
        }
    }
    
    private static class CookieInfo {
        String value;
        int maxAge;
        
        CookieInfo(String value, int maxAge) {
            this.value = value;
            this.maxAge = maxAge;
        }
    }
}

// Session管理器
class SessionManager {
    private Map<String, Map<String, Object>> sessions = new HashMap<>();
    private int sessionCounter = 1;
    
    public String createSession() {
        String sessionId = "SESSION_" + sessionCounter++;
        sessions.put(sessionId, new HashMap<>());
        return sessionId;
    }
    
    public void setAttribute(String sessionId, String name, Object value) {
        Map<String, Object> session = sessions.get(sessionId);
        if (session != null) {
            session.put(name, value);
        }
    }
    
    public Object getAttribute(String sessionId, String name) {
        Map<String, Object> session = sessions.get(sessionId);
        return session != null ? session.get(name) : null;
    }
    
    public void printSessionAttributes(String sessionId) {
        Map<String, Object> session = sessions.get(sessionId);
        if (session != null) {
            for (Map.Entry<String, Object> entry : session.entrySet()) {
                System.out.println("  " + entry.getKey() + ": " + entry.getValue());
            }
        }
    }
    
    public void invalidateSession(String sessionId) {
        sessions.remove(sessionId);
    }
}

// JWT管理器（简化版）
class JWTManager {
    private static final String SECRET = "mySecretKey";
    
    public String createToken(Map<String, Object> claims, int expirationSeconds) {
        // 简化的JWT创建（实际应用中应使用专业的JWT库）
        StringBuilder token = new StringBuilder();
        token.append("header."); // JWT头部
        
        // 添加claims
        for (Map.Entry<String, Object> entry : claims.entrySet()) {
            token.append(entry.getKey()).append(":").append(entry.getValue()).append(";");
        }
        
        token.append(".signature"); // JWT签名
        
        return token.toString();
    }
    
    public boolean validateToken(String token) {
        // 简化的JWT验证
        return token != null && token.contains("header.") && token.contains(".signature");
    }
    
    public Map<String, Object> parseToken(String token) {
        Map<String, Object> claims = new HashMap<>();
        
        // 简化的JWT解析
        if (validateToken(token)) {
            String[] parts = token.split("\\.");
            if (parts.length >= 2) {
                String claimsStr = parts[1];
                String[] claimPairs = claimsStr.split(";");
                
                for (String pair : claimPairs) {
                    if (pair.contains(":")) {
                        String[] keyValue = pair.split(":", 2);
                        claims.put(keyValue[0], keyValue[1]);
                    }
                }
            }
        }
        
        return claims;
    }
}

                } else if (path.matches("/api/users/\\d+")) {
                    System.out.println("Response: 200 OK");
                    System.out.println("Body: {\"id\":1,\"name\":\"Alice\",\"email\":\"alice@example.com\"}");
                }
                break;
            case "POST":
                System.out.println("Response: 201 Created");
                System.out.println("Body: {\"id\":3,\"name\":\"Alice\",\"email\":\"alice@example.com\"}");
                break;
            case "PUT":
                System.out.println("Response: 200 OK");
                System.out.println("Body: {\"id\":1,\"name\":\"Alice Smith\",\"email\":\"alice@example.com\"}");
                break;
            case "DELETE":
                System.out.println("Response: 204 No Content");
                break;
        }
    }
}

// API版本管理器
class APIVersionManager {
    public void handleVersionedRequest(String version, String method, String path) {
        System.out.println("\n处理API请求 - 版本: " + version + ", " + method + " " + path);
        
        switch (version) {
            case "v1":
                System.out.println("使用v1版本处理逻辑");
                System.out.println("返回基础用户信息");
                break;
            case "v2":
                System.out.println("使用v2版本处理逻辑");
                System.out.println("返回扩展用户信息（包含头像、偏好设置）");
                break;
            default:
                System.out.println("不支持的API版本: " + version);
                System.out.println("Response: 400 Bad Request");
        }
    }
}
