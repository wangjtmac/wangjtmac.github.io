        for (MSTEdge edge : edges) {
            System.out.println("  " + edge.source + "-" + edge.destination + 
                             " (æƒé‡: " + edge.weight + ")");
        }
        
        System.out.println("\né€‰æ‹©è¿‡ç¨‹:");

            if (uf.find(edge.source) != uf.find(edge.destination)) {
                uf.union(edge.source, edge.destination);
                mstEdges.add(edge);

                                 " (æƒé‡: " + edge.weight + ")");
                
                if (mstEdges.size() == graph.getVertices() - 1) {

            } else {

    // Primç®—æ³•
    public static PrimResult primMST(MSTGraph graph, int startVertex) {
        int vertices = graph.getVertices();
        boolean[] inMST = new boolean[vertices];
        int[] key = new int[vertices];
        int[] parent = new int[vertices];
        
        Arrays.fill(key, Integer.MAX_VALUE);
        Arrays.fill(parent, -1);
        key[startVertex] = 0;
        
        PriorityQueue<PrimNode> pq = new PriorityQueue<>((a, b) -> a.key - b.key);
        pq.offer(new PrimNode(startVertex, 0));
        
        List<MSTEdge> mstEdges = new ArrayList<>();
        int totalWeight = 0;
        
        System.out.println("ä»é¡¶ç‚¹ " + startVertex + " å¼€å§‹:");
        
        while (!pq.isEmpty()) {
            PrimNode current = pq.poll();
            int u = current.vertex;
            
            if (inMST[u]) continue;
            
            inMST[u] = true;
            
            if (parent[u] != -1) {
                MSTEdge edge = new MSTEdge(parent[u], u, key[u]);
                mstEdges.add(edge);
                totalWeight += key[u];
                
                System.out.println("æ·»åŠ è¾¹: " + parent[u] + "-" + u + 
                                 " (æƒé‡: " + key[u] + ")");
            }
            
            // æ›´æ–°ç›¸é‚»é¡¶ç‚¹çš„keyå€¼
            for (MSTEdge edge : graph.getAdjacencyList().get(u)) {
                int v = edge.destination;
                int weight = edge.weight;
                
                if (!inMST[v] && weight < key[v]) {
                    key[v] = weight;
                    parent[v] = u;
                    pq.offer(new PrimNode(v, key[v]));
                }
            }
        }
        
        return new PrimResult(mstEdges, totalWeight);
    }

    public static SecondMSTResult findSecondMinimumSpanningTree(MSTGraph graph) {
        // å…ˆæ‰¾åˆ°MST
        KruskalResult firstMST = kruskalMST(graph);
        int firstWeight = firstMST.totalWeight;
        
        // å°è¯•æ›¿æ¢MSTä¸­çš„æ¯æ¡è¾¹
        int secondWeight = Integer.MAX_VALUE;
        List<MSTEdge> allEdges = graph.getAllEdges();
        
        for (MSTEdge mstEdge : firstMST.mstEdges) {
            // ç§»é™¤MSTä¸­çš„ä¸€æ¡è¾¹ï¼Œå°è¯•ç”¨å…¶ä»–è¾¹æ›¿æ¢
            List<MSTEdge> candidateEdges = new ArrayList<>();
            for (MSTEdge edge : allEdges) {
                if (!edge.equals(mstEdge)) {
                    candidateEdges.add(edge);
                }
            }
            
            // ç”¨å‰©ä½™è¾¹æ„å»ºMST
            Collections.sort(candidateEdges, (a, b) -> a.weight - b.weight);
            UnionFind uf = new UnionFind(graph.getVertices());
            int weight = 0;
            int edgeCount = 0;
            
            for (MSTEdge edge : candidateEdges) {
                if (uf.find(edge.source) != uf.find(edge.destination)) {
                    uf.union(edge.source, edge.destination);
                    weight += edge.weight;
                    edgeCount++;
                    
                    if (edgeCount == graph.getVertices() - 1) {
                        break;
                    }
                }
            }
            
            if (edgeCount == graph.getVertices() - 1 && weight > firstWeight) {
                secondWeight = Math.min(secondWeight, weight);
            }
        }
        
        return new SecondMSTResult(firstWeight, secondWeight);
    }

    public static BottleneckMSTResult findBottleneckSpanningTree(MSTGraph graph) {
        List<MSTEdge> edges = graph.getAllEdges();
        Collections.sort(edges, (a, b) -> a.weight - b.weight);
        
        UnionFind uf = new UnionFind(graph.getVertices());
        List<MSTEdge> mstEdges = new ArrayList<>();
        int maxWeight = 0;
        
        for (MSTEdge edge : edges) {
            if (uf.find(edge.source) != uf.find(edge.destination)) {
                uf.union(edge.source, edge.destination);
                mstEdges.add(edge);
                maxWeight = Math.max(maxWeight, edge.weight);
                
                if (mstEdges.size() == graph.getVertices() - 1) {
                    break;
                }
            }
        }
        
        return new BottleneckMSTResult(mstEdges, maxWeight);
    }

class MSTGraph {
    private int vertices;
    private List<List<MSTEdge>> adjacencyList;
    
    public MSTGraph(int vertices) {
        this.vertices = vertices;
        this.adjacencyList = new ArrayList<>();
        
        for (int i = 0; i < vertices; i++) {
            adjacencyList.add(new ArrayList<>());
        }
    }
    
    public void addEdge(int source, int destination, int weight) {
        adjacencyList.get(source).add(new MSTEdge(destination, destination, weight));
        adjacencyList.get(destination).add(new MSTEdge(source, source, weight));
    }
    
    public int getVertices() {
        return vertices;
    }
    
    public List<List<MSTEdge>> getAdjacencyList() {
        return adjacencyList;
    }
    
    public List<MSTEdge> getAllEdges() {
        List<MSTEdge> edges = new ArrayList<>();
        Set<String> added = new HashSet<>();
        
        for (int i = 0; i < vertices; i++) {
            for (MSTEdge edge : adjacencyList.get(i)) {
                String edgeKey = Math.min(i, edge.destination) + "-" + 
                               Math.max(i, edge.destination);
                
                if (!added.contains(edgeKey)) {
                    edges.add(new MSTEdge(i, edge.destination, edge.weight));
                    added.add(edgeKey);
                }
            }
        }
        
        return edges;
    }
    
    public void printGraph() {
        for (int i = 0; i < vertices; i++) {
            System.out.print("é¡¶ç‚¹ " + i + ": ");
            for (MSTEdge edge : adjacencyList.get(i)) {
                System.out.print("(" + edge.destination + "," + edge.weight + ") ");
            }
            System.out.println();
        }
    }
}

class UnionFind {
    private int[] parent;
    private int[] rank;
    
    public UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            rank[i] = 0;
        }
    }
    
    public int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]); // è·¯å¾„å‹ç¼©
        }
        return parent[x];
    }
    
    public void union(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        
        if (rootX != rootY) {
            // æŒ‰ç§©åˆå¹¶
            if (rank[rootX] < rank[rootY]) {
                parent[rootX] = rootY;
            } else if (rank[rootX] > rank[rootY]) {
                parent[rootY] = rootX;
            } else {
                parent[rootY] = rootX;
                rank[rootX]++;
            }
        }
    }
}

class ClusterAnalysis {
    private List<DataPoint> dataPoints;
    private MSTGraph distanceGraph;
    
    public ClusterAnalysis() {
        this.dataPoints = new ArrayList<>();
    }
    
    public void addDataPoint(int id, double x, double y) {
        dataPoints.add(new DataPoint(id, x, y));
    }
    
    public void printDataPoints() {
        for (DataPoint point : dataPoints) {
            System.out.println("  ç‚¹" + point.id + ": (" + point.x + ", " + point.y + ")");
        }
    }
    
    public ClusterResult performClustering(int numClusters) {
        int n = dataPoints.size();
        distanceGraph = new MSTGraph(n);
        
        // è®¡ç®—æ‰€æœ‰ç‚¹å¯¹ä¹‹é—´çš„è·ç¦»
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                double distance = calculateDistance(dataPoints.get(i), dataPoints.get(j));
                distanceGraph.addEdge(i, j, (int) (distance * 100)); // æ”¾å¤§100å€é¿å…ç²¾åº¦é—®é¢˜
            }
        }
        
        // æ„å»ºMST
        KruskalResult mst = MinimumSpanningTreeDemo.kruskalMST(distanceGraph);
        
        // ç§»é™¤æœ€å¤§çš„k-1æ¡è¾¹æ¥å½¢æˆkä¸ªç°‡
        List<MSTEdge> sortedEdges = new ArrayList<>(mst.mstEdges);
        sortedEdges.sort((a, b) -> b.weight - a.weight);
        
        Set<MSTEdge> removedEdges = new HashSet<>();
        for (int i = 0; i < numClusters - 1; i++) {
            removedEdges.add(sortedEdges.get(i));
        }
        
        // ä½¿ç”¨å¹¶æŸ¥é›†æ‰¾åˆ°è¿é€šåˆ†é‡
        UnionFind uf = new UnionFind(n);
        for (MSTEdge edge : mst.mstEdges) {
            if (!removedEdges.contains(edge)) {
                uf.union(edge.source, edge.destination);
            }
        }
        
        // åˆ†ç»„
        Map<Integer, List<Integer>> clusterMap = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int root = uf.find(i);
            clusterMap.computeIfAbsent(root, k -> new ArrayList<>()).add(i);
        }
        
        List<List<Integer>> clusters = new ArrayList<>(clusterMap.values());
        return new ClusterResult(clusters);
    }
    
    private double calculateDistance(DataPoint p1, DataPoint p2) {
        double dx = p1.x - p2.x;
        double dy = p1.y - p2.y;
        return Math.sqrt(dx * dx + dy * dy);
    }
}

class TSPApproximation {
    private List<City> cities;
    
    public TSPApproximation() {
        this.cities = new ArrayList<>();
    }
    
    public void addCity(int id, double x, double y) {
        cities.add(new City(id, x, y));
    }
    
    public void printCities() {
        for (City city : cities) {
            System.out.println("  åŸå¸‚" + city.id + ": (" + city.x + ", " + city.y + ")");
        }
    }
    
    public TSPResult approximateTSP() {
        int n = cities.size();
        MSTGraph graph = new MSTGraph(n);
        
        // æ„å»ºå®Œå…¨å›¾
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                double distance = calculateDistance(cities.get(i), cities.get(j));
                graph.addEdge(i, j, (int) (distance * 100));
            }
        }
        
        // æ„å»ºMST
        PrimResult mst = MinimumSpanningTreeDemo.primMST(graph, 0);
        
        // DFSéå†MSTå¾—åˆ°è¿‘ä¼¼TSPè·¯å¾„
        List<Integer> tour = new ArrayList<>();
        boolean[] visited = new boolean[n];
        dfsTraversal(0, mst.mstEdges, visited, tour);
        tour.add(0); // å›åˆ°èµ·ç‚¹
        
        // è®¡ç®—æ€»è·ç¦»
        double totalDistance = 0;
        for (int i = 0; i < tour.size() - 1; i++) {
            totalDistance += calculateDistance(cities.get(tour.get(i)), cities.get(tour.get(i + 1)));
        }
        
        return new TSPResult(tour, totalDistance);
    }
    
    private void dfsTraversal(int current, List<MSTEdge> mstEdges, boolean[] visited, List<Integer> tour) {
        visited[current] = true;
        tour.add(current);
        
        for (MSTEdge edge : mstEdges) {
            int next = -1;
            if (edge.source == current && !visited[edge.destination]) {
                next = edge.destination;
            } else if (edge.destination == current && !visited[edge.source]) {
                next = edge.source;
            }
            
            if (next != -1) {
                dfsTraversal(next, mstEdges, visited, tour);
            }
        }
    }
    
    private double calculateDistance(City c1, City c2) {
        double dx = c1.x - c2.x;
        double dy = c1.y - c2.y;
        return Math.sqrt(dx * dx + dy * dy);
    }
}
