    public static int countIslands(char[][] grid) {
        if (grid == null || grid.length == 0) return 0;
        
        int count = 0;
        boolean[][] visited = new boolean[grid.length][grid[0].length];
        
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1' && !visited[i][j]) {
                    dfsIsland(grid, i, j, visited);
                    count++;
                }
            }
        }
        
        return count;
    }
    
    private static void dfsIsland(char[][] grid, int i, int j, boolean[][] visited) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length || 
            visited[i][j] || grid[i][j] == '0') {
            return;
        }
        
        visited[i][j] = true;
        
        // è®¿é—®å››ä¸ªæ–¹å‘
        dfsIsland(grid, i + 1, j, visited);
        dfsIsland(grid, i - 1, j, visited);
        dfsIsland(grid, i, j + 1, visited);
        dfsIsland(grid, i, j - 1, visited);
    }
    
    public static List<int[]> solveMaze(int[][] maze, int[] start, int[] end) {
        int rows = maze.length, cols = maze[0].length;
        boolean[][] visited = new boolean[rows][cols];
        List<int[]> path = new ArrayList<>();
        
        if (solveMazeDFS(maze, start[0], start[1], end[0], end[1], visited, path)) {
            return path;
        }
        
        return null;
    }
    
    private static boolean solveMazeDFS(int[][] maze, int x, int y, int endX, int endY, 
                                      boolean[][] visited, List<int[]> path) {
        if (x < 0 || x >= maze.length || y < 0 || y >= maze[0].length || 
            visited[x][y] || maze[x][y] == 1) {
            return false;
        }
        
        path.add(new int[]{x, y});
        visited[x][y] = true;
        
        if (x == endX && y == endY) {
            return true;
        }
        
        // å°è¯•å››ä¸ªæ–¹å‘
        int[][] directions = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        for (int[] dir : directions) {
            if (solveMazeDFS(maze, x + dir[0], y + dir[1], endX, endY, visited, path)) {
                return true;
            }
        }
        
        path.remove(path.size() - 1); // å›æº¯
        return false;
    }
    
    public static void performanceComparison(Graph graph) {
        int iterations = 1000;
        
        // DFSæ€§èƒ½æµ‹è¯•
        long startTime = System.nanoTime();
        for (int i = 0; i < iterations; i++) {

        }

        // BFSæ€§èƒ½æµ‹è¯•

            graph.bfs(0);
        }
        long bfsTime = (System.nanoTime() - startTime) / iterations;
        
        System.out.println("å¹³å‡æ‰§è¡Œæ—¶é—´ (" + iterations + " æ¬¡):");
        System.out.println("DFS: " + dfsTime + " çº³ç§’");
        System.out.println("BFS: " + bfsTime + " çº³ç§’");
        System.out.println("æ€§èƒ½æ¯”: " + (double) Math.max(dfsTime, bfsTime) / Math.min(dfsTime, bfsTime));
    }

class Graph {
    private Map<Integer, List<Integer>> adjacencyList;
    
    public Graph() {
        adjacencyList = new HashMap<>();
    }
    
    public void addVertex(int vertex) {
        adjacencyList.putIfAbsent(vertex, new ArrayList<>());
    }
    
    public void addEdge(int u, int v) {
        addVertex(u);
        addVertex(v);
        adjacencyList.get(u).add(v);
        adjacencyList.get(v).add(u);
    }
    
    public void printGraph() {
        for (Map.Entry<Integer, List<Integer>> entry : adjacencyList.entrySet()) {
            System.out.println("é¡¶ç‚¹ " + entry.getKey() + ": " + entry.getValue());
        }

    // DFSé€’å½’å®ç°
    public List<Integer> dfsRecursive(int start) {
        List<Integer> result = new ArrayList<>();
        Set<Integer> visited = new HashSet<>();
        dfsHelper(start, visited, result);
        return result;
    }
    
    private void dfsHelper(int vertex, Set<Integer> visited, List<Integer> result) {
        visited.add(vertex);
        result.add(vertex);
        
        for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
            if (!visited.contains(neighbor)) {
                dfsHelper(neighbor, visited, result);
            }
        }
    }
    
    // DFSè¿­ä»£å®ç°
    public List<Integer> dfsIterative(int start) {
        List<Integer> result = new ArrayList<>();
        Set<Integer> visited = new HashSet<>();
        Stack<Integer> stack = new Stack<>();
        
        stack.push(start);
        
        while (!stack.isEmpty()) {
            int vertex = stack.pop();
            
            if (!visited.contains(vertex)) {
                visited.add(vertex);
                result.add(vertex);
                
                // é€†åºæ·»åŠ é‚»å±…ä»¥ä¿æŒä¸é€’å½’ç‰ˆæœ¬ç›¸åŒçš„é¡ºåº
                List<Integer> neighbors = adjacencyList.getOrDefault(vertex, new ArrayList<>());
                for (int i = neighbors.size() - 1; i >= 0; i--) {
                    if (!visited.contains(neighbors.get(i))) {
                        stack.push(neighbors.get(i));
                    }
                }
            }
        }
        
        return result;
    }
    
    // BFSå®ç°
    public List<Integer> bfs(int start) {
        List<Integer> result = new ArrayList<>();
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        
        queue.offer(start);
        visited.add(start);
        
        while (!queue.isEmpty()) {
            int vertex = queue.poll();
            result.add(vertex);
            
            for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    queue.offer(neighbor);
                }
            }
        }
        
        return result;
    }
    
    // BFSå±‚æ¬¡éå†
    public List<List<Integer>> bfsLevels(int start) {
        List<List<Integer>> levels = new ArrayList<>();
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        
        queue.offer(start);
        visited.add(start);
        
        while (!queue.isEmpty()) {
            int size = queue.size();
            List<Integer> currentLevel = new ArrayList<>();
            
            for (int i = 0; i < size; i++) {
                int vertex = queue.poll();
                currentLevel.add(vertex);
                
                for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
                    if (!visited.contains(neighbor)) {
                        visited.add(neighbor);
                        queue.offer(neighbor);
                    }
                }
            }
            
            levels.add(currentLevel);
        }
        
        return levels;
    }
    
    // DFSè·¯å¾„æŸ¥æ‰¾
    public List<Integer> findPathDFS(int start, int end) {
        Set<Integer> visited = new HashSet<>();
        List<Integer> path = new ArrayList<>();
        
        if (findPathDFSHelper(start, end, visited, path)) {
            return path;
        }
        
        return null;
    }
    
    private boolean findPathDFSHelper(int current, int end, Set<Integer> visited, List<Integer> path) {
        visited.add(current);
        path.add(current);
        
        if (current == end) {
            return true;
        }
        
        for (int neighbor : adjacencyList.getOrDefault(current, new ArrayList<>())) {
            if (!visited.contains(neighbor)) {
                if (findPathDFSHelper(neighbor, end, visited, path)) {
                    return true;
                }
            }
        }
        
        path.remove(path.size() - 1); // å›æº¯
        return false;
    }
    
    // BFSæœ€çŸ­è·¯å¾„
    public List<Integer> shortestPathBFS(int start, int end) {
        Map<Integer, Integer> parent = new HashMap<>();
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        
        queue.offer(start);
        visited.add(start);
        parent.put(start, null);
        
        while (!queue.isEmpty()) {
            int vertex = queue.poll();
            
            if (vertex == end) {
                // é‡æ„è·¯å¾„
                List<Integer> path = new ArrayList<>();
                Integer current = end;
                while (current != null) {
                    path.add(current);
                    current = parent.get(current);
                }
                Collections.reverse(path);
                return path;
            }
            
            for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    parent.put(neighbor, vertex);
                    queue.offer(neighbor);
                }
            }
        }
        
        return null;
    }

    public Map<Integer, Integer> bfsDistances(int start) {
        Map<Integer, Integer> distances = new HashMap<>();
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        
        queue.offer(start);
        visited.add(start);
        distances.put(start, 0);
        
        while (!queue.isEmpty()) {
            int vertex = queue.poll();
            int currentDistance = distances.get(vertex);
            
            for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    distances.put(neighbor, currentDistance + 1);
                    queue.offer(neighbor);
                }
            }
        }
        
        return distances;
    }

    public Map<Integer, Integer> multiSourceBFS(List<Integer> sources) {
        Map<Integer, Integer> distances = new HashMap<>();
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        
        // åˆå§‹åŒ–æ‰€æœ‰æºç‚¹
        for (int source : sources) {
            queue.offer(source);
            visited.add(source);
            distances.put(source, 0);
        }
        
        while (!queue.isEmpty()) {
            int vertex = queue.poll();
            int currentDistance = distances.get(vertex);
            
            for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    distances.put(neighbor, currentDistance + 1);
                    queue.offer(neighbor);
                }
            }
        }
        
        return distances;
    }

    public List<List<Integer>> findConnectedComponents() {
        List<List<Integer>> components = new ArrayList<>();
        Set<Integer> globalVisited = new HashSet<>();
        
        for (int vertex : adjacencyList.keySet()) {
            if (!globalVisited.contains(vertex)) {
                List<Integer> component = new ArrayList<>();
                Set<Integer> visited = new HashSet<>();
                dfsComponent(vertex, visited, component);
                globalVisited.addAll(visited);
                components.add(component);
            }
        }
        
        return components;
    }
    
    private void dfsComponent(int vertex, Set<Integer> visited, List<Integer> component) {
        visited.add(vertex);
        component.add(vertex);
        
        for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
            if (!visited.contains(neighbor)) {
                dfsComponent(neighbor, visited, component);
            }
        }
    }

    public boolean hasCycleDFS() {
        Set<Integer> visited = new HashSet<>();
        
        for (int vertex : adjacencyList.keySet()) {
            if (!visited.contains(vertex)) {
                if (hasCycleDFSHelper(vertex, -1, visited)) {
                    return true;
                }
            }
        }
        
        return false;
    }
    
    private boolean hasCycleDFSHelper(int vertex, int parent, Set<Integer> visited) {
        visited.add(vertex);
        
        for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
            if (!visited.contains(neighbor)) {
                if (hasCycleDFSHelper(neighbor, vertex, visited)) {
                    return true;
                }
            } else if (neighbor != parent) {
                return true;
            }
        }
        
        return false;
    }

    public boolean isBipartite() {
        Map<Integer, Integer> colors = new HashMap<>();
        
        for (int vertex : adjacencyList.keySet()) {
            if (!colors.containsKey(vertex)) {
                if (!isBipartiteBFS(vertex, colors)) {
                    return false;
                }
            }
        }
        
        return true;
    }
    
    private boolean isBipartiteBFS(int start, Map<Integer, Integer> colors) {
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(start);
        colors.put(start, 0);
        
        while (!queue.isEmpty()) {
            int vertex = queue.poll();
            int currentColor = colors.get(vertex);
            
            for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
                if (!colors.containsKey(neighbor)) {
                    colors.put(neighbor, 1 - currentColor);
                    queue.offer(neighbor);
                } else if (colors.get(neighbor) == currentColor) {
                    return false;
                }
            }
        }
        
        return true;
    }
    
    public Map<Integer, Integer> getBipartiteColoring() {
        Map<Integer, Integer> colors = new HashMap<>();
        
        for (int vertex : adjacencyList.keySet()) {
            if (!colors.containsKey(vertex)) {
                isBipartiteBFS(vertex, colors);
            }
        }
        
        return colors;
    }

class DirectedGraph {
    private Map<Integer, List<Integer>> adjacencyList;
    
    public DirectedGraph() {
        adjacencyList = new HashMap<>();
    }
    
    public void addVertex(int vertex) {
        adjacencyList.putIfAbsent(vertex, new ArrayList<>());
    }
    
    public void addEdge(int from, int to) {
        addVertex(from);
        addVertex(to);
        adjacencyList.get(from).add(to);
    }
    
    public void printGraph() {
        for (Map.Entry<Integer, List<Integer>> entry : adjacencyList.entrySet()) {
            System.out.println("é¡¶ç‚¹ " + entry.getKey() + " -> " + entry.getValue());
        }
    }
    
    // æ‹“æ‰‘æ’åº
    public List<Integer> topologicalSort() {
        Stack<Integer> stack = new Stack<>();
        Set<Integer> visited = new HashSet<>();
        
        for (int vertex : adjacencyList.keySet()) {
            if (!visited.contains(vertex)) {
                topologicalSortDFS(vertex, visited, stack);
            }
        }
        
        List<Integer> result = new ArrayList<>();
        while (!stack.isEmpty()) {
            result.add(stack.pop());
        }
        
        return result;
    }
    
    private void topologicalSortDFS(int vertex, Set<Integer> visited, Stack<Integer> stack) {
        visited.add(vertex);
        
        for (int neighbor : adjacencyList.getOrDefault(vertex, new ArrayList<>())) {
            if (!visited.contains(neighbor)) {
                topologicalSortDFS(neighbor, visited, stack);
            }
        }
        
        stack.push(vertex);
    }
