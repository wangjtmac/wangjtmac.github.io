最短路径算法详解 ​

📚 概述 ​

最短路径算法是图论中最重要和最实用的算法之一，用于在加权图中找到两个节点之间的最短路径。这类算法在现实世界中有广泛应用，从GPS导航系统到网络路由，从游戏AI到社交网络分析，都离不开最短路径算法。根据问题的不同特点，发展出了多种经典算法：Dijkstra算法解决单源最短路径问题，Floyd-Warshall算法解决所有节点对之间的最短路径，Bellman-Ford算法可以处理负权边，A*算法在启发式搜索中表现优异。理解这些算法的原理、适用场景和实现细节，对于解决实际的路径优化问题具有重要意义。

🎯 学习目标 ​

  • 深入理解最短路径问题的本质和分类
  • 掌握Dijkstra、Floyd-Warshall、Bellman-Ford等经典算法
  • 学会分析不同算法的时间复杂度和适用场景
  • 理解负权边、负权回路对算法的影响
  • 掌握路径重构和实际路径输出技巧
  • 学会在实际问题中选择合适的最短路径算法
  • 了解最短路径算法的优化和变种
  • 培养图论思维和算法设计能力

    public static void demonstrateDijkstraAlgorithm() {
        System.out.println("=== Dijkstra算法演示 ===");

         * Dijkstra算法特点：
         * 1. 单源最短路径 - 从一个源点到所有其他点的最短路径
         * 2. 非负权重 - 要求所有边的权重非负
         * 3. 贪心策略 - 每次选择距离最小的未访问节点
         * 4. 时间复杂度 - O((V+E)logV) 使用优先队列
         * 5. 空间复杂度 - O(V) 存储距离和前驱信息
         */
        
        System.out.println("Dijkstra算法特点:");

        System.out.println("2. 非负权重限制 - 不能处理负权边");
        System.out.println("3. 贪心策略 - 逐步确定最短距离");
        System.out.println("4. 优先队列优化 - 提高算法效率");

        graph.addEdge(0, 1, 4);
        graph.addEdge(0, 2, 2);

        graph.addEdge(2, 4, 10);

        graph.addEdge(3, 5, 6);
        graph.addEdge(4, 5, 3);

        }

    // Dijkstra算法
    public static DijkstraResult dijkstra(WeightedGraph graph, int source) {
        int vertices = graph.getVertices();
        int[] distances = new int[vertices];
        int[] predecessors = new int[vertices];
        boolean[] visited = new boolean[vertices];
        
        Arrays.fill(distances, Integer.MAX_VALUE);
        Arrays.fill(predecessors, -1);
        distances[source] = 0;
        
        PriorityQueue<Node> pq = new PriorityQueue<>((a, b) -> a.distance - b.distance);
        pq.offer(new Node(source, 0));
        
        while (!pq.isEmpty()) {
            Node current = pq.poll();
            int u = current.vertex;
            
            if (visited[u]) continue;
            visited[u] = true;
            
            for (Edge edge : graph.getAdjacencyList().get(u)) {
                int v = edge.destination;
                int weight = edge.weight;
                
                if (!visited[v] && distances[u] + weight < distances[v]) {
                    distances[v] = distances[u] + weight;
                    predecessors[v] = u;
                    pq.offer(new Node(v, distances[v]));
                }
            }
        }
        
        return new DijkstraResult(distances, predecessors);
    }
    
    // 带步骤的Dijkstra算法
    public static void dijkstraWithSteps(WeightedGraph graph, int source) {
        int vertices = graph.getVertices();
        int[] distances = new int[vertices];
        boolean[] visited = new boolean[vertices];
        
        Arrays.fill(distances, Integer.MAX_VALUE);
        distances[source] = 0;
        
        System.out.println("初始状态: " + Arrays.toString(distances));
        
        for (int step = 0; step < vertices; step++) {
            // 找到未访问的最小距离节点
            int u = -1;
            for (int v = 0; v < vertices; v++) {
                if (!visited[v] && (u == -1 || distances[v] < distances[u])) {
                    u = v;
                }
            }
            
            if (u == -1 || distances[u] == Integer.MAX_VALUE) break;
            
            visited[u] = true;
            System.out.println("步骤 " + (step + 1) + ": 访问节点 " + u + 
                             " (距离: " + distances[u] + ")");
            
            // 松弛相邻节点
            for (Edge edge : graph.getAdjacencyList().get(u)) {
                int v = edge.destination;
                int weight = edge.weight;
                
                if (!visited[v] && distances[u] + weight < distances[v]) {
                    int oldDistance = distances[v];
                    distances[v] = distances[u] + weight;
                    System.out.println("  更新节点 " + v + " 的距离: " + 
                                     oldDistance + " -> " + distances[v]);
                }
            }
            
            System.out.println("  当前距离: " + Arrays.toString(distances));
        }
    }
    
    // Bellman-Ford算法
    public static BellmanFordResult bellmanFord(WeightedGraph graph, int source) {
        int vertices = graph.getVertices();
        int[] distances = new int[vertices];
        int[] predecessors = new int[vertices];
        
        Arrays.fill(distances, Integer.MAX_VALUE);
        Arrays.fill(predecessors, -1);
        distances[source] = 0;
        
        // 松弛所有边 V-1 次
        for (int i = 0; i < vertices - 1; i++) {
            for (int u = 0; u < vertices; u++) {
                if (distances[u] != Integer.MAX_VALUE) {
                    for (Edge edge : graph.getAdjacencyList().get(u)) {
                        int v = edge.destination;
                        int weight = edge.weight;
                        
                        if (distances[u] + weight < distances[v]) {
                            distances[v] = distances[u] + weight;
                            predecessors[v] = u;
                        }
                    }
                }
            }
        }
        
        // 检测负权回路
        boolean hasNegativeCycle = false;
        for (int u = 0; u < vertices; u++) {
            if (distances[u] != Integer.MAX_VALUE) {
                for (Edge edge : graph.getAdjacencyList().get(u)) {
                    int v = edge.destination;
                    int weight = edge.weight;
                    
                    if (distances[u] + weight < distances[v]) {
                        hasNegativeCycle = true;
                        break;
                    }
                }
            }
            if (hasNegativeCycle) break;
        }
        
        return new BellmanFordResult(distances, predecessors, hasNegativeCycle);
    }
    
    // Floyd-Warshall算法
    public static FloydWarshallResult floydWarshall(int[][] graph) {
        int vertices = graph.length;
        int[][] distances = new int[vertices][vertices];
        int[][] next = new int[vertices][vertices];
        
        // 初始化
        for (int i = 0; i < vertices; i++) {
            for (int j = 0; j < vertices; j++) {
                distances[i][j] = graph[i][j];
                if (graph[i][j] != Integer.MAX_VALUE && i != j) {
                    next[i][j] = j;
                } else {
                    next[i][j] = -1;
                }
            }
        }
        
        // Floyd-Warshall核心算法
        for (int k = 0; k < vertices; k++) {
            for (int i = 0; i < vertices; i++) {
                for (int j = 0; j < vertices; j++) {
                    if (distances[i][k] != Integer.MAX_VALUE && 
                        distances[k][j] != Integer.MAX_VALUE &&
                        distances[i][k] + distances[k][j] < distances[i][j]) {
                        distances[i][j] = distances[i][k] + distances[k][j];
                        next[i][j] = next[i][k];
                    }
                }
            }
        }
        
        return new FloydWarshallResult(distances, next);
    }
    
    // 带步骤的Floyd-Warshall算法
    public static void floydWarshallWithSteps(int[][] graph) {
        int vertices = graph.length;
        int[][] distances = new int[vertices][vertices];
        
        // 初始化
        for (int i = 0; i < vertices; i++) {
            for (int j = 0; j < vertices; j++) {
                distances[i][j] = graph[i][j];
            }
        }
        
        for (int k = 0; k < vertices; k++) {
            System.out.println("通过节点 " + k + " 作为中间节点:");
            
            for (int i = 0; i < vertices; i++) {
                for (int j = 0; j < vertices; j++) {
                    if (distances[i][k] != Integer.MAX_VALUE && 
                        distances[k][j] != Integer.MAX_VALUE &&
                        distances[i][k] + distances[k][j] < distances[i][j]) {
                        
                        int oldDistance = distances[i][j];
                        distances[i][j] = distances[i][k] + distances[k][j];
                        
                        System.out.println("  更新 d[" + i + "][" + j + "]: " + 
                                         oldDistance + " -> " + distances[i][j] + 
                                         " (通过节点 " + k + ")");
                    }
                }
            }
            
            printMatrix(distances, "第 " + (k + 1) + " 轮结果");
        }
    }
    
    // 路径重构
    public static List<Integer> reconstructPath(int[] predecessors, int source, int target) {
        List<Integer> path = new ArrayList<>();
        int current = target;
        
        while (current != -1) {
            path.add(current);
            current = predecessors[current];
        }
        
        Collections.reverse(path);
        
        if (path.get(0) != source) {
            return new ArrayList<>(); // 无路径
        }
        
        return path;
    }
    
    public static List<Integer> reconstructFloydPath(int[][] next, int source, int target) {
        List<Integer> path = new ArrayList<>();
        
        if (next[source][target] == -1) {
            return path; // 无路径
        }
        
        int current = source;
        path.add(current);
        
        while (current != target) {
            current = next[current][target];
            path.add(current);
        }
        
        return path;
    }
    
    // 性能比较
    public static void performanceComparison() {
        int[] sizes = {100, 200, 500};
        
        System.out.printf("%-8s %-12s %-12s %-12s%n", 
                         "图大小", "Dijkstra", "Bellman-Ford", "Floyd-Warshall");
        System.out.println("--------------------------------------------");
        
        for (int size : sizes) {
            WeightedGraph graph = generateRandomGraph(size, 0.3);
            
            // Dijkstra算法
            long startTime = System.nanoTime();
            dijkstra(graph, 0);
            long dijkstraTime = (System.nanoTime() - startTime) / 1000;
            
            // Bellman-Ford算法
            startTime = System.nanoTime();
            bellmanFord(graph, 0);
            long bellmanTime = (System.nanoTime() - startTime) / 1000;
            
            // Floyd-Warshall算法（小图才测试）
            long floydTime = 0;
            if (size <= 200) {
                int[][] matrix = graph.toAdjacencyMatrix();
                startTime = System.nanoTime();
                floydWarshall(matrix);
                floydTime = (System.nanoTime() - startTime) / 1000;
            }
            
            System.out.printf("%-8d %-12d %-12d %-12d%n", 
                             size, dijkstraTime, bellmanTime, 
                             size <= 200 ? floydTime : -1);
        }
        
        System.out.println("\n算法选择建议:");
        System.out.println("- 单源最短路径 + 非负权重: Dijkstra");
        System.out.println("- 单源最短路径 + 可能有负权边: Bellman-Ford");
        System.out.println("- 全对最短路径 + 小图: Floyd-Warshall");
        System.out.println("- 稀疏图: Dijkstra或Bellman-Ford");
        System.out.println("- 稠密图: Floyd-Warshall");
    }
    
    // 辅助方法
    public static void printMatrix(int[][] matrix, String title) {
        System.out.println(title + ":");
        for (int[] row : matrix) {
            for (int val : row) {
                if (val == Integer.MAX_VALUE) {
                    System.out.print("  ∞");
                } else {
                    System.out.printf("%3d", val);
                }
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static WeightedGraph generateRandomGraph(int vertices, double density) {
        WeightedGraph graph = new WeightedGraph(vertices);
        Random random = new Random();
        
        for (int i = 0; i < vertices; i++) {
            for (int j = i + 1; j < vertices; j++) {
                if (random.nextDouble() < density) {
                    int weight = random.nextInt(20) + 1;
                    graph.addEdge(i, j, weight);
                }
            }
        }
        
        return graph;
    }

class WeightedGraph {
    private int vertices;
    private List<List<Edge>> adjacencyList;
    
    public WeightedGraph(int vertices) {
        this.vertices = vertices;
        this.adjacencyList = new ArrayList<>();
        
        for (int i = 0; i < vertices; i++) {
            adjacencyList.add(new ArrayList<>());
        }
    }
    
    public void addEdge(int source, int destination, int weight) {
        adjacencyList.get(source).add(new Edge(destination, weight));
        adjacencyList.get(destination).add(new Edge(source, weight)); // 无向图
    }
    
    public int getVertices() {
        return vertices;
    }
    
    public List<List<Edge>> getAdjacencyList() {
        return adjacencyList;
    }
    
    public void printGraph() {
        for (int i = 0; i < vertices; i++) {
            System.out.print("节点 " + i + ": ");
            for (Edge edge : adjacencyList.get(i)) {
                System.out.print("(" + edge.destination + "," + edge.weight + ") ");
            }
            System.out.println();
        }
    }
    
    public int[][] toAdjacencyMatrix() {
        int[][] matrix = new int[vertices][vertices];
        
        for (int i = 0; i < vertices; i++) {
            Arrays.fill(matrix[i], Integer.MAX_VALUE);
            matrix[i][i] = 0;
        }
        
        for (int i = 0; i < vertices; i++) {
            for (Edge edge : adjacencyList.get(i)) {
                matrix[i][edge.destination] = edge.weight;
            }
        }
        
        return matrix;
    }
}

            }

        }
    }
    
    public NavigationResult findRoute(String startCity, String endCity) {
        int startId = cityToId.get(startCity);
        int endId = cityToId.get(endCity);
        
        DijkstraResult result = ShortestPathDemo.dijkstra(roadNetwork, startId);
        
        if (result.distances[endId] == Integer.MAX_VALUE) {
            return new NavigationResult(new ArrayList<>(), 0, 0);

        int totalDistance = result.distances[endId];

        return new NavigationResult(route, totalDistance, estimatedTime);

class GameMap {
    private boolean[][] obstacles;
    private int rows, cols;
    
    public GameMap(int rows, int cols) {
        this.rows = rows;
        this.cols = cols;
        this.obstacles = new boolean[rows][cols];
    }
    
    public void setObstacle(int row, int col) {
        obstacles[row][col] = true;
    }
    
    public void printMap() {
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                System.out.print(obstacles[i][j] ? " X " : " . ");
            }
            System.out.println();
        }
    }
    
    public GamePathResult findPath(int startRow, int startCol, int endRow, int endCol) {
        // 使用BFS找最短路径
        Queue<int[]> queue = new LinkedList<>();
        boolean[][] visited = new boolean[rows][cols];
        int[][] parent = new int[rows][cols];
        
        queue.offer(new int[]{startRow, startCol});
        visited[startRow][startCol] = true;
        
        int[][] directions = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        
        while (!queue.isEmpty()) {
            int[] current = queue.poll();
            int row = current[0], col = current[1];
            
            if (row == endRow && col == endCol) {
                break;
            }
            
            for (int[] dir : directions) {
                int newRow = row + dir[0];
                int newCol = col + dir[1];
                
                if (newRow >= 0 && newRow < rows && newCol >= 0 && newCol < cols &&
                    !obstacles[newRow][newCol] && !visited[newRow][newCol]) {
                    
                    visited[newRow][newCol] = true;
                    parent[newRow][newCol] = row * cols + col;
                    queue.offer(new int[]{newRow, newCol});
                }
            }
        }
        
        // 重构路径
        List<String> path = new ArrayList<>();
        int currentRow = endRow, currentCol = endCol;
        int steps = 0;
        
        while (!(currentRow == startRow && currentCol == startCol)) {
            path.add("(" + currentRow + "," + currentCol + ")");
            int parentPos = parent[currentRow][currentCol];
            currentRow = parentPos / cols;
            currentCol = parentPos % cols;
            steps++;
        }
        
        path.add("(" + startRow + "," + startCol + ")");
        Collections.reverse(path);
        
        return new GamePathResult(path, steps);
    }
}
