拓扑排序算法详解 ​

📚 概述 ​

拓扑排序是针对有向无环图（DAG）的一种线性排序算法，它将图中的顶点排列成一个线性序列，使得对于图中的每一条有向边(u,v)，顶点u在序列中都出现在顶点v之前。拓扑排序在实际应用中非常广泛，从项目调度到编译器的依赖分析，从课程安排到任务执行顺序，都需要用到拓扑排序。常见的拓扑排序算法有两种：基于DFS的算法和基于入度的Kahn算法。理解拓扑排序不仅能解决实际的调度问题，更能加深对图论和算法设计的理解，特别是对于处理具有依赖关系的问题具有重要意义。

🎯 学习目标 ​

  • 深入理解拓扑排序的概念和应用场景
  • 掌握基于DFS和Kahn算法的两种实现方式
  • 学会检测有向图中的环并处理环的情况
  • 理解拓扑排序在实际问题中的应用
  • 掌握拓扑排序的变种和扩展问题
  • 学会分析拓扑排序算法的时间复杂度
  • 了解拓扑排序在系统设计中的重要性
  • 培养处理依赖关系问题的思维能力

    // 基于DFS的拓扑排序
    public static DFSTopologicalResult dfsTopologicalSort(DirectedGraph graph) {
        int vertices = graph.getVertices();
        boolean[] visited = new boolean[vertices];
        boolean[] recursionStack = new boolean[vertices];
        Stack<Integer> stack = new Stack<>();
        
        System.out.println("DFS访问顺序:");
        
        // 对每个未访问的顶点进行DFS
        for (int i = 0; i < vertices; i++) {
            if (!visited[i]) {
                if (dfsUtil(graph, i, visited, recursionStack, stack)) {
                    return new DFSTopologicalResult(new ArrayList<>(), true);
                }
            }
        }
        
        // 从栈中弹出元素得到拓扑排序
        List<Integer> topologicalOrder = new ArrayList<>();
        while (!stack.isEmpty()) {
            topologicalOrder.add(stack.pop());
        }
        
        return new DFSTopologicalResult(topologicalOrder, false);
    }
    
    private static boolean dfsUtil(DirectedGraph graph, int vertex, boolean[] visited, 
                                  boolean[] recursionStack, Stack<Integer> stack) {
        visited[vertex] = true;
        recursionStack[vertex] = true;
        
        System.out.println("  访问顶点: " + graph.getVertexName(vertex));
        
        // 访问所有相邻顶点
        for (int neighbor : graph.getAdjacencyList().get(vertex)) {
            if (!visited[neighbor]) {
                if (dfsUtil(graph, neighbor, visited, recursionStack, stack)) {
                    return true; // 发现环
                }
            } else if (recursionStack[neighbor]) {
                System.out.println("  检测到环: " + graph.getVertexName(vertex) + 
                                 " -> " + graph.getVertexName(neighbor));
                return true; // 发现环
            }
        }
        
        recursionStack[vertex] = false;
        stack.push(vertex);
        System.out.println("  完成顶点: " + graph.getVertexName(vertex));
        
        return false;
    }
    
    // Kahn算法
    public static KahnResult kahnTopologicalSort(DirectedGraph graph) {
        int vertices = graph.getVertices();
        int[] inDegree = new int[vertices];
        
        // 计算每个顶点的入度
        for (int i = 0; i < vertices; i++) {
            for (int neighbor : graph.getAdjacencyList().get(i)) {
                inDegree[neighbor]++;
            }
        }
        
        System.out.println("初始入度:");
        for (int i = 0; i < vertices; i++) {
            System.out.println("  " + graph.getVertexName(i) + ": " + inDegree[i]);
        }
        
        // 将入度为0的顶点加入队列
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < vertices; i++) {
            if (inDegree[i] == 0) {
                queue.offer(i);
            }
        }
        
        List<Integer> topologicalOrder = new ArrayList<>();
        
        System.out.println("\n处理过程:");
        while (!queue.isEmpty()) {
            int current = queue.poll();
            topologicalOrder.add(current);
            
            System.out.println("  处理顶点: " + graph.getVertexName(current));
            
            // 移除当前顶点的所有出边
            for (int neighbor : graph.getAdjacencyList().get(current)) {
                inDegree[neighbor]--;
                System.out.println("    更新 " + graph.getVertexName(neighbor) + 
                                 " 入度: " + inDegree[neighbor]);
                
                if (inDegree[neighbor] == 0) {
                    queue.offer(neighbor);
                    System.out.println("    添加到队列: " + graph.getVertexName(neighbor));
                }
            }
        }
        
        // 检查是否存在环
        boolean hasCycle = topologicalOrder.size() != vertices;
        List<Integer> remainingNodes = new ArrayList<>();
        
        if (hasCycle) {
            for (int i = 0; i < vertices; i++) {
                if (inDegree[i] > 0) {
                    remainingNodes.add(i);
                }
            }
        }
        
        return new KahnResult(topologicalOrder, hasCycle, remainingNodes);
    }
    
    // 环检测
    public static CycleDetectionResult detectCycleWithDFS(DirectedGraph graph) {
        int vertices = graph.getVertices();
        boolean[] visited = new boolean[vertices];
        boolean[] recursionStack = new boolean[vertices];
        List<Integer> cyclePath = new ArrayList<>();
        
        for (int i = 0; i < vertices; i++) {
            if (!visited[i]) {
                if (detectCycleUtil(graph, i, visited, recursionStack, cyclePath)) {
                    return new CycleDetectionResult(true, cyclePath);
                }
            }
        }
        
        return new CycleDetectionResult(false, new ArrayList<>());
    }
    
    private static boolean detectCycleUtil(DirectedGraph graph, int vertex, boolean[] visited,
                                         boolean[] recursionStack, List<Integer> cyclePath) {
        visited[vertex] = true;
        recursionStack[vertex] = true;
        cyclePath.add(vertex);
        
        for (int neighbor : graph.getAdjacencyList().get(vertex)) {
            if (!visited[neighbor]) {
                if (detectCycleUtil(graph, neighbor, visited, recursionStack, cyclePath)) {
                    return true;
                }
            } else if (recursionStack[neighbor]) {
                // 找到环，添加回到环起点的路径
                cyclePath.add(neighbor);
                return true;
            }
        }
        
        recursionStack[vertex] = false;
        cyclePath.remove(cyclePath.size() - 1);
        return false;
    }

                long startTime = System.nanoTime();
                dfsTopologicalSort(graph);
                long dfsTime = (System.nanoTime() - startTime) / 1000;
                
                // Kahn算法
                startTime = System.nanoTime();
                kahnTopologicalSort(graph);
                long kahnTime = (System.nanoTime() - startTime) / 1000;
                
                System.out.printf("%-8d %-8.1f %-12d %-12d%n", 
                                 size, density, dfsTime, kahnTime);
            }
        }
        
        System.out.println("\n算法选择建议:");
        System.out.println("- 需要环检测: DFS算法");
        System.out.println("- 需要逐步处理: Kahn算法");
        System.out.println("- 内存受限: Kahn算法 (无递归)");
        System.out.println("- 理论分析: DFS算法 (更直观)");
    }
    
    // 辅助方法
    public static DirectedGraph generateRandomDAG(int vertices, double density) {
        DirectedGraph graph = new DirectedGraph(vertices);
        Random random = new Random();
        
        for (int i = 0; i < vertices; i++) {
            graph.addVertex(i, "V" + i);
        }
        
        // 确保生成的是DAG，只添加从小编号到大编号的边
        for (int i = 0; i < vertices; i++) {
            for (int j = i + 1; j < vertices; j++) {
                if (random.nextDouble() < density) {
                    graph.addEdge(i, j);
                }
            }
        }
        
        return graph;
    }
}

// 有向图数据结构
class DirectedGraph {
    private int vertices;
    private List<List<Integer>> adjacencyList;
    private Map<Integer, String> vertexNames;
    
    public DirectedGraph(int vertices) {
        this.vertices = vertices;
        this.adjacencyList = new ArrayList<>();
        this.vertexNames = new HashMap<>();
        
        for (int i = 0; i < vertices; i++) {
            adjacencyList.add(new ArrayList<>());
        }
    }
    
    public void addVertex(int id, String name) {
        vertexNames.put(id, name);
    }
    
    public void addEdge(int source, int destination) {
        adjacencyList.get(source).add(destination);
    }
    
    public int getVertices() {
        return vertices;
    }
    
    public List<List<Integer>> getAdjacencyList() {
        return adjacencyList;
    }
    
    public String getVertexName(int vertex) {
        return vertexNames.getOrDefault(vertex, "V" + vertex);
    }
    
    public void printGraph() {
        for (int i = 0; i < vertices; i++) {
            System.out.print(getVertexName(i) + " -> ");
            for (int neighbor : adjacencyList.get(i)) {
                System.out.print(getVertexName(neighbor) + " ");
            }
            System.out.println();
        }
    }
}

class CompilationSystem {
    private Map<String, Integer> fileToId;
    private Map<Integer, String> idToFile;
    private DirectedGraph dependencyGraph;
    private int fileCount;
    
    public CompilationSystem() {
        this.fileToId = new HashMap<>();
        this.idToFile = new HashMap<>();
        this.fileCount = 0;
    }
    
    public void addSourceFile(String fileName, int id) {
        fileToId.put(fileName, id);
        idToFile.put(id, fileName);
        fileCount = Math.max(fileCount, id + 1);
        
        if (dependencyGraph == null) {
            dependencyGraph = new DirectedGraph(20);
        }
        dependencyGraph.addVertex(id, fileName);
    }
    
    public void addDependency(String dependent, String dependency) {
        int dependentId = fileToId.get(dependent);
        int dependencyId = fileToId.get(dependency);
        dependencyGraph.addEdge(dependencyId, dependentId);
    }
    
    public void printDependencies() {
        dependencyGraph.printGraph();
    }
    
    public CompilationResult generateCompilationOrder() {
        KahnResult result = TopologicalSortDemo.kahnTopologicalSort(dependencyGraph);
        
        List<String> compilationOrder = new ArrayList<>();
        for (int id : result.topologicalOrder) {
            compilationOrder.add(idToFile.get(id));
        }
        
        return new CompilationResult(compilationOrder, result.hasCycle);
    }
}
