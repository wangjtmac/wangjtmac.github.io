    public static boolean isConnected(AdjacencyListGraph graph) {
        if (graph.getVertices().isEmpty()) return true;
        
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        
        Integer start = graph.getVertices().iterator().next();
        queue.offer(start);
        visited.add(start);
        
        while (!queue.isEmpty()) {
            Integer current = queue.poll();
            for (Integer neighbor : graph.getNeighbors(current)) {
                if (!visited.contains(neighbor)) {
                    visited.add(neighbor);
                    queue.offer(neighbor);
                }
            }
        }
        
        return visited.size() == graph.getVertices().size();
    }
    
    public static boolean hasCycle(AdjacencyListGraph graph) {
        Set<Integer> visited = new HashSet<>();
        
        for (Integer vertex : graph.getVertices()) {
            if (!visited.contains(vertex)) {
                if (hasCycleDFS(graph, vertex, -1, visited)) {
                    return true;
                }
            }
        }
        
        return false;
    }
    
    private static boolean hasCycleDFS(AdjacencyListGraph graph, Integer current, 
                                     Integer parent, Set<Integer> visited) {
        visited.add(current);
        
        for (Integer neighbor : graph.getNeighbors(current)) {
            if (!visited.contains(neighbor)) {
                if (hasCycleDFS(graph, neighbor, current, visited)) {
                    return true;
                }
            } else if (!neighbor.equals(parent)) {
                return true;
            }
        }
        
        return false;
    }
    
    public static boolean isBipartite(AdjacencyListGraph graph) {
        Map<Integer, Integer> colors = new HashMap<>();
        
        for (Integer vertex : graph.getVertices()) {
            if (!colors.containsKey(vertex)) {
                if (!isBipartiteBFS(graph, vertex, colors)) {
                    return false;
                }
            }
        }
        
        return true;
    }
    
    private static boolean isBipartiteBFS(AdjacencyListGraph graph, Integer start, 
                                        Map<Integer, Integer> colors) {
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(start);
        colors.put(start, 0);
        
        while (!queue.isEmpty()) {
            Integer current = queue.poll();
            int currentColor = colors.get(current);
            
            for (Integer neighbor : graph.getNeighbors(current)) {
                if (!colors.containsKey(neighbor)) {
                    colors.put(neighbor, 1 - currentColor);
                    queue.offer(neighbor);
                } else if (colors.get(neighbor) == currentColor) {
                    return false;
                }
            }
        }
        
        return true;
    }
    
    public static void analyzeGraphDegrees(AdjacencyListGraph graph) {
        Map<Integer, Integer> degreeCount = new HashMap<>();
        int totalDegree = 0;
        
        for (Integer vertex : graph.getVertices()) {
            int degree = graph.getNeighbors(vertex).size();
            totalDegree += degree;
            degreeCount.put(degree, degreeCount.getOrDefault(degree, 0) + 1);
        }
        
        System.out.println("åº¦æ•°åˆ†å¸ƒ:");
        for (Map.Entry<Integer, Integer> entry : degreeCount.entrySet()) {
            System.out.println("åº¦æ•° " + entry.getKey() + ": " + entry.getValue() + " ä¸ªé¡¶ç‚¹");
        }
        
        System.out.println("å¹³å‡åº¦æ•°: " + (double) totalDegree / graph.getVertices().size());
        System.out.println("æ€»åº¦æ•°: " + totalDegree + " (åº”ç­‰äº 2 * è¾¹æ•°)");
    }

class SimpleGraph {
    private Set<String> vertices;
    private Set<String> edges;
    private Map<String, Set<String>> adjacencyList;
    
    public SimpleGraph() {
        vertices = new HashSet<>();
        edges = new HashSet<>();
        adjacencyList = new HashMap<>();
    }
    
    public void addVertex(String vertex) {
        vertices.add(vertex);
        adjacencyList.putIfAbsent(vertex, new HashSet<>());
    }
    
    public void addEdge(String u, String v) {
        addVertex(u);
        addVertex(v);
        
        adjacencyList.get(u).add(v);
        adjacencyList.get(v).add(u);
        
        String edge = u.compareTo(v) < 0 ? u + "-" + v : v + "-" + u;
        edges.add(edge);
    }
    
    public Set<String> getVertices() { return vertices; }
    public Set<String> getEdges() { return edges; }
    
    public int getDegree(String vertex) {
        return adjacencyList.getOrDefault(vertex, new HashSet<>()).size();
    }
}

        this.matrix = new int[vertices][vertices];
    }
    
    public void addEdge(int u, int v) {
        matrix[u][v] = 1;
        matrix[v][u] = 1; // æ— å‘å›¾
    }
    
    public void printMatrix() {
        for (int i = 0; i < vertices; i++) {
            for (int j = 0; j < vertices; j++) {
                System.out.print(matrix[i][j] + " ");
            }
            System.out.println();
        }
    }
}

class AdjacencyListGraph {
    private Map<Integer, List<Integer>> adjacencyList;
    
    public AdjacencyListGraph() {
        adjacencyList = new HashMap<>();
    }
    
    public void addVertex(Integer vertex) {
        adjacencyList.putIfAbsent(vertex, new ArrayList<>());
    }
    
    public void addEdge(Integer u, Integer v) {
        addVertex(u);
        addVertex(v);
        adjacencyList.get(u).add(v);
        adjacencyList.get(v).add(u); // æ— å‘å›¾
    }
    
    public Set<Integer> getVertices() {
        return adjacencyList.keySet();
    }
    
    public List<Integer> getNeighbors(Integer vertex) {
        return adjacencyList.getOrDefault(vertex, new ArrayList<>());
    }
    
    public void printGraph() {
        for (Map.Entry<Integer, List<Integer>> entry : adjacencyList.entrySet()) {
            System.out.println(entry.getKey() + ": " + entry.getValue());
        }
    }
}

class WeightedGraph {
    private Map<String, List<Edge>> adjacencyList;
    
    public WeightedGraph() {
        adjacencyList = new HashMap<>();
    }
    
    public void addVertex(String vertex) {
        adjacencyList.putIfAbsent(vertex, new ArrayList<>());
    }
    
    public void addEdge(String from, String to, int weight) {
        addVertex(from);
        addVertex(to);
        adjacencyList.get(from).add(new Edge(to, weight));
        adjacencyList.get(to).add(new Edge(from, weight)); // æ— å‘å›¾
    }
    
    @Override
    public String toString() {
        return adjacencyList.toString();
    }
    
    static class Edge {
        String to;
        int weight;
        
        Edge(String to, int weight) {
            this.to = to;
            this.weight = weight;
        }
        
        @Override
        public String toString() {
            return to + "(" + weight + ")";
        }
    }
}
