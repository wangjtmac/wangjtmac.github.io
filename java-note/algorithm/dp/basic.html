基础DP问题详解 ​

📚 概述 ​

基础动态规划问题是学习动态规划算法的入门阶段，涵盖了最经典和最基础的DP问题类型。这些问题虽然相对简单，但包含了动态规划的核心思想：最优子结构、重叠子问题和状态转移。通过掌握这些基础问题，可以建立起对动态规划的直观理解，为后续学习更复杂的DP问题打下坚实基础。基础DP问题通常具有清晰的状态定义、简单的状态转移方程和明确的边界条件，是理解动态规划思维的最佳起点。

🎯 学习目标 ​

  • 深入理解动态规划的基本思想和解题步骤
  • 掌握经典基础DP问题的分析和实现方法
  • 学会识别问题的最优子结构和重叠子问题
  • 理解状态定义、状态转移和边界条件的设计
  • 掌握自顶向下和自底向上两种DP实现方式
  • 学会DP问题的空间优化技巧
  • 了解DP问题的变种和扩展
  • 培养动态规划的思维模式和解题能力

    // 算法实现
    
    // 斐波那契数列实现
    public static int fibonacciRecursive(int n) {
        if (n <= 1) return n;
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
    }
    
    public static int fibonacciMemo(int n) {
        int[] memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return fibMemoHelper(n, memo);
    }
    
    private static int fibMemoHelper(int n, int[] memo) {
        if (n <= 1) return n;
        if (memo[n] != -1) return memo[n];
        
        memo[n] = fibMemoHelper(n - 1, memo) + fibMemoHelper(n - 2, memo);
        return memo[n];
    }
    
    public static int fibonacciDP(int n) {
        if (n <= 1) return n;
        
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        
        return dp[n];
    }
    
    public static int fibonacciOptimized(int n) {
        if (n <= 1) return n;
        
        int prev2 = 0, prev1 = 1;
        for (int i = 2; i <= n; i++) {
            int current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
    
    public static long fibonacciMemoMap(int n, Map<Integer, Long> memo) {
        if (n <= 1) return n;
        if (memo.containsKey(n)) return memo.get(n);
        
        long result = fibonacciMemoMap(n - 1, memo) + fibonacciMemoMap(n - 2, memo);
        memo.put(n, result);
        return result;
    }
    
    // 最大子数组和
    public static MaxSubarrayResult maxSubarraySum(int[] nums) {
        int maxSum = nums[0];
        int currentSum = nums[0];
        int start = 0, end = 0, tempStart = 0;
        
        for (int i = 1; i < nums.length; i++) {
            if (currentSum < 0) {
                currentSum = nums[i];
                tempStart = i;
            } else {
                currentSum += nums[i];
            }
            
            if (currentSum > maxSum) {
                maxSum = currentSum;
                start = tempStart;
                end = i;
            }
        }
        
        return new MaxSubarrayResult(maxSum, start, end);
    }
    
    // 最长递增子序列
    public static LISResult longestIncreasingSubsequence(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        int[] parent = new int[n];
        Arrays.fill(dp, 1);
        Arrays.fill(parent, -1);
        
        int maxLength = 1;
        int maxIndex = 0;
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i] && dp[j] + 1 > dp[i]) {
                    dp[i] = dp[j] + 1;
                    parent[i] = j;
                }
            }
            
            if (dp[i] > maxLength) {
                maxLength = dp[i];
                maxIndex = i;
            }
        }
        
        // 重构序列
        List<Integer> sequence = new ArrayList<>();
        int current = maxIndex;
        while (current != -1) {
            sequence.add(nums[current]);
            current = parent[current];
        }
        Collections.reverse(sequence);
        
        return new LISResult(maxLength, sequence);
    }
    
    // 数字三角形最小路径和
    public static int minimumPathSum(int[][] triangle) {
        int n = triangle.length;
        int[][] dp = new int[n][];
        
        for (int i = 0; i < n; i++) {
            dp[i] = new int[triangle[i].length];
        }
        
        dp[0][0] = triangle[0][0];
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < triangle[i].length; j++) {
                dp[i][j] = Integer.MAX_VALUE;
                
                if (j > 0) {
                    dp[i][j] = Math.min(dp[i][j], dp[i-1][j-1] + triangle[i][j]);
                }
                if (j < triangle[i-1].length) {
                    dp[i][j] = Math.min(dp[i][j], dp[i-1][j] + triangle[i][j]);
                }
            }
        }
        
        int result = Integer.MAX_VALUE;
        for (int j = 0; j < triangle[n-1].length; j++) {
            result = Math.min(result, dp[n-1][j]);
        }
        
        return result;
    }
    
    public static List<Integer> getMinimumPath(int[][] triangle) {
        int n = triangle.length;
        int[][] dp = new int[n][];
        int[][] choice = new int[n][];
        
        for (int i = 0; i < n; i++) {
            dp[i] = new int[triangle[i].length];
            choice[i] = new int[triangle[i].length];
        }
        
        dp[0][0] = triangle[0][0];
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < triangle[i].length; j++) {
                dp[i][j] = Integer.MAX_VALUE;
                
                if (j > 0 && dp[i-1][j-1] + triangle[i][j] < dp[i][j]) {
                    dp[i][j] = dp[i-1][j-1] + triangle[i][j];
                    choice[i][j] = j - 1;
                }
                if (j < triangle[i-1].length && dp[i-1][j] + triangle[i][j] < dp[i][j]) {
                    dp[i][j] = dp[i-1][j] + triangle[i][j];
                    choice[i][j] = j;
                }
            }
        }
        
        // 找到最小值的位置
        int minIndex = 0;
        for (int j = 1; j < triangle[n-1].length; j++) {
            if (dp[n-1][j] < dp[n-1][minIndex]) {
                minIndex = j;
            }
        }
        
        // 重构路径
        List<Integer> path = new ArrayList<>();
        int currentIndex = minIndex;
        for (int i = n - 1; i >= 0; i--) {
            path.add(triangle[i][currentIndex]);
            if (i > 0) {
                currentIndex = choice[i][currentIndex];
            }
        }
        Collections.reverse(path);
        
        return path;
    }
    
    // 爬楼梯问题
    public static int climbStairs(int n) {
        if (n <= 2) return n;
        
        int prev2 = 1, prev1 = 2;
        for (int i = 3; i <= n; i++) {
            int current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
    
    public static int climbStairsWithSteps(int n, int[] steps) {
        int[] dp = new int[n + 1];
        dp[0] = 1;
        
        for (int i = 1; i <= n; i++) {
            for (int step : steps) {
                if (i >= step) {
                    dp[i] += dp[i - step];
                }
            }
        }
        
        return dp[n];
    }
    
    public static int minCostClimbingStairs(int[] cost) {
        int n = cost.length;
        int[] dp = new int[n + 1];
        
        for (int i = 2; i <= n; i++) {
            dp[i] = Math.min(dp[i-1] + cost[i-1], dp[i-2] + cost[i-2]);
        }
        
        return dp[n];
    }
    
    // 路径问题
    public static int uniquePaths(int m, int n) {
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[j] += dp[j - 1];
            }
        }
        
        return dp[n - 1];
    }
    
    public static int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        
        if (obstacleGrid[0][0] == 1) return 0;
        
        int[] dp = new int[n];
        dp[0] = 1;
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (obstacleGrid[i][j] == 1) {
                    dp[j] = 0;
                } else if (j > 0) {
                    dp[j] += dp[j - 1];
                }
            }
        }
        
        return dp[n - 1];
    }
    
    // 子序列问题
    public static LCSResult longestCommonSubsequence(String text1, String text2) {
        int m = text1.length(), n = text2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // 重构LCS
        StringBuilder lcs = new StringBuilder();
        int i = m, j = n;
        while (i > 0 && j > 0) {
            if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                lcs.append(text1.charAt(i - 1));
                i--;
                j--;
            } else if (dp[i - 1][j] > dp[i][j - 1]) {
                i--;
            } else {
                j--;
            }
        }
        
        return new LCSResult(dp[m][n], lcs.reverse().toString());
    }
    
    public static String longestCommonSubstring(String str1, String str2) {
        int m = str1.length(), n = str2.length();
        int[][] dp = new int[m + 1][n + 1];
        int maxLength = 0;
        int endIndex = 0;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                    if (dp[i][j] > maxLength) {
                        maxLength = dp[i][j];
                        endIndex = i;
                    }
                }
            }
        }
        
        return str1.substring(endIndex - maxLength, endIndex);
    }
    
    public static int minEditDistance(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) dp[i][0] = i;
        for (int j = 0; j <= n; j++) dp[0][j] = j;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = 1 + Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]);
                }
            }
        }
        
        return dp[m][n];
    }
    
    public static List<String> getEditOperations(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) dp[i][0] = i;
        for (int j = 0; j <= n; j++) dp[0][j] = j;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = 1 + Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]);
                }
            }
        }
        
        // 重构操作序列
        List<String> operations = new ArrayList<>();
        int i = m, j = n;
        
        while (i > 0 || j > 0) {
            if (i > 0 && j > 0 && word1.charAt(i - 1) == word2.charAt(j - 1)) {
                i--;
                j--;
            } else if (i > 0 && j > 0 && dp[i][j] == dp[i - 1][j - 1] + 1) {
                operations.add("替换 '" + word1.charAt(i - 1) + "' 为 '" + word2.charAt(j - 1) + "'");
                i--;
                j--;
            } else if (i > 0 && dp[i][j] == dp[i - 1][j] + 1) {
                operations.add("删除 '" + word1.charAt(i - 1) + "'");
                i--;
            } else {
                operations.add("插入 '" + word2.charAt(j - 1) + "'");
                j--;
            }
        }
        
        Collections.reverse(operations);
        return operations;
    }
    
    public static int longestPalindromeSubseq(String s) {
        int n = s.length();
        int[][] dp = new int[n][n];
        
        for (int i = 0; i < n; i++) {
            dp[i][i] = 1;
        }
        
        for (int len = 2; len <= n; len++) {
            for (int i = 0; i <= n - len; i++) {
                int j = i + len - 1;
                if (s.charAt(i) == s.charAt(j)) {
                    dp[i][j] = dp[i + 1][j - 1] + 2;
                } else {
                    dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
                }
            }
        }
        
        return dp[0][n - 1];
    }
    
    public static String getLongestPalindromeSubseq(String s) {
        int n = s.length();
        int[][] dp = new int[n][n];
        
        for (int i = 0; i < n; i++) {
            dp[i][i] = 1;
        }
        
        for (int len = 2; len <= n; len++) {
            for (int i = 0; i <= n - len; i++) {
                int j = i + len - 1;
                if (s.charAt(i) == s.charAt(j)) {
                    dp[i][j] = dp[i + 1][j - 1] + 2;
                } else {
                    dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // 重构回文子序列
        StringBuilder palindrome = new StringBuilder();
        int i = 0, j = n - 1;
        
        while (i <= j) {
            if (i == j) {
                palindrome.append(s.charAt(i));
                break;
            } else if (s.charAt(i) == s.charAt(j)) {
                palindrome.append(s.charAt(i));
                i++;
                j--;
            } else if (dp[i + 1][j] > dp[i][j - 1]) {
                i++;
            } else {
                j--;
            }
        }
        
        String firstHalf = palindrome.toString();
        String secondHalf = new StringBuilder(firstHalf).reverse().toString();
        
        if (firstHalf.length() * 2 == dp[0][n - 1]) {
            return firstHalf + secondHalf;
        } else {
            return firstHalf + secondHalf.substring(1);
        }
    }

    public static int lcsStandard(String text1, String text2) {
        int m = text1.length(), n = text2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        return dp[m][n];
    }
    
    public static int lcsOptimized(String text1, String text2) {
        if (text1.length() < text2.length()) {
            String temp = text1;
            text1 = text2;
            text2 = temp;
        }
        
        int m = text1.length(), n = text2.length();
        int[] prev = new int[n + 1];
        int[] curr = new int[n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                    curr[j] = prev[j - 1] + 1;
                } else {
                    curr[j] = Math.max(prev[j], curr[j - 1]);
                }
            }
            
            int[] temp = prev;
            prev = curr;
            curr = temp;
        }
        
        return prev[n];
    }

        int[] values = {1, 4, 5, 7};

    public static int knapsack2D(int[] weights, int[] values, int capacity) {
        int n = weights.length;
        int[][] dp = new int[n + 1][capacity + 1];
        
        for (int i = 1; i <= n; i++) {
            for (int w = 1; w <= capacity; w++) {
                if (weights[i - 1] <= w) {
                    dp[i][w] = Math.max(dp[i - 1][w], 
                                       dp[i - 1][w - weights[i - 1]] + values[i - 1]);
                } else {
                    dp[i][w] = dp[i - 1][w];
                }
            }
        }
        
        return dp[n][capacity];
    }
    
    public static int knapsack1D(int[] weights, int[] values, int capacity) {
        int[] dp = new int[capacity + 1];
        
        for (int i = 0; i < weights.length; i++) {
            for (int w = capacity; w >= weights[i]; w--) {
                dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
            }
        }
        
        return dp[capacity];
    }
    
    public static PathResult minPathSumWithPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        String[][] path = new String[m][n];
        
        dp[0][0] = grid[0][0];
        path[0][0] = "(0,0)";
        
        // 初始化第一行
        for (int j = 1; j < n; j++) {
            dp[0][j] = dp[0][j - 1] + grid[0][j];
            path[0][j] = path[0][j - 1] + " -> (0," + j + ")";
        }
        
        // 初始化第一列
        for (int i = 1; i < m; i++) {
            dp[i][0] = dp[i - 1][0] + grid[i][0];
            path[i][0] = path[i - 1][0] + " -> (" + i + ",0)";
        }
        
        // 填充DP表
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (dp[i - 1][j] < dp[i][j - 1]) {
                    dp[i][j] = dp[i - 1][j] + grid[i][j];
                    path[i][j] = path[i - 1][j] + " -> (" + i + "," + j + ")";
                } else {
                    dp[i][j] = dp[i][j - 1] + grid[i][j];
                    path[i][j] = path[i][j - 1] + " -> (" + i + "," + j + ")";
                }
            }
        }
        
        return new PathResult(dp[m - 1][n - 1], path[m - 1][n - 1]);
    }
