    public static LCSResult longestCommonSubsequence(String text1, String text2) {
        int m = text1.length(), n = text2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // é‡æ„LCS
        StringBuilder lcs = new StringBuilder();
        List<String> positions = new ArrayList<>();
        int i = m, j = n;
        
        while (i > 0 && j > 0) {
            if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                lcs.append(text1.charAt(i - 1));
                positions.add("(" + (i - 1) + "," + (j - 1) + ")");
                i--;
                j--;
            } else if (dp[i - 1][j] > dp[i][j - 1]) {
                i--;
            } else {
                j--;
            }
        }
        
        lcs.reverse();
        Collections.reverse(positions);
        
        return new LCSResult(dp[m][n], lcs.toString(), positions);
    }

    public static int weightedEditDistance(String source, String target, 
                                         int insertCost, int deleteCost, int replaceCost) {
        int m = source.length(), n = target.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) dp[i][0] = i * deleteCost;
        for (int j = 0; j <= n; j++) dp[0][j] = j * insertCost;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (source.charAt(i - 1) == target.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = Math.min(Math.min(
                        dp[i - 1][j] + deleteCost,
                        dp[i][j - 1] + insertCost),
                        dp[i - 1][j - 1] + replaceCost);
                }
            }
        }
        
        return dp[m][n];
    }

    public static boolean isOneEditDistance(String s, String t) {
        int m = s.length(), n = t.length();
        
        if (Math.abs(m - n) > 1) return false;
        if (s.equals(t)) return false;
        
        if (m > n) return isOneEditDistance(t, s);
        
        for (int i = 0; i < m; i++) {
            if (s.charAt(i) != t.charAt(i)) {
                if (m == n) {
                    return s.substring(i + 1).equals(t.substring(i + 1));
                } else {
                    return s.substring(i).equals(t.substring(i + 1));
                }
            }
        }
        
        return m + 1 == n;
    }

        int[][] dp = new int[n][n];
        
        for (int i = 0; i < n; i++) {
            dp[i][i] = 1;
        }
        
        for (int len = 2; len <= n; len++) {
            for (int i = 0; i <= n - len; i++) {
                int j = i + len - 1;
                if (s.charAt(i) == s.charAt(j)) {
                    dp[i][j] = dp[i + 1][j - 1] + 2;
                } else {
                    dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // é‡æ„å›æ–‡å­åºåˆ—

        while (i <= j) {

                break;
            } else if (s.charAt(i) == s.charAt(j)) {
                palindrome.append(s.charAt(i));
                i++;
                j--;

                i++;
            } else {
                j--;

        }
        
        String firstHalf = palindrome.toString();
        String secondHalf = new StringBuilder(firstHalf).reverse().toString();
        
        String result;
        if (firstHalf.length() * 2 == dp[0][n - 1]) {
            result = firstHalf + secondHalf;
        } else {
            result = firstHalf + secondHalf.substring(1);
        }
        
        return new PalindromeResult(dp[0][n - 1], result);

                                              Map<String, List<String>> memo) {
        if (memo.containsKey(s)) {

        }
        
        List<String> result = new ArrayList<>();
        
        if (s.length() == 0) {
            result.add("");
            return result;
        }
        
        for (String word : wordSet) {
            if (s.startsWith(word)) {
                List<String> sublist = wordBreakHelper(s.substring(word.length()), wordSet, memo);
                for (String sub : sublist) {
                    result.add(word + (sub.isEmpty() ? "" : " " + sub));
                }
            }
        }
        
        memo.put(s, result);
        return result;
    }

            }
        }

    private static void backtrackPalindrome(String s, int start, List<String> current, 
                                          List<List<String>> result) {
        if (start == s.length()) {
            result.add(new ArrayList<>(current));
            return;
        }
        
        for (int end = start; end < s.length(); end++) {
            if (isPalindromeString(s, start, end)) {
                current.add(s.substring(start, end + 1));
                backtrackPalindrome(s, end + 1, current, result);
                current.remove(current.size() - 1);
            }
        }
    }

    }

        }

    // æœ€çŸ­è¶…ä¸²
    public static String shortestSuperstring(String[] words) {
        int n = words.length;
        
        // è®¡ç®—é‡å é•¿åº¦
        int[][] overlap = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    overlap[i][j] = getOverlap(words[i], words[j]);
                }
            }
        }
        
        // DP with bitmask
        int[][] dp = new int[1 << n][n];
        int[][] parent = new int[1 << n][n];
        
        for (int mask = 0; mask < (1 << n); mask++) {
            Arrays.fill(dp[mask], Integer.MAX_VALUE);
            Arrays.fill(parent[mask], -1);
        }
        
        for (int i = 0; i < n; i++) {
            dp[1 << i][i] = words[i].length();
        }
        
        for (int mask = 1; mask < (1 << n); mask++) {
            for (int i = 0; i < n; i++) {
                if ((mask & (1 << i)) == 0) continue;
                
                int prevMask = mask ^ (1 << i);
                if (prevMask == 0) continue;
                
                for (int j = 0; j < n; j++) {
                    if (dp[prevMask][j] != Integer.MAX_VALUE) {
                        int newLen = dp[prevMask][j] + words[i].length() - overlap[j][i];
                        if (newLen < dp[mask][i]) {
                            dp[mask][i] = newLen;
                            parent[mask][i] = j;
                        }
                    }
                }
            }
        }
        
        // æ‰¾åˆ°æœ€çŸ­é•¿åº¦
        int finalMask = (1 << n) - 1;
        int minLen = Integer.MAX_VALUE;
        int lastNode = -1;
        
        for (int i = 0; i < n; i++) {
            if (dp[finalMask][i] < minLen) {
                minLen = dp[finalMask][i];
                lastNode = i;
            }
        }
        
        // é‡æ„è·¯å¾„
        List<Integer> path = new ArrayList<>();
        int mask = finalMask;
        int node = lastNode;
        
        while (node != -1) {
            path.add(node);
            int nextNode = parent[mask][node];
            mask ^= (1 << node);
            node = nextNode;
        }
        
        Collections.reverse(path);
        
        // æ„å»ºç»“æœå­—ç¬¦ä¸²
        StringBuilder result = new StringBuilder(words[path.get(0)]);
        for (int i = 1; i < path.size(); i++) {
            int prev = path.get(i - 1);
            int curr = path.get(i);
            int overlapLen = overlap[prev][curr];
            result.append(words[curr].substring(overlapLen));
        }
        
        return result.toString();
    }
    
    private static int getOverlap(String a, String b) {
        int maxOverlap = 0;
        int minLen = Math.min(a.length(), b.length());
        
        for (int i = 1; i <= minLen; i++) {
            if (a.substring(a.length() - i).equals(b.substring(0, i))) {
                maxOverlap = i;
            }
        }
        
        return maxOverlap;
    }

    public static String compressString(String s) {

        int count = 1;
        
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i - 1)) {

    public static void printEditDistanceTable(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) dp[i][0] = i;
        for (int j = 0; j <= n; j++) dp[0][j] = j;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = 1 + Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]);
