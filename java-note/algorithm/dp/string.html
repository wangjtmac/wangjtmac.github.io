字符串DP详解 ​

📚 概述 ​

字符串动态规划是DP算法在字符串处理领域的重要应用，专门解决涉及字符串匹配、编辑、变换等问题。字符串DP问题通常具有明确的子结构特性，通过比较字符串的前缀或后缀来构建状态转移关系。这类问题在文本处理、生物信息学、编译器设计、自然语言处理等领域有广泛应用。掌握字符串DP不仅能解决经典的算法问题，更能为实际的文本分析和处理任务提供高效的解决方案。字符串DP的核心在于理解字符串的结构特性，设计合适的状态表示和转移策略。

🎯 学习目标 ​

  • 深入理解字符串DP问题的特点和解题思路
  • 掌握经典字符串DP问题的分析和实现方法
  • 学会设计字符串状态和转移方程
  • 理解字符串匹配、编辑距离等核心概念
  • 掌握字符串DP的优化技巧和空间压缩
  • 学会处理多字符串和复杂约束的DP问题
  • 了解字符串DP在实际应用中的变种
  • 培养字符串算法设计和分析能力

    // 最长公共子序列
    public static LCSResult longestCommonSubsequence(String text1, String text2) {
        int m = text1.length(), n = text2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // 重构LCS
        StringBuilder lcs = new StringBuilder();
        List<String> positions = new ArrayList<>();
        int i = m, j = n;
        
        while (i > 0 && j > 0) {
            if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                lcs.append(text1.charAt(i - 1));
                positions.add("(" + (i - 1) + "," + (j - 1) + ")");
                i--;
                j--;
            } else if (dp[i - 1][j] > dp[i][j - 1]) {
                i--;
            } else {
                j--;
            }
        }
        
        lcs.reverse();
        Collections.reverse(positions);
        
        return new LCSResult(dp[m][n], lcs.toString(), positions);
    }
    
    // 最长公共子串
    public static SubstringResult longestCommonSubstring(String str1, String str2) {
        int m = str1.length(), n = str2.length();
        int[][] dp = new int[m + 1][n + 1];
        int maxLength = 0;
        int endIndex1 = 0, endIndex2 = 0;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                    if (dp[i][j] > maxLength) {
                        maxLength = dp[i][j];
                        endIndex1 = i;
                        endIndex2 = j;
                    }
                }
            }
        }
        
        String substring = str1.substring(endIndex1 - maxLength, endIndex1);
        int position1 = endIndex1 - maxLength;
        int position2 = endIndex2 - maxLength;
        
        return new SubstringResult(substring, maxLength, position1, position2);
    }
    
    // 通配符匹配
    public static boolean wildcardMatching(String text, String pattern) {
        int m = text.length(), n = pattern.length();
        boolean[][] dp = new boolean[m + 1][n + 1];
        
        dp[0][0] = true;
        
        // 处理模式开头的*
        for (int j = 1; j <= n; j++) {
            if (pattern.charAt(j - 1) == '*') {
                dp[0][j] = dp[0][j - 1];
            }
        }
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                char pChar = pattern.charAt(j - 1);
                char tChar = text.charAt(i - 1);
                
                if (pChar == '*') {
                    dp[i][j] = dp[i - 1][j] || dp[i][j - 1];
                } else if (pChar == '?' || pChar == tChar) {
                    dp[i][j] = dp[i - 1][j - 1];
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 正则表达式匹配
    public static boolean regexMatching(String text, String pattern) {
        int m = text.length(), n = pattern.length();
        boolean[][] dp = new boolean[m + 1][n + 1];
        
        dp[0][0] = true;
        
        // 处理a*b*这样的模式
        for (int j = 2; j <= n; j += 2) {
            if (pattern.charAt(j - 1) == '*') {
                dp[0][j] = dp[0][j - 2];
            }
        }
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                char tChar = text.charAt(i - 1);
                char pChar = pattern.charAt(j - 1);
                
                if (pChar == '*') {
                    char prevPChar = pattern.charAt(j - 2);
                    dp[i][j] = dp[i][j - 2]; // 匹配0次
                    
                    if (prevPChar == '.' || prevPChar == tChar) {
                        dp[i][j] = dp[i][j] || dp[i - 1][j]; // 匹配1次或多次
                    }
                } else if (pChar == '.' || pChar == tChar) {
                    dp[i][j] = dp[i - 1][j - 1];
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 编辑距离
    public static EditDistanceResult minEditDistance(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        // 初始化
        for (int i = 0; i <= m; i++) dp[i][0] = i;
        for (int j = 0; j <= n; j++) dp[0][j] = j;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = 1 + Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]);
                }
            }
        }
        
        // 重构操作序列
        List<String> operations = new ArrayList<>();
        int i = m, j = n;
        
        while (i > 0 || j > 0) {
            if (i > 0 && j > 0 && word1.charAt(i - 1) == word2.charAt(j - 1)) {
                i--;
                j--;
            } else if (i > 0 && j > 0 && dp[i][j] == dp[i - 1][j - 1] + 1) {
                operations.add("替换位置" + (i - 1) + "的'" + word1.charAt(i - 1) + "'为'" + word2.charAt(j - 1) + "'");
                i--;
                j--;
            } else if (i > 0 && dp[i][j] == dp[i - 1][j] + 1) {
                operations.add("删除位置" + (i - 1) + "的'" + word1.charAt(i - 1) + "'");
                i--;
            } else {
                operations.add("在位置" + i + "插入'" + word2.charAt(j - 1) + "'");
                j--;
            }
        }
        
        Collections.reverse(operations);
        return new EditDistanceResult(dp[m][n], operations);
    }
    
    // 只允许插入删除的编辑距离
    public static int minDistanceInsertDelete(String s1, String s2) {
        int lcsLength = longestCommonSubsequence(s1, s2).length;
        return s1.length() + s2.length() - 2 * lcsLength;
    }
    
    // 带权重的编辑距离
    public static int weightedEditDistance(String source, String target, 
                                         int insertCost, int deleteCost, int replaceCost) {
        int m = source.length(), n = target.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) dp[i][0] = i * deleteCost;
        for (int j = 0; j <= n; j++) dp[0][j] = j * insertCost;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (source.charAt(i - 1) == target.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = Math.min(Math.min(
                        dp[i - 1][j] + deleteCost,
                        dp[i][j - 1] + insertCost),
                        dp[i - 1][j - 1] + replaceCost);
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 一次编辑距离
    public static boolean isOneEditDistance(String s, String t) {
        int m = s.length(), n = t.length();
        
        if (Math.abs(m - n) > 1) return false;
        if (s.equals(t)) return false;
        
        if (m > n) return isOneEditDistance(t, s);
        
        for (int i = 0; i < m; i++) {
            if (s.charAt(i) != t.charAt(i)) {
                if (m == n) {
                    return s.substring(i + 1).equals(t.substring(i + 1));
                } else {
                    return s.substring(i).equals(t.substring(i + 1));
                }
            }
        }
        
        return m + 1 == n;
    }
    
    // 最长回文子序列
    public static PalindromeResult longestPalindromeSubsequence(String s) {
        int n = s.length();
        int[][] dp = new int[n][n];
        
        for (int i = 0; i < n; i++) {
            dp[i][i] = 1;
        }
        
        for (int len = 2; len <= n; len++) {
            for (int i = 0; i <= n - len; i++) {
                int j = i + len - 1;
                if (s.charAt(i) == s.charAt(j)) {
                    dp[i][j] = dp[i + 1][j - 1] + 2;
                } else {
                    dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // 重构回文子序列
        StringBuilder palindrome = new StringBuilder();
        int i = 0, j = n - 1;
        
        while (i <= j) {
            if (i == j) {
                palindrome.append(s.charAt(i));
                break;
            } else if (s.charAt(i) == s.charAt(j)) {
                palindrome.append(s.charAt(i));
                i++;
                j--;
            } else if (dp[i + 1][j] > dp[i][j - 1]) {
                i++;
            } else {
                j--;
            }
        }
        
        String firstHalf = palindrome.toString();
        String secondHalf = new StringBuilder(firstHalf).reverse().toString();
        
        String result;
        if (firstHalf.length() * 2 == dp[0][n - 1]) {
            result = firstHalf + secondHalf;
        } else {
            result = firstHalf + secondHalf.substring(1);
        }
        
        return new PalindromeResult(dp[0][n - 1], result);
    }
    
    // 最长重复子序列
    public static String longestRepeatedSubsequence(String s) {
        int n = s.length();
        int[][] dp = new int[n + 1][n + 1];
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (s.charAt(i - 1) == s.charAt(j - 1) && i != j) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        // 重构序列
        StringBuilder result = new StringBuilder();
        int i = n, j = n;
        
        while (i > 0 && j > 0) {
            if (dp[i][j] == dp[i - 1][j - 1] + 1) {
                result.append(s.charAt(i - 1));
                i--;
                j--;
            } else if (dp[i][j] == dp[i - 1][j]) {
                i--;
            } else {
                j--;
            }
        }
        
        return result.reverse().toString();
    }
    
    // 不同子序列计数
    public static int numDistinctSubsequences(String s, String t) {
        int m = s.length(), n = t.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) {
            dp[i][0] = 1;
        }
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = dp[i - 1][j];
                if (s.charAt(i - 1) == t.charAt(j - 1)) {
                    dp[i][j] += dp[i - 1][j - 1];
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 子序列匹配
    public static boolean isSubsequence(String s, String t) {
        int i = 0, j = 0;
        
        while (i < s.length() && j < t.length()) {
            if (s.charAt(i) == t.charAt(j)) {
                i++;
            }
            j++;
        }
        
        return i == s.length();
    }
    
    // 三字符串最长公共子序列
    public static int longestCommonSubsequence3(String s1, String s2, String s3) {
        int m = s1.length(), n = s2.length(), p = s3.length();
        int[][][] dp = new int[m + 1][n + 1][p + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                for (int k = 1; k <= p; k++) {
                    if (s1.charAt(i - 1) == s2.charAt(j - 1) && 
                        s2.charAt(j - 1) == s3.charAt(k - 1)) {
                        dp[i][j][k] = dp[i - 1][j - 1][k - 1] + 1;
                    } else {
                        dp[i][j][k] = Math.max(Math.max(dp[i - 1][j][k], dp[i][j - 1][k]), 
                                              dp[i][j][k - 1]);
                    }
                }
            }
        }
        
        return dp[m][n][p];
    }
    
    // 单词拆分
    public static boolean wordBreak(String s, List<String> wordDict) {
        Set<String> wordSet = new HashSet<>(wordDict);
        boolean[] dp = new boolean[s.length() + 1];
        dp[0] = true;
        
        for (int i = 1; i <= s.length(); i++) {
            for (int j = 0; j < i; j++) {
                if (dp[j] && wordSet.contains(s.substring(j, i))) {
                    dp[i] = true;
                    break;
                }
            }
        }
        
        return dp[s.length()];
    }
    
    // 单词拆分II
    public static List<String> wordBreakII(String s, List<String> wordDict) {
        Set<String> wordSet = new HashSet<>(wordDict);
        Map<String, List<String>> memo = new HashMap<>();
        return wordBreakHelper(s, wordSet, memo);
    }
    
    private static List<String> wordBreakHelper(String s, Set<String> wordSet, 
                                              Map<String, List<String>> memo) {
        if (memo.containsKey(s)) {
            return memo.get(s);
        }
        
        List<String> result = new ArrayList<>();
        
        if (s.length() == 0) {
            result.add("");
            return result;
        }
        
        for (String word : wordSet) {
            if (s.startsWith(word)) {
                List<String> sublist = wordBreakHelper(s.substring(word.length()), wordSet, memo);
                for (String sub : sublist) {
                    result.add(word + (sub.isEmpty() ? "" : " " + sub));
                }
            }
        }
        
        memo.put(s, result);
        return result;
    }
    
    // 回文分割最少次数
    public static int minCutPalindrome(String s) {
        int n = s.length();
        boolean[][] isPalindrome = new boolean[n][n];
        
        // 预处理回文信息
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= i; j++) {
                if (s.charAt(i) == s.charAt(j) && (i - j <= 2 || isPalindrome[j + 1][i - 1])) {
                    isPalindrome[j][i] = true;
                }
            }
        }
        
        int[] dp = new int[n];
        for (int i = 0; i < n; i++) {
            if (isPalindrome[0][i]) {
                dp[i] = 0;
            } else {
                dp[i] = i;
                for (int j = 1; j <= i; j++) {
                    if (isPalindrome[j][i]) {
                        dp[i] = Math.min(dp[i], dp[j - 1] + 1);
                    }
                }
            }
        }
        
        return dp[n - 1];
    }
    
    // 回文分割所有方案
    public static List<List<String>> palindromePartition(String s) {
        List<List<String>> result = new ArrayList<>();
        List<String> current = new ArrayList<>();
        backtrackPalindrome(s, 0, current, result);
        return result;
    }
    
    private static void backtrackPalindrome(String s, int start, List<String> current, 
                                          List<List<String>> result) {
        if (start == s.length()) {
            result.add(new ArrayList<>(current));
            return;
        }
        
        for (int end = start; end < s.length(); end++) {
            if (isPalindromeString(s, start, end)) {
                current.add(s.substring(start, end + 1));
                backtrackPalindrome(s, end + 1, current, result);
                current.remove(current.size() - 1);
            }
        }
    }
    
    private static boolean isPalindromeString(String s, int left, int right) {
        while (left < right) {
            if (s.charAt(left) != s.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
    
    // 交错字符串
    public static boolean isInterleavingString(String s1, String s2, String s3) {
        int m = s1.length(), n = s2.length();
        
        if (m + n != s3.length()) return false;
        
        boolean[][] dp = new boolean[m + 1][n + 1];
        dp[0][0] = true;
        
        for (int i = 1; i <= m; i++) {
            dp[i][0] = dp[i - 1][0] && s1.charAt(i - 1) == s3.charAt(i - 1);
        }
        
        for (int j = 1; j <= n; j++) {
            dp[0][j] = dp[0][j - 1] && s2.charAt(j - 1) == s3.charAt(j - 1);
        }
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = (dp[i - 1][j] && s1.charAt(i - 1) == s3.charAt(i + j - 1)) ||
                          (dp[i][j - 1] && s2.charAt(j - 1) == s3.charAt(i + j - 1));
            }
        }
        
        return dp[m][n];
    }
    
    // 不同的子序列II
    public static int distinctSubsequencesII(String s) {
        int MOD = 1000000007;
        int[] dp = new int[26];
        
        for (char c : s.toCharArray()) {
            int prev = 0;
            for (int count : dp) {
                prev = (prev + count) % MOD;
            }
            dp[c - 'a'] = (prev + 1) % MOD;
        }
        
        int result = 0;
        for (int count : dp) {
            result = (result + count) % MOD;
        }
        
        return result;
    }
    
    // 最短超串
    public static String shortestSuperstring(String[] words) {
        int n = words.length;
        
        // 计算重叠长度
        int[][] overlap = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    overlap[i][j] = getOverlap(words[i], words[j]);
                }
            }
        }
        
        // DP with bitmask
        int[][] dp = new int[1 << n][n];
        int[][] parent = new int[1 << n][n];
        
        for (int mask = 0; mask < (1 << n); mask++) {
            Arrays.fill(dp[mask], Integer.MAX_VALUE);
            Arrays.fill(parent[mask], -1);
        }
        
        for (int i = 0; i < n; i++) {
            dp[1 << i][i] = words[i].length();
        }
        
        for (int mask = 1; mask < (1 << n); mask++) {
            for (int i = 0; i < n; i++) {
                if ((mask & (1 << i)) == 0) continue;
                
                int prevMask = mask ^ (1 << i);
                if (prevMask == 0) continue;
                
                for (int j = 0; j < n; j++) {
                    if (dp[prevMask][j] != Integer.MAX_VALUE) {
                        int newLen = dp[prevMask][j] + words[i].length() - overlap[j][i];
                        if (newLen < dp[mask][i]) {
                            dp[mask][i] = newLen;
                            parent[mask][i] = j;
                        }
                    }
                }
            }
        }
        
        // 找到最短长度
        int finalMask = (1 << n) - 1;
        int minLen = Integer.MAX_VALUE;
        int lastNode = -1;
        
        for (int i = 0; i < n; i++) {
            if (dp[finalMask][i] < minLen) {
                minLen = dp[finalMask][i];
                lastNode = i;
            }
        }
        
        // 重构路径
        List<Integer> path = new ArrayList<>();
        int mask = finalMask;
        int node = lastNode;
        
        while (node != -1) {
            path.add(node);
            int nextNode = parent[mask][node];
            mask ^= (1 << node);
            node = nextNode;
        }
        
        Collections.reverse(path);
        
        // 构建结果字符串
        StringBuilder result = new StringBuilder(words[path.get(0)]);
        for (int i = 1; i < path.size(); i++) {
            int prev = path.get(i - 1);
            int curr = path.get(i);
            int overlapLen = overlap[prev][curr];
            result.append(words[curr].substring(overlapLen));
        }
        
        return result.toString();
    }
    
    private static int getOverlap(String a, String b) {
        int maxOverlap = 0;
        int minLen = Math.min(a.length(), b.length());
        
        for (int i = 1; i <= minLen; i++) {
            if (a.substring(a.length() - i).equals(b.substring(0, i))) {
                maxOverlap = i;
            }
        }
        
        return maxOverlap;
    }
    
    // 字符串压缩
    public static String compressString(String s) {
        StringBuilder compressed = new StringBuilder();
        int count = 1;
        
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == s.charAt(i - 1)) {
                count++;
            } else {
                compressed.append(s.charAt(i - 1));
                if (count > 1) {
                    compressed.append(count);
                }
                count = 1;
            }
        }
        
        // 处理最后一组
        if (s.length() > 0) {
            compressed.append(s.charAt(s.length() - 1));
            if (count > 1) {
                compressed.append(count);
            }
        }
        
        return compressed.length() < s.length() ? compressed.toString() : s;
    }
