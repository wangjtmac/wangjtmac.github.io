路径问题详解 ​

📚 概述 ​

路径问题是动态规划中的一个重要分支，主要研究在给定约束条件下从起点到终点的最优路径。这类问题在实际应用中非常广泛，从机器人路径规划到游戏AI，从网络路由到资源分配，都能看到路径DP的身影。路径问题的核心特点是状态转移具有明确的方向性，通常涉及二维或多维的状态空间，需要考虑移动的约束条件和优化目标。掌握路径DP问题不仅能提高算法设计能力，更能培养空间思维和系统优化的思想。

🎯 学习目标 ​

  • 深入理解路径DP问题的基本特征和解题思路
  • 掌握网格路径、图路径等经典问题的解法
  • 学会处理带约束条件的路径优化问题
  • 理解路径计数、最短路径、最优路径的区别
  • 掌握路径重构和方案输出的技巧
  • 学会路径DP的空间优化和时间优化
  • 了解路径问题在实际应用中的变种
  • 培养多维状态设计和转移的能力

    // 基础路径计数
    public static int uniquePaths(int m, int n) {
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[j] += dp[j - 1];
            }
        }
        
        return dp[n - 1];
    }
    
    public static List<String> getAllPaths(int m, int n) {
        List<String> result = new ArrayList<>();
        generatePaths(0, 0, m - 1, n - 1, "", result);
        return result;
    }
    
    private static void generatePaths(int x, int y, int targetX, int targetY, 
                                    String path, List<String> result) {
        if (x == targetX && y == targetY) {
            result.add(path + "(" + x + "," + y + ")");
            return;
        }
        
        String currentPath = path + "(" + x + "," + y + ") -> ";
        
        if (x < targetX) {
            generatePaths(x + 1, y, targetX, targetY, currentPath, result);
        }
        if (y < targetY) {
            generatePaths(x, y + 1, targetX, targetY, currentPath, result);
        }
    }
    
    // 带障碍物的路径
    public static int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;
        
        if (obstacleGrid[0][0] == 1) return 0;
        
        int[] dp = new int[n];
        dp[0] = 1;
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (obstacleGrid[i][j] == 1) {
                    dp[j] = 0;
                } else if (j > 0) {
                    dp[j] += dp[j - 1];
                }
            }
        }
        
        return dp[n - 1];
    }
    
    public static List<String> getValidPaths(int[][] obstacleGrid) {
        List<String> result = new ArrayList<>();
        if (obstacleGrid[0][0] == 1) return result;
        
        generateValidPaths(obstacleGrid, 0, 0, "", result);
        return result;
    }
    
    private static void generateValidPaths(int[][] grid, int x, int y, 
                                         String path, List<String> result) {
        int m = grid.length, n = grid[0].length;
        
        if (x == m - 1 && y == n - 1) {
            result.add(path + "(" + x + "," + y + ")");
            return;
        }
        
        String currentPath = path + "(" + x + "," + y + ") -> ";
        
        if (x + 1 < m && grid[x + 1][y] == 0) {
            generateValidPaths(grid, x + 1, y, currentPath, result);
        }
        if (y + 1 < n && grid[x][y + 1] == 0) {
            generateValidPaths(grid, x, y + 1, currentPath, result);
        }
    }
    
    // 最小路径和
    public static PathResult minPathSumWithPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        String[][] path = new String[m][n];
        
        dp[0][0] = grid[0][0];
        path[0][0] = "(0,0)";
        
        // 初始化第一行
        for (int j = 1; j < n; j++) {
            dp[0][j] = dp[0][j - 1] + grid[0][j];
            path[0][j] = path[0][j - 1] + " -> (0," + j + ")";
        }
        
        // 初始化第一列
        for (int i = 1; i < m; i++) {
            dp[i][0] = dp[i - 1][0] + grid[i][0];
            path[i][0] = path[i - 1][0] + " -> (" + i + ",0)";
        }
        
        // 填充DP表
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (dp[i - 1][j] < dp[i][j - 1]) {
                    dp[i][j] = dp[i - 1][j] + grid[i][j];
                    path[i][j] = path[i - 1][j] + " -> (" + i + "," + j + ")";
                } else {
                    dp[i][j] = dp[i][j - 1] + grid[i][j];
                    path[i][j] = path[i][j - 1] + " -> (" + i + "," + j + ")";
                }
            }
        }
        
        return new PathResult(dp[m - 1][n - 1], path[m - 1][n - 1]);
    }
    
    // 最大路径和
    public static PathResult maxPathSumWithPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        String[][] path = new String[m][n];
        
        dp[0][0] = grid[0][0];
        path[0][0] = "(0,0)";
        
        // 初始化第一行
        for (int j = 1; j < n; j++) {
            dp[0][j] = dp[0][j - 1] + grid[0][j];
            path[0][j] = path[0][j - 1] + " -> (0," + j + ")";
        }
        
        // 初始化第一列
        for (int i = 1; i < m; i++) {
            dp[i][0] = dp[i - 1][0] + grid[i][0];
            path[i][0] = path[i - 1][0] + " -> (" + i + ",0)";
        }
        
        // 填充DP表
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (dp[i - 1][j] > dp[i][j - 1]) {
                    dp[i][j] = dp[i - 1][j] + grid[i][j];
                    path[i][j] = path[i - 1][j] + " -> (" + i + "," + j + ")";
                } else {
                    dp[i][j] = dp[i][j - 1] + grid[i][j];
                    path[i][j] = path[i][j - 1] + " -> (" + i + "," + j + ")";
                }
            }
        }
        
        return new PathResult(dp[m - 1][n - 1], path[m - 1][n - 1]);
    }
    
    // 四方向移动的最短路径
    public static int shortestPathFourDirections(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dist = new int[m][n];
        
        for (int[] row : dist) {
            Arrays.fill(row, Integer.MAX_VALUE);
        }
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[2] - b[2]);
        pq.offer(new int[]{0, 0, grid[0][0]});
        dist[0][0] = grid[0][0];
        
        int[][] directions = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        
        while (!pq.isEmpty()) {
            int[] current = pq.poll();
            int x = current[0], y = current[1], d = current[2];
            
            if (x == m - 1 && y == n - 1) {
                return d;
            }
            
            if (d > dist[x][y]) continue;
            
            for (int[] dir : directions) {
                int nx = x + dir[0], ny = y + dir[1];
                
                if (nx >= 0 && nx < m && ny >= 0 && ny < n) {
                    int newDist = d + grid[nx][ny];
                    if (newDist < dist[nx][ny]) {
                        dist[nx][ny] = newDist;
                        pq.offer(new int[]{nx, ny, newDist});
                    }
                }
            }
        }
        
        return dist[m - 1][n - 1];
    }
    
    // 固定步数的路径
    public static int pathsWithFixedSteps(int m, int n, int steps) {
        int[][][] dp = new int[m][n][steps + 1];
        dp[0][0][0] = 1;
        
        for (int k = 1; k <= steps; k++) {
            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    if (i > 0) dp[i][j][k] += dp[i - 1][j][k - 1];
                    if (j > 0) dp[i][j][k] += dp[i][j - 1][k - 1];
                    if (i < m - 1) dp[i][j][k] += dp[i + 1][j][k - 1];
                    if (j < n - 1) dp[i][j][k] += dp[i][j + 1][k - 1];
                }
            }
        }
        
        return dp[m - 1][n - 1][steps];
    }
    
    // 收集物品的最大数量
    public static int maxItemsCollected(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        
        dp[0][0] = grid[0][0];
        
        for (int j = 1; j < n; j++) {
            dp[0][j] = dp[0][j - 1] + grid[0][j];
        }
        
        for (int i = 1; i < m; i++) {
            dp[i][0] = dp[i - 1][0] + grid[i][0];
        }
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]) + grid[i][j];
            }
        }
        
        return dp[m - 1][n - 1];
    }
    
    // 清扫所有脏污的最少时间
    public static int minTimeToCleanAll(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int totalDirty = 0;
        
        for (int[] row : grid) {
            for (int cell : row) {
                totalDirty += cell;
            }
        }
        
        // 使用BFS找到访问所有脏污位置的最短路径
        Queue<int[]> queue = new LinkedList<>();
        Set<String> visited = new HashSet<>();
        
        queue.offer(new int[]{0, 0, 0, 0}); // x, y, cleaned, time
        visited.add("0,0,0");
        
        int[][] directions = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        
        while (!queue.isEmpty()) {
            int[] current = queue.poll();
            int x = current[0], y = current[1], cleaned = current[2], time = current[3];
            
            if (cleaned == totalDirty) {
                return time;
            }
            
            for (int[] dir : directions) {
                int nx = x + dir[0], ny = y + dir[1];
                
                if (nx >= 0 && nx < m && ny >= 0 && ny < n) {
                    int newCleaned = cleaned + grid[nx][ny];
                    String state = nx + "," + ny + "," + newCleaned;
                    
                    if (!visited.contains(state)) {
                        visited.add(state);
                        queue.offer(new int[]{nx, ny, newCleaned, time + 1});
                    }
                }
            }
        }
        
        return -1; // 无法清扫所有脏污
    }
    
    // 最少初始能量
    public static int minInitialEnergyRequired(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        
        // 从终点开始反向计算
        dp[m - 1][n - 1] = Math.max(1, 1 - grid[m - 1][n - 1]);
        
        // 最后一行
        for (int j = n - 2; j >= 0; j--) {
            dp[m - 1][j] = Math.max(1, dp[m - 1][j + 1] - grid[m - 1][j]);
        }
        
        // 最后一列
        for (int i = m - 2; i >= 0; i--) {
            dp[i][n - 1] = Math.max(1, dp[i + 1][n - 1] - grid[i][n - 1]);
        }
        
        // 其他位置
        for (int i = m - 2; i >= 0; i--) {
            for (int j = n - 2; j >= 0; j--) {
                int minEnergyOnExit = Math.min(dp[i + 1][j], dp[i][j + 1]);
                dp[i][j] = Math.max(1, minEnergyOnExit - grid[i][j]);
            }
        }
        
        return dp[0][0];
    }
    
    // 樱桃收集（双路径问题）
    public static int cherryPickup(int[][] grid) {
        int n = grid.length;
        int[][][] dp = new int[n][n][n];
        
        for (int[][] layer : dp) {
            for (int[] row : layer) {
                Arrays.fill(row, -1);
            }
        }
        
        dp[0][0][0] = grid[0][0];
        
        for (int k = 1; k < 2 * n - 1; k++) {
            for (int i1 = Math.max(0, k - n + 1); i1 <= Math.min(k, n - 1); i1++) {
                for (int i2 = Math.max(0, k - n + 1); i2 <= Math.min(k, n - 1); i2++) {
                    int j1 = k - i1, j2 = k - i2;
                    
                    if (grid[i1][j1] == -1 || grid[i2][j2] == -1) {
                        dp[i1][j1][i2] = -1;
                        continue;
                    }
                    
                    int val = 0;
                    if (i1 == i2) {
                        val = grid[i1][j1];
                    } else {
                        val = grid[i1][j1] + grid[i2][j2];
                    }
                    
                    int maxPrev = -1;
                    if (i1 > 0 && i2 > 0 && dp[i1 - 1][j1][i2 - 1] != -1) {
                        maxPrev = Math.max(maxPrev, dp[i1 - 1][j1][i2 - 1]);
                    }
                    if (i1 > 0 && dp[i1 - 1][j1][i2] != -1) {
                        maxPrev = Math.max(maxPrev, dp[i1 - 1][j1][i2]);
                    }
                    if (i2 > 0 && dp[i1][j1 - 1][i2 - 1] != -1) {
                        maxPrev = Math.max(maxPrev, dp[i1][j1 - 1][i2 - 1]);
                    }
                    if (dp[i1][j1 - 1][i2] != -1) {
                        maxPrev = Math.max(maxPrev, dp[i1][j1 - 1][i2]);
                    }
                    
                    if (maxPrev != -1) {
                        dp[i1][j1][i2] = maxPrev + val;
                    }
                }
            }
        }
        
        return Math.max(0, dp[n - 1][n - 1][n - 1]);
    }
    
    // 下降路径最小和
    public static int minFallingPathSum(int[][] matrix) {
        int n = matrix.length;
        int[] dp = matrix[0].clone();
        
        for (int i = 1; i < n; i++) {
            int[] newDp = new int[n];
            for (int j = 0; j < n; j++) {
                newDp[j] = dp[j];
                if (j > 0) newDp[j] = Math.min(newDp[j], dp[j - 1]);
                if (j < n - 1) newDp[j] = Math.min(newDp[j], dp[j + 1]);
                newDp[j] += matrix[i][j];
            }
            dp = newDp;
        }
        
        return Arrays.stream(dp).min().orElse(0);
    }
    
    // 地牢游戏
    public static int calculateMinimumHP(int[][] dungeon) {
        int m = dungeon.length, n = dungeon[0].length;
        int[][] dp = new int[m][n];
        
        dp[m - 1][n - 1] = Math.max(1, 1 - dungeon[m - 1][n - 1]);
        
        for (int j = n - 2; j >= 0; j--) {
            dp[m - 1][j] = Math.max(1, dp[m - 1][j + 1] - dungeon[m - 1][j]);
        }
        
        for (int i = m - 2; i >= 0; i--) {
            dp[i][n - 1] = Math.max(1, dp[i + 1][n - 1] - dungeon[i][n - 1]);
        }
        
        for (int i = m - 2; i >= 0; i--) {
            for (int j = n - 2; j >= 0; j--) {
                int minHealthOnExit = Math.min(dp[i + 1][j], dp[i][j + 1]);
                dp[i][j] = Math.max(1, minHealthOnExit - dungeon[i][j]);
            }
        }
        
        return dp[0][0];
    }
    
    // 最大正方形
    public static int maximalSquare(char[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        int[][] dp = new int[m + 1][n + 1];
        int maxSide = 0;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (matrix[i - 1][j - 1] == '1') {
                    dp[i][j] = Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1;
                    maxSide = Math.max(maxSide, dp[i][j]);
                }
            }
        }
        
        return maxSide * maxSide;
    }
    
    // 不同路径III
    public static int uniquePathsIII(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int startX = 0, startY = 0, emptyCount = 0;
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    startX = i;
                    startY = j;
                    emptyCount++;
                } else if (grid[i][j] == 0) {
                    emptyCount++;
                }
            }
        }
        
        return dfsUniquePathsIII(grid, startX, startY, emptyCount);
    }
    
    private static int dfsUniquePathsIII(int[][] grid, int x, int y, int remaining) {
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || grid[x][y] == -1) {
            return 0;
        }
        
        if (grid[x][y] == 2) {
            return remaining == 1 ? 1 : 0;
        }
        
        grid[x][y] = -1; // 标记为已访问
        
        int paths = dfsUniquePathsIII(grid, x + 1, y, remaining - 1) +
                   dfsUniquePathsIII(grid, x - 1, y, remaining - 1) +
                   dfsUniquePathsIII(grid, x, y + 1, remaining - 1) +
                   dfsUniquePathsIII(grid, x, y - 1, remaining - 1);
        
        grid[x][y] = 0; // 回溯
        
        return paths;
    }
    
    // 股票交易策略
    public static TradingResult optimalTradingStrategy(int[] prices, int maxTransactions) {
        int n = prices.length;
        if (n <= 1 || maxTransactions == 0) {
            return new TradingResult(0, new ArrayList<>());
        }
        
        // 如果交易次数足够多，可以进行任意次交易
        if (maxTransactions >= n / 2) {
            int profit = 0;
            List<String> transactions = new ArrayList<>();
            
            for (int i = 1; i < n; i++) {
                if (prices[i] > prices[i - 1]) {
                    profit += prices[i] - prices[i - 1];
                    transactions.add("买入价格" + prices[i - 1] + "，卖出价格" + prices[i] + 
                                   "，利润" + (prices[i] - prices[i - 1]));
                }
            }
            
            return new TradingResult(profit, transactions);
        }
        
        // DP解法
        int[][] buy = new int[n][maxTransactions + 1];
        int[][] sell = new int[n][maxTransactions + 1];
        
        for (int i = 0; i <= maxTransactions; i++) {
            buy[0][i] = -prices[0];
        }
        
        for (int i = 1; i < n; i++) {
            for (int j = maxTransactions; j >= 1; j--) {
                sell[i][j] = Math.max(sell[i - 1][j], buy[i - 1][j] + prices[i]);
                buy[i][j] = Math.max(buy[i - 1][j], sell[i - 1][j - 1] - prices[i]);
            }
        }
        
        return new TradingResult(sell[n - 1][maxTransactions], new ArrayList<>());
    }
    
    // 项目资源分配
    public static AllocationResult optimalProjectAllocation(int[] costs, int[] values, int budget) {
        int n = costs.length;
        int[][] dp = new int[n + 1][budget + 1];
        
        for (int i = 1; i <= n; i++) {
            for (int w = 1; w <= budget; w++) {
                if (costs[i - 1] <= w) {
                    dp[i][w] = Math.max(dp[i - 1][w], 
                                       dp[i - 1][w - costs[i - 1]] + values[i - 1]);
                } else {
                    dp[i][w] = dp[i - 1][w];
                }
            }
        }
        
        // 重构选择的项目
        List<Integer> selectedProjects = new ArrayList<>();
        int w = budget;
        int usedBudget = 0;
        
        for (int i = n; i > 0 && w > 0; i--) {
            if (dp[i][w] != dp[i - 1][w]) {
                selectedProjects.add(i - 1);
                w -= costs[i - 1];
                usedBudget += costs[i - 1];
            }
        }
        
        Collections.reverse(selectedProjects);
        
        return new AllocationResult(dp[n][budget], selectedProjects, usedBudget);
    }

class GameMap {
    private int[][] map;
    private int rows, cols;
    
    public GameMap(int rows, int cols) {
        this.rows = rows;
        this.cols = cols;
        this.map = new int[rows][cols];
    }
    
    public void addObstacle(int x, int y) {
        map[x][y] = -1;
    }
    
    public void addTreasure(int x, int y, int value) {
        map[x][y] = value;
    }
    
    public void printMap() {
        for (int[] row : map) {
            System.out.println("  " + Arrays.toString(row));
        }
    }
    
    public GamePathResult findOptimalPath(int startX, int startY, int endX, int endY) {
        int[][] dp = new int[rows][cols];
        String[][] path = new String[rows][cols];
        
        dp[startX][startY] = Math.max(0, map[startX][startY]);
        path[startX][startY] = "(" + startX + "," + startY + ")";
        
        for (int i = startX; i <= endX; i++) {
            for (int j = startY; j <= endY; j++) {
                if (i == startX && j == startY) continue;
                if (map[i][j] == -1) continue;
                
                int maxValue = Integer.MIN_VALUE;
                String bestPath = "";
                
                if (i > startX && map[i - 1][j] != -1) {
                    if (dp[i - 1][j] > maxValue) {
                        maxValue = dp[i - 1][j];
                        bestPath = path[i - 1][j];
                    }
                }
                
                if (j > startY && map[i][j - 1] != -1) {
                    if (dp[i][j - 1] > maxValue) {
                        maxValue = dp[i][j - 1];
                        bestPath = path[i][j - 1];
                    }
                }
                
                if (maxValue != Integer.MIN_VALUE) {
                    dp[i][j] = maxValue + Math.max(0, map[i][j]);
                    path[i][j] = bestPath + " -> (" + i + "," + j + ")";
                }
            }
        }
        
        return new GamePathResult(dp[endX][endY], path[endX][endY]);
    }
}

class NetworkTopology {
    private int[][] latencyMatrix;
    private int nodes;
    
    public NetworkTopology(int nodes, int maxLatency) {
        this.nodes = nodes;
        this.latencyMatrix = new int[nodes][nodes];
        
        for (int i = 0; i < nodes; i++) {
            Arrays.fill(latencyMatrix[i], maxLatency);
            latencyMatrix[i][i] = 0;
        }
    }
    
    public void setLatency(int from, int to, int latency) {
        latencyMatrix[from][to] = latency;
        latencyMatrix[to][from] = latency;
    }
    
    public void printTopology() {
        for (int[] row : latencyMatrix) {
            System.out.println("  " + Arrays.toString(row));
        }
    }
    
    public NetworkPathResult findMinLatencyPath(int source, int destination) {
        int[] dist = new int[nodes];
        int[] parent = new int[nodes];
        boolean[] visited = new boolean[nodes];
        
        Arrays.fill(dist, Integer.MAX_VALUE);
        Arrays.fill(parent, -1);
        dist[source] = 0;
        
        for (int i = 0; i < nodes - 1; i++) {
            int u = -1;
            for (int v = 0; v < nodes; v++) {
                if (!visited[v] && (u == -1 || dist[v] < dist[u])) {
                    u = v;
                }
            }
            
            visited[u] = true;
            
            for (int v = 0; v < nodes; v++) {
                if (!visited[v] && latencyMatrix[u][v] != Integer.MAX_VALUE) {
                    int newDist = dist[u] + latencyMatrix[u][v];
                    if (newDist < dist[v]) {
                        dist[v] = newDist;
                        parent[v] = u;
                    }
                }
            }
        }
        
        // 重构路径
        List<Integer> path = new ArrayList<>();
        int current = destination;
        while (current != -1) {
            path.add(current);
            current = parent[current];
        }
        Collections.reverse(path);
        
        StringBuilder pathStr = new StringBuilder();
        for (int i = 0; i < path.size(); i++) {
            pathStr.append("节点").append(path.get(i));
            if (i < path.size() - 1) {
                pathStr.append(" -> ");
            }
        }
        
        return new NetworkPathResult(dist[destination], pathStr.toString());
    }
