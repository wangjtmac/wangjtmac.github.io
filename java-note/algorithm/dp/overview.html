动态规划概述 ​

📚 概述 ​

动态规划（Dynamic Programming，简称DP）是算法设计中的一种重要思想，它通过将复杂问题分解为子问题，并存储子问题的解来避免重复计算，从而大大提高算法效率。动态规划适用于具有最优子结构和重叠子问题特性的问题，是解决优化问题的强有力工具。从简单的斐波那契数列到复杂的背包问题、路径规划问题，动态规划都能提供优雅而高效的解决方案。掌握动态规划不仅能帮助我们解决各种算法问题，更能培养我们分析问题、分解问题的思维能力。本章将系统介绍动态规划的核心思想、设计方法、经典问题以及优化技巧。

🎯 学习目标 ​

  • 深入理解动态规划的核心思想和原理
  • 掌握动态规划问题的识别和分析方法
  • 学会设计状态转移方程
  • 熟悉动态规划的实现技巧和优化方法
  • 掌握经典动态规划问题的解法
  • 理解动态规划与其他算法思想的区别

  • 了解动态规划在面试和实际开发中的应用

    public static void demonstrateDPConcepts() {

        /*
         * 动态规划的核心要素：
         * 1. 最优子结构 (Optimal Substructure)
         * 2. 重叠子问题 (Overlapping Subproblems)
         * 3. 状态转移方程 (State Transition Equation)
         * 4. 边界条件 (Base Cases)
         */
        
        System.out.println("动态规划的核心要素:");
        System.out.println("1. 最优子结构: 问题的最优解包含子问题的最优解");
        System.out.println("2. 重叠子问题: 递归过程中会重复计算相同的子问题");
        System.out.println("3. 状态转移方程: 描述状态之间转换关系的方程");
        System.out.println("4. 边界条件: 递归的终止条件，最小子问题的解");
        
        // 斐波那契数列演示DP概念
        System.out.println("\n斐波那契数列演示:");
        int n = 10;
        
        // 1. 暴力递归 - 展示重叠子问题
        System.out.println("1. 暴力递归 (展示重叠子问题):");
        long startTime = System.nanoTime();
        long result1 = fibonacciRecursive(n);
        long time1 = System.nanoTime() - startTime;
        System.out.println("   fib(" + n + ") = " + result1 + ", 用时: " + time1 + "ns");
        System.out.println("   问题: 大量重复计算，时间复杂度O(2^n)");
        
        // 2. 记忆化递归 - 解决重叠子问题
        System.out.println("\n2. 记忆化递归 (解决重叠子问题):");
        Map<Integer, Long> memo = new HashMap<>();
        startTime = System.nanoTime();
        long result2 = fibonacciMemoized(n, memo);
        long time2 = System.nanoTime() - startTime;
        System.out.println("   fib(" + n + ") = " + result2 + ", 用时: " + time2 + "ns");
        System.out.println("   优化: 存储子问题结果，时间复杂度O(n)");
        
        // 3. 自底向上DP - 表格法
        System.out.println("\n3. 自底向上DP (表格法):");
        startTime = System.nanoTime();
        long result3 = fibonacciDP(n);
        long time3 = System.nanoTime() - startTime;
        System.out.println("   fib(" + n + ") = " + result3 + ", 用时: " + time3 + "ns");
        System.out.println("   特点: 迭代计算，避免递归开销");

         * 最优子结构：问题的最优解包含子问题的最优解
         * 如果一个问题具有最优子结构，就可以用动态规划求解
         */
        
        System.out.println("最优子结构示例:");
        
        // 1. 最短路径问题
        System.out.println("\n1. 最短路径问题:");
        System.out.println("   如果A到C的最短路径经过B，那么A到B和B到C都必须是最短路径");
        System.out.println("   这体现了最优子结构性质");
        
        // 2. 最长递增子序列
        System.out.println("\n2. 最长递增子序列 (LIS):");

        System.out.println("\n3. 最大子数组和:");

        System.out.println("   最大子数组和: " + maxSum);
        System.out.println("   最优子结构: maxSum(i) = max(nums[i], maxSum(i-1) + nums[i])");

        System.out.println("- fib(2) 计算了 3 次");

    // 记忆化 vs 表格法演示
    public static void demonstrateMemoizationVsTabulation() {
        System.out.println("=== 记忆化 vs 表格法演示 ===");
        
        /*
         * 动态规划的两种实现方式：
         * 1. 记忆化 (Memoization) - 自顶向下
         * 2. 表格法 (Tabulation) - 自底向上
         */
        
        System.out.println("两种DP实现方式:");
        
        int n = 20;
        
        // 1. 记忆化 (自顶向下)
        System.out.println("\n1. 记忆化 (Memoization) - 自顶向下:");
        System.out.println("   特点: 递归 + 缓存，按需计算");
        
        Map<Integer, Long> memo = new HashMap<>();
        long startTime = System.nanoTime();
        long result1 = fibonacciMemoized(n, memo);
        long time1 = System.nanoTime() - startTime;
        
        System.out.println("   结果: fib(" + n + ") = " + result1);
        System.out.println("   用时: " + time1 + "ns");
        System.out.println("   缓存大小: " + memo.size());
        
        // 2. 表格法 (自底向上)
        System.out.println("\n2. 表格法 (Tabulation) - 自底向上:");
        System.out.println("   特点: 迭代填表，计算所有子问题");
        
        startTime = System.nanoTime();
        long result2 = fibonacciDP(n);
        long time2 = System.nanoTime() - startTime;
        
        System.out.println("   结果: fib(" + n + ") = " + result2);
        System.out.println("   用时: " + time2 + "ns");
        
        // 3. 空间优化的表格法
        System.out.println("\n3. 空间优化的表格法:");
        System.out.println("   特点: 只保存必要的状态，节省空间");
        
        startTime = System.nanoTime();
        long result3 = fibonacciOptimized(n);
        long time3 = System.nanoTime() - startTime;
        
        System.out.println("   结果: fib(" + n + ") = " + result3);
        System.out.println("   用时: " + time3 + "ns");
        System.out.println("   空间复杂度: O(1)");
        
        // 比较分析
        System.out.println("\n比较分析:");
        System.out.println("记忆化 vs 表格法:");
        System.out.println("- 记忆化: 递归实现，直观易懂，按需计算");
        System.out.println("- 表格法: 迭代实现，效率更高，计算所有子问题");
        System.out.println("- 空间优化: 在表格法基础上优化空间使用");
        
        System.out.println();
    }
    
    // 算法实现
    
    // 暴力递归
    public static long fibonacciRecursive(int n) {
        if (n <= 1) return n;
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
    }
    
    // 记忆化递归
    public static long fibonacciMemoized(int n, Map<Integer, Long> memo) {
        if (n <= 1) return n;
        
        if (memo.containsKey(n)) {
            return memo.get(n);
        }
        
        long result = fibonacciMemoized(n - 1, memo) + fibonacciMemoized(n - 2, memo);
        memo.put(n, result);
        return result;
    }
    
    // 表格法DP
    public static long fibonacciDP(int n) {
        if (n <= 1) return n;
        
        long[] dp = new long[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        
        return dp[n];
    }
    
    // 空间优化DP
    public static long fibonacciOptimized(int n) {
        if (n <= 1) return n;
        
        long prev2 = 0, prev1 = 1;
        for (int i = 2; i <= n; i++) {
            long current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
    
    // 最长递增子序列
    public static int longestIncreasingSubsequence(int[] nums) {
        if (nums.length == 0) return 0;
        
        int[] dp = new int[nums.length];
        Arrays.fill(dp, 1);
        
        for (int i = 1; i < nums.length; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                }
            }
        }
        
        return Arrays.stream(dp).max().orElse(0);
    }
    
    // 最大子数组和
    public static int maxSubarraySum(int[] nums) {
        int maxSoFar = nums[0];
        int maxEndingHere = nums[0];
        
        for (int i = 1; i < nums.length; i++) {
            maxEndingHere = Math.max(nums[i], maxEndingHere + nums[i]);
            maxSoFar = Math.max(maxSoFar, maxEndingHere);
        }
        
        return maxSoFar;
    }
    
    // 统计调用次数的斐波那契
    public static long fibonacciWithCount(int n, Map<Integer, Integer> callCount) {
        callCount.put(n, callCount.getOrDefault(n, 0) + 1);
        
        if (n <= 1) return n;
        return fibonacciWithCount(n - 1, callCount) + fibonacciWithCount(n - 2, callCount);
    }
}

💡 最佳实践总结 ​

问题识别 ​

  1. 最优子结构：问题的最优解包含子问题的最优解
  2. 重叠子问题：递归过程中存在重复计算
  3. 无后效性：当前状态只依赖于之前的状态
  4. 最优化问题：求最大值、最小值、方案数等

设计方法 ​

  1. 状态定义：明确dp数组每个位置的含义
  2. 转移方程：找出状态之间的递推关系
  3. 边界条件：确定最小子问题的解
  4. 计算顺序：保证依赖关系的正确性

实现技巧 ​

  1. 自顶向下：递归+记忆化，直观易懂
  2. 自底向上：迭代填表，效率更高
  3. 空间优化：滚动数组、状态压缩
  4. 时间优化：减少不必要的状态转移

调试验证 ​

  1. 小规模测试：用小数据验证算法正确性
  2. 边界检查：确保边界条件处理正确
  3. 状态打印：输出中间状态帮助调试
  4. 复杂度分析：验证时间空间复杂度

📊 动态规划问题分类 ​

线性DP ​

  • 特征：状态转移呈线性关系
  • 典型问题：爬楼梯、打家劫舍、最大子数组和
  • 状态定义：通常是一维数组dp[i]
  • 转移方程：dp[i] = f(dp[i-1], dp[i-2], ...)

区间DP ​

  • 特征：在区间上进行状态转移
  • 典型问题：矩阵链乘法、回文子串
  • 状态定义：二维数组dp[i][j]表示区间[i,j]
  • 转移方程：dp[i][j] = f(dp[i][k], dp[k+1][j])

背包DP ​

  • 特征：在容量限制下选择物品
  • 典型问题：0-1背包、完全背包、多重背包
  • 状态定义：dp[i][w]表示前i个物品重量不超过w的最大价值
  • 转移方程：dp[i][w] = max(dp[i-1][w], dp[i-1][w-weight[i]] + value[i])

树形DP ​

  • 特征：在树结构上进行状态转移
  • 典型问题：树的直径、树上背包
  • 状态定义：dp[node]表示以node为根的子树的最优解
  • 转移方程：dp[node] = f(dp[child1], dp[child2], ...)

状态机DP ​

  • 特征：状态之间有明确的转移条件
  • 典型问题：股票买卖、状态转换
  • 状态定义：dp[i][state]表示第i天处于某状态的最优解
  • 转移方程：根据状态转移图确定

📈 复杂度分析 ​

时间复杂度 ​

  • 状态数量：确定总共有多少个状态
  • 转移复杂度：计算每个状态的转移时间
  • 总时间复杂度：状态数量 × 转移复杂度

空间复杂度 ​

  • 存储空间：dp数组占用的空间
  • 优化可能：是否可以用滚动数组优化
  • 实际考虑：递归栈空间、临时变量等

2. 动态规划设计步骤 ​

java
// 动态规划设计步骤演示
import java.util.*;

public class DPDesignStepsDemo {
    
    public static void main(String[] args) {
        demonstrateDesignSteps();
        demonstrateStateDefinition();
        demonstrateTransitionEquation();
        demonstrateImplementationStrategies();
    }
    
    // 设计步骤演示
    public static void demonstrateDesignSteps() {
        System.out.println("=== 动态规划设计步骤 ===");
        
        /*
         * 动态规划问题的设计步骤：
         * 1. 确定状态 (Define State)
         * 2. 找出状态转移方程 (Find Transition Equation)
         * 3. 确定边界条件 (Determine Base Cases)
         * 4. 确定计算顺序 (Determine Computation Order)
         * 5. 优化空间复杂度 (Optimize Space)
         */
        
        System.out.println("动态规划设计的5个步骤:");
        System.out.println("1. 确定状态: 定义dp[i]或dp[i][j]表示什么");
        System.out.println("2. 状态转移方程: 找出状态之间的递推关系");
        System.out.println("3. 边界条件: 确定最小子问题的解");
        System.out.println("4. 计算顺序: 确保计算当前状态时所需状态已计算");
        System.out.println("5. 空间优化: 如果可能，优化空间复杂度");
        
        // 以爬楼梯问题为例演示设计步骤
        System.out.println("\n示例：爬楼梯问题");
        System.out.println("问题：有n级台阶，每次可以爬1级或2级，有多少种爬法？");
        
        int n = 5;
        
        System.out.println("\n步骤1 - 确定状态:");
        System.out.println("dp[i] = 爬到第i级台阶的方法数");
        
        System.out.println("\n步骤2 - 状态转移方程:");
        System.out.println("dp[i] = dp[i-1] + dp[i-2]");
        System.out.println("(可以从第i-1级爬1级，或从第i-2级爬2级)");
        
        System.out.println("\n步骤3 - 边界条件:");
        System.out.println("dp[0] = 1 (0级台阶有1种方法：不爬)");
        System.out.println("dp[1] = 1 (1级台阶有1种方法：爬1级)");
        
        System.out.println("\n步骤4 - 计算顺序:");
        System.out.println("从小到大计算：dp[0] -> dp[1] -> ... -> dp[n]");
        
        int result = climbStairs(n);
        System.out.println("\n结果：爬" + n + "级台阶有 " + result + " 种方法");
        
        System.out.println("\n步骤5 - 空间优化:");
        int optimizedResult = climbStairsOptimized(n);
        System.out.println("空间优化后结果: " + optimizedResult + " (空间复杂度从O(n)优化到O(1))");
        
        System.out.println();
    }

    public static void demonstrateTransitionEquation() {
        System.out.println("=== 状态转移方程演示 ===");
        
        /*
         * 状态转移方程描述了状态之间的关系
         * 常见的转移模式：
         * 1. 线性转移：dp[i] = f(dp[i-1], dp[i-2], ...)
         * 2. 区间转移：dp[i][j] = f(dp[i][k], dp[k+1][j])
         * 3. 树形转移：dp[node] = f(dp[child1], dp[child2], ...)
         */
        
        System.out.println("常见状态转移模式:");
        
        // 1. 线性转移示例：打家劫舍
        System.out.println("\n1. 线性转移 - 打家劫舍:");
        int[] houses = {2, 7, 9, 3, 1};
        System.out.println("   房屋金额: " + Arrays.toString(houses));
        System.out.println("   状态转移: dp[i] = max(dp[i-1], dp[i-2] + nums[i])");
        System.out.println("   含义: 第i家，要么不偷(取dp[i-1])，要么偷(取dp[i-2]+当前金额)");
        
        int maxMoney = rob(houses);
        System.out.println("   最大金额: " + maxMoney);
        
        // 2. 区间转移示例：矩阵链乘法
        System.out.println("\n2. 区间转移 - 矩阵链乘法:");
        int[] dimensions = {1, 2, 3, 4, 5}; // 4个矩阵：1x2, 2x3, 3x4, 4x5
        System.out.println("   矩阵维度: " + Arrays.toString(dimensions));
        System.out.println("   状态转移: dp[i][j] = min(dp[i][k] + dp[k+1][j] + cost(i,k,j))");
        System.out.println("   含义: 计算矩阵i到j的最小乘法次数");
        
        int minMultiplications = matrixChainOrder(dimensions);
        System.out.println("   最小乘法次数: " + minMultiplications);
        
        System.out.println();
    }

    // 算法实现
    
    // 爬楼梯
    public static int climbStairs(int n) {
        if (n <= 1) return 1;
        
        int[] dp = new int[n + 1];
        dp[0] = 1;
        dp[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        
        return dp[n];
    }
    
    public static int climbStairsOptimized(int n) {
        if (n <= 1) return 1;
        
        int prev2 = 1, prev1 = 1;
        for (int i = 2; i <= n; i++) {
            int current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
    
    // 最大子数组和（带步骤）
    public static int maxSubarrayWithSteps(int[] nums) {
        System.out.println("   DP过程:");
        
        int[] dp = new int[nums.length];
        dp[0] = nums[0];
        int maxSum = dp[0];
        
        System.out.println("   dp[0] = " + dp[0]);
        
        for (int i = 1; i < nums.length; i++) {
            dp[i] = Math.max(nums[i], dp[i - 1] + nums[i]);
            maxSum = Math.max(maxSum, dp[i]);
            System.out.println("   dp[" + i + "] = max(" + nums[i] + ", " + dp[i-1] + " + " + nums[i] + ") = " + dp[i]);
        }
        
        return maxSum;
    }
    
    // 最长公共子序列
    public static int longestCommonSubsequence(String text1, String text2) {
        int m = text1.length(), n = text2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 打家劫舍
    public static int rob(int[] nums) {
        if (nums.length == 0) return 0;
        if (nums.length == 1) return nums[0];
        
        int[] dp = new int[nums.length];
        dp[0] = nums[0];
        dp[1] = Math.max(nums[0], nums[1]);
        
        for (int i = 2; i < nums.length; i++) {
            dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
        }
        
        return dp[nums.length - 1];
    }
    
    // 矩阵链乘法
    public static int matrixChainOrder(int[] dimensions) {
        int n = dimensions.length - 1;
        int[][] dp = new int[n][n];
        
        for (int len = 2; len <= n; len++) {
            for (int i = 0; i <= n - len; i++) {
                int j = i + len - 1;
                dp[i][j] = Integer.MAX_VALUE;
                
                for (int k = i; k < j; k++) {
                    int cost = dp[i][k] + dp[k + 1][j] + 
                              dimensions[i] * dimensions[k + 1] * dimensions[j + 1];
                    dp[i][j] = Math.min(dp[i][j], cost);
                }
            }
        }
        
        return dp[0][n - 1];
    }
    
    // 编辑距离 - 记忆化
    public static int editDistanceMemoized(String word1, String word2) {
        Map<String, Integer> memo = new HashMap<>();
        return editDistanceHelper(word1, word2, 0, 0, memo);
    }
    
    private static int editDistanceHelper(String word1, String word2, int i, int j, Map<String, Integer> memo) {
        if (i == word1.length()) return word2.length() - j;
        if (j == word2.length()) return word1.length() - i;
        
        String key = i + "," + j;
        if (memo.containsKey(key)) {
            return memo.get(key);
        }
        
        int result;
        if (word1.charAt(i) == word2.charAt(j)) {
            result = editDistanceHelper(word1, word2, i + 1, j + 1, memo);
        } else {
            int insert = editDistanceHelper(word1, word2, i, j + 1, memo);
            int delete = editDistanceHelper(word1, word2, i + 1, j, memo);
            int replace = editDistanceHelper(word1, word2, i + 1, j + 1, memo);
            result = 1 + Math.min(Math.min(insert, delete), replace);
        }
        
        memo.put(key, result);
        return result;
    }
    
    // 编辑距离 - 表格法
    public static int editDistanceTabulation(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 0; i <= m; i++) dp[i][0] = i;
        for (int j = 0; j <= n; j++) dp[0][j] = j;
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = 1 + Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]);
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 编辑距离 - 空间优化
    public static int editDistanceOptimized(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        
        // 确保word1是较短的字符串
        if (m > n) {
            return editDistanceOptimized(word2, word1);
        }
        
        int[] dp = new int[m + 1];
        for (int i = 0; i <= m; i++) dp[i] = i;
        
        for (int j = 1; j <= n; j++) {
            int prev = dp[0];
            dp[0] = j;
            
            for (int i = 1; i <= m; i++) {
                int temp = dp[i];
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i] = prev;
                } else {
                    dp[i] = 1 + Math.min(Math.min(dp[i], dp[i - 1]), prev);
                }
                prev = temp;
            }
        }
        
        return dp[m];
    }
