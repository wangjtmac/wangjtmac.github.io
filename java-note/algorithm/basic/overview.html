算法基础概述 ​

📚 概述 ​

算法是解决问题的一系列明确指令，是计算机科学的核心基础。一个好的算法不仅能正确解决问题，还要在时间和空间上具有良好的效率。算法的学习对于程序员来说至关重要，它不仅能提高编程能力，还能培养逻辑思维和问题解决能力。本章将从算法的基本概念开始，逐步深入到算法分析、设计方法和实际应用。我们将学习如何评估算法的好坏，掌握常见的算法设计技巧，并通过大量的实例来加深理解。无论是面试准备还是实际开发，扎实的算法基础都是不可或缺的技能。

🎯 学习目标 ​

  • 理解算法的基本概念和特征
  • 掌握算法复杂度分析方法
  • 学会常见的算法设计策略
  • 熟悉算法的实现和优化技巧
  • 培养算法思维和问题解决能力
  • 了解算法在实际开发中的应用
  • 掌握算法面试的常见题型和解题思路

    // 欧几里得算法求最大公约数
    public static int euclideanGCD(int a, int b) {
        System.out.println("计算 gcd(" + a + ", " + b + ")");
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
            if (b != 0) {
                System.out.println("  -> gcd(" + a + ", " + b + ")");
            }
        }
        return a;
    }
    
    // 线性查找
    public static int linearSearch(int[] array, int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                return i;
            }
        }
        return -1;
    }
    
    // 冒泡排序
    public static void bubbleSort(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    // 交换元素
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
        System.out.println("排序后: " + java.util.Arrays.toString(array));
    }
    
    // 斐波那契数列
    public static int fibonacci(int n) {
        if (n <= 1) {
            return n;
        }
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
    
    // 回文检查
    public static boolean isPalindrome(String str) {
        int left = 0;
        int right = str.length() - 1;
        
        while (left < right) {
            if (str.charAt(left) != str.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
    
    // 二分查找
    public static int binarySearch(int[] array, int target) {
        int left = 0;
        int right = array.length - 1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] == target) {
                return mid;
            } else if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return -1;
    }
    
    // 数组求和
    public static int arraySum(int[] array) {
        int sum = 0;
        for (int num : array) {
            sum += num;
        }
        return sum;
    }
}

import java.util.*;

public class AlgorithmDesignStrategiesDemo {
    
    public static void main(String[] args) {
        demonstrateDivideAndConquer();
        demonstrateGreedyAlgorithm();
        demonstrateDynamicProgramming();
        demonstrateBacktracking();
    }
    
    // 分治算法演示
    public static void demonstrateDivideAndConquer() {
        System.out.println("=== 分治算法演示 ===");
        
        /*
         * 分治算法 (Divide and Conquer)：
         * 1. 分解 - 将问题分解为若干个规模较小的相同问题
         * 2. 解决 - 若子问题规模足够小，直接求解
         * 3. 合并 - 将子问题的解合并为原问题的解
         */
        
        System.out.println("分治算法 - 归并排序:");
        
        int[] array = {38, 27, 43, 3, 9, 82, 10};
        System.out.println("排序前: " + Arrays.toString(array));
        
        mergeSort(array, 0, array.length - 1);
        System.out.println("排序后: " + Arrays.toString(array));
        
        // 分治算法 - 快速幂
        System.out.println("\n分治算法 - 快速幂:");
        int base = 2, exponent = 10;
        long result = fastPower(base, exponent);
        System.out.println(base + "^" + exponent + " = " + result);
        
        System.out.println();
    }
    
    // 贪心算法演示
    public static void demonstrateGreedyAlgorithm() {
        System.out.println("=== 贪心算法演示 ===");
        
        /*
         * 贪心算法 (Greedy Algorithm)：
         * 在每一步选择中都采取在当前状态下最好或最优的选择，
         * 从而希望导致结果是最好或最优的算法
         */
        
        System.out.println("贪心算法 - 找零钱问题:");
        
        int[] coins = {25, 10, 5, 1}; // 硬币面值
        int amount = 67; // 需要找零的金额
        
        List<Integer> result = greedyCoinChange(coins, amount);
        System.out.println("找零 " + amount + " 分，使用硬币: " + result);
        
        // 贪心算法 - 活动选择问题
        System.out.println("\n贪心算法 - 活动选择问题:");
        
        int[] start = {1, 3, 0, 5, 8, 5};
        int[] finish = {2, 4, 6, 7, 9, 9};
        
        List<Integer> selectedActivities = activitySelection(start, finish);
        System.out.println("选择的活动: " + selectedActivities);
        
        System.out.println();
    }
    
    // 动态规划演示
    public static void demonstrateDynamicProgramming() {
        System.out.println("=== 动态规划演示 ===");
        
        /*
         * 动态规划 (Dynamic Programming)：
         * 通过把原问题分解为相对简单的子问题的方式求解复杂问题的方法
         * 核心思想：避免重复计算
         */
        
        System.out.println("动态规划 - 斐波那契数列:");
        
        int n = 10;
        System.out.println("递归方式 - 第" + n + "个斐波那契数: " + fibonacciRecursive(n));
        System.out.println("动态规划 - 第" + n + "个斐波那契数: " + fibonacciDP(n));
        
        // 动态规划 - 最长公共子序列
        System.out.println("\n动态规划 - 最长公共子序列:");
        
        String str1 = "ABCDGH";
        String str2 = "AEDFHR";
        int lcsLength = longestCommonSubsequence(str1, str2);
        System.out.println("字符串 \"" + str1 + "\" 和 \"" + str2 + "\" 的最长公共子序列长度: " + lcsLength);
        
        System.out.println();
    }
    
    // 回溯算法演示
    public static void demonstrateBacktracking() {
        System.out.println("=== 回溯算法演示 ===");
        
        /*
         * 回溯算法 (Backtracking)：
         * 采用试错的思想，它尝试分步的去解决一个问题。
         * 在分步解决问题的过程中，当它通过尝试发现现有的分步答案不能得到有效的正确的解答的时候，
         * 它将取消上一步甚至是上几步的计算，再通过其它的可能的分步解答再次尝试寻找问题的答案。
         */
        
        System.out.println("回溯算法 - N皇后问题 (4皇后):");
        
        int n = 4;
        List<List<String>> solutions = solveNQueens(n);
        
        System.out.println("4皇后问题的解决方案数量: " + solutions.size());
        for (int i = 0; i < solutions.size(); i++) {
            System.out.println("解决方案 " + (i + 1) + ":");
            for (String row : solutions.get(i)) {
                System.out.println(row);
            }
            System.out.println();
        }
        
        System.out.println();
    }
    
    // 分治算法实现
    
    // 归并排序
    public static void mergeSort(int[] array, int left, int right) {
        if (left < right) {
            int mid = left + (right - left) / 2;
            
            mergeSort(array, left, mid);
            mergeSort(array, mid + 1, right);
            
            merge(array, left, mid, right);
        }
    }
    
    public static void merge(int[] array, int left, int mid, int right) {
        int[] temp = new int[right - left + 1];
        int i = left, j = mid + 1, k = 0;
        
        while (i <= mid && j <= right) {
            if (array[i] <= array[j]) {
                temp[k++] = array[i++];
            } else {
                temp[k++] = array[j++];
            }
        }
        
        while (i <= mid) {
            temp[k++] = array[i++];
        }
        
        while (j <= right) {
            temp[k++] = array[j++];
        }
        
        System.arraycopy(temp, 0, array, left, temp.length);
    }
    
    // 快速幂
    public static long fastPower(int base, int exponent) {
        if (exponent == 0) return 1;
        if (exponent == 1) return base;
        
        if (exponent % 2 == 0) {
            long half = fastPower(base, exponent / 2);
            return half * half;
        } else {
            return base * fastPower(base, exponent - 1);
        }
    }
    
    // 贪心算法实现
    
    // 贪心找零钱
    public static List<Integer> greedyCoinChange(int[] coins, int amount) {
        List<Integer> result = new ArrayList<>();
        
        for (int coin : coins) {
            while (amount >= coin) {
                result.add(coin);
                amount -= coin;
            }
        }
        
        return result;
    }
    
    // 活动选择问题
    public static List<Integer> activitySelection(int[] start, int[] finish) {
        List<Integer> result = new ArrayList<>();
        int n = start.length;
        
        // 按结束时间排序
        Integer[] indices = new Integer[n];
        for (int i = 0; i < n; i++) {
            indices[i] = i;
        }
        Arrays.sort(indices, (a, b) -> Integer.compare(finish[a], finish[b]));
        
        result.add(indices[0]);
        int lastSelected = 0;
        
        for (int i = 1; i < n; i++) {
            if (start[indices[i]] >= finish[indices[lastSelected]]) {
                result.add(indices[i]);
                lastSelected = i;
            }
        }
        
        return result;
    }
    
    // 动态规划实现
    
    // 递归斐波那契（效率低）
    public static int fibonacciRecursive(int n) {
        if (n <= 1) return n;
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
    }
    
    // 动态规划斐波那契
    public static int fibonacciDP(int n) {
        if (n <= 1) return n;
        
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        
        return dp[n];
    }
    
    // 最长公共子序列
    public static int longestCommonSubsequence(String str1, String str2) {
        int m = str1.length();
        int n = str2.length();
        int[][] dp = new int[m + 1][n + 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        
        return dp[m][n];
    }
    
    // 回溯算法实现
    
    // N皇后问题
    public static List<List<String>> solveNQueens(int n) {
        List<List<String>> result = new ArrayList<>();
        char[][] board = new char[n][n];
        
        // 初始化棋盘
        for (int i = 0; i < n; i++) {
            Arrays.fill(board[i], '.');
        }
        
        backtrackNQueens(board, 0, result);
        return result;
    }
    
    private static void backtrackNQueens(char[][] board, int row, List<List<String>> result) {
        if (row == board.length) {
            result.add(constructSolution(board));
            return;
        }
        
        for (int col = 0; col < board.length; col++) {
            if (isValidNQueens(board, row, col)) {
                board[row][col] = 'Q';
                backtrackNQueens(board, row + 1, result);
                board[row][col] = '.';
            }
        }
    }
    
    private static boolean isValidNQueens(char[][] board, int row, int col) {
        int n = board.length;
        
        // 检查列
        for (int i = 0; i < row; i++) {
            if (board[i][col] == 'Q') {
                return false;
            }
        }
        
        // 检查左上对角线
        for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
            if (board[i][j] == 'Q') {
                return false;
            }
        }
        
        // 检查右上对角线
        for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
            if (board[i][j] == 'Q') {
                return false;
            }
        }
        
        return true;
    }
    
    private static List<String> constructSolution(char[][] board) {
        List<String> solution = new ArrayList<>();
        for (char[] row : board) {
            solution.add(new String(row));
        }
        return solution;
    }
}

  3. 多做练习：通过大量练习加深对算法的理解
  4. 总结归纳：定期总结算法的共同特点和规律

算法实现技巧 ​

  1. 代码规范：保持代码清晰、注释完整
  2. 边界处理：注意处理特殊情况和边界条件
  3. 测试验证：编写测试用例验证算法正确性
  4. 性能优化：在保证正确性的前提下优化性能

复杂度分析要点 ​

  1. 准确分析：正确分析时间和空间复杂度
  2. 最坏情况：通常分析最坏情况下的复杂度
  3. 实际测试：理论分析结合实际性能测试
  4. 权衡选择：根据实际需求选择合适的算法

问题解决思路 ​

  1. 理解问题：充分理解问题的要求和约束
  2. 分析数据：分析输入数据的特点和规模
  3. 选择策略：根据问题特点选择合适的算法策略
  4. 实现优化：实现算法并进行必要的优化
