数组与链表详解 ​

📚 概述 ​

数组和链表是两种最基础也是最重要的线性数据结构，它们在计算机科学中扮演着核心角色。数组以其连续的内存布局提供了高效的随机访问能力，而链表则以其灵活的动态内存分配支持高效的插入和删除操作。理解这两种数据结构的特点、优缺点以及适用场景，对于选择合适的数据结构解决实际问题至关重要。从底层的内存管理到高层的算法设计，数组和链表的概念无处不在。本章将深入探讨数组和链表的实现原理、操作复杂度、优化技巧以及在实际开发中的应用，帮助您全面掌握这两种基础数据结构。

🎯 学习目标 ​

  • 深入理解数组和链表的底层实现原理
  • 掌握各种操作的时间和空间复杂度
  • 学会选择合适的数据结构解决问题
  • 熟悉数组和链表的优化技巧
  • 理解动态数组的扩容机制
  • 掌握链表的各种变种和应用
  • 学会实现高效的数组和链表操作
  • 了解数组和链表在实际开发中的应用

    public static void demonstrateArrayOperations() {
        System.out.println("=== 数组操作演示 ===");
        
        /*
         * 数组的基本操作及其复杂度：
         * 1. 访问 - O(1)
         * 2. 搜索 - O(n)
         * 3. 插入 - O(n)
         * 4. 删除 - O(n)

        int[] array = {1, 3, 5, 7, 9, 11, 13, 15};
        System.out.println("原始数组: " + Arrays.toString(array));
        
        // 1. 搜索操作
        System.out.println("\n1. 搜索操作:");
        
        // 线性搜索 - O(n)
        int target = 7;
        long startTime = System.nanoTime();
        int linearIndex = linearSearch(array, target);
        long linearTime = System.nanoTime() - startTime;
        System.out.println("线性搜索 " + target + ": 索引 " + linearIndex + ", 耗时: " + linearTime + "ns");
        
        // 二分搜索 - O(log n) (需要有序数组)
        startTime = System.nanoTime();
        int binaryIndex = binarySearch(array, target);
        long binaryTime = System.nanoTime() - startTime;
        System.out.println("二分搜索 " + target + ": 索引 " + binaryIndex + ", 耗时: " + binaryTime + "ns");
        
        // 2. 插入操作
        System.out.println("\n2. 插入操作:");
        int[] insertArray = Arrays.copyOf(array, array.length + 1);
        
        // 在指定位置插入 - O(n)
        startTime = System.nanoTime();
        insertAt(insertArray, 2, 4);
        long insertTime = System.nanoTime() - startTime;
        System.out.println("在索引2插入4: " + Arrays.toString(insertArray) + ", 耗时: " + insertTime + "ns");
        
        // 3. 删除操作
        System.out.println("\n3. 删除操作:");
        
        // 删除指定位置元素 - O(n)
        startTime = System.nanoTime();
        int[] deleteArray = deleteAt(insertArray, 2);
        long deleteTime = System.nanoTime() - startTime;
        System.out.println("删除索引2: " + Arrays.toString(deleteArray) + ", 耗时: " + deleteTime + "ns");
        
        // 4. 批量操作
        System.out.println("\n4. 批量操作:");
        
        // 数组复制 - O(n)
        startTime = System.nanoTime();
        int[] copiedArray = Arrays.copyOf(array, array.length);
        long copyTime = System.nanoTime() - startTime;
        System.out.println("数组复制: " + Arrays.toString(copiedArray) + ", 耗时: " + copyTime + "ns");
        
        // 数组排序 - O(n log n)

    // 数组优化演示
    public static void demonstrateArrayOptimizations() {
        System.out.println("=== 数组优化演示 ===");
        
        /*
         * 数组优化技巧：
         * 1. 缓存局部性
         * 2. 预分配空间
         * 3. 批量操作
         * 4. 内存对齐
         */
        
        System.out.println("数组优化技巧:");
        
        // 1. 缓存局部性
        System.out.println("\n1. 缓存局部性优化:");
        int size = 1000;
        int[][] matrix = new int[size][size];
        
        // 按行访问 (缓存友好)
        long startTime = System.nanoTime();
        for (int i = 0; i < size; i++) {
            for (int j = 0; j < size; j++) {
                matrix[i][j] = i + j;
            }
        }
        long rowTime = System.nanoTime() - startTime;
        
        // 按列访问 (缓存不友好)
        startTime = System.nanoTime();
        for (int j = 0; j < size; j++) {
            for (int i = 0; i < size; i++) {
                matrix[i][j] = i + j;
            }
        }
        long colTime = System.nanoTime() - startTime;
        
        System.out.println("按行访问耗时: " + rowTime + "ns");
        System.out.println("按列访问耗时: " + colTime + "ns");
        System.out.println("性能差异: " + (colTime / (double) rowTime) + "倍");
        
        // 2. 预分配空间
        System.out.println("\n2. 预分配空间优化:");
        
        // 不预分配
        List<Integer> list1 = new ArrayList<>();
        startTime = System.nanoTime();
        for (int i = 0; i < 10000; i++) {
            list1.add(i);
        }
        long noPreallocTime = System.nanoTime() - startTime;
        
        // 预分配
        List<Integer> list2 = new ArrayList<>(10000);
        startTime = System.nanoTime();
        for (int i = 0; i < 10000; i++) {
            list2.add(i);
        }
        long preallocTime = System.nanoTime() - startTime;
        
        System.out.println("不预分配耗时: " + noPreallocTime + "ns");
        System.out.println("预分配耗时: " + preallocTime + "ns");
        System.out.println("性能提升: " + (noPreallocTime / (double) preallocTime) + "倍");
        
        // 3. 批量操作
        System.out.println("\n3. 批量操作优化:");
        demonstrateBatchOperations();
        
        System.out.println();
    }
    
    // 算法实现
    
    public static int linearSearch(int[] array, int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                return i;
            }
        }
        return -1;
    }
    
    public static int binarySearch(int[] array, int target) {
        int left = 0, right = array.length - 1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] == target) {
                return mid;
            } else if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return -1;
    }
    
    public static void insertAt(int[] array, int index, int value) {
        // 向右移动元素
        for (int i = array.length - 1; i > index; i--) {
            array[i] = array[i - 1];
        }
        array[index] = value;
    }
    
    public static int[] deleteAt(int[] array, int index) {
        int[] newArray = new int[array.length - 1];
        
        // 复制index之前的元素
        System.arraycopy(array, 0, newArray, 0, index);
        
        // 复制index之后的元素
        System.arraycopy(array, index + 1, newArray, index, array.length - index - 1);
        
        return newArray;
    }
    
    public static void compareArrayPerformance() {
        int size = 100000;
        
        // 数组 vs ArrayList 访问性能
        int[] array = new int[size];
        List<Integer> arrayList = new ArrayList<>(size);
        
        for (int i = 0; i < size; i++) {
            array[i] = i;
            arrayList.add(i);
        }
        
        // 数组访问
        long startTime = System.nanoTime();
        long sum1 = 0;
        for (int i = 0; i < size; i++) {
            sum1 += array[i];
        }
        long arrayTime = System.nanoTime() - startTime;
        
        // ArrayList访问
        startTime = System.nanoTime();
        long sum2 = 0;
        for (int i = 0; i < size; i++) {
            sum2 += arrayList.get(i);
        }
        long listTime = System.nanoTime() - startTime;
        
        System.out.println("数组访问耗时: " + arrayTime + "ns");
        System.out.println("ArrayList访问耗时: " + listTime + "ns");
        System.out.println("性能差异: " + (listTime / (double) arrayTime) + "倍");
    }
    
    public static void demonstrateBatchOperations() {
        int[] source = new int[10000];
        for (int i = 0; i < source.length; i++) {
            source[i] = i;
        }
        
        // 逐个复制
        int[] dest1 = new int[source.length];
        long startTime = System.nanoTime();
        for (int i = 0; i < source.length; i++) {
            dest1[i] = source[i];
        }
        long individualTime = System.nanoTime() - startTime;
        
        // 批量复制
        int[] dest2 = new int[source.length];
        startTime = System.nanoTime();
        System.arraycopy(source, 0, dest2, 0, source.length);
        long batchTime = System.nanoTime() - startTime;
        
        System.out.println("逐个复制耗时: " + individualTime + "ns");
        System.out.println("批量复制耗时: " + batchTime + "ns");
        System.out.println("性能提升: " + (individualTime / (double) batchTime) + "倍");
    }
}

// 简单动态数组实现
class SimpleArrayList<T> {
    private Object[] elements;
    private int size;
    private static final int DEFAULT_CAPACITY = 10;
    
    public SimpleArrayList() {
        elements = new Object[DEFAULT_CAPACITY];
        size = 0;
    }
    
    public void add(T element) {
        if (size >= elements.length) {
            resize();
        }
        elements[size++] = element;
    }
    
    private void resize() {
        int newCapacity = elements.length * 2;
        elements = Arrays.copyOf(elements, newCapacity);
        System.out.println("扩容: " + (newCapacity / 2) + " -> " + newCapacity);
    }
    
    @SuppressWarnings("unchecked")
    public T get(int index) {
        if (index >= size) {
            throw new IndexOutOfBoundsException();
        }
        return (T) elements[index];
    }
    
    public int size() {
        return size;
    }
    
    public int capacity() {
        return elements.length;
    }
}

    // 链表操作演示
    public static void demonstrateLinkedListOperations() {
        System.out.println("=== 链表操作演示 ===");

        /*
         * 链表的基本操作及其复杂度：
         * 1. 访问 - O(n)
         * 2. 搜索 - O(n)
         * 3. 插入 - O(1) 已知位置，O(n) 未知位置
         * 4. 删除 - O(1) 已知位置，O(n) 未知位置
         */

        SinglyLinkedList<Integer> list = new SinglyLinkedList<>();

        // 初始化链表
        for (int i = 1; i <= 8; i++) {
            list.addLast(i * 10);
        }
        System.out.println("初始链表: " + list);

        // 1. 搜索操作
        System.out.println("\n1. 搜索操作:");
        int target = 50;
        long startTime = System.nanoTime();
        int index = list.indexOf(target);
        long searchTime = System.nanoTime() - startTime;
        System.out.println("搜索 " + target + ": 索引 " + index + ", 耗时: " + searchTime + "ns");

        // 2. 插入操作
        System.out.println("\n2. 插入操作:");

        // 头部插入 - O(1)
        startTime = System.nanoTime();
        list.addFirst(5);
        long headInsertTime = System.nanoTime() - startTime;
        System.out.println("头部插入5: " + list + ", 耗时: " + headInsertTime + "ns");

        // 尾部插入 - O(n) 单链表，O(1) 如果有尾指针
        startTime = System.nanoTime();
        list.addLast(95);
        long tailInsertTime = System.nanoTime() - startTime;
        System.out.println("尾部插入95: " + list + ", 耗时: " + tailInsertTime + "ns");

        // 中间插入 - O(n)
        startTime = System.nanoTime();
        list.insertAt(3, 25);
        long middleInsertTime = System.nanoTime() - startTime;
        System.out.println("索引3插入25: " + list + ", 耗时: " + middleInsertTime + "ns");

        // 3. 删除操作
        System.out.println("\n3. 删除操作:");

        // 头部删除 - O(1)
        startTime = System.nanoTime();
        Integer removed1 = list.removeFirst();
        long headRemoveTime = System.nanoTime() - startTime;
        System.out.println("头部删除: " + removed1 + ", 结果: " + list + ", 耗时: " + headRemoveTime + "ns");

        // 按值删除 - O(n)
        startTime = System.nanoTime();
        boolean removed2 = list.remove(Integer.valueOf(50));
        long valueRemoveTime = System.nanoTime() - startTime;
        System.out.println("删除值50: " + removed2 + ", 结果: " + list + ", 耗时: " + valueRemoveTime + "ns");

        // 按索引删除 - O(n)
        startTime = System.nanoTime();

    public static void compareArrayVsLinkedList() {
        int size = 10000;

        // 数组
        List<Integer> arrayList = new ArrayList<>();

        // 链表
        List<Integer> linkedList = new LinkedList<>();

        // 1. 顺序插入性能
        long startTime = System.nanoTime();

            arrayList.add(i);
        }
        long arrayAddTime = System.nanoTime() - startTime;

        startTime = System.nanoTime();
        for (int i = 0; i < size; i++) {
            linkedList.add(i);
        }
        long linkedAddTime = System.nanoTime() - startTime;

        // 2. 随机访问性能
        Random random = new Random();
        startTime = System.nanoTime();
        for (int i = 0; i < 1000; i++) {
            arrayList.get(random.nextInt(size));
        }
        long arrayGetTime = System.nanoTime() - startTime;

        startTime = System.nanoTime();
        for (int i = 0; i < 1000; i++) {
            linkedList.get(random.nextInt(size));
        }
        long linkedGetTime = System.nanoTime() - startTime;

        // 3. 头部插入性能
        startTime = System.nanoTime();
        for (int i = 0; i < 1000; i++) {
            arrayList.add(0, i);
        }
        long arrayInsertTime = System.nanoTime() - startTime;

        startTime = System.nanoTime();
        for (int i = 0; i < 1000; i++) {
            linkedList.add(0, i);
        }
        long linkedInsertTime = System.nanoTime() - startTime;

        System.out.println("性能对比结果:");
        System.out.println("顺序插入 - 数组: " + arrayAddTime + "ns, 链表: " + linkedAddTime + "ns");
        System.out.println("随机访问 - 数组: " + arrayGetTime + "ns, 链表: " + linkedGetTime + "ns");
        System.out.println("头部插入 - 数组: " + arrayInsertTime + "ns, 链表: " + linkedInsertTime + "ns");

        System.out.println("\n结论:");
        System.out.println("- 顺序插入: 数组略优");
        System.out.println("- 随机访问: 数组大幅优于链表");
        System.out.println("- 头部插入: 链表大幅优于数组");
    }

// 单链表实现
class SinglyLinkedList<T> {
    private Node<T> head;
    private int size;

    private static class Node<T> {
        T data;
        Node<T> next;

        Node(T data) {
            this.data = data;
        }
    }

    public void addFirst(T data) {
        Node<T> newNode = new Node<>(data);
        newNode.next = head;
        head = newNode;
        size++;
    }

    public void addLast(T data) {
        Node<T> newNode = new Node<>(data);
        if (head == null) {
            head = newNode;
        } else {
            Node<T> current = head;
            while (current.next != null) {
                current = current.next;
            }
            current.next = newNode;
        }
        size++;
    }

    public void insertAt(int index, T data) {
        if (index == 0) {
            addFirst(data);
            return;
        }

        Node<T> newNode = new Node<>(data);
        Node<T> current = head;
        for (int i = 0; i < index - 1; i++) {
            current = current.next;
        }

        newNode.next = current.next;
        current.next = newNode;
        size++;
    }

    public T get(int index) {
        Node<T> current = head;
        for (int i = 0; i < index; i++) {
            current = current.next;
        }
        return current.data;
    }

    public int indexOf(T data) {
        Node<T> current = head;
        int index = 0;

        while (current != null) {
            if (current.data.equals(data)) {
                return index;
            }
            current = current.next;
            index++;
        }

        return -1;
    }

    public T removeFirst() {
        if (head == null) return null;

        T data = head.data;
        head = head.next;
        size--;
        return data;
    }

    public boolean remove(T data) {
        if (head == null) return false;

        if (head.data.equals(data)) {
            head = head.next;
            size--;
            return true;
        }

        Node<T> current = head;
        while (current.next != null) {
            if (current.next.data.equals(data)) {
                current.next = current.next.next;
                size--;
                return true;
            }
            current = current.next;
        }

        return false;
    }

    public T removeAt(int index) {
        if (index == 0) {
            return removeFirst();
        }

        Node<T> current = head;
        for (int i = 0; i < index - 1; i++) {
            current = current.next;
        }

        T data = current.next.data;
        current.next = current.next.next;
        size--;
        return data;
    }

    public int size() {
        return size;
    }

    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder("[");
        Node<T> current = head;
        while (current != null) {
            sb.append(current.data);
            if (current.next != null) {
                sb.append(", ");
            }
            current = current.next;
        }
        sb.append("]");
        return sb.toString();
    }
}

// 双链表实现
class DoublyLinkedList<T> {
    private Node<T> head;
    private Node<T> tail;
    private int size;

    private static class Node<T> {
        T data;
        Node<T> next;
        Node<T> prev;

        Node(T data) {
            this.data = data;
        }
    }

    public void addLast(T data) {
        Node<T> newNode = new Node<>(data);

        if (tail == null) {
            head = tail = newNode;
        } else {
            tail.next = newNode;
            newNode.prev = tail;
            tail = newNode;
        }
        size++;
    }

    public List<T> forwardTraversal() {
        List<T> result = new ArrayList<>();
        Node<T> current = head;
        while (current != null) {
            result.add(current.data);
            current = current.next;
        }
        return result;
    }

    public List<T> backwardTraversal() {
        List<T> result = new ArrayList<>();
        Node<T> current = tail;
        while (current != null) {
            result.add(current.data);
            current = current.prev;
        }
        return result;
    }

    @Override
    public String toString() {
        return forwardTraversal().toString();
    }
}

// 循环链表实现
class CircularLinkedList<T> {
    private Node<T> tail;
    private int size;

    private static class Node<T> {
        T data;
        Node<T> next;

        Node(T data) {
            this.data = data;
        }
    }

    public void add(T data) {
        Node<T> newNode = new Node<>(data);

        if (tail == null) {
            tail = newNode;
            tail.next = tail;
        } else {
            newNode.next = tail.next;
            tail.next = newNode;
            tail = newNode;
        }
        size++;
    }

    public List<T> circularTraversal(int count) {
        List<T> result = new ArrayList<>();
        if (tail == null) return result;

        Node<T> current = tail.next;
        for (int i = 0; i < count; i++) {
            result.add(current.data);
            current = current.next;
        }
        return result;
    }

    @Override
    public String toString() {
        if (tail == null) return "[]";

        List<T> result = new ArrayList<>();
        Node<T> current = tail.next;
        do {
            result.add(current.data);
            current = current.next;
        } while (current != tail.next);

        return result.toString();
    }
}

class SentinelLinkedList<T> {
    private Node<T> sentinel;
    private int size;

    private static class Node<T> {
        T data;
        Node<T> next;

        Node(T data) {
            this.data = data;
        }
    }

    public SentinelLinkedList() {
        sentinel = new Node<>(null);
        sentinel.next = sentinel;
    }

    public void add(T data) {
        Node<T> newNode = new Node<>(data);
        newNode.next = sentinel.next;
        sentinel.next = newNode;
        size++;
    }

    @Override
    public String toString() {
        List<T> result = new ArrayList<>();
        Node<T> current = sentinel.next;
        while (current != sentinel) {
            result.add(current.data);
            current = current.next;
        }
        return result.toString();
    }
}

💡 最佳实践总结 ​

数组使用建议 ​

  1. 适用场景：

    • 需要频繁随机访问元素
    • 数据大小相对固定
    • 对内存使用效率要求高
    • 需要利用缓存局部性

  2. 优化技巧：

    • 预分配足够的空间避免频繁扩容
    • 利用缓存局部性优化访问模式
    • 使用批量操作提高效率
    • 考虑内存对齐优化性能

链表使用建议 ​

  1. 适用场景：

    • 需要频繁插入删除操作
    • 数据大小变化很大
    • 不需要随机访问
    • 实现其他数据结构的基础

  2. 优化技巧：

    • 使用尾指针优化尾部操作
    • 考虑双链表支持双向操作
    • 使用哨兵节点简化边界处理
    • 注意内存管理避免泄漏

选择决策 ​

  1. 性能考虑：

    • 访问模式：随机访问选数组，顺序访问可选链表
    • 操作频率：频繁插入删除选链表，频繁访问选数组
    • 内存限制：内存紧张选数组，内存充足可选链表
    • 缓存效率：重视缓存性能选数组

  2. 实际应用：

    • 数据库索引：B+树结合数组和链表优势
    • 操作系统：进程链表管理，内存数组分配
    • 编译器：符号表用哈希表，语法树用链表
    • 图形处理：顶点数组，邻接链表
