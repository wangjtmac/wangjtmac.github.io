    public static BinaryTreeNode<Integer> createSampleTree() {

        root.right = new BinaryTreeNode<>(3);

    public static int getHeight(BinaryTreeNode<?> node) {
        if (node == null) return -1;
        return 1 + Math.max(getHeight(node.left), getHeight(node.right));
    }
    
    public static int getNodeCount(BinaryTreeNode<?> node) {
        if (node == null) return 0;
        return 1 + getNodeCount(node.left) + getNodeCount(node.right);
    }
    
    public static int getLeafCount(BinaryTreeNode<?> node) {
        if (node == null) return 0;
        if (node.left == null && node.right == null) return 1;
        return getLeafCount(node.left) + getLeafCount(node.right);
    }
    
    public static int getInternalNodeCount(BinaryTreeNode<?> node) {
        return getNodeCount(node) - getLeafCount(node);
    }

    public static List<Integer> preorderTraversal(BinaryTreeNode<Integer> root) {
        List<Integer> result = new ArrayList<>();

    private static void preorderHelper(BinaryTreeNode<Integer> node, List<Integer> result) {
        if (node == null) return;
        result.add(node.data);
        preorderHelper(node.left, result);
        preorderHelper(node.right, result);
    }
    
    public static List<Integer> inorderTraversal(BinaryTreeNode<Integer> root) {

    private static void inorderHelper(BinaryTreeNode<Integer> node, List<Integer> result) {
        if (node == null) return;
        inorderHelper(node.left, result);
        result.add(node.data);
        inorderHelper(node.right, result);
    }
    
    public static List<Integer> postorderTraversal(BinaryTreeNode<Integer> root) {

        return result;

    private static void postorderHelper(BinaryTreeNode<Integer> node, List<Integer> result) {
        if (node == null) return;
        postorderHelper(node.left, result);
        postorderHelper(node.right, result);
        result.add(node.data);
    }
    
    public static List<Integer> levelOrderTraversal(BinaryTreeNode<Integer> root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) return result;
        
        Queue<BinaryTreeNode<Integer>> queue = new LinkedList<>();
        queue.offer(root);
        
        while (!queue.isEmpty()) {
            BinaryTreeNode<Integer> node = queue.poll();
            result.add(node.data);
            
            if (node.left != null) queue.offer(node.left);
            if (node.right != null) queue.offer(node.right);
        }
        
        return result;
    }

    public static List<Integer> preorderIterative(BinaryTreeNode<Integer> root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) return result;
        
        Stack<BinaryTreeNode<Integer>> stack = new Stack<>();
        stack.push(root);
        
        while (!stack.isEmpty()) {
            BinaryTreeNode<Integer> node = stack.pop();
            result.add(node.data);
            
            if (node.right != null) stack.push(node.right);
            if (node.left != null) stack.push(node.left);
        }
        
        return result;
    }
    
    public static List<Integer> inorderIterative(BinaryTreeNode<Integer> root) {
        List<Integer> result = new ArrayList<>();
        Stack<BinaryTreeNode<Integer>> stack = new Stack<>();
        BinaryTreeNode<Integer> current = root;
        
        while (current != null || !stack.isEmpty()) {
            while (current != null) {
                stack.push(current);
                current = current.left;
            }
            
            current = stack.pop();
            result.add(current.data);
            current = current.right;
        }
        
        return result;
    }
    
    public static List<Integer> postorderIterative(BinaryTreeNode<Integer> root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) return result;
        
        Stack<BinaryTreeNode<Integer>> stack = new Stack<>();
        BinaryTreeNode<Integer> lastVisited = null;
        BinaryTreeNode<Integer> current = root;
        
        while (current != null || !stack.isEmpty()) {
            if (current != null) {
                stack.push(current);
                current = current.left;
            } else {
                BinaryTreeNode<Integer> peekNode = stack.peek();
                if (peekNode.right != null && lastVisited != peekNode.right) {
                    current = peekNode.right;
                } else {
                    result.add(peekNode.data);
                    lastVisited = stack.pop();
                }
            }
        }
        
        return result;
    }

        preorderHelperString(node.left, result);
        preorderHelperString(node.right, result);
    }
    
    public static List<String> inorderTraversalString(BinaryTreeNode<String> root) {
        List<String> result = new ArrayList<>();
        inorderHelperString(root, result);
        return result;
    }
    
    private static void inorderHelperString(BinaryTreeNode<String> node, List<String> result) {
        if (node == null) return;
        inorderHelperString(node.left, result);
        result.add(node.data);
        inorderHelperString(node.right, result);
    }
    
    public static List<String> postorderTraversalString(BinaryTreeNode<String> root) {
        List<String> result = new ArrayList<>();
        postorderHelperString(root, result);
        return result;
    }
    
    private static void postorderHelperString(BinaryTreeNode<String> node, List<String> result) {
        if (node == null) return;
        postorderHelperString(node.left, result);
        postorderHelperString(node.right, result);
        result.add(node.data);
    }
}

class BinarySearchTree<T extends Comparable<T>> {
    private BinaryTreeNode<T> root;
    
    public void insert(T data) {
        root = insertHelper(root, data);
    }
    
    private BinaryTreeNode<T> insertHelper(BinaryTreeNode<T> node, T data) {
        if (node == null) {
            return new BinaryTreeNode<>(data);
        }
        
        if (data.compareTo(node.data) < 0) {
            node.left = insertHelper(node.left, data);
        } else if (data.compareTo(node.data) > 0) {
            node.right = insertHelper(node.right, data);
        }
        
        return node;
    }
    
    public boolean search(T data) {
        return searchHelper(root, data);
    }
    
    private boolean searchHelper(BinaryTreeNode<T> node, T data) {
        if (node == null) return false;
        
        if (data.equals(node.data)) return true;
        
        if (data.compareTo(node.data) < 0) {
            return searchHelper(node.left, data);
        } else {
            return searchHelper(node.right, data);
        }
    }
    
    public void delete(T data) {
        root = deleteHelper(root, data);
    }
    
    private BinaryTreeNode<T> deleteHelper(BinaryTreeNode<T> node, T data) {
        if (node == null) return null;
        
        if (data.compareTo(node.data) < 0) {
            node.left = deleteHelper(node.left, data);
        } else if (data.compareTo(node.data) > 0) {
            node.right = deleteHelper(node.right, data);
        } else {
            // æ‰¾åˆ°è¦åˆ é™¤çš„èŠ‚ç‚¹
            if (node.left == null) return node.right;
            if (node.right == null) return node.left;
            
            // æœ‰ä¸¤ä¸ªå­èŠ‚ç‚¹ï¼Œæ‰¾åˆ°å³å­æ ‘çš„æœ€å°å€¼
            T minValue = findMinValue(node.right);
            node.data = minValue;
            node.right = deleteHelper(node.right, minValue);
        }
        
        return node;
    }
    
    private T findMinValue(BinaryTreeNode<T> node) {
        while (node.left != null) {
            node = node.left;
        }
        return node.data;
    }
    
    public T findMin() {
        if (root == null) return null;
        return findMinValue(root);
    }
    
    public T findMax() {
        if (root == null) return null;
        BinaryTreeNode<T> node = root;
        while (node.right != null) {
            node = node.right;
        }
        return node.data;
    }
    
    public int getHeight() {
        return getHeightHelper(root);
    }
    
    private int getHeightHelper(BinaryTreeNode<T> node) {
        if (node == null) return -1;
        return 1 + Math.max(getHeightHelper(node.left), getHeightHelper(node.right));
    }
    
    public List<T> inorderTraversal() {
        List<T> result = new ArrayList<>();
        inorderHelper(root, result);
        return result;
    }
    
    private void inorderHelper(BinaryTreeNode<T> node, List<T> result) {
        if (node == null) return;
        inorderHelper(node.left, result);
        result.add(node.data);
        inorderHelper(node.right, result);
    }
}

class ExpressionTree {
    
    public BinaryTreeNode<String> buildTree(String expression) {
        // ç®€åŒ–å®ç°ï¼Œå‡è®¾è¡¨è¾¾å¼æ ¼å¼æ­£ç¡®
        Stack<BinaryTreeNode<String>> stack = new Stack<>();
        
        for (char c : expression.toCharArray()) {
            if (Character.isDigit(c)) {
                stack.push(new BinaryTreeNode<>(String.valueOf(c)));
            } else if (c == '+' || c == '-' || c == '*' || c == '/') {
                BinaryTreeNode<String> node = new BinaryTreeNode<>(String.valueOf(c));
                node.right = stack.pop();
                node.left = stack.pop();
                stack.push(node);
            }
        }
        
        return stack.isEmpty() ? null : stack.pop();
    }
    
    public int evaluate(BinaryTreeNode<String> node) {
        if (node == null) return 0;
        
        if (Character.isDigit(node.data.charAt(0))) {
            return Integer.parseInt(node.data);
        }
        
        int left = evaluate(node.left);
        int right = evaluate(node.right);
        
        switch (node.data) {
            case "+": return left + right;
            case "-": return left - right;
            case "*": return left * right;
            case "/": return left / right;
            default: return 0;
        }
    }
}

class FileSystemTree {
    private Map<String, List<String>> directories;
    private Set<String> files;
    
    public FileSystemTree() {
        directories = new HashMap<>();
        files = new HashSet<>();
    }
    
    public void createDirectory(String parent, String name) {
        String path = parent.equals("/") ? "/" + name : parent + "/" + name;
        directories.putIfAbsent(parent, new ArrayList<>());
        directories.get(parent).add(name + "/");
        directories.putIfAbsent(path, new ArrayList<>());
    }
    
    public void createFile(String parent, String name) {
        directories.putIfAbsent(parent, new ArrayList<>());
        directories.get(parent).add(name);
        files.add(parent + "/" + name);
    }
    
    public void printStructure() {
        printDirectory("/", 0);
    }
    
    private void printDirectory(String path, int depth) {
        String indent = "  ".repeat(depth);
        System.out.println(indent + (depth == 0 ? "/" : path.substring(path.lastIndexOf("/") + 1)));
        
        List<String> contents = directories.get(path);
        if (contents != null) {
            for (String item : contents) {
                if (item.endsWith("/")) {
                    String subPath = path.equals("/") ? "/" + item.substring(0, item.length() - 1) 
                                                     : path + "/" + item.substring(0, item.length() - 1);
                    printDirectory(subPath, depth + 1);
                } else {
                    System.out.println("  ".repeat(depth + 1) + item);
                }
            }
        }
    }
}

class MinHeap<T extends Comparable<T>> {
    private List<T> heap;
    
    public MinHeap() {
        heap = new ArrayList<>();
    }
    
    public void insert(T item) {
        heap.add(item);
        heapifyUp(heap.size() - 1);
    }
    
    public T extractMin() {
        if (heap.isEmpty()) return null;
        
        T min = heap.get(0);
        T last = heap.remove(heap.size() - 1);
        
        if (!heap.isEmpty()) {
            heap.set(0, last);
            heapifyDown(0);
        }
        
        return min;
    }
    
    public T peek() {
        return heap.isEmpty() ? null : heap.get(0);
    }
    
    public boolean isEmpty() {
        return heap.isEmpty();
    }
    
    public List<T> getHeapArray() {
        return new ArrayList<>(heap);
    }
    
    private void heapifyUp(int index) {
        while (index > 0) {
            int parentIndex = (index - 1) / 2;
            if (heap.get(index).compareTo(heap.get(parentIndex)) >= 0) {
                break;
            }
            
            Collections.swap(heap, index, parentIndex);
            index = parentIndex;
        }
    }
    
    private void heapifyDown(int index) {
        while (true) {
            int leftChild = 2 * index + 1;
            int rightChild = 2 * index + 2;
            int smallest = index;
            
            if (leftChild < heap.size() && 
                heap.get(leftChild).compareTo(heap.get(smallest)) < 0) {
                smallest = leftChild;
            }
            
            if (rightChild < heap.size() && 
                heap.get(rightChild).compareTo(heap.get(smallest)) < 0) {
                smallest = rightChild;
            }
            
            if (smallest == index) break;
            
            Collections.swap(heap, index, smallest);
            index = smallest;
        }
    }
}
