选择排序详解 ​

📚 概述 ​

选择排序是一种简单直观的排序算法，它的工作原理是在未排序序列中找到最小（或最大）元素，存放到排序序列的起始位置，然后再从剩余未排序元素中继续寻找最小（或最大）元素，然后放到已排序序列的末尾。选择排序的主要优点是交换次数最少，每次都能确定一个元素的最终位置，这在某些特定场景下很有价值。虽然选择排序的时间复杂度为O(n²)，但它的实现简单，理解容易，是学习排序算法的重要基础。通过深入理解选择排序，可以更好地理解排序算法的设计思想和优化策略。

🎯 学习目标 ​

  • 深入理解选择排序的基本原理和工作机制
  • 掌握选择排序的标准实现和优化版本
  • 学会分析选择排序的时间和空间复杂度
  • 理解选择排序的不稳定性特征
  • 掌握选择排序的各种变种实现
  • 学会选择排序的优化技巧
  • 了解选择排序的实际应用场景
  • 培养算法分析和比较的思维能力

    // 基础选择排序
    public static void selectionSortBasic(int[] array) {
        int n = array.length;
        System.out.println("开始选择排序...");
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            // 在未排序部分找最小值
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            
            // 交换最小值到正确位置
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
                System.out.println("第 " + (i + 1) + " 轮: 交换 " + array[i] + 
                                 " 和 " + array[minIndex] + " -> " + Arrays.toString(array));
            } else {
                System.out.println("第 " + (i + 1) + " 轮: 最小值 " + array[i] + 
                                 " 已在正确位置 -> " + Arrays.toString(array));
            }
        }
        
        System.out.println("排序完成: " + Arrays.toString(array));
    }
    
    // 带步骤的选择排序
    public static void selectionSortWithSteps(int[] array) {
        int n = array.length;
        
        for (int i = 0; i < n - 1; i++) {
            System.out.println("第 " + (i + 1) + " 轮开始:");
            System.out.println("  未排序部分: " + 
                             Arrays.toString(Arrays.copyOfRange(array, i, n)));
            
            int minIndex = i;
            int minValue = array[i];
            
            // 寻找最小值
            for (int j = i + 1; j < n; j++) {
                System.out.println("  比较 " + array[j] + " 与当前最小值 " + minValue);
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                    minValue = array[j];
                    System.out.println("    发现更小值: " + minValue);
                }
            }
            
            // 交换
            if (minIndex != i) {
                array[minIndex] = array[i];
                array[i] = minValue;
                System.out.println("  交换位置 " + i + " 和 " + minIndex);
            } else {
                System.out.println("  最小值已在正确位置");
            }
            
            System.out.println("  第 " + (i + 1) + " 轮结束: " + Arrays.toString(array));
            System.out.println();
        }
    }
    
    // 带计数的选择排序
    public static int selectionSortWithCount(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                comparisons++;
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
        
        return comparisons;
    }
    
    // 带交换计数的选择排序
    public static int selectionSortWithSwapCount(int[] array) {

        int swaps = 0;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {

                }

                array[i] = array[minIndex];
                array[minIndex] = temp;
                swaps++;
            }
        }

    }
    
    // 双向选择排序
    public static int bidirectionalSelectionSort(int[] array) {
        int left = 0;
        int right = array.length - 1;
        int comparisons = 0;
        
        while (left < right) {
            int minIndex = left;
            int maxIndex = right;
            
            // 同时找最小值和最大值
            for (int i = left; i <= right; i++) {
                comparisons++;
                if (array[i] < array[minIndex]) {
                    minIndex = i;
                }
                if (array[i] > array[maxIndex]) {
                    maxIndex = i;
                }
            }
            
            // 交换最小值到左边
            if (minIndex != left) {
                int temp = array[left];
                array[left] = array[minIndex];
                array[minIndex] = temp;
                
                // 如果最大值在left位置，更新maxIndex
                if (maxIndex == left) {
                    maxIndex = minIndex;
                }
            }
            
            // 交换最大值到右边
            if (maxIndex != right) {
                int temp = array[right];
                array[right] = array[maxIndex];
                array[maxIndex] = temp;
            }
            
            left++;
            right--;
        }
        
        return comparisons;
    }
    
    // 递归选择排序
    public static void selectionSortRecursive(int[] array, int start) {
        if (start >= array.length - 1) return;
        
        int minIndex = start;
        for (int i = start + 1; i < array.length; i++) {
            if (array[i] < array[minIndex]) {
                minIndex = i;
            }
        }
        
        if (minIndex != start) {
            int temp = array[start];
            array[start] = array[minIndex];
            array[minIndex] = temp;
        }
        
        selectionSortRecursive(array, start + 1);
    }
    
    // 稳定选择排序
    public static void stableSelectionSort(Element[] array) {
        int n = array.length;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                if (array[j].value < array[minIndex].value) {
                    minIndex = j;
                }
            }
            
            // 通过移动而不是交换来保持稳定性
            if (minIndex != i) {
                Element minElement = array[minIndex];
                for (int k = minIndex; k > i; k--) {
                    array[k] = array[k - 1];
                }
                array[i] = minElement;
            }
        }
    }
    
    // 优化的选择排序
    public static void selectionSortOptimized(int[] array) {
        int n = array.length;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            
            // 只在需要时交换
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
    }
    
    // 早期终止选择排序
    public static int selectionSortEarlyTermination(int[] array) {
        int n = array.length;
        int iterations = 0;
        
        for (int i = 0; i < n - 1; i++) {
            iterations++;
            int minIndex = i;
            boolean found = false;
            
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                    found = true;
                }
            }
            
            if (!found) {
                // 剩余部分已经排序
                break;
            }
            
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
        
        return iterations;
    }
    
    // 混合排序
    public static void hybridSelectionSort(int[] array) {
        int n = array.length;
        
        // 对于小数组使用选择排序
        if (n <= 10) {
            selectionSortOptimized(array);
            return;
        }
        
        // 对于大数组，先用选择排序处理一部分，再用插入排序
        int threshold = n / 4;
        for (int i = 0; i < threshold; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
        
        // 对剩余部分使用插入排序
        insertionSort(array, threshold);
    }
    
    // 基础计时版本
    public static void selectionSortBasicTimed(int[] array) {
        int n = array.length;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            
            int temp = array[i];
            array[i] = array[minIndex];
            array[minIndex] = temp;
        }
    }
    
    // 元素排序
    public static void selectionSortElements(Element[] elements) {
        int n = elements.length;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            
            for (int j = i + 1; j < n; j++) {
                if (elements[j].value < elements[minIndex].value) {
                    minIndex = j;
                }
            }
            
            if (minIndex != i) {
                Element temp = elements[i];
                elements[i] = elements[minIndex];
                elements[minIndex] = temp;
            }
        }
    }
    
    // 辅助方法
    
    public static void insertionSort(int[] array, int start) {
        for (int i = start + 1; i < array.length; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (j >= start && array[j] > key) {
                array[j + 1] = array[j];
                j--;
            }
            array[j + 1] = key;
        }
    }
    
    public static int[] generateSortedArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = i;
        }
        return array;
    }
    
    public static int[] generateRandomArray(int size) {
        int[] array = new int[size];
        Random random = new Random();
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(size);
        }
        return array;
    }
    
    public static int[] generateReverseArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = size - 1 - i;
        }
        return array;
    }
    
    public static void compareWithOtherSorts() {
        int size = 1000;
        int[] testData = generateRandomArray(size);
        
        // 选择排序
        int[] array1 = testData.clone();
        long startTime = System.nanoTime();
        selectionSortOptimized(array1);
        long selectionTime = System.nanoTime() - startTime;
        
        // 冒泡排序
        int[] array2 = testData.clone();
        startTime = System.nanoTime();
        bubbleSort(array2);
        long bubbleTime = System.nanoTime() - startTime;
        
        // 插入排序
        int[] array3 = testData.clone();
        startTime = System.nanoTime();
        insertionSort(array3, 0);
        long insertionTime = System.nanoTime() - startTime;
        
        System.out.println("排序算法性能比较 (" + size + " 个元素):");
        System.out.println("选择排序: " + selectionTime / 1000 + " 微秒");
        System.out.println("冒泡排序: " + bubbleTime / 1000 + " 微秒");
        System.out.println("插入排序: " + insertionTime / 1000 + " 微秒");
    }
    
    public static void bubbleSort(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
    }
