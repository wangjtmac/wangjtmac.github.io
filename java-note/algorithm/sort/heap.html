堆排序详解 ​

📚 概述 ​

堆排序是一种基于堆数据结构的高效排序算法，由J.W.J. Williams在1964年发明。它利用堆这种近似完全二叉树的特殊数据结构来进行排序，具有时间复杂度稳定为O(n log n)的优秀特性。堆排序的核心思想是将待排序数组构建成一个最大堆，然后重复执行"取出堆顶元素"和"重新调整堆"的操作。虽然堆排序在最坏情况下的性能仍然很好，但由于其不稳定性和相对复杂的实现，在实际应用中不如快速排序和归并排序常见。然而，理解堆排序对于掌握堆数据结构和优先队列的实现具有重要意义。

🎯 学习目标 ​

  • 深入理解堆数据结构的性质和操作
  • 掌握堆排序的基本原理和实现步骤
  • 学会分析堆排序的时间和空间复杂度
  • 理解堆排序的不稳定性特征
  • 掌握堆的构建和调整算法
  • 学会堆排序的优化技巧和变种
  • 了解堆在其他算法中的应用
  • 培养对树形数据结构的理解能力

    // 标准堆排序
    public static void heapSort(int[] array) {
        int n = array.length;
        
        // 构建最大堆
        buildMaxHeap(array);
        
        // 排序过程
        for (int i = n - 1; i > 0; i--) {
            // 将堆顶（最大值）与末尾元素交换
            swap(array, 0, i);
            
            // 重新调整堆（堆大小减1）
            maxHeapify(array, 0, i);
        }
    }
    
    // 带步骤的堆排序
    public static void heapSortWithSteps(int[] array) {
        int n = array.length;
        
        System.out.println("步骤1: 构建最大堆");
        buildMaxHeapWithSteps(array);
        System.out.println("最大堆: " + Arrays.toString(array));
        
        System.out.println("\n步骤2: 排序过程");
        for (int i = n - 1; i > 0; i--) {
            System.out.println("第 " + (n - i) + " 轮:");
            System.out.println("  交换堆顶 " + array[0] + " 与末尾 " + array[i]);
            
            swap(array, 0, i);
            System.out.println("  交换后: " + Arrays.toString(array));
            
            maxHeapify(array, 0, i);
            System.out.println("  调整堆后: " + Arrays.toString(array));
            System.out.println("  已排序部分: " + 
                             Arrays.toString(Arrays.copyOfRange(array, i, n)));
        }
        
        System.out.println("最终结果: " + Arrays.toString(array));
    }
    
    // 构建最大堆
    public static void buildMaxHeap(int[] array) {
        int n = array.length;
        
        // 从最后一个非叶子节点开始，向上调整
        for (int i = n / 2 - 1; i >= 0; i--) {
            maxHeapify(array, i, n);
        }
    }
    
    // 带步骤的构建最大堆
    public static void buildMaxHeapWithSteps(int[] array) {
        int n = array.length;
        
        System.out.println("  从最后一个非叶子节点开始调整:");
        for (int i = n / 2 - 1; i >= 0; i--) {
            System.out.println("  调整节点 " + i + " (值: " + array[i] + ")");
            maxHeapify(array, i, n);
            System.out.println("    调整后: " + Arrays.toString(array));
        }
    }
    
    // 最大堆调整
    public static void maxHeapify(int[] array, int i, int heapSize) {
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        int largest = i;
        
        // 找到最大值的索引
        if (left < heapSize && array[left] > array[largest]) {
            largest = left;
        }
        
        if (right < heapSize && array[right] > array[largest]) {
            largest = right;
        }
        
        // 如果最大值不是当前节点，交换并继续调整
        if (largest != i) {
            swap(array, i, largest);
            maxHeapify(array, largest, heapSize);
        }
    }
    
    // 最小堆排序（降序）
    public static void heapSortDescending(int[] array) {
        int n = array.length;
        
        // 构建最小堆
        buildMinHeap(array);
        
        // 排序过程
        for (int i = n - 1; i > 0; i--) {
            swap(array, 0, i);
            minHeapify(array, 0, i);
        }
    }
    
    // 构建最小堆
    public static void buildMinHeap(int[] array) {
        int n = array.length;
        for (int i = n / 2 - 1; i >= 0; i--) {
            minHeapify(array, i, n);
        }
    }
    
    // 最小堆调整
    public static void minHeapify(int[] array, int i, int heapSize) {
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        int smallest = i;
        
        if (left < heapSize && array[left] < array[smallest]) {
            smallest = left;
        }
        
        if (right < heapSize && array[right] < array[smallest]) {
            smallest = right;
        }
        
        if (smallest != i) {
            swap(array, i, smallest);
            minHeapify(array, smallest, heapSize);
        }
    }
    
    // 部分排序（前k个最大元素）
    public static void partialHeapSort(int[] array, int k) {
        int n = array.length;
        k = Math.min(k, n);
        
        // 构建最大堆
        buildMaxHeap(array);
        
        // 只排序前k个元素
        for (int i = n - 1; i >= n - k; i--) {
            swap(array, 0, i);
            maxHeapify(array, 0, i);
        }
    }
    
    // Top K 问题
    public static int[] findTopK(int[] array, int k) {
        // 使用最小堆来找Top K
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        
        for (int num : array) {
            if (minHeap.size() < k) {
                minHeap.offer(num);
            } else if (num > minHeap.peek()) {
                minHeap.poll();
                minHeap.offer(num);
            }
        }
        
        int[] result = new int[k];
        for (int i = k - 1; i >= 0; i--) {
            result[i] = minHeap.poll();
        }
        
        return result;
    }
    
    // 元素排序
    public static void heapSortElements(Element[] elements) {
        int n = elements.length;
        
        // 构建最大堆
        for (int i = n / 2 - 1; i >= 0; i--) {
            maxHeapifyElements(elements, i, n);
        }
        
        // 排序过程
        for (int i = n - 1; i > 0; i--) {
            Element temp = elements[0];
            elements[0] = elements[i];
            elements[i] = temp;
            
            maxHeapifyElements(elements, 0, i);
        }
    }
    
    // 元素堆调整
    public static void maxHeapifyElements(Element[] elements, int i, int heapSize) {
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        int largest = i;
        
        if (left < heapSize && elements[left].value > elements[largest].value) {
            largest = left;
        }
        
        if (right < heapSize && elements[right].value > elements[largest].value) {
            largest = right;
        }
        
        if (largest != i) {
            Element temp = elements[i];
            elements[i] = elements[largest];
            elements[largest] = temp;
            
            maxHeapifyElements(elements, largest, heapSize);
        }
    }
    
    // 辅助方法
    
    public static void swap(int[] array, int i, int j) {
        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }
    
    public static void printHeapStructure(int[] heap) {
        int n = heap.length;
        int level = 0;
        int levelNodes = 1;
        int index = 0;
        
        while (index < n) {
            System.out.print("Level " + level + ": ");
            
            for (int i = 0; i < levelNodes && index < n; i++) {
                System.out.print(heap[index] + " ");
                index++;
            }
            
            System.out.println();
            level++;
            levelNodes *= 2;
        }
    }

    // 快速排序（用于性能比较）
    public static void quickSort(int[] array, int low, int high) {
        if (low < high) {
            int pi = partition(array, low, high);
            quickSort(array, low, pi - 1);
            quickSort(array, pi + 1, high);
        }
    }
    
    private static int partition(int[] array, int low, int high) {
        int pivot = array[high];
        int i = low - 1;
        
        for (int j = low; j < high; j++) {
            if (array[j] <= pivot) {
                i++;
                swap(array, i, j);
            }
        }
        
        swap(array, i + 1, high);
        return i + 1;
    }
    
    // 归并排序（用于性能比较）
    public static void mergeSort(int[] array, int left, int right) {
        if (left < right) {
            int mid = left + (right - left) / 2;
            mergeSort(array, left, mid);
            mergeSort(array, mid + 1, right);
            merge(array, left, mid, right);
        }
    }
    
    private static void merge(int[] array, int left, int mid, int right) {
        int[] temp = new int[right - left + 1];
        int i = left, j = mid + 1, k = 0;
        
        while (i <= mid && j <= right) {
            if (array[i] <= array[j]) {
                temp[k++] = array[i++];
            } else {
                temp[k++] = array[j++];
            }
        }
        
        while (i <= mid) temp[k++] = array[i++];
        while (j <= right) temp[k++] = array[j++];
        
        System.arraycopy(temp, 0, array, left, temp.length);
    }
}

// 最大堆实现
class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;
    
    public MaxHeap(int capacity) {
        this.capacity = capacity;
        this.heap = new int[capacity];
        this.size = 0;
    }
    
    public void insert(int value) {
        if (size >= capacity) {
            throw new RuntimeException("Heap is full");
        }
        
        heap[size] = value;
        heapifyUp(size);
        size++;
    }
    
    public int extractMax() {
        if (size == 0) {
            throw new RuntimeException("Heap is empty");
        }
        
        int max = heap[0];
        heap[0] = heap[size - 1];
        size--;
        heapifyDown(0);
        
        return max;
    }
    
    public int getMax() {
        if (size == 0) {
            throw new RuntimeException("Heap is empty");
        }
        return heap[0];
    }
    
    public boolean isEmpty() {
        return size == 0;
    }
    
    public int size() {
        return size;
    }
    
    private void heapifyUp(int index) {
        while (index > 0) {
            int parentIndex = (index - 1) / 2;
            if (heap[index] <= heap[parentIndex]) {
                break;
            }
            
            swap(index, parentIndex);
            index = parentIndex;
        }
    }
    
    private void heapifyDown(int index) {
        while (true) {
            int leftChild = 2 * index + 1;
            int rightChild = 2 * index + 2;
            int largest = index;
            
            if (leftChild < size && heap[leftChild] > heap[largest]) {
                largest = leftChild;
            }
            
            if (rightChild < size && heap[rightChild] > heap[largest]) {
                largest = rightChild;
            }
            
            if (largest == index) {
                break;
            }
            
            swap(index, largest);
            index = largest;
        }
    }
    
    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }
    
    @Override
    public String toString() {
        return Arrays.toString(Arrays.copyOf(heap, size));
    }
}
