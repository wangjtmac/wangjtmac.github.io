冒泡排序详解 ​

📚 概述 ​

冒泡排序是最简单直观的排序算法之一，它通过重复遍历待排序数组，比较相邻元素并交换位置，使较大的元素像气泡一样"冒泡"到数组末尾。虽然冒泡排序的时间复杂度为O(n²)，在实际应用中效率不高，但它是理解排序算法原理的绝佳起点。冒泡排序的思想简单易懂，实现直观，是学习算法设计和分析的经典案例。通过深入理解冒泡排序，可以为学习更复杂的排序算法奠定坚实基础，同时培养算法优化的思维。

🎯 学习目标 ​

  • 深入理解冒泡排序的基本原理和工作机制
  • 掌握冒泡排序的标准实现和优化版本
  • 学会分析冒泡排序的时间和空间复杂度
  • 理解冒泡排序的稳定性特征
  • 掌握冒泡排序的各种优化技巧
  • 学会冒泡排序的变种实现
  • 了解冒泡排序的实际应用场景
  • 培养算法分析和优化的思维能力

    // 基础冒泡排序
    public static void bubbleSortBasic(int[] array) {
        int n = array.length;
        System.out.println("开始冒泡排序...");
        
        for (int i = 0; i < n - 1; i++) {
            System.out.println("\n第 " + (i + 1) + " 轮:");
            for (int j = 0; j < n - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    // 交换元素
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    System.out.println("  交换 " + array[j + 1] + " 和 " + array[j] + 
                                     ": " + Arrays.toString(array));
                }
            }
            System.out.println("  第 " + (i + 1) + " 轮结束: " + Arrays.toString(array));
        }
        
        System.out.println("排序完成: " + Arrays.toString(array));
    }
    
    // 带步骤的冒泡排序
    public static void bubbleSortWithSteps(int[] array) {
        int n = array.length;
        
        for (int i = 0; i < n - 1; i++) {
            System.out.println("第 " + (i + 1) + " 轮开始:");
            boolean swapped = false;
            
            for (int j = 0; j < n - 1 - i; j++) {
                System.out.print("  比较 " + array[j] + " 和 " + array[j + 1] + ": ");
                
                if (array[j] > array[j + 1]) {
                    // 交换
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swapped = true;
                    System.out.println("交换 -> " + Arrays.toString(array));
                } else {
                    System.out.println("不交换");
                }
            }
            
            System.out.println("  第 " + (i + 1) + " 轮结束: " + Arrays.toString(array));
            
            if (!swapped) {
                System.out.println("  没有交换发生，排序完成!");
                break;
            }
        }
    }
    
    // 基础冒泡排序（带计数）
    public static int bubbleSortBasicWithCount(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {
                comparisons++;
                if (array[j] > array[j + 1]) {
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
        
        return comparisons;
    }
    
    // 优化的冒泡排序（提前终止）
    public static int bubbleSortOptimized(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            
            for (int j = 0; j < n - 1 - i; j++) {
                comparisons++;
                if (array[j] > array[j + 1]) {
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swapped = true;
                }
            }
            
            if (!swapped) {
                break; // 如果没有交换，说明已经排序完成
            }
        }
        
        return comparisons;
    }
    
    // 边界记录优化的冒泡排序
    public static int bubbleSortBoundaryOptimized(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        while (n > 1) {
            int newN = 0;
            
            for (int i = 1; i < n; i++) {
                comparisons++;
                if (array[i - 1] > array[i]) {
                    int temp = array[i - 1];
                    array[i - 1] = array[i];
                    array[i] = temp;
                    newN = i; // 记录最后交换的位置
                }
            }
            
            n = newN; // 下次只需要检查到最后交换的位置
        }
        
        return comparisons;
    }
    
    // 鸡尾酒排序（双向冒泡）
    public static int cocktailSort(int[] array) {
        int left = 0;
        int right = array.length - 1;
        int comparisons = 0;
        
        while (left < right) {
            // 从左到右冒泡
            boolean swapped = false;
            for (int i = left; i < right; i++) {
                comparisons++;
                if (array[i] > array[i + 1]) {
                    int temp = array[i];
                    array[i] = array[i + 1];
                    array[i + 1] = temp;
                    swapped = true;
                }
            }
            right--;
            
            if (!swapped) break;
            
            // 从右到左冒泡
            swapped = false;
            for (int i = right; i > left; i--) {
                comparisons++;
                if (array[i] < array[i - 1]) {
                    int temp = array[i];
                    array[i] = array[i - 1];
                    array[i - 1] = temp;
                    swapped = true;
                }
            }
            left++;
            
            if (!swapped) break;
        }
        
        return comparisons;
    }
    
    // 递归冒泡排序
    public static void bubbleSortRecursive(int[] array, int n) {
        if (n == 1) return;
        
        // 一轮冒泡，将最大元素移到末尾
        for (int i = 0; i < n - 1; i++) {
            if (array[i] > array[i + 1]) {
                int temp = array[i];
                array[i] = array[i + 1];
                array[i + 1] = temp;
            }
        }
        
        // 递归排序前n-1个元素
        bubbleSortRecursive(array, n - 1);
    }
    
    // 奇偶排序
    public static void oddEvenSort(int[] array) {
        int n = array.length;
        boolean sorted = false;
        
        while (!sorted) {
            sorted = true;
            
            // 奇数位置的比较
            for (int i = 1; i < n - 1; i += 2) {
                if (array[i] > array[i + 1]) {
                    int temp = array[i];
                    array[i] = array[i + 1];
                    array[i + 1] = temp;
                    sorted = false;
                }
            }
            
            // 偶数位置的比较
            for (int i = 0; i < n - 1; i += 2) {
                if (array[i] > array[i + 1]) {
                    int temp = array[i];
                    array[i] = array[i + 1];
                    array[i + 1] = temp;
                    sorted = false;
                }
            }
        }
    }
    
    // 梳排序
    public static void combSort(int[] array) {
        int n = array.length;
        int gap = n;
        boolean swapped = true;
        
        while (gap != 1 || swapped) {
            gap = getNextGap(gap);
            swapped = false;
            
            for (int i = 0; i < n - gap; i++) {
                if (array[i] > array[i + gap]) {
                    int temp = array[i];
                    array[i] = array[i + gap];
                    array[i + gap] = temp;
                    swapped = true;
                }
            }
        }
    }
    
    private static int getNextGap(int gap) {
        gap = (gap * 10) / 13;
        return gap < 1 ? 1 : gap;
    }
    
    // 元素排序（稳定性测试）
    public static void bubbleSortElements(Element[] elements) {
        int n = elements.length;
        
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {
                if (elements[j].value > elements[j + 1].value) {
                    Element temp = elements[j];
                    elements[j] = elements[j + 1];
                    elements[j + 1] = temp;
                }
            }
        }
    }
    
    // 辅助方法
    
    public static int[] generateSortedArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = i;
        }
        return array;
    }
    
    public static int[] generateRandomArray(int size) {
        int[] array = new int[size];
        Random random = new Random();
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(size);
        }
        return array;
    }
    
    public static int[] generateReverseSortedArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = size - 1 - i;
        }
        return array;
    }
    
    public static void performanceComparison() {
        int[] sizes = {100, 500, 1000};
        
        System.out.printf("%-8s %-12s %-12s %-12s %-12s%n", 
                         "大小", "基础冒泡", "优化冒泡", "鸡尾酒", "梳排序");
        System.out.println("----------------------------------------------------");
        
        for (int size : sizes) {
            int[] testData = generateRandomArray(size);
            
            // 基础冒泡排序
            int[] array1 = testData.clone();
            long startTime = System.nanoTime();
            bubbleSortBasicWithCount(array1);
            long basicTime = (System.nanoTime() - startTime) / 1000;
            
            // 优化冒泡排序
            int[] array2 = testData.clone();
            startTime = System.nanoTime();
            bubbleSortOptimized(array2);
            long optimizedTime = (System.nanoTime() - startTime) / 1000;
            
            // 鸡尾酒排序
            int[] array3 = testData.clone();
            startTime = System.nanoTime();
            cocktailSort(array3);
            long cocktailTime = (System.nanoTime() - startTime) / 1000;
            
            // 梳排序
            int[] array4 = testData.clone();
            startTime = System.nanoTime();
            combSort(array4);
            long combTime = (System.nanoTime() - startTime) / 1000;
            
            System.out.printf("%-8d %-12d %-12d %-12d %-12d%n", 
                             size, basicTime, optimizedTime, cocktailTime, combTime);
        }
    }
}

    public Element(int value, String label) {
        this.value = value;
        this.label = label;
    }
    
    @Override
    public String toString() {
        return value + label;
    }
