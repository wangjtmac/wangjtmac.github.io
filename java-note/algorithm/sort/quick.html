快速排序详解 ​

📚 概述 ​

快速排序是最重要的排序算法之一，由英国计算机科学家Tony Hoare在1960年提出。它采用分治策略，通过选择一个基准元素将数组分为两部分，使得左部分的所有元素都小于等于基准，右部分的所有元素都大于等于基准，然后递归地对两部分进行排序。快速排序的平均时间复杂度为O(n log n)，在实际应用中通常比其他O(n log n)算法更快，因此被广泛使用。虽然最坏情况下时间复杂度为O(n²)，但通过合理的优化可以有效避免。理解快速排序的原理、实现和优化对于掌握算法设计思想具有重要意义。

🎯 学习目标 ​

  • 深入理解快速排序的分治思想
  • 掌握快速排序的基本实现和优化版本
  • 理解分区操作的核心原理
  • 学会分析快速排序的时间和空间复杂度
  • 掌握快速排序的各种优化技巧
  • 了解快速排序的变种和应用
  • 学会处理快速排序的边界情况
  • 理解快速排序在实际开发中的应用

        /*
         * 快速排序算法步骤：
         * 1. 选择基准元素 (pivot)
         * 2. 分区操作 (partition) - 重排数组
         * 3. 递归排序左右子数组
         * 
         * 特点：
         * - 分治算法
         * - 原地排序
         * - 不稳定排序
         * - 平均 O(n log n)，最坏 O(n²)

        System.out.println("3. 不稳定排序 - 相等元素可能改变相对位置");
        System.out.println("4. 实际性能优秀 - 常数因子小");
        
        // 基础快速排序演示

    }

    // 算法实现
    
    // 基础快速排序
    public static void quickSortBasic(int[] array, int low, int high) {
        if (low < high) {
            int pivotIndex = partition(array, low, high);
            quickSortBasic(array, low, pivotIndex - 1);
            quickSortBasic(array, pivotIndex + 1, high);
        }
    }
    
    // 基础分区操作
    public static int partition(int[] array, int low, int high) {
        int pivot = array[high]; // 选择最后一个元素作为基准
        int i = low - 1; // 小于基准的元素的索引
        
        for (int j = low; j < high; j++) {
            if (array[j] <= pivot) {
                i++;
                swap(array, i, j);
            }
        }
        
        swap(array, i + 1, high);
        return i + 1;
    }
    
    // 带步骤的快速排序
    public static void quickSortWithSteps(int[] array, int low, int high, int depth) {
        if (low < high) {
            String indent = "  ".repeat(depth);
            System.out.println(indent + "排序范围 [" + low + ", " + high + "]: " + 
                             Arrays.toString(Arrays.copyOfRange(array, low, high + 1)));
            
            int pivotIndex = partition(array, low, high);
            System.out.println(indent + "分区完成，基准位置: " + pivotIndex + 
                             ", 基准值: " + array[pivotIndex]);
            
            quickSortWithSteps(array, low, pivotIndex - 1, depth + 1);
            quickSortWithSteps(array, pivotIndex + 1, high, depth + 1);
        }
    }
    
    // 详细分区过程
    public static int partitionWithSteps(int[] array, int low, int high) {
        int pivot = array[high];
        int i = low - 1;
        
        System.out.println("分区过程:");
        System.out.println("基准值: " + pivot);
        
        for (int j = low; j < high; j++) {
            System.out.print("比较 " + array[j] + " 与 " + pivot + ": ");
            if (array[j] <= pivot) {
                i++;
                swap(array, i, j);
                System.out.println("≤ 基准，交换到位置 " + i);
            } else {
                System.out.println("> 基准，不交换");
            }
            System.out.println("当前数组: " + Arrays.toString(array));
        }
        
        swap(array, i + 1, high);
        System.out.println("基准归位: " + Arrays.toString(array));
        
        return i + 1;
    }
    
    // 三数取中优化
    public static void quickSortMedianOfThree(int[] array, int low, int high) {
        if (low < high) {
            // 三数取中选择基准
            int mid = low + (high - low) / 2;
            if (array[mid] < array[low]) swap(array, low, mid);
            if (array[high] < array[low]) swap(array, low, high);
            if (array[high] < array[mid]) swap(array, mid, high);
            swap(array, mid, high); // 将中位数放到末尾
            
            int pivotIndex = partition(array, low, high);
            quickSortMedianOfThree(array, low, pivotIndex - 1);
            quickSortMedianOfThree(array, pivotIndex + 1, high);
        }
    }
    
    // 三路快排
    public static void quickSort3Way(int[] array, int low, int high) {
        if (low >= high) return;
        
        int pivot = array[low];
        int lt = low;      // array[low..lt-1] < pivot
        int gt = high;     // array[gt+1..high] > pivot
        int i = low + 1;   // array[lt..i-1] == pivot
        
        while (i <= gt) {
            if (array[i] < pivot) {
                swap(array, lt++, i++);
            } else if (array[i] > pivot) {
                swap(array, i, gt--);
            } else {
                i++;
            }
        }
        
        quickSort3Way(array, low, lt - 1);
        quickSort3Way(array, gt + 1, high);
    }
    
    // 混合排序
    public static void hybridQuickSort(int[] array, int low, int high) {
        if (high - low < 10) {
            // 小数组使用插入排序
            insertionSort(array, low, high);
        } else {
            int pivotIndex = partition(array, low, high);
            hybridQuickSort(array, low, pivotIndex - 1);
            hybridQuickSort(array, pivotIndex + 1, high);
        }
    }
    
    // 迭代版本快排
    public static void quickSortIterative(int[] array) {
        Stack<Integer> stack = new Stack<>();
        stack.push(0);
        stack.push(array.length - 1);
        
        while (!stack.isEmpty()) {
            int high = stack.pop();
            int low = stack.pop();
            
            if (low < high) {
                int pivotIndex = partition(array, low, high);
                
                // 将子数组范围压入栈
                stack.push(low);
                stack.push(pivotIndex - 1);
                stack.push(pivotIndex + 1);
                stack.push(high);
            }
        }
    }
    
    // 辅助方法
    
    public static void swap(int[] array, int i, int j) {
        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }
    
    public static void insertionSort(int[] array, int low, int high) {
        for (int i = low + 1; i <= high; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (j >= low && array[j] > key) {
                array[j + 1] = array[j];
                j--;
            }
            array[j + 1] = key;
        }
    }
    
    public static int[] generateRandomArray(int size) {
        Random random = new Random();
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(100);
        }
        return array;
    }
    
    public static void analyzeQuickSortPerformance(int[] array, String caseType) {
        long startTime = System.nanoTime();
        quickSortBasic(array, 0, array.length - 1);
        long endTime = System.nanoTime();
        
        System.out.println(caseType + " 耗时: " + (endTime - startTime) + " 纳秒");
    }
    
    public static void comparePartitionStrategies() {
        int[] testArray = {64, 34, 25, 12, 22, 11, 90};
        
        System.out.println("原始数组: " + Arrays.toString(testArray));
        
        // 策略1: 选择第一个元素
        int[] array1 = testArray.clone();
        int pivot1 = partitionFirst(array1, 0, array1.length - 1);
        System.out.println("选择第一个元素作基准: " + Arrays.toString(array1) + 
                         ", 基准位置: " + pivot1);
        
        // 策略2: 选择最后一个元素
        int[] array2 = testArray.clone();
        int pivot2 = partition(array2, 0, array2.length - 1);
        System.out.println("选择最后一个元素作基准: " + Arrays.toString(array2) + 
                         ", 基准位置: " + pivot2);
        
        // 策略3: 选择中间元素
        int[] array3 = testArray.clone();
        int pivot3 = partitionMiddle(array3, 0, array3.length - 1);
        System.out.println("选择中间元素作基准: " + Arrays.toString(array3) + 
                         ", 基准位置: " + pivot3);
    }
    
    public static int partitionFirst(int[] array, int low, int high) {
        int pivot = array[low];
        int i = low + 1;
        int j = high;
        
        while (true) {
            while (i <= j && array[i] <= pivot) i++;
            while (i <= j && array[j] >= pivot) j--;
            
            if (i > j) break;
            swap(array, i, j);
        }
        
        swap(array, low, j);
        return j;
    }
    
    public static int partitionMiddle(int[] array, int low, int high) {
        int mid = low + (high - low) / 2;
        swap(array, mid, high); // 将中间元素移到末尾
        return partition(array, low, high);
    }
    
    public static void performanceTest() {
        int[] sizes = {1000, 5000, 10000, 20000};
        
        System.out.printf("%-8s %-12s %-12s %-12s%n", "大小", "基础快排", "优化快排", "性能提升");
        System.out.println("--------------------------------------------");
        
        for (int size : sizes) {
            // 基础快排
            int[] array1 = generateRandomArray(size);
            long startTime = System.nanoTime();
            quickSortBasic(array1, 0, array1.length - 1);
            long basicTime = System.nanoTime() - startTime;
            
            // 优化快排
            int[] array2 = generateRandomArray(size);
            startTime = System.nanoTime();
            hybridQuickSort(array2, 0, array2.length - 1);
            long optimizedTime = System.nanoTime() - startTime;
            
            double improvement = (double) basicTime / optimizedTime;
            
            System.out.printf("%-8d %-12d %-12d %-12.2f%n", 
                             size, basicTime / 1000, optimizedTime / 1000, improvement);
        }
    }
