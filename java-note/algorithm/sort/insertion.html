插入排序详解 ​

📚 概述 ​

插入排序是一种简单直观的排序算法，它的工作原理类似于我们手中整理扑克牌的过程。算法将数组分为已排序和未排序两部分，每次从未排序部分取出一个元素，在已排序部分找到合适的位置插入。插入排序虽然时间复杂度为O(n²)，但它具有很多优秀的特性：稳定性、原地排序、对小数据集效率高、对部分有序数据有很好的性能。在实际应用中，插入排序常被用作其他高级排序算法的子程序，特别是在处理小规模数据时。理解插入排序对于掌握排序算法的本质和优化思想具有重要意义。

🎯 学习目标 ​

  • 深入理解插入排序的基本原理和工作机制
  • 掌握插入排序的标准实现和优化版本
  • 学会分析插入排序的时间和空间复杂度
  • 理解插入排序的稳定性和自适应性特征
  • 掌握插入排序的各种变种实现
  • 学会插入排序的优化技巧和应用场景
  • 了解插入排序在混合算法中的作用
  • 培养算法分析和实际应用的思维能力

    // 基础插入排序
    public static void insertionSortBasic(int[] array) {
        int n = array.length;
        System.out.println("开始插入排序...");
        
        for (int i = 1; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            System.out.println("\n第 " + i + " 轮: 插入元素 " + key);
            System.out.println("已排序部分: " + 
                             Arrays.toString(Arrays.copyOfRange(array, 0, i)));
            
            // 向后移动较大的元素
            while (j >= 0 && array[j] > key) {
                array[j + 1] = array[j];
                j--;
                System.out.println("  移动 " + array[j + 2] + " 到位置 " + (j + 2));
            }
            
            array[j + 1] = key;
            System.out.println("  插入 " + key + " 到位置 " + (j + 1));
            System.out.println("  当前状态: " + Arrays.toString(array));
        }
        
        System.out.println("排序完成: " + Arrays.toString(array));
    }
    
    // 带步骤的插入排序
    public static void insertionSortWithSteps(int[] array) {
        int n = array.length;
        
        for (int i = 1; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            System.out.println("步骤 " + i + ": 处理元素 " + key);
            System.out.print("  寻找插入位置: ");
            
            while (j >= 0 && array[j] > key) {
                System.out.print(array[j] + ">" + key + " ");
                array[j + 1] = array[j];
                j--;
            }
            
            array[j + 1] = key;
            System.out.println();
            System.out.println("  插入后: " + Arrays.toString(array));
        }
    }
    
    // 带计数的插入排序
    public static int insertionSortWithCount(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        for (int i = 1; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (j >= 0) {
                comparisons++;
                if (array[j] > key) {
                    array[j + 1] = array[j];
                    j--;
                } else {
                    break;
                }
            }
            
            array[j + 1] = key;
        }
        
        return comparisons;
    }
    
    // 二分查找插入排序
    public static int binaryInsertionSort(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        for (int i = 1; i < n; i++) {
            int key = array[i];
            
            // 使用二分查找找到插入位置
            int left = 0, right = i;
            while (left < right) {
                comparisons++;
                int mid = left + (right - left) / 2;
                if (array[mid] > key) {
                    right = mid;
                } else {
                    left = mid + 1;
                }
            }
            
            // 移动元素
            for (int j = i; j > left; j--) {
                array[j] = array[j - 1];
            }
            
            array[left] = key;
        }
        
        return comparisons;
    }
    
    // 哨兵优化插入排序
    public static void insertionSortWithSentinel(int[] array) {
        int n = array.length;
        
        // 找到最小值作为哨兵
        int minIndex = 0;
        for (int i = 1; i < n; i++) {
            if (array[i] < array[minIndex]) {
                minIndex = i;
            }
        }
        
        // 将最小值移到第一位作为哨兵
        if (minIndex != 0) {
            int temp = array[0];
            array[0] = array[minIndex];
            array[minIndex] = temp;
        }
        
        // 插入排序（无需边界检查）
        for (int i = 2; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (array[j] > key) {
                array[j + 1] = array[j];
                j--;
            }
            
            array[j + 1] = key;
        }
    }
    
    // 希尔排序
    public static void shellSort(int[] array) {
        int n = array.length;
        
        // 使用Knuth序列: 1, 4, 13, 40, 121, ...
        int gap = 1;
        while (gap < n / 3) {
            gap = gap * 3 + 1;
        }
        
        while (gap >= 1) {
            // 对每个子序列进行插入排序
            for (int i = gap; i < n; i++) {
                int key = array[i];
                int j = i;
                
                while (j >= gap && array[j - gap] > key) {
                    array[j] = array[j - gap];
                    j -= gap;
                }
                
                array[j] = key;
            }
            
            gap /= 3;
        }
    }
    
    // 递归插入排序
    public static void insertionSortRecursive(int[] array, int n) {
        if (n <= 1) return;
        
        // 递归排序前n-1个元素
        insertionSortRecursive(array, n - 1);
        
        // 插入第n个元素
        int key = array[n - 1];
        int j = n - 2;
        
        while (j >= 0 && array[j] > key) {
            array[j + 1] = array[j];
            j--;
        }
        
        array[j + 1] = key;
    }
    
    // 链表插入排序
    public static ListNode insertionSortLinkedList(ListNode head) {
        if (head == null || head.next == null) return head;
        
        ListNode dummy = new ListNode(0);
        ListNode current = head;
        
        while (current != null) {
            ListNode next = current.next;
            
            // 在已排序部分找到插入位置
            ListNode prev = dummy;
            while (prev.next != null && prev.next.val < current.val) {
                prev = prev.next;
            }
            
            // 插入节点
            current.next = prev.next;
            prev.next = current;
            
            current = next;
        }
        
        return dummy.next;
    }
    
    // 多关键字排序
    public static void insertionSortStudents(Student[] students) {
        int n = students.length;
        
        for (int i = 1; i < n; i++) {
            Student key = students[i];
            int j = i - 1;
            
            while (j >= 0 && compareStudents(students[j], key) > 0) {
                students[j + 1] = students[j];
                j--;
            }
            
            students[j + 1] = key;
        }
    }
    
    private static int compareStudents(Student a, Student b) {
        // 首先按成绩降序
        if (a.score != b.score) {
            return b.score - a.score;
        }
        // 成绩相同则按年龄升序
        return a.age - b.age;
    }
    
    // 混合排序
    public static void hybridSort(int[] array) {
        hybridSortHelper(array, 0, array.length - 1);
    }
    
    private static void hybridSortHelper(int[] array, int left, int right) {
        if (right - left <= 10) {
            // 小数组使用插入排序
            insertionSortRange(array, left, right);
        } else {
            // 大数组使用快速排序
            int pivot = partition(array, left, right);
            hybridSortHelper(array, left, pivot - 1);
            hybridSortHelper(array, pivot + 1, right);
        }
    }
    
    private static void insertionSortRange(int[] array, int left, int right) {
        for (int i = left + 1; i <= right; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (j >= left && array[j] > key) {
                array[j + 1] = array[j];
                j--;
            }
            
            array[j + 1] = key;
        }
    }
    
    private static int partition(int[] array, int left, int right) {
        int pivot = array[right];
        int i = left - 1;
        
        for (int j = left; j < right; j++) {
            if (array[j] <= pivot) {
                i++;
                int temp = array[i];
                array[i] = array[j];
                array[j] = temp;
            }
        }
        
        int temp = array[i + 1];
        array[i + 1] = array[right];
        array[right] = temp;
        
        return i + 1;
    }
    
    // 元素排序
    public static void insertionSortElements(Element[] elements) {
        int n = elements.length;
        
        for (int i = 1; i < n; i++) {
            Element key = elements[i];
            int j = i - 1;
            
            while (j >= 0 && elements[j].value > key.value) {
                elements[j + 1] = elements[j];
                j--;
            }
            
            elements[j + 1] = key;
        }
    }

        System.out.println("自适应性测试:");
        
        // 完全有序
        int[] sorted = {1, 2, 3, 4, 5};
        System.out.println("完全有序: " + Arrays.toString(sorted));
        int comp1 = insertionSortWithCount(sorted.clone());
        System.out.println("比较次数: " + comp1);
        
        // 部分有序
        int[] partialSorted = {1, 2, 5, 3, 4};
        System.out.println("部分有序: " + Arrays.toString(partialSorted));
        int comp2 = insertionSortWithCount(partialSorted.clone());
        System.out.println("比较次数: " + comp2);
        
        // 完全逆序
        int[] reverse = {5, 4, 3, 2, 1};
        System.out.println("完全逆序: " + Arrays.toString(reverse));
        int comp3 = insertionSortWithCount(reverse.clone());
        System.out.println("比较次数: " + comp3);
    }
    
    public static int[] generateSortedArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = i;
        }
        return array;
    }
    
    public static int[] generateRandomArray(int size) {
        int[] array = new int[size];
        Random random = new Random();
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(size);
        }
        return array;
    }
    
    public static int[] generateReverseArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = size - 1 - i;
        }
        return array;
    }
    
    public static int[] generatePartiallyOrderedArray(int size, double orderRatio) {
        int[] array = generateSortedArray(size);
        Random random = new Random();
        
        int swaps = (int) ((1 - orderRatio) * size / 2);
        for (int i = 0; i < swaps; i++) {
            int idx1 = random.nextInt(size);
            int idx2 = random.nextInt(size);
            int temp = array[idx1];
            array[idx1] = array[idx2];
            array[idx2] = temp;
        }
        
        return array;
    }
    
    public static void compareWithOtherSorts() {
        int size = 1000;
        int[] testData = generateRandomArray(size);
        
        // 插入排序
        int[] array1 = testData.clone();
        long startTime = System.nanoTime();
        insertionSortWithCount(array1);
        long insertionTime = System.nanoTime() - startTime;
        
        // 选择排序
        int[] array2 = testData.clone();
        startTime = System.nanoTime();
        selectionSort(array2);
        long selectionTime = System.nanoTime() - startTime;
        
        // 冒泡排序
        int[] array3 = testData.clone();
        startTime = System.nanoTime();
        bubbleSort(array3);
        long bubbleTime = System.nanoTime() - startTime;
        
        System.out.println("排序算法性能比较 (" + size + " 个元素):");
        System.out.println("插入排序: " + insertionTime / 1000 + " 微秒");
        System.out.println("选择排序: " + selectionTime / 1000 + " 微秒");
        System.out.println("冒泡排序: " + bubbleTime / 1000 + " 微秒");
    }
    
    public static void selectionSort(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
    }
    
    public static void bubbleSort(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
    }
    
    // 链表相关方法
    public static ListNode createLinkedList(int[] values) {
        ListNode dummy = new ListNode(0);
        ListNode current = dummy;
        for (int value : values) {
            current.next = new ListNode(value);
            current = current.next;
        }
        return dummy.next;
    }
    
    public static String printLinkedList(ListNode head) {
        StringBuilder sb = new StringBuilder();
        while (head != null) {
            sb.append(head.val);
            if (head.next != null) sb.append(" -> ");
            head = head.next;
        }
        return sb.toString();
    }
