排序算法概述 ​

📚 概述 ​

排序算法是计算机科学中最基础也是最重要的算法之一，它将一组数据按照特定的顺序进行排列。排序算法不仅在实际开发中应用广泛，也是算法学习和面试的重点内容。不同的排序算法有着不同的时间复杂度、空间复杂度和稳定性特点，适用于不同的场景。掌握各种排序算法的原理、实现和特点，能够帮助我们在实际问题中选择最合适的排序方法。本章将系统介绍各种经典的排序算法，包括简单排序、高效排序和特殊排序，通过详细的代码实现和性能分析，让您全面掌握排序算法的精髓。

🎯 学习目标 ​

  • 理解排序算法的基本概念和分类
  • 掌握各种排序算法的实现原理
  • 学会分析排序算法的时间和空间复杂度
  • 理解算法稳定性的概念和重要性
  • 掌握排序算法的优化技巧
  • 学会根据场景选择合适的排序算法
  • 了解排序算法在实际开发中的应用
  • 熟悉排序算法的面试常见问题

    // 选择排序 (原地、不稳定)
    public static void selectionSort(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
    }
    
    // 插入排序 (原地、稳定、自适应)
    public static void insertionSort(int[] array) {
        int n = array.length;
        for (int i = 1; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (j >= 0 && array[j] > key) {
                array[j + 1] = array[j];
                j--;
            }
            array[j + 1] = key;
        }
    }
    
    // 归并排序 (非原地、稳定)
    public static int[] mergeSort(int[] array) {
        if (array.length <= 1) {
            return array;
        }
        
        int mid = array.length / 2;
        int[] left = Arrays.copyOfRange(array, 0, mid);
        int[] right = Arrays.copyOfRange(array, mid, array.length);
        
        left = mergeSort(left);
        right = mergeSort(right);
        
        return merge(left, right);
    }
    
    private static int[] merge(int[] left, int[] right) {
        int[] result = new int[left.length + right.length];
        int i = 0, j = 0, k = 0;
        
        while (i < left.length && j < right.length) {
            if (left[i] <= right[j]) {
                result[k++] = left[i++];
            } else {
                result[k++] = right[j++];
            }
        }
        
        while (i < left.length) {
            result[k++] = left[i++];
        }
        
        while (j < right.length) {
            result[k++] = right[j++];
        }
        
        return result;
    }

    // 快速排序实现
    public static void quickSortWithSteps(int[] array, int low, int high, int depth) {
        if (low < high) {
            String indent = "  ".repeat(depth);
            System.out.println(indent + "排序范围 [" + low + ", " + high + "]: " +
                             Arrays.toString(Arrays.copyOfRange(array, low, high + 1)));

            int pivotIndex = partition(array, low, high);
            System.out.println(indent + "基准位置: " + pivotIndex + ", 基准值: " + array[pivotIndex]);

            quickSortWithSteps(array, low, pivotIndex - 1, depth + 1);
            quickSortWithSteps(array, pivotIndex + 1, high, depth + 1);
        }
    }

    public static void quickSort(int[] array, int low, int high) {
        if (low < high) {
            int pivotIndex = partition(array, low, high);
            quickSort(array, low, pivotIndex - 1);
            quickSort(array, pivotIndex + 1, high);
        }
    }

    private static int partition(int[] array, int low, int high) {
        int pivot = array[high];
        int i = low - 1;

        for (int j = low; j < high; j++) {
            if (array[j] <= pivot) {
                i++;
                swap(array, i, j);
            }
        }

        swap(array, i + 1, high);
        return i + 1;
    }

        if (low < high) {
            int[] result = partitionWithCount(array, low, high);
            int pivotIndex = result[0];
            comparisons = result[1];

            comparisons += quickSortFirstPivot(array, low, pivotIndex - 1);
            comparisons += quickSortFirstPivot(array, pivotIndex + 1, high);
        }
        return comparisons;
    }

    public static int quickSortRandomPivot(int[] array, int low, int high) {
        int comparisons = 0;
        if (low < high) {
            // 随机选择基准
            Random random = new Random();
            int randomIndex = low + random.nextInt(high - low + 1);
            swap(array, randomIndex, high);

            int[] result = partitionWithCount(array, low, high);
            int pivotIndex = result[0];
            comparisons = result[1];

            comparisons += quickSortRandomPivot(array, low, pivotIndex - 1);
            comparisons += quickSortRandomPivot(array, pivotIndex + 1, high);
        }
        return comparisons;
    }

    public static int quickSortMedianOfThree(int[] array, int low, int high) {
        int comparisons = 0;
        if (low < high) {
            // 三数取中
            int mid = low + (high - low) / 2;
            if (array[mid] < array[low]) swap(array, low, mid);
            if (array[high] < array[low]) swap(array, low, high);
            if (array[high] < array[mid]) swap(array, mid, high);
            swap(array, mid, high);

            int[] result = partitionWithCount(array, low, high);
            int pivotIndex = result[0];
            comparisons = result[1];

            comparisons += quickSortMedianOfThree(array, low, pivotIndex - 1);
            comparisons += quickSortMedianOfThree(array, pivotIndex + 1, high);
        }
        return comparisons;
    }

    private static int[] partitionWithCount(int[] array, int low, int high) {
        int pivot = array[high];
        int i = low - 1;
        int comparisons = 0;

        for (int j = low; j < high; j++) {
            comparisons++;
            if (array[j] <= pivot) {
                i++;
                swap(array, i, j);
            }
        }

        swap(array, i + 1, high);
        return new int[]{i + 1, comparisons};
    }

    // 归并排序实现
    public static void mergeSortWithSteps(int[] array, int left, int right, int depth) {
        if (left < right) {
            String indent = "  ".repeat(depth);
            System.out.println(indent + "分割 [" + left + ", " + right + "]: " +
                             Arrays.toString(Arrays.copyOfRange(array, left, right + 1)));

            int mid = left + (right - left) / 2;

            mergeSortWithSteps(array, left, mid, depth + 1);
            mergeSortWithSteps(array, mid + 1, right, depth + 1);

            merge(array, left, mid, right);
            System.out.println(indent + "合并结果: " +
                             Arrays.toString(Arrays.copyOfRange(array, left, right + 1)));
        }
    }

    public static void mergeSort(int[] array, int left, int right) {
        if (left < right) {
            int mid = left + (right - left) / 2;

            mergeSort(array, left, mid);
            mergeSort(array, mid + 1, right);

            merge(array, left, mid, right);
        }
    }

    private static void merge(int[] array, int left, int mid, int right) {
        int[] temp = new int[right - left + 1];
        int i = left, j = mid + 1, k = 0;

        while (i <= mid && j <= right) {
            if (array[i] <= array[j]) {
                temp[k++] = array[i++];
            } else {
                temp[k++] = array[j++];
            }
        }

        while (i <= mid) temp[k++] = array[i++];
        while (j <= right) temp[k++] = array[j++];

        System.arraycopy(temp, 0, array, left, temp.length);
    }

    // 学生归并排序 (稳定性演示)
    public static void mergeSortStudents(Student[] array, int left, int right) {
        if (left < right) {
            int mid = left + (right - left) / 2;

            mergeSortStudents(array, left, mid);
            mergeSortStudents(array, mid + 1, right);

            mergeStudents(array, left, mid, right);
        }
    }

    private static void mergeStudents(Student[] array, int left, int mid, int right) {
        Student[] temp = new Student[right - left + 1];
        int i = left, j = mid + 1, k = 0;

        while (i <= mid && j <= right) {
            if (array[i].getScore() <= array[j].getScore()) {
                temp[k++] = array[i++];
            } else {
                temp[k++] = array[j++];
            }
        }

        while (i <= mid) temp[k++] = array[i++];
        while (j <= right) temp[k++] = array[j++];

        System.arraycopy(temp, 0, array, left, temp.length);
    }

    // 堆排序实现
    public static void heapSortWithSteps(int[] array) {
        int n = array.length;

        // 构建最大堆
        System.out.println("构建最大堆:");
        for (int i = n / 2 - 1; i >= 0; i--) {
            heapify(array, n, i);
            System.out.println("调整节点 " + i + ": " + Arrays.toString(array));
        }

        // 排序过程
        System.out.println("\n排序过程:");
        for (int i = n - 1; i > 0; i--) {
            swap(array, 0, i);
            System.out.println("交换堆顶和位置 " + i + ": " + Arrays.toString(array));

            heapify(array, i, 0);
            System.out.println("重新调整堆: " + Arrays.toString(array));
        }
    }

    public static void heapSort(int[] array) {
        int n = array.length;

        buildMaxHeap(array);

        for (int i = n - 1; i > 0; i--) {
            swap(array, 0, i);
            heapify(array, i, 0);
        }
    }

    public static void buildMaxHeap(int[] array) {
        int n = array.length;
        for (int i = n / 2 - 1; i >= 0; i--) {
            heapify(array, n, i);
        }
    }

    private static void heapify(int[] array, int n, int i) {
        int largest = i;
        int left = 2 * i + 1;
        int right = 2 * i + 2;

        if (left < n && array[left] > array[largest]) {
            largest = left;
        }

        if (right < n && array[right] > array[largest]) {
            largest = right;
        }

        if (largest != i) {
            swap(array, i, largest);
            heapify(array, n, largest);
        }
    }

    // 辅助方法
    private static void swap(int[] array, int i, int j) {
        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }

        }
    }

    public static int[] countingSortWithSteps(int[] array) {
        int max = Arrays.stream(array).max().orElse(0);
        int min = Arrays.stream(array).min().orElse(0);
        int range = max - min + 1;

        // 计数数组
        int[] count = new int[range];
        int[] output = new int[array.length];

        // 统计每个元素出现次数
        System.out.println("1. 统计元素出现次数:");
        for (int num : array) {
            count[num - min]++;
        }
        System.out.println("计数数组: " + Arrays.toString(count));

        // 计算累积计数
        System.out.println("2. 计算累积计数:");
        for (int i = 1; i < range; i++) {
            count[i] += count[i - 1];
        }
        System.out.println("累积计数: " + Arrays.toString(count));

        // 构建输出数组
        System.out.println("3. 构建输出数组:");
        for (int i = array.length - 1; i >= 0; i--) {
            output[count[array[i] - min] - 1] = array[i];
            count[array[i] - min]--;
            System.out.println("放置 " + array[i] + " 到位置 " + count[array[i] - min] +
                             ": " + Arrays.toString(output));
        }

        return output;
    }

    public static int[] countingSort(int[] array, int max) {
        int[] count = new int[max + 1];
        int[] output = new int[array.length];

        for (int num : array) {
            count[num]++;
        }

        for (int i = 1; i <= max; i++) {
            count[i] += count[i - 1];
        }

        for (int i = array.length - 1; i >= 0; i--) {
            output[count[array[i]] - 1] = array[i];
            count[array[i]]--;
        }

        return output;
    }

    // 基数排序
    public static void radixSortWithSteps(int[] array) {
        int max = Arrays.stream(array).max().orElse(0);

        for (int exp = 1; max / exp > 0; exp *= 10) {
            System.out.println("按第 " + getDigitPosition(exp) + " 位排序:");
            System.out.println("排序前: " + Arrays.toString(array));
            countingSortByDigit(array, exp);
            System.out.println("排序后: " + Arrays.toString(array));
        }

    private static void countingSortByDigit(int[] array, int exp) {
        int[] output = new int[array.length];
        int[] count = new int[10];

        for (int num : array) {
            count[(num / exp) % 10]++;
        }

        for (int i = 1; i < 10; i++) {
            count[i] += count[i - 1];
        }

        for (int i = array.length - 1; i >= 0; i--) {
            output[count[(array[i] / exp) % 10] - 1] = array[i];
            count[(array[i] / exp) % 10]--;
        }

        System.arraycopy(output, 0, array, 0, array.length);
    }

    private static int getDigitPosition(int exp) {
        int position = 1;
        while (exp > 1) {
            exp /= 10;
            position++;
        }
        return position;
    }

    private static void countingSortStringsByChar(String[] array, int pos) {

        int[] count = new int[256]; // ASCII字符

        for (String str : array) {
            char ch = pos < str.length() ? str.charAt(pos) : 0;
            count[ch]++;
        }

        for (int i = 1; i < 256; i++) {
            count[i] += count[i - 1];
        }

        for (int i = array.length - 1; i >= 0; i--) {
            char ch = pos < array[i].length() ? array[i].charAt(pos) : 0;
            output[count[ch] - 1] = array[i];
            count[ch]--;

    // 桶排序
    public static void bucketSortWithSteps(double[] array, int bucketCount) {
        List<List<Double>> buckets = new ArrayList<>();
        for (int i = 0; i < bucketCount; i++) {
            buckets.add(new ArrayList<>());
        }

        // 分配到桶
        System.out.println("1. 分配到桶:");
        for (double num : array) {
            int bucketIndex = (int) (num * bucketCount);
            if (bucketIndex >= bucketCount) bucketIndex = bucketCount - 1;
            buckets.get(bucketIndex).add(num);
            System.out.println("  " + num + " -> 桶 " + bucketIndex);
        }

        // 显示桶内容
        System.out.println("2. 桶内容:");
        for (int i = 0; i < bucketCount; i++) {
            System.out.println("  桶 " + i + ": " + buckets.get(i));
        }

        // 桶内排序
        System.out.println("3. 桶内排序:");
        for (int i = 0; i < bucketCount; i++) {
            Collections.sort(buckets.get(i));
            System.out.println("  桶 " + i + " 排序后: " + buckets.get(i));
        }

        // 合并桶
        System.out.println("4. 合并桶:");
        int index = 0;
        for (List<Double> bucket : buckets) {
            for (double num : bucket) {
                array[index++] = num;
            }
        }
    }

    public static void bucketSortIntegers(int[] array, int bucketCount) {
        int max = Arrays.stream(array).max().orElse(0);
        int min = Arrays.stream(array).min().orElse(0);
        int range = max - min + 1;

        List<List<Integer>> buckets = new ArrayList<>();
        for (int i = 0; i < bucketCount; i++) {
            buckets.add(new ArrayList<>());
        }

        for (int num : array) {
            int bucketIndex = (num - min) * bucketCount / range;
            if (bucketIndex >= bucketCount) bucketIndex = bucketCount - 1;
            buckets.get(bucketIndex).add(num);
        }

        for (List<Integer> bucket : buckets) {
            Collections.sort(bucket);
        }

        int index = 0;
        for (List<Integer> bucket : buckets) {
            for (int num : bucket) {
                array[index++] = num;
            }
        }
    }

算法选择指南 ​

  1. 数据量小(n<50)：插入排序
  2. 一般情况：快速排序
  3. 稳定性要求：归并排序
  4. 内存限制：堆排序
  5. 整数小范围：计数排序
  6. 均匀分布：桶排序

优化技巧 ​

  1. 混合排序：大数组用快排，小数组用插排
  2. 三数取中：优化快排的基准选择
  3. 尾递归优化：减少递归栈深度
  4. 并行化：利用多核处理器优势

实际应用考虑 ​

  1. 数据特征：分布、范围、重复度
  2. 性能要求：时间、空间、稳定性
  3. 系统环境：内存限制、并发需求
  4. 维护成本：代码复杂度、可读性

// 简单排序算法详细实现
import java.util.*;

public class SimpleSortingAlgorithmsDemo {
    
    public static void main(String[] args) {
        demonstrateBubbleSort();
        demonstrateSelectionSort();
        demonstrateInsertionSort();
        demonstratePerformanceComparison();
    }
    
    // 冒泡排序演示
    public static void demonstrateBubbleSort() {
        System.out.println("=== 冒泡排序演示 ===");
        
        /*
         * 冒泡排序 (Bubble Sort)：
         * 1. 比较相邻元素，如果顺序错误就交换
         * 2. 每次遍历都会将最大元素"冒泡"到末尾
         * 3. 重复直到没有需要交换的元素
         * 
         * 特点：
         * - 时间复杂度: O(n²)
         * - 空间复杂度: O(1)
         * - 稳定排序
         * - 自适应排序
         */
        
        int[] array = {64, 34, 25, 12, 22, 11, 90};
        System.out.println("原始数组: " + Arrays.toString(array));
        
        // 基础冒泡排序
        int[] basicBubble = array.clone();
        System.out.println("\n基础冒泡排序过程:");
        bubbleSortBasic(basicBubble);
        
        // 优化冒泡排序
        int[] optimizedBubble = array.clone();
        System.out.println("\n优化冒泡排序:");
        int comparisons = bubbleSortOptimized(optimizedBubble);
        System.out.println("排序结果: " + Arrays.toString(optimizedBubble));
        System.out.println("比较次数: " + comparisons);
        
        // 最优情况测试
        int[] sortedArray = {1, 2, 3, 4, 5, 6, 7};
        System.out.println("\n已排序数组测试: " + Arrays.toString(sortedArray));
        int optimalComparisons = bubbleSortOptimized(sortedArray.clone());
        System.out.println("比较次数: " + optimalComparisons + " (体现自适应性)");
        
        System.out.println();
    }
    
    // 选择排序演示
    public static void demonstrateSelectionSort() {
        System.out.println("=== 选择排序演示 ===");
        
        /*
         * 选择排序 (Selection Sort)：
         * 1. 找到数组中最小元素，与第一个元素交换
         * 2. 在剩余元素中找到最小元素，与第二个元素交换
         * 3. 重复直到排序完成
         * 
         * 特点：
         * - 时间复杂度: O(n²)
         * - 空间复杂度: O(1)
         * - 不稳定排序
         * - 非自适应排序
         */
        
        int[] array = {64, 34, 25, 12, 22, 11, 90};
        System.out.println("原始数组: " + Arrays.toString(array));
        
        System.out.println("\n选择排序过程:");
        int[] selectionArray = array.clone();
        int swaps = selectionSortWithSteps(selectionArray);
        System.out.println("排序结果: " + Arrays.toString(selectionArray));
        System.out.println("交换次数: " + swaps);
        
        // 稳定性测试
        System.out.println("\n稳定性测试:");
        Pair[] pairs = {
            new Pair(3, 'A'), new Pair(1, 'B'), new Pair(3, 'C'), new Pair(2, 'D')
        };
        System.out.println("原始: " + Arrays.toString(pairs));
        selectionSortPairs(pairs);
        System.out.println("排序后: " + Arrays.toString(pairs));
        System.out.println("注意: 相同值的相对位置可能改变 (不稳定)");
        
        System.out.println();
    }
    
    // 插入排序演示
    public static void demonstrateInsertionSort() {
        System.out.println("=== 插入排序演示 ===");
        
        /*
         * 插入排序 (Insertion Sort)：
         * 1. 将数组分为已排序和未排序两部分
         * 2. 取未排序部分的第一个元素
         * 3. 在已排序部分找到合适位置插入
         * 4. 重复直到所有元素都被插入
         * 
         * 特点：
         * - 时间复杂度: O(n²)
         * - 空间复杂度: O(1)
         * - 稳定排序
         * - 自适应排序
         */
        
        int[] array = {64, 34, 25, 12, 22, 11, 90};
        System.out.println("原始数组: " + Arrays.toString(array));
        
        System.out.println("\n插入排序过程:");
        int[] insertionArray = array.clone();
        int shifts = insertionSortWithSteps(insertionArray);
        System.out.println("排序结果: " + Arrays.toString(insertionArray));
        System.out.println("移动次数: " + shifts);
        
        // 自适应性测试
        System.out.println("\n自适应性测试:");
        
        // 随机数组
        int[] randomArray = {5, 2, 8, 1, 9, 3};
        System.out.println("随机数组: " + Arrays.toString(randomArray));
        long startTime = System.nanoTime();
        int randomShifts = insertionSortOptimized(randomArray.clone());
        long randomTime = System.nanoTime() - startTime;
        System.out.println("移动次数: " + randomShifts + ", 用时: " + randomTime + "ns");
        
        // 部分有序数组
        int[] partialArray = {1, 2, 5, 3, 4, 6};
        System.out.println("部分有序: " + Arrays.toString(partialArray));
        startTime = System.nanoTime();
        int partialShifts = insertionSortOptimized(partialArray.clone());
        long partialTime = System.nanoTime() - startTime;
        System.out.println("移动次数: " + partialShifts + ", 用时: " + partialTime + "ns");
        
        // 已排序数组
        int[] sortedArray = {1, 2, 3, 4, 5, 6};
        System.out.println("已排序: " + Arrays.toString(sortedArray));
        startTime = System.nanoTime();
        int sortedShifts = insertionSortOptimized(sortedArray.clone());
        long sortedTime = System.nanoTime() - startTime;
        System.out.println("移动次数: " + sortedShifts + ", 用时: " + sortedTime + "ns");
        
        System.out.println();
    }

    public static void demonstratePerformanceComparison() {
        System.out.println("=== 简单排序算法性能比较 ===");
        
        int[] sizes = {100, 500, 1000};
        
        System.out.printf("%-8s %-12s %-12s %-12s%n", "数组大小", "冒泡排序(ms)", "选择排序(ms)", "插入排序(ms)");
        System.out.println("------------------------------------------------");
        
        for (int size : sizes) {
            int[] testArray = generateRandomArray(size);
            
            // 冒泡排序测试

            long bubbleTime = System.currentTimeMillis() - startTime;
            
            // 选择排序测试
            startTime = System.currentTimeMillis();
            selectionSort(testArray.clone());
            long selectionTime = System.currentTimeMillis() - startTime;
            
            // 插入排序测试

        }
        
        System.out.println("\n性能分析:");
        System.out.println("1. 插入排序通常表现最好，特别是在小数组和部分有序数组上");
        System.out.println("2. 选择排序的交换次数最少，但比较次数固定");
        System.out.println("3. 冒泡排序在最坏情况下性能最差，但具有自适应性");

    // 算法实现
    
    // 基础冒泡排序
    public static void bubbleSortBasic(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            System.out.println("第 " + (i + 1) + " 轮:");
            for (int j = 0; j < n - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    // 交换
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    System.out.println("  交换 " + array[j + 1] + " 和 " + array[j] + ": " + Arrays.toString(array));
                }
            }
            System.out.println("  本轮结果: " + Arrays.toString(array));
        }
    }
    
    // 优化冒泡排序
    public static int bubbleSortOptimized(int[] array) {
        int n = array.length;
        int comparisons = 0;
        
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < n - 1 - i; j++) {
                comparisons++;
                if (array[j] > array[j + 1]) {
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swapped = true;
                }
            }
            if (!swapped) break; // 提前终止优化
        }
        
        return comparisons;
    }
    
    // 选择排序
    public static int selectionSortWithSteps(int[] array) {
        int n = array.length;
        int swaps = 0;
        
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
                swaps++;
                System.out.println("第 " + (i + 1) + " 轮: 交换 " + array[i] + " 和 " + array[minIndex] + 
                                 " -> " + Arrays.toString(array));
            }
        }
        
        return swaps;
    }
    
    public static void selectionSort(int[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (array[j] < array[minIndex]) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                int temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
    }
    
    // 插入排序
    public static int insertionSortWithSteps(int[] array) {
        int n = array.length;
        int shifts = 0;
        
        for (int i = 1; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            System.out.println("插入 " + key + ":");
            
            while (j >= 0 && array[j] > key) {
                array[j + 1] = array[j];
                j--;
                shifts++;
            }
            array[j + 1] = key;
            
            System.out.println("  结果: " + Arrays.toString(array));
        }
        
        return shifts;
    }
    
    public static int insertionSortOptimized(int[] array) {
        int n = array.length;
        int shifts = 0;
        
        for (int i = 1; i < n; i++) {
            int key = array[i];
            int j = i - 1;
            
            while (j >= 0 && array[j] > key) {
                array[j + 1] = array[j];
                j--;
                shifts++;
            }
            array[j + 1] = key;
        }
        
        return shifts;
    }
    
    // 选择排序稳定性测试
    public static void selectionSortPairs(Pair[] array) {
        int n = array.length;
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (array[j].value < array[minIndex].value) {
                    minIndex = j;
                }
            }
            if (minIndex != i) {
                Pair temp = array[i];
                array[i] = array[minIndex];
                array[minIndex] = temp;
            }
        }
    }
    
    // 辅助方法
    public static int[] generateRandomArray(int size) {
        Random random = new Random();
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(1000);
        }
        return array;
    }
}

    }
