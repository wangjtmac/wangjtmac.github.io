哈希查找详解 ​

📚 概述 ​

哈希查找是基于哈希表数据结构的高效查找算法，通过哈希函数将键直接映射到存储位置，实现平均O(1)时间复杂度的查找操作。哈希查找的核心思想是用空间换时间，通过预先计算和存储键值对的映射关系，避免了传统查找算法中的逐一比较过程。虽然哈希查找在最坏情况下可能退化到O(n)，但在实际应用中，通过合理的哈希函数设计和冲突处理策略，可以保持优秀的平均性能。哈希查找广泛应用于数据库索引、缓存系统、编译器符号表等需要快速查找的场景。

🎯 学习目标 ​

  • 深入理解哈希查找的基本原理和实现机制
  • 掌握哈希函数的设计原则和常见实现
  • 学会分析哈希查找的时间和空间复杂度
  • 理解哈希冲突的处理策略和优化方法
  • 掌握哈希表的动态扩容和负载因子控制
  • 学会哈希查找在实际问题中的应用
  • 了解现代哈希技术的发展和优化
  • 培养哈希思维和系统设计能力

    // 哈希函数实现
    public static int simpleHash(String key, int tableSize) {
        int hash = 0;
        for (char c : key.toCharArray()) {
            hash += c;
        }
        return hash % tableSize;
    }
    
    public static int djbHash(String key, int tableSize) {
        int hash = 5381;
        for (char c : key.toCharArray()) {
            hash = ((hash << 5) + hash) + c;
        }
        return Math.abs(hash) % tableSize;
    }
    
    public static int fnvHash(String key, int tableSize) {
        final int FNV_PRIME = 16777619;
        final int FNV_OFFSET_BASIS = (int) 2166136261L;
        
        int hash = FNV_OFFSET_BASIS;
        for (char c : key.toCharArray()) {
            hash ^= c;
            hash *= FNV_PRIME;
        }
        return Math.abs(hash) % tableSize;
    }
    
    public static int javaHash(String key, int tableSize) {
        return Math.abs(key.hashCode()) % tableSize;
    }
    
    // 哈希函数分析
    public static void analyzeHashFunction(String[] keys, int tableSize, HashFunction hashFunc) {
        int[] buckets = new int[tableSize];
        int collisions = 0;
        
        System.out.println("键 -> 哈希值:");
        for (String key : keys) {
            int hash = hashFunc.hash(key, tableSize);
            System.out.println("  " + key + " -> " + hash);
            
            if (buckets[hash] > 0) {
                collisions++;
            }
            buckets[hash]++;
        }
        
        System.out.println("桶分布: " + Arrays.toString(buckets));
        System.out.println("冲突数: " + collisions);
        
        // 计算分布均匀性
        double average = (double) keys.length / tableSize;
        double variance = 0;
        for (int count : buckets) {
            variance += Math.pow(count - average, 2);
        }
        variance /= tableSize;
        
        System.out.println("平均每桶: " + String.format("%.2f", average));
        System.out.println("方差: " + String.format("%.2f", variance));
    }

    public static Map<String, Integer> countWordFrequency(String text) {
        Map<String, Integer> frequency = new HashMap<>();
        String[] words = text.toLowerCase().split("\\s+");
        
        for (String word : words) {
            frequency.put(word, frequency.getOrDefault(word, 0) + 1);
        }
        
        return frequency;
    }
    
    // 两数之和
    public static int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        
        for (int i = 0; i < nums.length; i++) {
            int complement = target - nums[i];
            if (map.containsKey(complement)) {
                return new int[]{map.get(complement), i};
            }
            map.put(nums[i], i);
        }
        
        return null;
    }
    
    // 字母异位词分组
    public static List<List<String>> groupAnagrams(String[] words) {
        Map<String, List<String>> groups = new HashMap<>();
        
        for (String word : words) {
            char[] chars = word.toCharArray();
            Arrays.sort(chars);
            String key = new String(chars);
            
            groups.computeIfAbsent(key, k -> new ArrayList<>()).add(word);
        }
        
        return new ArrayList<>(groups.values());
    }

// 简单哈希表实现
class SimpleHashTable<K, V> {
    private static class Entry<K, V> {
        K key;
        V value;
        Entry<K, V> next;
        
        Entry(K key, V value) {
            this.key = key;
            this.value = value;
        }
    }
    
    private Entry<K, V>[] buckets;
    private int size;
    private int capacity;
    private int collisions;
    private int lookups;
    
    @SuppressWarnings("unchecked")
    public SimpleHashTable() {
        this.capacity = 16;
        this.buckets = new Entry[capacity];
        this.size = 0;
        this.collisions = 0;
        this.lookups = 0;
    }
    
    private int hash(K key) {
        return Math.abs(key.hashCode()) % capacity;
    }
    
    public void put(K key, V value) {
        int index = hash(key);
        Entry<K, V> entry = buckets[index];
        
        if (entry != null) {
            collisions++;
        }
        
        while (entry != null) {
            if (entry.key.equals(key)) {
                entry.value = value;
                return;
            }
            entry = entry.next;
        }
        
        Entry<K, V> newEntry = new Entry<>(key, value);
        newEntry.next = buckets[index];
        buckets[index] = newEntry;
        size++;
    }
    
    public V get(K key) {
        lookups++;
        int index = hash(key);
        Entry<K, V> entry = buckets[index];
        
        while (entry != null) {
            if (entry.key.equals(key)) {
                return entry.value;
            }
            entry = entry.next;
        }
        
        return null;
    }
    
    public int size() {
        return size;
    }
    
    public void printStatistics() {
        System.out.println("容量: " + capacity);
        System.out.println("大小: " + size);
        System.out.println("负载因子: " + String.format("%.2f", (double) size / capacity));
        System.out.println("冲突次数: " + collisions);
        System.out.println("查找次数: " + lookups);
    }
    
    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder("{");
        boolean first = true;
        
        for (Entry<K, V> entry : buckets) {
            while (entry != null) {
                if (!first) sb.append(", ");
                sb.append(entry.key).append("=").append(entry.value);
                first = false;
                entry = entry.next;
            }
        }
        
        sb.append("}");
        return sb.toString();

// 链地址法哈希表
class ChainingHashTable<K, V> {
    private List<Entry<K, V>>[] buckets;
    private int capacity;
    private int size;
    
    private static class Entry<K, V> {
        K key;
        V value;
        
        Entry(K key, V value) {
            this.key = key;
            this.value = value;
        }
    }
    
    @SuppressWarnings("unchecked")
    public ChainingHashTable(int capacity) {
        this.capacity = capacity;
        this.buckets = new List[capacity];
        for (int i = 0; i < capacity; i++) {
            buckets[i] = new ArrayList<>();
        }
        this.size = 0;
    }
    
    private int hash(K key) {
        return Math.abs(key.hashCode()) % capacity;
    }
    
    public void put(K key, V value) {
        int index = hash(key);
        List<Entry<K, V>> bucket = buckets[index];
        
        for (Entry<K, V> entry : bucket) {
            if (entry.key.equals(key)) {
                entry.value = value;
                return;
            }
        }
        
        bucket.add(new Entry<>(key, value));
        size++;
    }
    
    public V get(K key) {
        int index = hash(key);
        List<Entry<K, V>> bucket = buckets[index];
        
        for (Entry<K, V> entry : bucket) {
            if (entry.key.equals(key)) {
                return entry.value;
            }
        }
        
        return null;
    }
    
    public void printStructure() {
        for (int i = 0; i < capacity; i++) {
            System.out.print("桶" + i + ": ");
            List<Entry<K, V>> bucket = buckets[i];
            if (bucket.isEmpty()) {
                System.out.println("空");
            } else {
                for (int j = 0; j < bucket.size(); j++) {
                    Entry<K, V> entry = bucket.get(j);
                    System.out.print("[" + entry.key + "=" + entry.value + "]");
                    if (j < bucket.size() - 1) System.out.print(" -> ");
                }
                System.out.println();
            }
        }
    }
}

// 线性探测哈希表
class LinearProbingHashTable<K, V> {
    private K[] keys;
    private V[] values;
    private boolean[] deleted;
    private int capacity;
    private int size;
    
    @SuppressWarnings("unchecked")
    public LinearProbingHashTable(int capacity) {
        this.capacity = capacity;
        this.keys = (K[]) new Object[capacity];
        this.values = (V[]) new Object[capacity];
        this.deleted = new boolean[capacity];
        this.size = 0;
    }
    
    private int hash(K key) {
        return Math.abs(key.hashCode()) % capacity;
    }
    
    public void put(K key, V value) {
        int index = hash(key);
        
        while (keys[index] != null && !deleted[index]) {
            if (keys[index].equals(key)) {
                values[index] = value;
                return;
            }
            index = (index + 1) % capacity;
        }
        
        keys[index] = key;
        values[index] = value;
        deleted[index] = false;
        size++;
    }
    
    public V get(K key) {
        int index = hash(key);
        
        while (keys[index] != null) {
            if (!deleted[index] && keys[index].equals(key)) {
                return values[index];
            }
            index = (index + 1) % capacity;
        }
        
        return null;
    }
    
    public void printStructure() {
        for (int i = 0; i < capacity; i++) {
            if (keys[i] == null || deleted[i]) {
                System.out.println("位置" + i + ": 空");
            } else {
                System.out.println("位置" + i + ": [" + keys[i] + "=" + values[i] + "]");
            }
        }
    }
}
