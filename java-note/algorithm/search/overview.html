查找算法概述 ​

📚 概述 ​

查找算法是计算机科学中另一个重要的基础算法类别，它用于在数据集合中寻找特定的元素。查找算法的效率直接影响到程序的性能，特别是在处理大量数据时。不同的查找算法适用于不同的数据结构和应用场景，从简单的线性查找到复杂的树查找和哈希查找，每种算法都有其独特的优势和适用条件。掌握各种查找算法的原理、实现和性能特点，能够帮助我们在实际开发中选择最合适的查找策略。本章将系统介绍各种经典的查找算法，包括基于线性结构的查找、基于树结构的查找以及基于哈希的查找，通过详细的代码实现和性能分析，让您全面掌握查找算法的精髓。

🎯 学习目标 ​

  • 理解查找算法的基本概念和分类
  • 掌握各种查找算法的实现原理
  • 学会分析查找算法的时间和空间复杂度
  • 理解不同数据结构对查找效率的影响
  • 掌握查找算法的优化技巧
  • 学会根据场景选择合适的查找算法
  • 了解查找算法在实际开发中的应用
  • 熟悉查找算法的面试常见问题

    // 基础查找算法实现
    
    public static int linearSearch(int[] array, int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                return i;
            }
        }
        return -1;
    }
    
    public static int binarySearch(int[] array, int target) {
        int left = 0, right = array.length - 1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] == target) {
                return mid;
            } else if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return -1;
    }
    
    public static List<Integer> rangeSearch(int[] array, int lower, int upper) {
        List<Integer> result = new ArrayList<>();
        for (int num : array) {
            if (num >= lower && num <= upper) {
                result.add(num);
            }
        }
        return result;
    }
    
    public static int findClosest(int[] array, int target) {
        int left = 0, right = array.length - 1;
        
        while (left < right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        
        // 检查left和left-1哪个更接近
        if (left > 0 && Math.abs(array[left - 1] - target) < Math.abs(array[left] - target)) {
            return left - 1;
        }
        
        return left;
    }
    
    public static int findKthSmallest(int[] array, int k) {
        if (k <= 0 || k > array.length) {
            throw new IllegalArgumentException("k超出范围");
        }
        return array[k - 1]; // 假设数组已排序
    }
    
    public static int findFirstOccurrence(int[] array, int target) {
        int left = 0, right = array.length - 1;
        int result = -1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] == target) {
                result = mid;
                right = mid - 1; // 继续在左半部分查找
            } else if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return result;
    }
    
    public static int findLastOccurrence(int[] array, int target) {
        int left = 0, right = array.length - 1;
        int result = -1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] == target) {
                result = mid;
                left = mid + 1; // 继续在右半部分查找
            } else if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return result;
    }
    
    // 辅助方法
    public static int[] generateSortedArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = i;
        }
        return array;
    }
    
    public static int[] generateRandomArray(int size) {
        Random random = new Random();
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(1000);
        }
        return array;
    }

    public static int[] findMinMax(int[] array) {
        if (array.length == 0) return new int[]{0, 0};
        
        int min = array[0], max = array[0];
        for (int i = 1; i < array.length; i++) {
            if (array[i] < min) min = array[i];
            if (array[i] > max) max = array[i];
        }
        return new int[]{min, max};
    }
    
    public static int findSecondLargest(int[] array) {
        if (array.length < 2) return -1;
        
        int largest = Integer.MIN_VALUE;
        int secondLargest = Integer.MIN_VALUE;
        
        for (int num : array) {
            if (num > largest) {
                secondLargest = largest;
                largest = num;
            } else if (num > secondLargest && num != largest) {
                secondLargest = num;
            }
        }
        
        return secondLargest == Integer.MIN_VALUE ? -1 : secondLargest;
    }
    
    public static List<Integer> findByCondition(int[] array, Predicate<Integer> condition) {
        List<Integer> result = new ArrayList<>();
        for (int num : array) {
            if (condition.test(num)) {
                result.add(num);
            }
        }
        return result;
    }

    // 哨兵查找
    public static int sentinelLinearSearch(int[] array, int target) {
        int n = array.length;
        int last = array[n - 1];
        array[n - 1] = target; // 设置哨兵
        
        int i = 0;
        while (array[i] != target) {
            i++;
        }
        
        array[n - 1] = last; // 恢复原值
        
        if (i < n - 1 || array[n - 1] == target) {
            return i;
        }
        return -1;
    }
    
    // 移动到前面
    public static int moveToFrontSearch(int[] array, int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                // 将找到的元素移动到前面
                if (i > 0) {
                    int temp = array[i];
                    System.arraycopy(array, 0, array, 1, i);
                    array[0] = temp;
                    return 0;
                }
                return i;
            }
        }
        return -1;
    }
    
    // 转置法
    public static int transposeSearch(int[] array, int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                // 与前一个元素交换位置
                if (i > 0) {
                    int temp = array[i];
                    array[i] = array[i - 1];
                    array[i - 1] = temp;
                    return i - 1;
                }
                return i;
            }
        }
        return -1;
    }

    public static int findPeakElement(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            boolean isPeak = true;
            if (i > 0 && nums[i] <= nums[i - 1]) isPeak = false;
            if (i < nums.length - 1 && nums[i] <= nums[i + 1]) isPeak = false;
            if (isPeak) return i;
        }
        return -1;
    }
    
    public static int findMissingNumber(int[] nums) {
        int n = nums.length + 1; // 应该有n个数字
        int expectedSum = n * (n - 1) / 2;
        int actualSum = 0;
        for (int num : nums) {
            actualSum += num;
        }
        return expectedSum - actualSum;
    }

class FrequencyBasedSearch {
    private List<Integer> data;
    private Map<Integer, Integer> frequency;
    
    public FrequencyBasedSearch(int[] array) {
        this.data = new ArrayList<>();
        this.frequency = new HashMap<>();
        
        for (int num : array) {
            data.add(num);
            frequency.put(num, frequency.getOrDefault(num, 0) + 1);
        }
    }
    
    public int search(int target) {
        int index = data.indexOf(target);
        if (index >= 0) {
            // 增加频率计数
            frequency.put(target, frequency.get(target) + 1);
            System.out.println("找到 " + target + " 在位置 " + index + ", 频率: " + frequency.get(target));
        } else {
            System.out.println("未找到 " + target);
        }
        return index;
    }
    
    public void printFrequencies() {
        System.out.println("元素频率统计: " + frequency);
    }
}

算法选择指南 ​

  1. 无序小数据：线性查找
  2. 有序数据：二分查找
  3. 频繁查找：哈希表
  4. 范围查询：平衡树
  5. 字符串匹配：KMP或Boyer-Moore
  6. 近似查找：插值查找

优化策略 ​

  1. 数据预处理：排序、建索引
  2. 缓存机制：LRU、频率统计
  3. 并行查找：分治、多线程
  4. 数据结构选择：根据查找模式选择

实际应用考虑 ​

  1. 数据特征：大小、分布、有序性
  2. 查找模式：频率、类型、范围
  3. 性能要求：时间、空间、实时性
  4. 系统环境：内存限制、并发需求
