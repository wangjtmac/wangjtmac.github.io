    public static int sentinelLinearSearch(int[] array, int target) {
        int n = array.length;
        int last = array[n - 1];
        array[n - 1] = target;
        
        int i = 0;
        while (array[i] != target) {
            i++;
        }
        
        array[n - 1] = last;
        
        if (i < n - 1 || array[n - 1] == target) {
            return i;
        }
        return -1;
    }

    public static int recursiveLinearSearch(int[] array, int target, int index) {
        if (index >= array.length) {
            return -1;
        }
        
        if (array[index] == target) {
            return index;
        }
        
        return recursiveLinearSearch(array, target, index + 1);
    }

    public static int bidirectionalLinearSearch(int[] array, int target) {
        int left = 0;
        int right = array.length - 1;
        
        while (left <= right) {
            if (array[left] == target) {
                return left;
            }
            if (array[right] == target) {
                return right;
            }
            
            left++;
            right--;
        }
        
        return -1;
    }

    public static int binarySearch(int[] array, int target) {
        int left = 0, right = array.length - 1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (array[mid] == target) {
                return mid;
            } else if (array[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return -1;
    }

    public static ListNode createLinkedList(int[] values) {
        ListNode dummy = new ListNode(0);
        ListNode current = dummy;
        for (int value : values) {
            current.next = new ListNode(value);
            current = current.next;
        }
        return dummy.next;
    }

        }
        return sb.toString();
    }

class FrequencyOptimizedSearch {
    private int[] array;
    private int[] frequency;
    
    public FrequencyOptimizedSearch(int[] array) {
        this.array = array.clone();
        this.frequency = new int[array.length];
    }
    
    public int search(int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                frequency[i]++;
                reorganize();
                return findNewPosition(target);
            }
        }
        return -1;
    }
    
    private void reorganize() {
        // æ ¹æ®é¢‘ç‡é‡æ–°æ’åˆ—ï¼ˆç®€å•çš„å†’æ³¡æ’åºï¼‰
        for (int i = 0; i < array.length - 1; i++) {
            for (int j = 0; j < array.length - 1 - i; j++) {
                if (frequency[j] < frequency[j + 1]) {
                    // äº¤æ¢å…ƒç´ å’Œé¢‘ç‡
                    int tempVal = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = tempVal;
                    
                    int tempFreq = frequency[j];
                    frequency[j] = frequency[j + 1];
                    frequency[j + 1] = tempFreq;
                }
            }
        }
    }
    
    private int findNewPosition(int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                return i;
            }
        }
        return -1;
    }
    
    @Override
    public String toString() {
        return Arrays.toString(array);
    }
}
