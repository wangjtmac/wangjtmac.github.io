进阶项目系统 ​

📚 概述 ​

进阶项目系统是Java学习的高级阶段，旨在培养学习者的系统设计能力、架构思维和工程实践技能。这些项目不再是简单的功能实现，而是完整的软件系统，涉及复杂的业务逻辑、分布式架构、高并发处理、数据一致性等企业级开发的核心问题。通过这些项目的实践，学习者将从编程新手成长为具备系统思维的软件工程师。每个项目都经过精心设计，既体现了现代软件开发的最佳实践，又涵盖了主流技术栈的应用，为学习者进入企业级开发做好充分准备。

🎯 学习目标 ​

  • 掌握企业级应用的架构设计
  • 学会使用主流Java框架和技术栈
  • 培养分布式系统的设计思维
  • 掌握高并发和高可用的解决方案
  • 学会数据库设计和优化
  • 理解微服务架构的设计原则
  • 掌握DevOps和部署运维技能
  • 培养团队协作和项目管理能力

        System.out.println("1. 微服务架构 - 服务拆分和治理");
        System.out.println("2. 分布式事务 - 保证数据一致性");
        System.out.println("3. 高并发处理 - 秒杀、抢购场景");
        System.out.println("4. 缓存策略 - Redis多级缓存");
        System.out.println("5. 消息队列 - 异步处理和解耦");

        System.out.println("用户登录: " + (token != null ? "成功" : "失败"));

        System.out.println("\n2. 商品管理演示:");
        Product product = ecommerce.addProduct("iPhone 15", "最新款iPhone", 8999.0, 100);
        System.out.println("添加商品: " + product);
        
        List<Product> searchResults = ecommerce.searchProducts("iPhone");
        System.out.println("搜索结果: " + searchResults.size() + " 个商品");

        System.out.println("\n3. 购物车演示:");

        System.out.println("购物车商品数: " + cart.getItems().size());
        System.out.println("购物车总价: ¥" + cart.getTotalPrice());

        System.out.println("\n4. 订单处理演示:");

        ecommerce.startSeckill(product.getId(), 10, 6999.0);
        
        // 模拟多用户同时抢购
        ExecutorService executor = Executors.newFixedThreadPool(20);
        CountDownLatch latch = new CountDownLatch(20);
        
        for (int i = 0; i < 20; i++) {
            final int userId = i + 1;
            executor.submit(() -> {
                try {
                    boolean success = ecommerce.seckillPurchase(product.getId(), userId);
                    System.out.println("用户" + userId + " 秒杀: " + (success ? "成功" : "失败"));
                } finally {
                    latch.countDown();
                }
            });
        }
        
        try {
            latch.await();
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        executor.shutdown();
        
        System.out.println();
    }

class ECommerceSystem {
    private Map<String, User> users = new HashMap<>();
    private Map<String, Product> products = new HashMap<>();
    private Map<String, Cart> carts = new HashMap<>();
    private Map<String, Order> orders = new HashMap<>();
    private Map<String, Integer> productStock = new HashMap<>();
    private Map<String, SeckillInfo> seckillProducts = new HashMap<>();
    
    public User registerUser(String email, String password, String name) {
        User user = new User(UUID.randomUUID().toString(), email, name);
        users.put(user.getId(), user);
        carts.put(user.getId(), new Cart(user.getId()));
        return user;
    }
    
    public String login(String email, String password) {
        return "token_" + UUID.randomUUID().toString();
    }
    
    public Product addProduct(String name, String description, double price, int stock) {
        Product product = new Product(UUID.randomUUID().toString(), name, description, price);
        products.put(product.getId(), product);
        productStock.put(product.getId(), stock);
        return product;
    }
    
    public List<Product> searchProducts(String keyword) {
        return products.values().stream()
                .filter(p -> p.getName().contains(keyword))
                .collect(ArrayList::new, (list, item) -> list.add(item), ArrayList::addAll);
    }
    
    public void addToCart(String userId, String productId, int quantity) {
        Cart cart = carts.get(userId);
        cart.addItem(productId, quantity);
    }
    
    public Cart getCart(String userId) {
        return carts.get(userId);
    }
    
    public Order createOrder(String userId, Cart cart) {
        Order order = new Order(UUID.randomUUID().toString(), userId, cart.getTotalPrice());
        orders.put(order.getOrderId(), order);
        return order;
    }
    
    public boolean processPayment(String orderId, String paymentMethod) {
        return true; // 模拟支付成功
    }
    
    public int getProductStock(String productId) {
        return productStock.getOrDefault(productId, 0);
    }
    
    public void startSeckill(String productId, int seckillStock, double seckillPrice) {
        SeckillInfo seckill = new SeckillInfo(productId, seckillStock, seckillPrice);
        seckillProducts.put(productId, seckill);
    }
    
    public synchronized boolean seckillPurchase(String productId, int userId) {
        SeckillInfo seckill = seckillProducts.get(productId);
        if (seckill != null && seckill.getStock() > 0) {
            seckill.decreaseStock();
            return true;
        }
        return false;
    }
}

class User {
    private String id;
    private String email;
    private String name;
    
    public User(String id, String email, String name) {
        this.id = id;
        this.email = email;
        this.name = name;
    }
    
    public String getId() { return id; }
    public String getEmail() { return email; }
    public String getName() { return name; }
    
    @Override
    public String toString() {
        return "User{id='" + id + "', email='" + email + "', name='" + name + "'}";
    }
}

class Product {
    private String id;
    private String name;
    private String description;
    private double price;
    
    public Product(String id, String name, String description, double price) {
        this.id = id;
        this.name = name;
        this.description = description;
        this.price = price;
    }
    
    public String getId() { return id; }
    public String getName() { return name; }
    public String getDescription() { return description; }
    public double getPrice() { return price; }
    
    @Override
    public String toString() {
        return "Product{id='" + id + "', name='" + name + "', price=" + price + "}";
    }
}

class Cart {
    private String userId;
    private Map<String, Integer> items = new HashMap<>();
    
    public Cart(String userId) {
        this.userId = userId;
    }
    
    public void addItem(String productId, int quantity) {
        items.put(productId, items.getOrDefault(productId, 0) + quantity);
    }
    
    public Map<String, Integer> getItems() { return items; }
    
    public double getTotalPrice() {
        return items.size() * 100.0; // 简化计算
    }
}

class Order {
    private String orderId;
    private String userId;
    private double totalAmount;
    private String status = "PENDING";
    
    public Order(String orderId, String userId, double totalAmount) {
        this.orderId = orderId;
        this.userId = userId;
        this.totalAmount = totalAmount;
    }
    
    public String getOrderId() { return orderId; }
    public String getUserId() { return userId; }
    public double getTotalAmount() { return totalAmount; }
    public String getStatus() { return status; }
}

class SeckillInfo {
    private String productId;
    private int stock;
    private double price;
    
    public SeckillInfo(String productId, int stock, double price) {
        this.productId = productId;
        this.stock = stock;
        this.price = price;
    }
    
    public int getStock() { return stock; }
    public void decreaseStock() { stock--; }
}

// CRM系统相关类
class CRMSystem {
    private Map<String, Customer> customers = new HashMap<>();
    private Map<String, List<String>> customerTags = new HashMap<>();
    private Map<String, Opportunity> opportunities = new HashMap<>();
    private Map<String, ServiceTicket> tickets = new HashMap<>();
    
    public Customer addCustomer(String name, String email, String phone, String type) {
        Customer customer = new Customer(UUID.randomUUID().toString(), name, email, phone, type);
        customers.put(customer.getId(), customer);
        return customer;
    }
    
    public void addCustomerTag(String customerId, String tag) {
        customerTags.computeIfAbsent(customerId, k -> new ArrayList<>()).add(tag);
    }
    
    public List<String> getCustomerTags(String customerId) {
        return customerTags.getOrDefault(customerId, new ArrayList<>());
    }
    
    public Opportunity createOpportunity(String customerId, String name, double value) {
        Opportunity opportunity = new Opportunity(UUID.randomUUID().toString(), customerId, name, value);
        opportunities.put(opportunity.getId(), opportunity);
        return opportunity;
    }
    
    public void updateOpportunityStage(String opportunityId, String stage) {
        Opportunity opportunity = opportunities.get(opportunityId);
        if (opportunity != null) {

class ContentManagementSystem {
    private Map<String, CMSUser> users = new HashMap<>();
    private Map<String, Article> articles = new HashMap<>();
    private Map<String, List<ArticleVersion>> articleVersions = new HashMap<>();
    
    public CMSUser createUser(String email, String username, String role) {
        CMSUser user = new CMSUser(UUID.randomUUID().toString(), email, username, role);
        users.put(user.getId(), user);
        return user;
    }
    
    public Article createArticle(String authorId, String title, String content, String category) {
        Article article = new Article(UUID.randomUUID().toString(), authorId, title, content, category);
        articles.put(article.getId(), article);
        
        // 创建初始版本
        ArticleVersion version = new ArticleVersion(1, title, content, new Date());
        articleVersions.computeIfAbsent(article.getId(), k -> new ArrayList<>()).add(version);
        
        return article;
    }
    
    public void addArticleTag(String articleId, String tag) {
        Article article = articles.get(articleId);
        if (article != null) {
            article.addTag(tag);
        }
    }
    
    public void submitForReview(String articleId) {
        Article article = articles.get(articleId);
        if (article != null) {
            article.setStatus("PENDING_REVIEW");
        }
    }
    
    public void reviewArticle(String articleId, String reviewerId, boolean approved, String comment) {
        Article article = articles.get(articleId);
        if (article != null) {
            article.setStatus(approved ? "APPROVED" : "REJECTED");
            article.setReviewComment(comment);
        }
    }
    
    public void publishArticle(String articleId) {
        Article article = articles.get(articleId);
        if (article != null) {
            article.setStatus("PUBLISHED");
            article.setPublishedAt(new Date());
        }
    }
    
    public String getArticleStatus(String articleId) {
        Article article = articles.get(articleId);
        return article != null ? article.getStatus() : null;
    }
    
    public void updateArticle(String articleId, String title, String content) {
        Article article = articles.get(articleId);
        if (article != null) {
            article.setTitle(title);
            article.setContent(content);
            
            // 创建新版本
            List<ArticleVersion> versions = articleVersions.get(articleId);
            int newVersion = versions.size() + 1;
            versions.add(new ArticleVersion(newVersion, title, content, new Date()));
        }
    }
    
    public List<ArticleVersion> getArticleVersions(String articleId) {
        return articleVersions.getOrDefault(articleId, new ArrayList<>());
    }
    
    public ContentStats getContentStats() {
        long total = articles.size();
        long published = articles.values().stream().filter(a -> "PUBLISHED".equals(a.getStatus())).count();
        long pending = articles.values().stream().filter(a -> "PENDING_REVIEW".equals(a.getStatus())).count();
        long draft = articles.values().stream().filter(a -> "DRAFT".equals(a.getStatus())).count();
        
        return new ContentStats((int)total, (int)published, (int)pending, (int)draft);
    }
}

class ProjectManagementSystem {
    private Map<String, Project> projects = new HashMap<>();
    private Map<String, List<TeamMember>> projectTeams = new HashMap<>();
    private Map<String, List<Task>> projectTasks = new HashMap<>();
    private Map<String, List<WorkLog>> workLogs = new HashMap<>();
    
    public Project createProject(String name, String description, String startDate, String endDate) {
        Project project = new Project(UUID.randomUUID().toString(), name, description, startDate, endDate);
        projects.put(project.getId(), project);
        projectTeams.put(project.getId(), new ArrayList<>());
        projectTasks.put(project.getId(), new ArrayList<>());
        return project;
    }
    
    public TeamMember addTeamMember(String projectId, String name, String role) {
        TeamMember member = new TeamMember(UUID.randomUUID().toString(), name, role);
        projectTeams.get(projectId).add(member);
        return member;
    }
    
    public List<TeamMember> getTeamMembers(String projectId) {
        return projectTeams.getOrDefault(projectId, new ArrayList<>());
    }
    
    public Task createTask(String projectId, String name, String description, String assigneeId, 
                          String startDate, String endDate) {
        Task task = new Task(UUID.randomUUID().toString(), projectId, name, description, 
                           assigneeId, startDate, endDate);
        projectTasks.get(projectId).add(task);
        return task;
    }
    
    public void updateTaskStatus(String taskId, String status) {
        for (List<Task> tasks : projectTasks.values()) {
            for (Task task : tasks) {
                if (task.getId().equals(taskId)) {
                    task.setStatus(status);
                    return;
                }
            }
        }
    }
    
    public void updateTaskProgress(String taskId, int progress) {
        for (List<Task> tasks : projectTasks.values()) {
            for (Task task : tasks) {
                if (task.getId().equals(taskId)) {
                    task.setProgress(progress);
                    return;
                }
            }
        }
    }
    
    public String getTaskStatus(String taskId) {
        for (List<Task> tasks : projectTasks.values()) {
            for (Task task : tasks) {
                if (task.getId().equals(taskId)) {
                    return task.getStatus();
                }
            }
        }
        return null;
    }
    
    public int getTaskProgress(String taskId) {
        for (List<Task> tasks : projectTasks.values()) {
            for (Task task : tasks) {
                if (task.getId().equals(taskId)) {
                    return task.getProgress();
                }
            }
        }
        return 0;
    }
    
    public void logWorkTime(String taskId, String memberId, double hours, String description) {
        WorkLog log = new WorkLog(taskId, memberId, hours, description, new Date());
        workLogs.computeIfAbsent(taskId, k -> new ArrayList<>()).add(log);
    }
    
    public double getTotalWorkHours(String projectId) {
        List<Task> tasks = projectTasks.get(projectId);
        double total = 0;
        for (Task task : tasks) {
            List<WorkLog> logs = workLogs.getOrDefault(task.getId(), new ArrayList<>());
            total += logs.stream().mapToDouble(WorkLog::getHours).sum();
        }
        return total;
    }
    
    public ProjectReport generateProjectReport(String projectId) {
        List<Task> tasks = projectTasks.get(projectId);
        int total = tasks.size();
        int completed = (int) tasks.stream().filter(t -> "COMPLETED".equals(t.getStatus())).count();
        int inProgress = (int) tasks.stream().filter(t -> "IN_PROGRESS".equals(t.getStatus())).count();
        
        double overallProgress = tasks.stream().mapToInt(Task::getProgress).average().orElse(0);
        
        return new ProjectReport(total, completed, inProgress, overallProgress, "2024-05-15");
    }
}
