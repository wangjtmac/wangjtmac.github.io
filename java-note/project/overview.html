实践项目是将理论知识转化为实际技能的重要桥梁。通过动手实现各种项目，我们不仅能够巩固Java基础知识、算法与数据结构概念，更能培养解决实际问题的能力和工程思维。本系列实践项目从简单的基础练习开始，逐步深入到复杂的系统设计，涵盖了Java开发的各个方面。每个项目都经过精心设计，既注重技术深度，又兼顾实用性，帮助学习者在实践中成长。通过这些项目的实现，您将获得扎实的编程基础、良好的代码习惯、系统的设计思维，以及面对复杂问题的分析和解决能力。

// 排序性能测试器
class SortingPerformanceTester {
    public void compareAlgorithms(int size) {
        int[] data = generateRandomArray(size);
        
        // 冒泡排序
        int[] bubbleData = data.clone();
        long startTime = System.nanoTime();
        bubbleSort(bubbleData);
        long bubbleTime = System.nanoTime() - startTime;
        
        // 快速排序
        int[] quickData = data.clone();
        startTime = System.nanoTime();
        quickSort(quickData, 0, quickData.length - 1);
        long quickTime = System.nanoTime() - startTime;
        
        System.out.println("数组大小: " + size);
        System.out.println("冒泡排序耗时: " + bubbleTime + "ns");
        System.out.println("快速排序耗时: " + quickTime + "ns");
        System.out.println("性能提升: " + (bubbleTime / (double) quickTime) + "倍");
    }
    
    private int[] generateRandomArray(int size) {
        Random random = new Random();
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = random.nextInt(1000);
        }
        return array;
    }
    
    private void bubbleSort(int[] array) {
        for (int i = 0; i < array.length - 1; i++) {
            for (int j = 0; j < array.length - 1 - i; j++) {
                if (array[j] > array[j + 1]) {
                    int temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
    }
    
    private void quickSort(int[] array, int low, int high) {
        if (low < high) {
            int pivotIndex = partition(array, low, high);
            quickSort(array, low, pivotIndex - 1);
            quickSort(array, pivotIndex + 1, high);
        }
    }
    
    private int partition(int[] array, int low, int high) {
        int pivot = array[high];
        int i = low - 1;
        
        for (int j = low; j < high; j++) {
            if (array[j] <= pivot) {
                i++;
                int temp = array[i];
                array[i] = array[j];
                array[j] = temp;
            }
        }
        
        int temp = array[i + 1];
        array[i + 1] = array[high];
        array[high] = temp;
        
        return i + 1;
    }
}

// 自定义栈
class CustomStack<T> {
    private List<T> elements;
    
    public CustomStack() {
        elements = new ArrayList<>();
    }
    
    public void push(T element) {
        elements.add(element);
    }
    
    public T pop() {
        if (isEmpty()) {
            throw new IllegalStateException("栈为空");
        }
        return elements.remove(elements.size() - 1);
    }
    
    public T peek() {
        if (isEmpty()) {
            throw new IllegalStateException("栈为空");
        }
        return elements.get(elements.size() - 1);
    }
    
    public boolean isEmpty() {
        return elements.isEmpty();
    }
    
    public int size() {
        return elements.size();
    }
}

// 文件统计工具
class FileStatistics {
    private Map<String, Long> files;
    private long totalSize;
    
    public FileStatistics() {
        files = new HashMap<>();
        totalSize = 0;
    }
    
    public void addFile(String filename, long size) {
        files.put(filename, size);
        totalSize += size;
    }
    
    public void printStatistics() {
        System.out.println("文件统计信息:");
        System.out.println("文件总数: " + files.size());
        System.out.println("总大小: " + formatSize(totalSize));
        System.out.println("平均大小: " + formatSize(totalSize / files.size()));
        
        System.out.println("\n文件列表:");
        files.entrySet().stream()
            .sorted(Map.Entry.<String, Long>comparingByValue().reversed())
            .forEach(entry -> 
                System.out.println("  " + entry.getKey() + ": " + formatSize(entry.getValue()))
            );
    }
    
    private String formatSize(long bytes) {
        if (bytes < 1024) return bytes + " B";
        if (bytes < 1024 * 1024) return (bytes / 1024) + " KB";
        return (bytes / (1024 * 1024)) + " MB";
    }
}

// 数据库连接池
class DatabaseConnectionPool {
    private Queue<Connection> availableConnections;
    private Set<Connection> usedConnections;
    private int maxPoolSize;

    public DatabaseConnectionPool() {
        this.maxPoolSize = 10;
        this.availableConnections = new LinkedList<>();
        this.usedConnections = new HashSet<>();
        initializePool();
    }

    private void initializePool() {
        for (int i = 0; i < maxPoolSize; i++) {
            availableConnections.offer(new MockConnection("conn-" + i));
        }
    }

    public synchronized Connection getConnection() {
        if (availableConnections.isEmpty()) {
            return null; // 或等待
        }

        Connection connection = availableConnections.poll();
        usedConnections.add(connection);
        return connection;
    }

    public synchronized void releaseConnection(Connection connection) {
        if (usedConnections.remove(connection)) {
            availableConnections.offer(connection);
        }
    }

    public void demonstrateUsage() {
        System.out.println("数据库连接池演示:");
        System.out.println("初始可用连接: " + availableConnections.size());

        // 获取连接
        Connection conn1 = getConnection();
        Connection conn2 = getConnection();

        System.out.println("获取2个连接后:");
        System.out.println("可用连接: " + availableConnections.size());
        System.out.println("使用中连接: " + usedConnections.size());

        // 释放连接
        releaseConnection(conn1);

        System.out.println("释放1个连接后:");
        System.out.println("可用连接: " + availableConnections.size());
        System.out.println("使用中连接: " + usedConnections.size());
    }

    // 模拟数据库连接
    private static class MockConnection implements Connection {
        private String id;

        public MockConnection(String id) {
            this.id = id;
        }

        @Override
        public String toString() {
            return "Connection[" + id + "]";
        }
    }

    private interface Connection {
        // 简化的连接接口
    }
}

// 线程池任务处理器
class ThreadPoolTaskProcessor {
    private ExecutorService executor;

    public ThreadPoolTaskProcessor() {
        this.executor = Executors.newFixedThreadPool(4);
    }

    public void demonstrateProcessing() {
        System.out.println("线程池任务处理演示:");

        List<Future<String>> futures = new ArrayList<>();

        // 提交任务
        for (int i = 1; i <= 8; i++) {
            final int taskId = i;
            Future<String> future = executor.submit(() -> {
                Thread.sleep(1000); // 模拟耗时操作
                return "任务" + taskId + "完成 - 线程: " + Thread.currentThread().getName();
            });
            futures.add(future);
        }

        System.out.println("已提交8个任务到4个线程的线程池");

        // 获取结果
        try {
            for (Future<String> future : futures) {
                System.out.println(future.get());
            }
        } catch (Exception e) {
            e.printStackTrace();
        }

        executor.shutdown();
        System.out.println("所有任务完成，线程池关闭");
    }
}

// 简单依赖注入容器
class SimpleDIContainer {
    private Map<Class<?>, Object> instances;
    private Map<Class<?>, Class<?>> bindings;

    public SimpleDIContainer() {
        instances = new HashMap<>();
        bindings = new HashMap<>();
        setupBindings();
    }

    private void setupBindings() {
        // 模拟服务绑定
        bindings.put(UserService.class, UserServiceImpl.class);
        bindings.put(EmailService.class, EmailServiceImpl.class);
    }

    @SuppressWarnings("unchecked")
    public <T> T getInstance(Class<T> type) {
        if (instances.containsKey(type)) {
            return (T) instances.get(type);
        }

        Class<?> implClass = bindings.get(type);
        if (implClass == null) {
            implClass = type;
        }

        try {
            T instance = (T) implClass.getDeclaredConstructor().newInstance();
            instances.put(type, instance);
            return instance;
        } catch (Exception e) {
            throw new RuntimeException("无法创建实例: " + type, e);
        }
    }

    public void demonstrateFeatures() {
        System.out.println("依赖注入容器演示:");

        UserService userService = getInstance(UserService.class);
        EmailService emailService = getInstance(EmailService.class);

        System.out.println("获取服务实例:");
        System.out.println("UserService: " + userService.getClass().getSimpleName());
        System.out.println("EmailService: " + emailService.getClass().getSimpleName());

        // 验证单例
        UserService userService2 = getInstance(UserService.class);
        System.out.println("单例验证: " + (userService == userService2));

        System.out.println("容器特性:");
        System.out.println("- 依赖注入");
        System.out.println("- 单例管理");
        System.out.println("- 接口绑定");
        System.out.println("- 生命周期管理");
    }

    // 模拟服务接口和实现
    interface UserService {}
    interface EmailService {}

    static class UserServiceImpl implements UserService {}
    static class EmailServiceImpl implements EmailService {}
}
