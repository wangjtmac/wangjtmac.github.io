    public double evaluateExpression(String expression) {
        // ç®€åŒ–å®ç°ï¼Œå®é™…é¡¹ç›®ä¸­å¯ä»¥ä½¿ç”¨æ›´å¤æ‚çš„è§£æå™¨
        expression = expression.replaceAll("\\s+", "");
        
        // å¤„ç†ä¹˜é™¤æ³•
        while (expression.contains("*") || expression.contains("/")) {
            for (int i = 0; i < expression.length(); i++) {
                char c = expression.charAt(i);
                if (c == '*' || c == '/') {
                    // æ‰¾åˆ°æ“ä½œæ•°
                    int start = i - 1;
                    while (start > 0 && Character.isDigit(expression.charAt(start - 1))) {
                        start--;
                    }
                    
                    int end = i + 1;
                    while (end < expression.length() && Character.isDigit(expression.charAt(end))) {
                        end++;
                    }
                    
                    double left = Double.parseDouble(expression.substring(start, i));
                    double right = Double.parseDouble(expression.substring(i + 1, end));
                    double result = (c == '*') ? left * right : left / right;
                    
                    expression = expression.substring(0, start) + result + expression.substring(end);
                    break;
                }
            }
        }
        
        // å¤„ç†åŠ å‡æ³•
        double result = 0;
        double current = 0;
        char operation = '+';
        
        for (int i = 0; i <= expression.length(); i++) {
            if (i == expression.length() || expression.charAt(i) == '+' || expression.charAt(i) == '-') {
                if (operation == '+') {
                    result += current;
                } else {
                    result -= current;
                }
                
                if (i < expression.length()) {
                    operation = expression.charAt(i);
                }
                current = 0;
            } else {
                current = current * 10 + (expression.charAt(i) - '0');
            }
        }
        
        return result;
    }

class NumberGuessingGame {
    private int targetNumber;
    private int attempts;
    private int maxAttempts;
    private boolean gameWon;
    
    public NumberGuessingGame() {
        this(1, 100, 10);
    }
    
    public NumberGuessingGame(int min, int max, int maxAttempts) {
        Random random = new Random();
        this.targetNumber = random.nextInt(max - min + 1) + min;
        this.maxAttempts = maxAttempts;
        this.attempts = 0;
        this.gameWon = false;
    }
    
    public String makeGuess(int guess) {
        if (gameWon || attempts >= maxAttempts) {
            return "æ¸¸æˆå·²ç»“æŸ";
        }
        
        attempts++;
        
        if (guess == targetNumber) {
            gameWon = true;
            return "æ­å–œï¼çŒœå¯¹äº†ï¼ç”¨äº† " + attempts + " æ¬¡å°è¯•ã€‚";
        } else if (guess < targetNumber) {
            return "å¤ªå°äº†ï¼è¿˜æœ‰ " + (maxAttempts - attempts) + " æ¬¡æœºä¼šã€‚";
        } else {
            return "å¤ªå¤§äº†ï¼è¿˜æœ‰ " + (maxAttempts - attempts) + " æ¬¡æœºä¼šã€‚";
        }
    }
    
    public int getTargetNumber() { return targetNumber; }
    
    public String getStatistics() {
        return String.format("å°è¯•æ¬¡æ•°: %d/%d, æ¸¸æˆçŠ¶æ€: %s", 
                           attempts, maxAttempts, gameWon ? "èƒœåˆ©" : "è¿›è¡Œä¸­");
    }
}

class PasswordGenerator {
    private static final String LOWERCASE = "abcdefghijklmnopqrstuvwxyz";
    private static final String UPPERCASE = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    private static final String DIGITS = "0123456789";
    private static final String SPECIAL = "!@#$%^&*()_+-=[]{}|;:,.<>?";
    
    private Random random = new Random();
    
    public String generateSimple(int length) {
        String chars = LOWERCASE + DIGITS;
        return generateFromChars(chars, length);
    }
    
    public String generateComplex(int length) {
        String chars = LOWERCASE + UPPERCASE + DIGITS + SPECIAL;
        return generateFromChars(chars, length);
    }
    
    public String generateCustom(int length, boolean useLower, boolean useUpper, 
                                boolean useDigits, boolean useSpecial) {
        StringBuilder chars = new StringBuilder();
        if (useLower) chars.append(LOWERCASE);
        if (useUpper) chars.append(UPPERCASE);
        if (useDigits) chars.append(DIGITS);
        if (useSpecial) chars.append(SPECIAL);
        
        if (chars.length() == 0) {
            throw new IllegalArgumentException("è‡³å°‘é€‰æ‹©ä¸€ç§å­—ç¬¦ç±»å‹");
        }
        
        return generateFromChars(chars.toString(), length);
    }
    
    private String generateFromChars(String chars, int length) {
        StringBuilder password = new StringBuilder();
        for (int i = 0; i < length; i++) {
            int index = random.nextInt(chars.length());
            password.append(chars.charAt(index));
        }
        return password.toString();
    }
    
    public String checkStrength(String password) {
        int score = 0;
        
        if (password.length() >= 8) score++;
        if (password.length() >= 12) score++;
        if (password.matches(".*[a-z].*")) score++;
        if (password.matches(".*[A-Z].*")) score++;
        if (password.matches(".*[0-9].*")) score++;
        if (password.matches(".*[!@#$%^&*()_+\\-=\\[\\]{}|;:,.<>?].*")) score++;
        
        if (score <= 2) return "å¼±";
        if (score <= 4) return "ä¸­ç­‰";
        return "å¼º";
    }
}

class TemperatureConverter {
    
    public double celsiusToFahrenheit(double celsius) {
        return celsius * 9.0 / 5.0 + 32;
    }
    
    public double fahrenheitToCelsius(double fahrenheit) {
        return (fahrenheit - 32) * 5.0 / 9.0;
    }
    
    public double celsiusToKelvin(double celsius) {
        return celsius + 273.15;
    }
    
    public double kelvinToCelsius(double kelvin) {
        return kelvin - 273.15;
    }
    
    public double fahrenheitToKelvin(double fahrenheit) {
        return celsiusToKelvin(fahrenheitToCelsius(fahrenheit));
    }
    
    public double kelvinToFahrenheit(double kelvin) {
        return celsiusToFahrenheit(kelvinToCelsius(kelvin));
    }
    
    public void printConversionTable() {
        System.out.printf("%-8s %-8s %-8s%n", "Â°C", "Â°F", "K");
        System.out.println("------------------------");
        
        for (int c = -10; c <= 40; c += 10) {
            double f = celsiusToFahrenheit(c);
            double k = celsiusToKelvin(c);
            System.out.printf("%-8d %-8.1f %-8.1f%n", c, f, k);
        }
    }
