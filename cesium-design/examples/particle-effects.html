粒子系统与天气效果 ​

本示例展示如何在 Cesium 中使用粒子系统创建各种视觉效果，包括火焰、烟雾、爆炸、雨雪等天气效果。

核心功能 ​

1. 粒子系统基础 ​

  • ParticleSystem - Cesium 的粒子系统 API
  • 粒子发射器 - 控制粒子的生成位置和方向
  • 粒子生命周期 - 控制粒子的生成、运动和消亡
  • 粒子外观 - 纹理、颜色、大小等视觉属性

2. 火焰效果 ​

  • 火焰粒子 - 模拟火焰的燃烧效果
  • 颜色渐变 - 从黄色到红色的渐变
  • 上升运动 - 粒子向上飘动
  • 随机扰动 - 增加真实感

3. 烟雾效果 ​

  • 烟雾粒子 - 模拟烟雾扩散
  • 透明度变化 - 逐渐变淡消失
  • 扩散运动 - 向四周扩散
  • 风力影响 - 模拟风的作用

4. 天气效果 ​

  • 雨效果 - 模拟降雨
  • 雪效果 - 模拟降雪
  • 雾效果 - 模拟雾气

5. 爆炸效果 ​

  • 爆炸粒子 - 模拟爆炸瞬间
  • 冲击波 - 向外扩散的粒子
  • 火花飞溅 - 高速飞出的粒子

技术要点 ​

ParticleSystem 基础 ​

javascript
// 创建粒子系统
const particleSystem = viewer.scene.primitives.add(new Cesium.ParticleSystem({
  // 粒子图像
  image: 'path/to/particle.png',
  
  // 发射器位置
  modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(
    Cesium.Cartesian3.fromDegrees(116.4, 39.9, 100)
  ),
  
  // 发射器类型
  emitter: new Cesium.CircleEmitter(2.0), // 圆形发射器，半径2米
  
  // 粒子生命周期
  lifetime: 5.0, // 粒子存活时间（秒）
  
  // 发射速率
  emissionRate: 50, // 每秒发射50个粒子
  
  // 粒子初始速度
  speed: 5.0, // 5米/秒
  minimumSpeed: 3.0,
  maximumSpeed: 7.0,
  
  // 粒子大小
  imageSize: new Cesium.Cartesian2(20, 20),
  minimumImageSize: new Cesium.Cartesian2(10, 10),
  maximumImageSize: new Cesium.Cartesian2(30, 30),
  
  // 粒子颜色
  startColor: Cesium.Color.WHITE,
  endColor: Cesium.Color.WHITE.withAlpha(0),
  
  // 粒子缩放
  startScale: 1.0,
  endScale: 3.0,
  
  // 重力加速度
  forces: [
    function(particle) {
      return new Cesium.Cartesian3(0, 0, -9.8); // 重力
    }
  ]
}));

火焰效果 ​

javascript
// 创建火焰粒子系统
function createFireEffect(position) {
  return new Cesium.ParticleSystem({
    image: createFireTexture(),
    
    modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
    
    emitter: new Cesium.CircleEmitter(0.5),
    
    lifetime: 2.0,
    emissionRate: 50,
    
    speed: 3.0,
    minimumSpeed: 2.0,
    maximumSpeed: 4.0,
    
    imageSize: new Cesium.Cartesian2(20, 20),
    minimumImageSize: new Cesium.Cartesian2(15, 15),
    maximumImageSize: new Cesium.Cartesian2(25, 25),
    
    // 火焰颜色：从黄色到红色再到透明
    startColor: Cesium.Color.YELLOW,
    endColor: Cesium.Color.RED.withAlpha(0),
    
    startScale: 1.0,
    endScale: 2.0,
    
    // 向上的力
    forces: [
      function(particle) {
        const age = 1.0 - particle.normalizedAge;
        return new Cesium.Cartesian3(0, 0, age * 5.0);
      }
    ]
  });
}

// 创建火焰纹理
function createFireTexture() {
  const canvas = document.createElement('canvas');
  canvas.width = 32;
  canvas.height = 32;
  const ctx = canvas.getContext('2d');
  
  // 创建径向渐变
  const gradient = ctx.createRadialGradient(16, 16, 0, 16, 16, 16);
  gradient.addColorStop(0, 'rgba(255, 255, 255, 1)');
  gradient.addColorStop(0.5, 'rgba(255, 200, 0, 0.8)');
  gradient.addColorStop(1, 'rgba(255, 100, 0, 0)');
  
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, 32, 32);
  
  return canvas.toDataURL();
}

烟雾效果 ​

javascript
// 创建烟雾粒子系统
function createSmokeEffect(position) {
  return new Cesium.ParticleSystem({
    image: createSmokeTexture(),
    
    modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
    
    emitter: new Cesium.CircleEmitter(1.0),
    
    lifetime: 5.0,
    emissionRate: 20,
    
    speed: 2.0,
    minimumSpeed: 1.0,
    maximumSpeed: 3.0,
    
    imageSize: new Cesium.Cartesian2(30, 30),
    minimumImageSize: new Cesium.Cartesian2(20, 20),
    maximumImageSize: new Cesium.Cartesian2(50, 50),
    
    // 烟雾颜色：从灰色到透明
    startColor: Cesium.Color.GRAY.withAlpha(0.8),
    endColor: Cesium.Color.GRAY.withAlpha(0),
    
    startScale: 1.0,
    endScale: 4.0,
    
    // 向上飘动并扩散
    forces: [
      function(particle) {
        const age = 1.0 - particle.normalizedAge;
        return new Cesium.Cartesian3(
          Math.random() - 0.5, // 随机水平扰动
          Math.random() - 0.5,
          age * 2.0 // 向上
        );
      }
    ]
  });
}

// 创建烟雾纹理
function createSmokeTexture() {
  const canvas = document.createElement('canvas');
  canvas.width = 64;
  canvas.height = 64;
  const ctx = canvas.getContext('2d');
  
  // 创建径向渐变
  const gradient = ctx.createRadialGradient(32, 32, 0, 32, 32, 32);
  gradient.addColorStop(0, 'rgba(200, 200, 200, 1)');
  gradient.addColorStop(0.5, 'rgba(150, 150, 150, 0.5)');
  gradient.addColorStop(1, 'rgba(100, 100, 100, 0)');
  
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, 64, 64);
  
  return canvas.toDataURL();
}

雨效果 ​

javascript
// 创建雨效果
function createRainEffect(position) {
  return new Cesium.ParticleSystem({
    image: createRainTexture(),
    
    modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
    
    // 盒子发射器，覆盖一定区域
    emitter: new Cesium.BoxEmitter(new Cesium.Cartesian3(50, 50, 10)),
    
    lifetime: 3.0,
    emissionRate: 1000,
    
    speed: 20.0,
    
    imageSize: new Cesium.Cartesian2(2, 20), // 细长的雨滴
    
    startColor: Cesium.Color.LIGHTBLUE.withAlpha(0.7),
    endColor: Cesium.Color.LIGHTBLUE.withAlpha(0),
    
    // 重力加速度
    forces: [
      function(particle) {
        return new Cesium.Cartesian3(0, 0, -30.0); // 快速下落
      }
    ]
  });
}

// 创建雨滴纹理
function createRainTexture() {
  const canvas = document.createElement('canvas');
  canvas.width = 4;
  canvas.height = 20;
  const ctx = canvas.getContext('2d');
  
  const gradient = ctx.createLinearGradient(0, 0, 0, 20);
  gradient.addColorStop(0, 'rgba(255, 255, 255, 0)');
  gradient.addColorStop(0.5, 'rgba(255, 255, 255, 1)');
  gradient.addColorStop(1, 'rgba(255, 255, 255, 0)');
  
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, 4, 20);
  
  return canvas.toDataURL();
}

雪效果 ​

javascript
// 创建雪效果
function createSnowEffect(position) {
  return new Cesium.ParticleSystem({
    image: createSnowTexture(),
    
    modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
    
    emitter: new Cesium.BoxEmitter(new Cesium.Cartesian3(50, 50, 10)),
    
    lifetime: 10.0,
    emissionRate: 200,
    
    speed: 2.0,
    minimumSpeed: 1.0,
    maximumSpeed: 3.0,
    
    imageSize: new Cesium.Cartesian2(10, 10),
    
    startColor: Cesium.Color.WHITE,
    endColor: Cesium.Color.WHITE.withAlpha(0),
    
    // 缓慢下落并飘动
    forces: [
      function(particle) {
        return new Cesium.Cartesian3(
          Math.sin(Date.now() * 0.001) * 0.5, // 左右飘动
          Math.cos(Date.now() * 0.001) * 0.5,
          -2.0 // 缓慢下落
        );
      }
    ]
  });
}

// 创建雪花纹理
function createSnowTexture() {
  const canvas = document.createElement('canvas');
  canvas.width = 16;
  canvas.height = 16;
  const ctx = canvas.getContext('2d');
  
  const gradient = ctx.createRadialGradient(8, 8, 0, 8, 8, 8);
  gradient.addColorStop(0, 'rgba(255, 255, 255, 1)');
  gradient.addColorStop(1, 'rgba(255, 255, 255, 0)');
  
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, 16, 16);
  
  return canvas.toDataURL();
}

爆炸效果 ​

javascript
// 创建爆炸效果
function createExplosionEffect(position) {
  const particleSystem = new Cesium.ParticleSystem({
    image: createExplosionTexture(),
    
    modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
    
    emitter: new Cesium.SphereEmitter(0.1), // 球形发射器
    
    lifetime: 1.0,
    emissionRate: 0, // 初始不发射
    
    bursts: [
      // 爆炸瞬间发射大量粒子
      new Cesium.ParticleBurst({
        time: 0.0,
        minimum: 100,
        maximum: 200
      })
    ],
    
    speed: 20.0,
    minimumSpeed: 10.0,
    maximumSpeed: 30.0,
    
    imageSize: new Cesium.Cartesian2(20, 20),
    
    // 爆炸颜色：从白色到橙色到黑色
    startColor: Cesium.Color.WHITE,
    endColor: Cesium.Color.BLACK.withAlpha(0),
    
    startScale: 1.0,
    endScale: 3.0,
    
    // 重力影响
    forces: [
      function(particle) {
        return new Cesium.Cartesian3(0, 0, -9.8);
      }
    ]
  });
  
  // 1秒后停止粒子系统
  setTimeout(() => {
    particleSystem.emissionRate = 0;
  }, 1000);
  
  return particleSystem;
}

// 创建爆炸纹理
function createExplosionTexture() {
  const canvas = document.createElement('canvas');
  canvas.width = 32;
  canvas.height = 32;
  const ctx = canvas.getContext('2d');
  
  const gradient = ctx.createRadialGradient(16, 16, 0, 16, 16, 16);
  gradient.addColorStop(0, 'rgba(255, 255, 255, 1)');
  gradient.addColorStop(0.3, 'rgba(255, 150, 0, 0.8)');
  gradient.addColorStop(0.6, 'rgba(255, 50, 0, 0.5)');
  gradient.addColorStop(1, 'rgba(50, 50, 50, 0)');
  
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, 32, 32);
  
  return canvas.toDataURL();
}

发射器类型 ​

Cesium 提供多种粒子发射器类型：

javascript
// 1. 圆形发射器
new Cesium.CircleEmitter(radius)

// 2. 圆锥发射器
new Cesium.ConeEmitter(angle)

// 3. 盒子发射器
new Cesium.BoxEmitter(dimensions)

// 4. 球形发射器
new Cesium.SphereEmitter(radius)
