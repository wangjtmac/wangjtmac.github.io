数据样式定制示例 ​

这是一个演示如何对 Cesium 中的地理数据进行高级样式定制的综合示例。

功能说明 ​

  • 演示基于数据属性的动态样式
  • 实现条件样式和分级符号化
  • 支持颜色渐变和热力图效果
  • 提供样式模板和预设方案
  • 实现实时样式调整和预览

核心 API ​

  • Cesium.CallbackProperty - 动态属性回调
  • Cesium.ColorBlendMode - 颜色混合模式
  • Cesium.Material - 材质系统
  • Cesium.Color.lerp() - 颜色插值

  methods: {
    async loadCesium() {
      return new Promise((resolve, reject) => {
        if (window.Cesium) {
          resolve()
          return

        const link = document.createElement('link')
        link.rel = 'stylesheet'

        script.onerror = () => reject(new Error('Cesium 加载失败'))
        document.head.appendChild(script)

    async loadSampleData(Cesium) {
      this.sampleData.forEach(data => {
        const entity = this.dataSource.entities.add({
          name: data.name,
          position: Cesium.Cartesian3.fromDegrees(data.longitude, data.latitude),

            pixelSize: 15,
            color: Cesium.Color.YELLOW,
            outlineColor: Cesium.Color.BLACK,
            outlineWidth: 2,
            heightReference: Cesium.HeightReference.CLAMP_TO_GROUND
          },
          label: {
            text: data.name,
            font: '12pt sans-serif',
            pixelOffset: new Cesium.Cartesian2(0, -40),
            fillColor: Cesium.Color.WHITE,
            outlineColor: Cesium.Color.BLACK,
            outlineWidth: 2,
            style: Cesium.LabelStyle.FILL_AND_OUTLINE
          },
          properties: {
            population: data.population,
            gdp: data.gdp,
            area: data.area,
            aqi: data.aqi
          }
        })

      this.dataCount = this.sampleData.length
      this.updateStats()

    applyTemplate() {
      const template = this.styleTemplates[this.selectedTemplate]
      if (!template) return

      this.currentTemplate = template
      this.currentProperty = template.property
      this.styleType = template.type

      // 更新图例
      this.updateLegend(template)

      // 应用样式
      this.updateStyle()

      // 更新统计信息
      this.updateStats()
    },

    async updateStyle() {
      if (!this.viewer || !this.dataSource) return

      const Cesium = await import('cesium')
      const template = this.currentTemplate
      const entities = this.dataSource.entities.values

      switch (this.styleType) {
        case 'gradient':
          this.applyGradientStyle(Cesium, entities, template)
          break
        case 'classification':
          this.applyClassificationStyle(Cesium, entities, template)
          break
        case 'heatmap':
          this.applyHeatmapStyle(Cesium, entities, template)
          break
      }
    },

    applyGradientStyle(Cesium, entities, template) {
      // 获取属性值范围
      const values = entities.map(e => e.properties[template.property]?.getValue() || 0)
      const minValue = Math.min(...values)
      const maxValue = Math.max(...values)

      // 创建颜色渐变函数
      const colorGradient = this.createColorGradient(Cesium, minValue, maxValue, template.colorStops)

      entities.forEach(entity => {
        const value = entity.properties[template.property]?.getValue() || 0

        if (entity.point) {
          const color = colorGradient(value)
          const size = this.interpolateSize(value, minValue, maxValue, template.sizeRange || [8, 24])

          if (this.enableAnimation) {
            // 动态颜色效果
            entity.point.color = new Cesium.CallbackProperty((time, result) => {
              const factor = Math.sin(Cesium.JulianDate.secondsDifference(time, this.viewer.clock.currentTime) * 0.002) * 0.3 + 0.7
              return Cesium.Color.lerp(color, Cesium.Color.WHITE, 1 - factor, result)
            }, false)

            // 动态大小效果
            entity.point.pixelSize = new Cesium.CallbackProperty((time) => {
              const factor = Math.sin(Cesium.JulianDate.secondsDifference(time, this.viewer.clock.currentTime) * 0.003) * 0.2 + 1
              return size * factor
            }, false)
          } else {
            entity.point.color = color
            entity.point.pixelSize = size
          }
        }
      })
    },

    applyClassificationStyle(Cesium, entities, template) {
      entities.forEach(entity => {
        const value = entity.properties[template.property]?.getValue() || 0

        const styleClass = template.classes.find(cls => value >= cls.min && value < cls.max)
        if (styleClass && entity.point) {
          const color = Cesium.Color.fromCssColorString(styleClass.color)

          if (this.enableAnimation) {
            entity.point.color = new Cesium.CallbackProperty((time, result) => {
              const factor = Math.sin(Cesium.JulianDate.secondsDifference(time, this.viewer.clock.currentTime) * 0.002) * 0.2 + 0.8
              return Cesium.Color.lerp(color, Cesium.Color.WHITE, 1 - factor, result)
            }, false)
          } else {
            entity.point.color = color
          }

          entity.point.pixelSize = styleClass.size
        }
      })
    },

    applyHeatmapStyle(Cesium, entities, template) {
      // 清除现有的热力图效果
      this.clearHeatmapEffects()

      entities.forEach(entity => {
        const value = entity.properties[template.property]?.getValue() || 0
        const position = entity.position.getValue()

        // 创建热力图圆圈
        const radius = this.interpolateSize(value, 0, 200, [20000, 100000])
        const alpha = Math.min(value / 200 * 0.6, 0.6)

        const heatCircle = this.dataSource.entities.add({
          position: position,
          ellipse: {
            semiMajorAxis: radius,
            semiMinorAxis: radius,
            material: new Cesium.ColorMaterialProperty(
              Cesium.Color.RED.withAlpha(alpha)
            ),
            height: 0,
            outline: false
          }
        })

        // 标记为热力图效果，便于清理
        heatCircle.isHeatmapEffect = true

        // 设置原始点的样式
        if (entity.point) {
          entity.point.color = this.getHeatmapColor(Cesium, value, template.colorStops)
          entity.point.pixelSize = 8
        }
      })
    },

    createColorGradient(Cesium, minValue, maxValue, colorStops) {
      return (value) => {
        const normalizedValue = (value - minValue) / (maxValue - minValue)

        for (let i = 0; i < colorStops.length - 1; i++) {
          const currentStop = colorStops[i]
          const nextStop = colorStops[i + 1]

          const currentPos = (currentStop.value - minValue) / (maxValue - minValue)
          const nextPos = (nextStop.value - minValue) / (maxValue - minValue)

          if (normalizedValue >= currentPos && normalizedValue <= nextPos) {
            const localPosition = (normalizedValue - currentPos) / (nextPos - currentPos)
            const currentColor = Cesium.Color.fromCssColorString(currentStop.color)
            const nextColor = Cesium.Color.fromCssColorString(nextStop.color)
            return Cesium.Color.lerp(currentColor, nextColor, localPosition, new Cesium.Color())
          }
        }

        return Cesium.Color.fromCssColorString(colorStops[colorStops.length - 1].color)
      }
    },

    getHeatmapColor(Cesium, value, colorStops) {
      for (let i = 0; i < colorStops.length - 1; i++) {
        if (value >= colorStops[i].value && value < colorStops[i + 1].value) {
          return Cesium.Color.fromCssColorString(colorStops[i].color)
        }
      }
      return Cesium.Color.fromCssColorString(colorStops[colorStops.length - 1].color)
    },

    interpolateSize(value, minValue, maxValue, sizeRange) {
      const normalizedValue = (value - minValue) / (maxValue - minValue)
      return sizeRange[0] + normalizedValue * (sizeRange[1] - sizeRange[0])
    },

    clearHeatmapEffects() {
      if (!this.dataSource) return

      const entitiesToRemove = []
      this.dataSource.entities.values.forEach(entity => {
        if (entity.isHeatmapEffect) {
          entitiesToRemove.push(entity)
        }
      })

      entitiesToRemove.forEach(entity => {
        this.dataSource.entities.remove(entity)
      })
    },

    async toggleAnimation() {
      this.updateStyle()
    },

    resetStyle() {
      this.selectedTemplate = 'population'
      this.styleType = 'gradient'
      this.enableAnimation = false
      this.applyTemplate()
    },

    exportConfig() {
      const config = {
        template: this.currentTemplate,
        styleType: this.styleType,
        enableAnimation: this.enableAnimation,
        timestamp: new Date().toISOString(),
        version: '1.0'
      }

      const blob = new Blob([JSON.stringify(config, null, 2)], {
        type: 'application/json'
      })

      const url = URL.createObjectURL(blob)
      const a = document.createElement('a')
      a.href = url
      a.download = 'cesium-style-config.json'
      a.click()

      URL.revokeObjectURL(url)
    },

    updateLegend(template) {
      this.legendItems = []

      if (template.colorStops) {
        this.legendItems = template.colorStops.map(stop => ({
          color: stop.color,
          label: stop.label || `${stop.value}`
        }))
      } else if (template.classes) {
        this.legendItems = template.classes.map(cls => ({
          color: cls.color,
          label: cls.label || `${cls.min}-${cls.max}`
        }))
      }
    },

    updateStats() {
      if (!this.currentTemplate.property) return

      const values = this.sampleData.map(data => data[this.currentTemplate.property])
      const minValue = Math.min(...values)
      const maxValue = Math.max(...values)

      this.valueRange = `${minValue.toLocaleString()} - ${maxValue.toLocaleString()}`
    },

    setupClickHandler(Cesium) {
      this.viewer.selectedEntityChanged.addEventListener((selectedEntity) => {
        if (selectedEntity && selectedEntity.properties) {
          const properties = selectedEntity.properties
          const name = selectedEntity.name

          // 创建信息窗口内容
          let description = `<h3>${name}</h3><table>`
          description += `<tr><td>人口：</td><td>${properties.population?.getValue().toLocaleString()} 人</td></tr>`
          description += `<tr><td>GDP：</td><td>${properties.gdp?.getValue().toLocaleString()} 亿元</td></tr>`
          description += `<tr><td>面积：</td><td>${properties.area?.getValue().toLocaleString()} km²</td></tr>`
          description += `<tr><td>AQI：</td><td>${properties.aqi?.getValue()}</td></tr>`
          description += `</table>`

          selectedEntity.description = description
        }
      })
    }
  }
}
</script>

2. 颜色渐变映射 ​

javascript
// 创建颜色渐变函数
const createColorGradient = (minValue, maxValue, colorStops) => {
  return (value) => {
    // 标准化值到 0-1 范围
    const normalizedValue = (value - minValue) / (maxValue - minValue)
    
    // 找到对应的颜色区间
    for (let i = 0; i < colorStops.length - 1; i++) {
      const currentStop = colorStops[i]
      const nextStop = colorStops[i + 1]
      
      if (normalizedValue >= currentStop.position && normalizedValue <= nextStop.position) {
        // 在两个颜色之间插值
        const localPosition = (normalizedValue - currentStop.position) / 
                             (nextStop.position - currentStop.position)
        return Cesium.Color.lerp(currentStop.color, nextStop.color, localPosition, new Cesium.Color())
      }
    }
    
    return colorStops[colorStops.length - 1].color
  }
}

// 使用示例
const populationGradient = createColorGradient(0, 30000000, [
  { position: 0.0, color: Cesium.Color.GREEN },
  { position: 0.3, color: Cesium.Color.YELLOW },
  { position: 0.7, color: Cesium.Color.ORANGE },
  { position: 1.0, color: Cesium.Color.RED }
])

4. 分级符号化 ​

javascript
// 根据数据值分级设置符号大小
const createSizeClassification = (values, sizes) => {
  // 计算分位数
  const sortedValues = [...values].sort((a, b) => a - b)
  const quantiles = []
  
  for (let i = 0; i < sizes.length; i++) {
    const index = Math.floor((i / (sizes.length - 1)) * (sortedValues.length - 1))
    quantiles.push(sortedValues[index])
  }
  
  return (value) => {
    for (let i = quantiles.length - 1; i >= 0; i--) {
      if (value >= quantiles[i]) {
        return sizes[i]
      }
    }
    return sizes[0]
  }
}

// 应用分级符号化
const sizeClassifier = createSizeClassification(populationValues, [8, 12, 16, 20, 24])
entity.point.pixelSize = sizeClassifier(population)

2. 样式应用引擎 ​

javascript
class StyleEngine {
  constructor(viewer) {
    this.viewer = viewer
    this.currentStyle = null
  }
  
  applyTemplate(entities, templateName) {
    const template = styleTemplates[templateName]
    if (!template) return
    
    this.currentStyle = template
    
    switch (template.type) {
      case 'gradient':
        this.applyGradientStyle(entities, template)
        break
      case 'classification':
        this.applyClassificationStyle(entities, template)
        break
      case 'heatmap':
        this.applyHeatmapStyle(entities, template)
        break
    }
  }
  
  applyGradientStyle(entities, template) {
    const values = entities.map(e => e.properties[template.property]?.getValue() || 0)
    const minValue = Math.min(...values)
    const maxValue = Math.max(...values)
    
    const colorGradient = this.createColorGradient(minValue, maxValue, template.colorStops)
    
    entities.forEach(entity => {
      const value = entity.properties[template.property]?.getValue() || 0
      
      if (entity.point) {
        entity.point.color = colorGradient(value)
        entity.point.pixelSize = this.interpolateSize(value, minValue, maxValue, template.sizeRange)
      }
    })
  }
  
  applyClassificationStyle(entities, template) {
    entities.forEach(entity => {
      const value = entity.properties[template.property]?.getValue() || 0
      
      const styleClass = template.classes.find(cls => value >= cls.min && value < cls.max)
      if (styleClass && entity.point) {
        entity.point.color = styleClass.color
        entity.point.pixelSize = styleClass.size
      }
    })
  }
  
  interpolateSize(value, minValue, maxValue, sizeRange) {
    const normalizedValue = (value - minValue) / (maxValue - minValue)
    return sizeRange[0] + normalizedValue * (sizeRange[1] - sizeRange[0])
  }
}

1. 样式缓存 ​

javascript
class StyleCache {
  constructor() {
    this.cache = new Map()
  }
  
  getColor(value, gradient) {
    const key = `${value}_${gradient.id}`
    
    if (!this.cache.has(key)) {
      const color = gradient.getColor(value)
      this.cache.set(key, color)
    }
    
    return this.cache.get(key)
  }
  
  clear() {
    this.cache.clear()
  }
}
