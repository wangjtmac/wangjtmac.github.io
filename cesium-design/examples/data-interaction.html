  • 点击查看实体属性信息
  • 弹窗显示详细数据
  • 实体高亮和选择效果
  • 属性面板动态更新
  • 多种交互模式切换
  • 空间查询和筛选功能

核心 API ​

  • Cesium.ScreenSpaceEventHandler - 屏幕空间事件处理
  • Cesium.Scene.pick() - 实体拾取
  • Cesium.InfoBox - 信息框显示

// 实体高亮管理器
class EntityHighlighter {
  constructor(viewer) {
    this.viewer = viewer
    this.highlightedEntities = new Set()
    this.originalStyles = new Map()
  }

  highlightEntity(entity) {
    if (this.highlightedEntities.has(entity)) return

    this.saveOriginalStyle(entity)

    if (entity.point) {
      entity.point.color = window.Cesium.Color.YELLOW
      entity.point.pixelSize = (entity.point.pixelSize?.getValue() || 15) * 1.5
      entity.point.outlineColor = window.Cesium.Color.RED
      entity.point.outlineWidth = 3
    }

    this.highlightedEntities.add(entity)
  }

  unhighlightEntity(entity) {
    if (!this.highlightedEntities.has(entity)) return

    this.restoreOriginalStyle(entity)
    this.highlightedEntities.delete(entity)
  }

  clearAllHighlights() {
    this.highlightedEntities.forEach(entity => {
      this.restoreOriginalStyle(entity)
    })
    this.highlightedEntities.clear()
  }

  saveOriginalStyle(entity) {
    const style = {}

    if (entity.point) {
      style.point = {
        color: entity.point.color?.getValue(),
        pixelSize: entity.point.pixelSize?.getValue(),
        outlineColor: entity.point.outlineColor?.getValue(),
        outlineWidth: entity.point.outlineWidth?.getValue()
      }
    }

    this.originalStyles.set(entity, style)
  }

  restoreOriginalStyle(entity) {
    const style = this.originalStyles.get(entity)
    if (!style) return

    if (style.point && entity.point) {
      entity.point.color = style.point.color || window.Cesium.Color.BLUE
      entity.point.pixelSize = style.point.pixelSize || 15
      entity.point.outlineColor = style.point.outlineColor || window.Cesium.Color.WHITE
      entity.point.outlineWidth = style.point.outlineWidth || 2
    }

    this.originalStyles.delete(entity)
  }
}

// 多选管理器
class MultiSelectionManager {
  constructor() {
    this.selectedEntities = new Set()
    this.selectionMode = 'single'
  }

  setSelectionMode(mode) {
    this.selectionMode = mode
    if (mode === 'single') {
      this.clearSelection()
    }
  }

  addSelection(entity) {
    if (this.selectionMode === 'single') {
      this.clearSelection()
    }

    this.selectedEntities.add(entity)
    if (highlighter) {
      highlighter.highlightEntity(entity)
    }
  }

  removeSelection(entity) {
    this.selectedEntities.delete(entity)
    if (highlighter) {
      highlighter.unhighlightEntity(entity)
    }
  }

  toggleSelection(entity) {
    if (this.selectedEntities.has(entity)) {
      this.removeSelection(entity)
    } else {
      this.addSelection(entity)
    }
  }

  clearSelection() {
    this.selectedEntities.forEach(entity => {
      if (highlighter) {
        highlighter.unhighlightEntity(entity)
      }
    })
    this.selectedEntities.clear()
  }

  getSelectedEntities() {
    return Array.from(this.selectedEntities)
  }
}

// 空间查询管理器
class SpatialQueryManager {
  constructor(viewer) {
    this.viewer = viewer
  }

  queryByCircle(center, radius) {
    const results = []
    const centerCartesian = window.Cesium.Cartesian3.fromDegrees(center.longitude, center.latitude)

    this.viewer.dataSources.get(0).entities.values.forEach(entity => {
      const entityPosition = entity.position?.getValue()
      if (entityPosition) {
        const distance = window.Cesium.Cartesian3.distance(centerCartesian, entityPosition)
        if (distance <= radius) {
          results.push({
            entity: entity,
            distance: distance
          })
        }
      }
    })

    return results.sort((a, b) => a.distance - b.distance)
  }

  queryByProperty(propertyName, operator, value) {
    const results = []

    this.viewer.dataSources.get(0).entities.values.forEach(entity => {
      const properties = entity.properties
      if (properties && properties[propertyName]) {
        const propertyValue = properties[propertyName].getValue()

        let matches = false
        switch (operator) {
          case '=':
            matches = propertyValue === value
            break
          case '>':
            matches = propertyValue > value
            break
          case '<':
            matches = propertyValue < value
            break
          case '>=':
            matches = propertyValue >= value
            break
          case '<=':
            matches = propertyValue <= value
            break
        }

        if (matches) {
          results.push(entity)
        }
      }
    })

    return results
  }
}

const loadCesium = () => {
  return new Promise((resolve, reject) => {
    if (window.Cesium) {
      resolve(window.Cesium)
      return
    }

    const script = document.createElement('script')
    script.src = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Cesium.js'
    script.onload = () => {
      const link = document.createElement('link')
      link.rel = 'stylesheet'
      link.href = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Widgets/widgets.css'
      link.onload = () => resolve(window.Cesium)
      link.onerror = () => reject(new Error('加载 Cesium CSS 失败'))
      document.head.appendChild(link)
    }
    script.onerror = () => reject(new Error('加载 Cesium JS 失败'))
    document.head.appendChild(script)
  })
}

const getColorByCategory = (Cesium, category) => {
  switch (category) {
    case 'capital':
      return Cesium.Color.RED

// 设置事件处理
const setupEventHandlers = (Cesium) => {
  handler = new Cesium.ScreenSpaceEventHandler(viewer.scene.canvas)

  // 左键点击事件
  handler.setInputAction((click) => {
    if (isCircleQueryActive) {
      executeCircleQuery(click.position)
      return
    }

    const pickedObject = viewer.scene.pick(click.position)

    if (pickedObject && pickedObject.id) {
      const entity = pickedObject.id
      handleEntityClick(entity)
    } else {
      if (selectionMode.value === 'single') {
        clearAllSelections()
      }
    }
  }, Cesium.ScreenSpaceEventType.LEFT_CLICK)

  // 鼠标移动事件
  handler.setInputAction((movement) => {
    if (!showHoverEffect.value) return

    const pickedObject = viewer.scene.pick(movement.endPosition)

    if (pickedObject && pickedObject.id) {
      const entity = pickedObject.id
      handleEntityHover(entity, movement.endPosition)
      viewer.canvas.style.cursor = 'pointer'
    } else {
      handleMouseLeave()
      viewer.canvas.style.cursor = 'default'
    }
  }, Cesium.ScreenSpaceEventType.MOUSE_MOVE)
}

// 处理实体点击
const handleEntityClick = (entity) => {
  if (selectionMode.value === 'multiple') {
    multiSelection.toggleSelection(entity)
    selectedEntities.value = multiSelection.getSelectedEntities()
  } else {
    multiSelection.clearSelection()
    multiSelection.addSelection(entity)
    selectedEntities.value = [entity]
  }

  // 显示实体详情
  showEntityDetails(entity)
}

// 处理实体悬停
const handleEntityHover = (entity, screenPosition) => {
  if (currentHoveredEntity !== entity) {
    if (currentHoveredEntity && !multiSelection.selectedEntities.has(currentHoveredEntity)) {
      highlighter.unhighlightEntity(currentHoveredEntity)
    }

    if (!multiSelection.selectedEntities.has(entity)) {
      highlighter.highlightEntity(entity)
    }

    currentHoveredEntity = entity

    // 显示提示框
    showEntityTooltip(entity, screenPosition)
  }
}

// 处理鼠标离开
const handleMouseLeave = () => {
  if (currentHoveredEntity && !multiSelection.selectedEntities.has(currentHoveredEntity)) {
    highlighter.unhighlightEntity(currentHoveredEntity)
  }
  currentHoveredEntity = null
  hideTooltip()
}

// 显示实体详情
const showEntityDetails = (entity) => {
  selectedEntity.value = entity
  entityProperties.value = {}

  if (entity.properties) {
    const propertyNames = entity.properties.propertyNames
    propertyNames.forEach(name => {
      const value = entity.properties[name]?.getValue()
      if (value !== undefined) {
        entityProperties.value[name] = value
      }
    })
  }
}

// 显示实体提示框
const showEntityTooltip = (entity, screenPosition) => {
  tooltipData.name = entity.name || '未命名实体'

  const properties = entity.properties
  if (properties && properties.population) {
    const population = properties.population.getValue()
    tooltipData.info = `人口: ${population.toLocaleString()}`
  } else {
    tooltipData.info = '点击查看详情'
  }

  tooltipPosition.x = screenPosition.x + 10
  tooltipPosition.y = screenPosition.y - 10
  showTooltip.value = true
}

// 隐藏提示框
const hideTooltip = () => {
  showTooltip.value = false
}

// 设置交互模式
const setInteractionMode = (mode) => {
  interactionMode.value = mode
  isCircleQueryActive = false

  if (mode !== 'query') {
    clearQueryResults()
  }
}

// 更新选择模式
const updateSelectionMode = () => {
  multiSelection.setSelectionMode(selectionMode.value)
  selectedEntities.value = multiSelection.getSelectedEntities()
}

// 切换悬停效果
const toggleHoverEffect = () => {
  if (!showHoverEffect.value && currentHoveredEntity) {
    handleMouseLeave()
  }
}

// 清除所有选择
const clearAllSelections = () => {
  multiSelection.clearSelection()
  selectedEntities.value = []
  selectedEntity.value = null

// 实体高亮管理
class EntityHighlighter {
  constructor(viewer) {
    this.viewer = viewer
    this.highlightedEntities = new Set()
    this.originalStyles = new Map()
  }
  
  // 高亮实体
  highlightEntity(entity) {
    if (this.highlightedEntities.has(entity)) return
    
    // 保存原始样式
    this.saveOriginalStyle(entity)
    
    // 应用高亮样式
    if (entity.point) {
      entity.point.color = Cesium.Color.YELLOW
      entity.point.pixelSize = entity.point.pixelSize.getValue() * 1.5
      entity.point.outlineColor = Cesium.Color.RED
      entity.point.outlineWidth = 3
    }
    
    if (entity.polygon) {
      entity.polygon.material = Cesium.Color.YELLOW.withAlpha(0.7)
      entity.polygon.outlineColor = Cesium.Color.RED
    }
    
    if (entity.polyline) {
      entity.polyline.material = Cesium.Color.YELLOW
      entity.polyline.width = entity.polyline.width.getValue() * 2
    }
    
    this.highlightedEntities.add(entity)
  }
  
  // 取消高亮
  unhighlightEntity(entity) {
    if (!this.highlightedEntities.has(entity)) return
    
    // 恢复原始样式
    this.restoreOriginalStyle(entity)
    
    this.highlightedEntities.delete(entity)
  }
  
  // 清除所有高亮
  clearAllHighlights() {
    this.highlightedEntities.forEach(entity => {
      this.restoreOriginalStyle(entity)
    })
    this.highlightedEntities.clear()
  }
  
  // 保存原始样式
  saveOriginalStyle(entity) {
    const style = {}
    
    if (entity.point) {
      style.point = {
        color: entity.point.color?.getValue(),
        pixelSize: entity.point.pixelSize?.getValue(),
        outlineColor: entity.point.outlineColor?.getValue(),
        outlineWidth: entity.point.outlineWidth?.getValue()
      }
    }
    
    if (entity.polygon) {
      style.polygon = {
        material: entity.polygon.material?.getValue(),
        outlineColor: entity.polygon.outlineColor?.getValue()
      }
    }
    
    if (entity.polyline) {
      style.polyline = {
        material: entity.polyline.material?.getValue(),
        width: entity.polyline.width?.getValue()
      }
    }
    
    this.originalStyles.set(entity, style)
  }
  
  // 恢复原始样式
  restoreOriginalStyle(entity) {
    const style = this.originalStyles.get(entity)
    if (!style) return
    
    if (style.point && entity.point) {
      entity.point.color = style.point.color
      entity.point.pixelSize = style.point.pixelSize
      entity.point.outlineColor = style.point.outlineColor
      entity.point.outlineWidth = style.point.outlineWidth
    }
    
    if (style.polygon && entity.polygon) {
      entity.polygon.material = style.polygon.material
      entity.polygon.outlineColor = style.polygon.outlineColor
    }
    
    if (style.polyline && entity.polyline) {
      entity.polyline.material = style.polyline.material
      entity.polyline.width = style.polyline.width
    }
    
    this.originalStyles.delete(entity)
  }
}

// 空间查询管理器
class SpatialQueryManager {
  constructor(viewer) {
    this.viewer = viewer
    this.queryResults = []
  }
  
  // 圆形范围查询
  queryByCircle(center, radius) {
    const results = []
    const centerCartesian = Cesium.Cartesian3.fromDegrees(center.longitude, center.latitude)
    
    this.viewer.dataSources.get(0).entities.values.forEach(entity => {
      const entityPosition = entity.position?.getValue()
      if (entityPosition) {
        const distance = Cesium.Cartesian3.distance(centerCartesian, entityPosition)
        if (distance <= radius) {
          results.push({
            entity: entity,
            distance: distance
          })
        }
      }
    })
    
    return results.sort((a, b) => a.distance - b.distance)
  }
  
  // 矩形范围查询
  queryByRectangle(west, south, east, north) {
    const results = []
    
    this.viewer.dataSources.get(0).entities.values.forEach(entity => {
      const position = entity.position?.getValue()
      if (position) {
        const cartographic = Cesium.Cartographic.fromCartesian(position)
        const longitude = Cesium.Math.toDegrees(cartographic.longitude)
        const latitude = Cesium.Math.toDegrees(cartographic.latitude)
        
        if (longitude >= west && longitude <= east && 
            latitude >= south && latitude <= north) {
          results.push(entity)
        }
      }
    })
    
    return results
  }
  
  // 属性查询
  queryByProperty(propertyName, operator, value) {
    const results = []
    
    this.viewer.dataSources.get(0).entities.values.forEach(entity => {
      const properties = entity.properties
      if (properties && properties[propertyName]) {
        const propertyValue = properties[propertyName].getValue()
        
        let matches = false
        switch (operator) {
          case '=':
            matches = propertyValue === value
            break
          case '>':
            matches = propertyValue > value
            break
          case '<':
            matches = propertyValue < value
            break
          case '>=':
            matches = propertyValue >= value
            break
          case '<=':
            matches = propertyValue <= value
            break
          case 'contains':
            matches = propertyValue.toString().includes(value.toString())
            break
        }
        
        if (matches) {
          results.push(entity)
        }
      }
    })
    
    return results
  }
}

// 多选管理器
class MultiSelectionManager {
  constructor(viewer) {
    this.viewer = viewer
    this.selectedEntities = new Set()
    this.selectionMode = 'single' // 'single' | 'multiple'
  }
  
  // 切换选择模式
  setSelectionMode(mode) {
    this.selectionMode = mode
    if (mode === 'single') {
      this.clearSelection()
    }
  }
  
  // 添加选择
  addSelection(entity) {
    if (this.selectionMode === 'single') {
      this.clearSelection()
    }
    
    this.selectedEntities.add(entity)
    this.highlightEntity(entity)
    this.updateSelectionInfo()
  }
  
  // 移除选择
  removeSelection(entity) {
    this.selectedEntities.delete(entity)
    this.unhighlightEntity(entity)
    this.updateSelectionInfo()
  }
  
  // 切换选择状态
  toggleSelection(entity) {
    if (this.selectedEntities.has(entity)) {
      this.removeSelection(entity)
    } else {
      this.addSelection(entity)
    }
  }
  
  // 清除所有选择
  clearSelection() {
    this.selectedEntities.forEach(entity => {
      this.unhighlightEntity(entity)
    })
    this.selectedEntities.clear()
    this.updateSelectionInfo()
  }
  
  // 获取选中实体
  getSelectedEntities() {
    return Array.from(this.selectedEntities)
  }
  
  // 更新选择信息显示
  updateSelectionInfo() {
    const count = this.selectedEntities.size
    const infoElement = document.getElementById('selection-info')
    if (infoElement) {
      infoElement.textContent = `已选择 ${count} 个实体`
    }
  }
}

      content += '<div class="properties-list">'
      const propertyNames = properties.propertyNames
      
      propertyNames.forEach(propertyName => {
        const value = properties[propertyName]?.getValue()
        if (value !== undefined) {
          content += `
            <div style="margin-bottom: 8px; display: flex; justify-content: space-between;">
              <span style="font-weight: bold;">${propertyName}:</span>
              <span style="color: #ecf0f1;">${this.formatValue(value)}</span>
            </div>
          `
        }
      })
      
      content += '</div>'
    }
    
    return content
  }
  
  // 格式化值
  formatValue(value) {
    if (typeof value === 'number') {
      return value.toLocaleString()
    }
    return value.toString()
  }
}

// 防抖函数
function debounce(func, wait) {
  let timeout
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout)
      func(...args)
    }
    clearTimeout(timeout)
    timeout = setTimeout(later, wait)
  }
}
