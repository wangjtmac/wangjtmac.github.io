地形数据配置示例 ​

这是一个演示如何在 Cesium 中配置和管理地形数据的综合示例。

功能说明 ​

  • 支持多种地形提供者配置
  • 地形夸张效果控制
  • 地形采样和高程查询
  • 地形分析工具
  • 自定义地形数据加载
  • 地形与实体的交互

核心 API ​

  • Cesium.TerrainProvider - 地形提供者基类
  • Cesium.CesiumTerrainProvider - Cesium 地形提供者
  • Cesium.EllipsoidTerrainProvider - 椭球地形提供者
  • Cesium.sampleTerrain() - 地形采样
  • Cesium.sampleTerrainMostDetailed() - 最详细地形采样

<script setup>
import { ref, onMounted, onUnmounted, getCurrentInstance } from 'vue'

const { proxy } = getCurrentInstance()

const cesiumContainer = ref(null)
const loading = ref(true)
const error = ref('')

let viewer = null
let handler = null

// 地形状态
const selectedProvider = ref('world')
const exaggeration = ref(1.0)
const heightQueryEnabled = ref(false)
const sampledPoints = ref([])
const profileData = ref(null)
const providerName = ref('Cesium World Terrain')

// 地形采样管理器
class TerrainSamplingManager {
  constructor(viewer) {
    this.viewer = viewer
  }

  async sampleHeight(longitude, latitude, level = 11) {
    const position = window.Cesium.Cartographic.fromDegrees(longitude, latitude)

    try {
      const sampledPositions = await window.Cesium.sampleTerrain(
        this.viewer.terrainProvider,
        level,
        [position]
      )

      if (sampledPositions.length > 0) {
        return sampledPositions[0].height
      }
    } catch (error) {
      console.error('地形采样失败:', error)
      return null
    }
  }

  async sampleHeights(positions, level = 11) {
    const cartographics = positions.map(pos =>
      window.Cesium.Cartographic.fromDegrees(pos.longitude, pos.latitude)
    )

    try {
      const sampledPositions = await window.Cesium.sampleTerrain(
        this.viewer.terrainProvider,
        level,
        cartographics
      )

      return sampledPositions.map((pos, index) => ({
        longitude: window.Cesium.Math.toDegrees(pos.longitude),
        latitude: window.Cesium.Math.toDegrees(pos.latitude),
        height: pos.height,
        originalIndex: index
      }))
    } catch (error) {
      console.error('批量地形采样失败:', error)
      return []
    }
  }

  async sampleAlongPath(startPos, endPos, numSamples = 50) {
    const positions = []

    for (let i = 0; i <= numSamples; i++) {
      const t = i / numSamples
      const longitude = startPos.longitude + (endPos.longitude - startPos.longitude) * t
      const latitude = startPos.latitude + (endPos.latitude - startPos.latitude) * t
      positions.push({ longitude, latitude })
    }

    return await this.sampleHeights(positions)
  }
}

let samplingManager = null

// 加载 Cesium
const loadCesium = () => {
  return new Promise((resolve, reject) => {
    if (window.Cesium) {
      resolve(window.Cesium)
      return
    }

    const script = document.createElement('script')
    script.src = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Cesium.js'
    script.onload = () => {
      const link = document.createElement('link')
      link.rel = 'stylesheet'
      link.href = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Widgets/widgets.css'
      link.onload = () => resolve(window.Cesium)
      link.onerror = () => reject(new Error('加载 Cesium CSS 失败'))
      document.head.appendChild(link)
    }
    script.onerror = () => reject(new Error('加载 Cesium JS 失败'))
    document.head.appendChild(script)
  })
}

// 初始化 Cesium
const initCesium = async () => {
  try {
    loading.value = true
    const Cesium = await loadCesium()

    Cesium.Ion.defaultAccessToken = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJqdGkiOiI5N2UyMjcwOS00MDY1LTQxYjEtYjZjMy00YTU0ZTg1YmJjMGEiLCJpZCI6MjU5MCwic2NvcGVzIjpbImFzciIsImdjIl0sImlhdCI6MTU0Mjg2MTUzMX0.Hvl3oBH8t1pJOOkOXJJ0M0pHcJJ8Gx8Hb3Ks3k7Ks3k'

    viewer = new Cesium.Viewer(cesiumContainer.value, {
      animation: false,
      timeline: false,
      baseLayerPicker: false,
      geocoder: false,
      homeButton: true,
      sceneModePicker: false,
      navigationHelpButton: false,
      fullscreenButton: false,
      terrainProvider: await Cesium.CesiumTerrainProvider.fromIonAssetId(1, {
        requestVertexNormals: true,
        requestWaterMask: true
      })
    })

    // 设置初始视角到有地形起伏的区域（喜马拉雅山脉）
    viewer.camera.setView({
      destination: Cesium.Cartesian3.fromDegrees(86.9250, 27.9881, 50000),
      orientation: {
        heading: Cesium.Math.toRadians(0),
        pitch: Cesium.Math.toRadians(-45),
        roll: 0.0
      }
    })

    // 创建采样管理器
    samplingManager = new TerrainSamplingManager(viewer)

    // 设置事件处理器
    handler = new Cesium.ScreenSpaceEventHandler(viewer.scene.canvas)

    loading.value = false
  } catch (err) {
    error.value = err.message
    loading.value = false
  }
}

// 切换地形提供者
const switchTerrain = async () => {
  if (!viewer) return

  const Cesium = window.Cesium

  try {
    if (selectedProvider.value === 'ellipsoid') {
      viewer.terrainProvider = new Cesium.EllipsoidTerrainProvider()
      providerName.value = '椭球地形'
    } else if (selectedProvider.value === 'world') {
      viewer.terrainProvider = await Cesium.CesiumTerrainProvider.fromIonAssetId(1, {
        requestVertexNormals: true,
        requestWaterMask: true
      })
      providerName.value = 'Cesium World Terrain'
    }

    // 重新创建采样管理器
    samplingManager = new TerrainSamplingManager(viewer)
  } catch (err) {
    console.error('切换地形失败:', err)
  }
}

// 更新地形夸张
const updateExaggeration = () => {
  if (!viewer) return
  viewer.scene.verticalExaggeration = exaggeration.value
}

// 重置地形夸张
const resetExaggeration = () => {
  exaggeration.value = 1.0
  updateExaggeration()
}

// 动画过渡地形夸张
const animateExaggeration = () => {
  const targetValue = exaggeration.value === 1.0 ? 3.0 : 1.0
  const startValue = exaggeration.value
  const duration = 2000
  const startTime = Date.now()

  const animate = () => {
    const elapsed = Date.now() - startTime
    const progress = Math.min(elapsed / duration, 1)

    // 缓动函数
    const easeProgress = progress * progress * (3 - 2 * progress)
    exaggeration.value = startValue + (targetValue - startValue) * easeProgress

    updateExaggeration()

    if (progress < 1) {
      requestAnimationFrame(animate)
    }
  }

  animate()
}

// 启用高程查询
const enableHeightQuery = () => {
  if (!viewer || !handler) return

  heightQueryEnabled.value = !heightQueryEnabled.value

  if (heightQueryEnabled.value) {
    const Cesium = window.Cesium

    handler.setInputAction(async (click) => {
      const cartesian = viewer.camera.pickEllipsoid(click.position, viewer.scene.globe.ellipsoid)

      if (cartesian) {
        const cartographic = Cesium.Cartographic.fromCartesian(cartesian)
        const longitude = Cesium.Math.toDegrees(cartographic.longitude)
        const latitude = Cesium.Math.toDegrees(cartographic.latitude)

        const height = await samplingManager.sampleHeight(longitude, latitude)

        if (height !== null) {
          // 添加标记
          viewer.entities.add({
            position: Cesium.Cartesian3.fromDegrees(longitude, latitude, height),
            point: {
              pixelSize: 10,
              color: Cesium.Color.RED,
              outlineColor: Cesium.Color.WHITE,
              outlineWidth: 2
            },
            label: {
              text: `${height.toFixed(1)}m`,
              font: '14px sans-serif',
              fillColor: Cesium.Color.WHITE,
              outlineColor: Cesium.Color.BLACK,
              outlineWidth: 2,
              style: Cesium.LabelStyle.FILL_AND_OUTLINE,
              verticalOrigin: Cesium.VerticalOrigin.BOTTOM,
              pixelOffset: new Cesium.Cartesian2(0, -15)
            }
          })

          sampledPoints.value.push({ longitude, latitude, height })
        }
      }
    }, Cesium.ScreenSpaceEventType.LEFT_CLICK)
  } else {
    handler.removeInputAction(window.Cesium.ScreenSpaceEventType.LEFT_CLICK)
  }
}

// 随机采样点
const sampleRandomPoints = async () => {
  if (!viewer || !samplingManager) return

  const Cesium = window.Cesium
  const camera = viewer.camera
  const center = camera.positionCartographic

  const centerLon = Cesium.Math.toDegrees(center.longitude)
  const centerLat = Cesium.Math.toDegrees(center.latitude)

  const positions = []
  for (let i = 0; i < 10; i++) {
    const lon = centerLon + (Math.random() - 0.5) * 0.5
    const lat = centerLat + (Math.random() - 0.5) * 0.5
    positions.push({ longitude: lon, latitude: lat })
  }

  const sampledHeights = await samplingManager.sampleHeights(positions)

  sampledHeights.forEach(pos => {
    viewer.entities.add({
      position: Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.height),
      point: {
        pixelSize: 8,
        color: Cesium.Color.BLUE,
        outlineColor: Cesium.Color.WHITE,
        outlineWidth: 2
      }
    })

    sampledPoints.value.push(pos)
  })
}

// 清除采样
const clearSamples = () => {
  if (!viewer) return

  viewer.entities.removeAll()
  sampledPoints.value = []
  profileData.value = null
}

// 创建剖面线
const createProfile = async () => {
  if (!viewer || !samplingManager) return

  const Cesium = window.Cesium
  const camera = viewer.camera
  const center = camera.positionCartographic

  const centerLon = Cesium.Math.toDegrees(center.longitude)
  const centerLat = Cesium.Math.toDegrees(center.latitude)

  const startPos = { longitude: centerLon - 0.2, latitude: centerLat }
  const endPos = { longitude: centerLon + 0.2, latitude: centerLat }

  const profile = await samplingManager.sampleAlongPath(startPos, endPos, 50)

  // 创建剖面线
  const positions = profile.map(pos =>
    Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.height)
  )

  viewer.entities.add({
    name: '地形剖面',
    polyline: {
      positions: positions,
      width: 5,
      material: new Cesium.PolylineGlowMaterialProperty({
        glowPower: 0.2,
        color: Cesium.Color.CYAN
      })
    }
  })

  // 添加端点标记
  profile.forEach((pos, index) => {
    if (index % 10 === 0) {
      viewer.entities.add({
        position: Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.height),
        point: {
          pixelSize: 6,
          color: Cesium.Color.YELLOW
        }
      })
    }
  })

  // 计算统计信息
  const maxHeight = Math.max(...profile.map(p => p.height))
  const minHeight = Math.min(...profile.map(p => p.height))

  profileData.value = {
    maxHeight,
    minHeight,
    elevationGain: maxHeight - minHeight
  }

  // 飞行到剖面线
  viewer.flyTo(viewer.entities)
}

// 坡度分析
const analyzeSlope = async () => {
  if (!viewer || !samplingManager) return

  const Cesium = window.Cesium
  const camera = viewer.camera
  const center = camera.positionCartographic

  const centerLon = Cesium.Math.toDegrees(center.longitude)
  const centerLat = Cesium.Math.toDegrees(center.latitude)

  const gridSize = 5
  const range = 0.1

  for (let i = 0; i < gridSize; i++) {
    for (let j = 0; j < gridSize; j++) {
      const lon = centerLon - range / 2 + (i / gridSize) * range
      const lat = centerLat - range / 2 + (j / gridSize) * range

      // 简化的坡度计算
      const positions = [
        { longitude: lon, latitude: lat },
        { longitude: lon + 0.01, latitude: lat },
        { longitude: lon, latitude: lat + 0.01 }
      ]

      const heights = await samplingManager.sampleHeights(positions)

      if (heights.length === 3) {
        const dh1 = heights[1].height - heights[0].height
        const dh2 = heights[2].height - heights[0].height
        const slope = Math.sqrt(dh1 * dh1 + dh2 * dh2) / 1000

        let color
        if (slope < 0.05) {
          color = Cesium.Color.GREEN
        } else if (slope < 0.15) {
          color = Cesium.Color.YELLOW
        } else if (slope < 0.3) {
          color = Cesium.Color.ORANGE
        } else {
          color = Cesium.Color.RED
        }

        viewer.entities.add({
          position: Cesium.Cartesian3.fromDegrees(lon, lat, heights[0].height),
          point: {
            pixelSize: 15,
            color: color.withAlpha(0.7)
          }
        })
      }
    }
  }
}

// 清除分析
const clearAnalysis = () => {
  clearSamples()
}

// 清理资源
const cleanup = () => {
  if (handler) {
    handler.destroy()
  }

  if (viewer) {
    viewer.destroy()
    viewer = null
  }
}

onMounted(() => {
  initCesium()
})

onUnmounted(() => {
  cleanup()
})
</script>

// 地形提供者管理器
class TerrainProviderManager {
  constructor(viewer) {
    this.viewer = viewer
    this.providers = new Map()
    this.initializeProviders()
  }

  // 初始化地形提供者
  async initializeProviders() {
    // 1. 椭球地形（无地形起伏）
    this.providers.set('ellipsoid', new Cesium.EllipsoidTerrainProvider())

    // 2. Cesium World Terrain（全球地形）
    try {
      const worldTerrain = await Cesium.CesiumTerrainProvider.fromIonAssetId(1, {
        requestVertexNormals: true,
        requestWaterMask: true
      })
      this.providers.set('world', worldTerrain)
    } catch (error) {
      console.warn('无法加载 Cesium World Terrain:', error)
    }

    // 3. 自定义地形提供者
    try {
      const customTerrain = await Cesium.CesiumTerrainProvider.fromUrl(
        'https://assets.agi.com/stk-terrain/v1/tilesets/world/tiles',
        {
          requestVertexNormals: true,
          requestWaterMask: false
        }
      )
      this.providers.set('custom', customTerrain)
    } catch (error) {
      console.warn('无法加载自定义地形:', error)
    }
  }

  // 切换地形提供者
  switchTerrain(providerName) {
    const provider = this.providers.get(providerName)
    if (provider) {
      this.viewer.terrainProvider = provider
      console.log(`已切换到地形: ${providerName}`)
    } else {
      console.warn(`地形提供者 ${providerName} 不存在`)
    }
  }

  // 获取当前地形提供者
  getCurrentProvider() {
    return this.viewer.terrainProvider
  }

  // 获取所有可用的地形提供者
  getAvailableProviders() {
    return Array.from(this.providers.keys())
  }
}

// 地形夸张管理器
class TerrainExaggerationManager {
  constructor(viewer) {
    this.viewer = viewer
    this.currentExaggeration = 1.0
    this.minExaggeration = 0.1
    this.maxExaggeration = 10.0
  }

  // 设置地形夸张
  setExaggeration(value) {
    // 限制在合理范围内
    const exaggeration = Math.max(
      this.minExaggeration,
      Math.min(this.maxExaggeration, value)
    )

    this.viewer.scene.verticalExaggeration = exaggeration
    this.currentExaggeration = exaggeration

    console.log(`地形夸张设置为: ${exaggeration}`)
    return exaggeration
  }

  // 增加地形夸张
  increaseExaggeration(step = 0.5) {
    return this.setExaggeration(this.currentExaggeration + step)
  }

  // 减少地形夸张
  decreaseExaggeration(step = 0.5) {
    return this.setExaggeration(this.currentExaggeration - step)
  }

  // 重置地形夸张
  resetExaggeration() {
    return this.setExaggeration(1.0)
  }

  // 获取当前地形夸张值
  getExaggeration() {
    return this.currentExaggeration
  }

  // 动画过渡地形夸张
  animateExaggeration(targetValue, duration = 2000) {
    const startValue = this.currentExaggeration
    const startTime = Date.now()

    const animate = () => {
      const elapsed = Date.now() - startTime
      const progress = Math.min(elapsed / duration, 1)

      // 使用缓动函数
      const easeProgress = progress * progress * (3 - 2 * progress)
      const currentValue = startValue + (targetValue - startValue) * easeProgress

      this.setExaggeration(currentValue)

      if (progress < 1) {
        requestAnimationFrame(animate)
      }
    }

    animate()
  }
}

// 地形采样管理器
class TerrainSamplingManager {
  constructor(viewer) {
    this.viewer = viewer
    this.sampledPositions = []
  }

  // 采样单个位置的高程
  async sampleHeight(longitude, latitude, level = 11) {
    const position = Cesium.Cartographic.fromDegrees(longitude, latitude)
    
    try {
      const sampledPositions = await Cesium.sampleTerrain(
        this.viewer.terrainProvider,
        level,
        [position]
      )

      if (sampledPositions.length > 0) {
        const height = sampledPositions[0].height
        console.log(`位置 (${longitude}, ${latitude}) 的高程: ${height}m`)
        return height
      }
    } catch (error) {
      console.error('地形采样失败:', error)
      return null
    }
  }

  // 采样多个位置的高程
  async sampleHeights(positions, level = 11) {
    const cartographics = positions.map(pos =>
      Cesium.Cartographic.fromDegrees(pos.longitude, pos.latitude)
    )

    try {
      const sampledPositions = await Cesium.sampleTerrain(
        this.viewer.terrainProvider,
        level,
        cartographics
      )

      return sampledPositions.map((pos, index) => ({
        longitude: Cesium.Math.toDegrees(pos.longitude),
        latitude: Cesium.Math.toDegrees(pos.latitude),
        height: pos.height,
        originalIndex: index
      }))
    } catch (error) {
      console.error('批量地形采样失败:', error)
      return []
    }
  }

  // 采样最详细的高程
  async sampleMostDetailed(positions) {
    const cartographics = positions.map(pos =>
      Cesium.Cartographic.fromDegrees(pos.longitude, pos.latitude)
    )

    try {
      const sampledPositions = await Cesium.sampleTerrainMostDetailed(
        this.viewer.terrainProvider,
        cartographics
      )

      return sampledPositions.map((pos, index) => ({
        longitude: Cesium.Math.toDegrees(pos.longitude),
        latitude: Cesium.Math.toDegrees(pos.latitude),
        height: pos.height,
        originalIndex: index
      }))
    } catch (error) {
      console.error('最详细地形采样失败:', error)
      return []
    }
  }

  // 沿路径采样高程
  async sampleAlongPath(startPos, endPos, numSamples = 50) {
    const positions = []
    
    for (let i = 0; i <= numSamples; i++) {
      const t = i / numSamples
      const longitude = startPos.longitude + (endPos.longitude - startPos.longitude) * t
      const latitude = startPos.latitude + (endPos.latitude - startPos.latitude) * t
      positions.push({ longitude, latitude })
    }

    return await this.sampleHeights(positions)
  }

  // 获取鼠标点击位置的高程
  async getHeightAtClick(clickPosition) {
    const cartesian = this.viewer.camera.pickEllipsoid(
      clickPosition,
      this.viewer.scene.globe.ellipsoid
    )

    if (cartesian) {
      const cartographic = Cesium.Cartographic.fromCartesian(cartesian)
      const longitude = Cesium.Math.toDegrees(cartographic.longitude)
      const latitude = Cesium.Math.toDegrees(cartographic.latitude)

      return await this.sampleHeight(longitude, latitude)
    }

    return null
  }
}

// 地形分析管理器
class TerrainAnalysisManager {
  constructor(viewer) {
    this.viewer = viewer
    this.samplingManager = new TerrainSamplingManager(viewer)
  }

  // 计算坡度
  async calculateSlope(longitude, latitude, sampleDistance = 0.001) {
    // 采样中心点和周围四个点
    const positions = [
      { longitude, latitude }, // 中心
      { longitude: longitude + sampleDistance, latitude }, // 东
      { longitude: longitude - sampleDistance, latitude }, // 西
      { longitude, latitude: latitude + sampleDistance }, // 北
      { longitude, latitude: latitude - sampleDistance }  // 南
    ]

    const sampledHeights = await this.samplingManager.sampleHeights(positions)

    if (sampledHeights.length === 5) {
      const center = sampledHeights[0].height
      const east = sampledHeights[1].height
      const west = sampledHeights[2].height
      const north = sampledHeights[3].height
      const south = sampledHeights[4].height

      // 计算东西和南北方向的坡度
      const dzdx = (east - west) / (2 * sampleDistance * 111000) // 转换为米
      const dzdy = (north - south) / (2 * sampleDistance * 111000)

      // 计算总坡度（度数）
      const slope = Math.atan(Math.sqrt(dzdx * dzdx + dzdy * dzdy)) * (180 / Math.PI)

      // 计算坡向（度数，0=北，90=东，180=南，270=西）
      let aspect = Math.atan2(dzdy, dzdx) * (180 / Math.PI)
      aspect = (90 - aspect + 360) % 360

      return {
        slope: slope,
        aspect: aspect,
        center: center
      }
    }

    return null
  }

  // 生成等高线
  async generateContours(bounds, interval = 100, level = 11) {
    const { west, south, east, north } = bounds
    const gridSize = 20

    const positions = []
    for (let lat = south; lat <= north; lat += (north - south) / gridSize) {
      for (let lon = west; lon <= east; lon += (east - west) / gridSize) {
        positions.push({ longitude: lon, latitude: lat })
      }
    }

    const sampledHeights = await this.samplingManager.sampleHeights(positions, level)

    // 按高程分组
    const contourGroups = new Map()
    sampledHeights.forEach(pos => {
      const contourLevel = Math.floor(pos.height / interval) * interval
      if (!contourGroups.has(contourLevel)) {
        contourGroups.set(contourLevel, [])
      }
      contourGroups.get(contourLevel).push(pos)
    })

    return contourGroups
  }

  // 计算剖面线
  async generateProfile(startPos, endPos, numSamples = 100) {
    const sampledHeights = await this.samplingManager.sampleAlongPath(
      startPos,
      endPos,
      numSamples
    )

    // 计算距离
    const startCartographic = Cesium.Cartographic.fromDegrees(
      startPos.longitude,
      startPos.latitude
    )
    const endCartographic = Cesium.Cartographic.fromDegrees(
      endPos.longitude,
      endPos.latitude
    )

    const geodesic = new Cesium.EllipsoidGeodesic(startCartographic, endCartographic)
    const totalDistance = geodesic.surfaceDistance

    return sampledHeights.map((pos, index) => ({
      distance: (index / numSamples) * totalDistance,
      height: pos.height,
      longitude: pos.longitude,
      latitude: pos.latitude
    }))
  }

  // 计算可视域
  async calculateViewshed(observerPos, observerHeight, radius, numRays = 36) {
    const visiblePoints = []
    const hiddenPoints = []

    const observerCartographic = Cesium.Cartographic.fromDegrees(
      observerPos.longitude,
      observerPos.latitude
    )

    // 采样观察点高程
    const sampledObserver = await this.samplingManager.sampleHeight(
      observerPos.longitude,
      observerPos.latitude
    )
    const observerElevation = sampledObserver + observerHeight

    // 在观察点周围采样
    for (let i = 0; i < numRays; i++) {
      const angle = (i / numRays) * 2 * Math.PI
      const targetLon = observerPos.longitude + Math.cos(angle) * radius
      const targetLat = observerPos.latitude + Math.sin(angle) * radius

      const targetHeight = await this.samplingManager.sampleHeight(targetLon, targetLat)

      if (targetHeight !== null) {
        // 简化的可视性判断
        const distance = Cesium.Cartesian3.distance(
          Cesium.Cartesian3.fromDegrees(observerPos.longitude, observerPos.latitude, observerElevation),
          Cesium.Cartesian3.fromDegrees(targetLon, targetLat, targetHeight)
        )

        const isVisible = targetHeight < observerElevation + distance * 0.1

        if (isVisible) {
          visiblePoints.push({ longitude: targetLon, latitude: targetLat, height: targetHeight })
        } else {
          hiddenPoints.push({ longitude: targetLon, latitude: targetLat, height: targetHeight })
        }
      }
    }

    return { visiblePoints, hiddenPoints }
  }
}

// 地形实体交互管理器
class TerrainEntityInteraction {
  constructor(viewer) {
    this.viewer = viewer
    this.samplingManager = new TerrainSamplingManager(viewer)
    this.clampedEntities = new Set()
  }

  // 将实体贴合到地形
  async clampEntityToTerrain(entity, offset = 0) {
    const position = entity.position.getValue(Cesium.JulianDate.now())
    const cartographic = Cesium.Cartographic.fromCartesian(position)

    const longitude = Cesium.Math.toDegrees(cartographic.longitude)
    const latitude = Cesium.Math.toDegrees(cartographic.latitude)

    const terrainHeight = await this.samplingManager.sampleHeight(longitude, latitude)

    if (terrainHeight !== null) {
      entity.position = Cesium.Cartesian3.fromDegrees(
        longitude,
        latitude,
        terrainHeight + offset
      )

      this.clampedEntities.add(entity)
      return terrainHeight
    }

    return null
  }

  // 沿地形创建路径
  async createTerrainPath(positions, offset = 2) {
    const sampledPositions = await this.samplingManager.sampleMostDetailed(positions)

    const pathPositions = sampledPositions.map(pos =>
      Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.height + offset)
    )

    const entity = this.viewer.entities.add({
      name: '地形路径',
      polyline: {
        positions: pathPositions,
        width: 3,
        material: Cesium.Color.YELLOW,
        clampToGround: false
      }
    })

    return entity
  }

  // 创建地形剖面可视化
  async visualizeTerrainProfile(startPos, endPos, numSamples = 50) {
    const profile = await this.samplingManager.sampleAlongPath(
      startPos,
      endPos,
      numSamples
    )

    // 创建剖面线
    const positions = profile.map(pos =>
      Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.height)
    )

    const profileEntity = this.viewer.entities.add({
      name: '地形剖面',
      polyline: {
        positions: positions,
        width: 5,
        material: new Cesium.PolylineGlowMaterialProperty({
          glowPower: 0.2,
          color: Cesium.Color.CYAN
        })
      }
    })

    // 添加高程标记
    const maxHeight = Math.max(...profile.map(p => p.height))
    const minHeight = Math.min(...profile.map(p => p.height))

    profile.forEach((pos, index) => {
      if (index % 10 === 0) {
        this.viewer.entities.add({
          position: Cesium.Cartesian3.fromDegrees(
            pos.longitude,
            pos.latitude,
            pos.height
          ),
          point: {
            pixelSize: 8,
            color: Cesium.Color.RED,
            outlineColor: Cesium.Color.WHITE,
            outlineWidth: 2
          },
          label: {
            text: `${pos.height.toFixed(1)}m`,
            font: '12px sans-serif',
            fillColor: Cesium.Color.WHITE,
            outlineColor: Cesium.Color.BLACK,
            outlineWidth: 2,
            style: Cesium.LabelStyle.FILL_AND_OUTLINE,
            verticalOrigin: Cesium.VerticalOrigin.BOTTOM,
            pixelOffset: new Cesium.Cartesian2(0, -10)
          }
        })
      }
    })

    return {
      entity: profileEntity,
      profile: profile,
      statistics: {
        maxHeight,
        minHeight,
        elevationGain: maxHeight - minHeight,
        numSamples: profile.length
      }
    }
  }

  // 创建坡度可视化
  async visualizeSlope(bounds, gridSize = 10) {
    const { west, south, east, north } = bounds
    const analysisManager = new TerrainAnalysisManager(this.viewer)

    const entities = []

    for (let lat = south; lat <= north; lat += (north - south) / gridSize) {
      for (let lon = west; lon <= east; lon += (east - west) / gridSize) {
        const slopeData = await analysisManager.calculateSlope(lon, lat)

        if (slopeData) {
          // 根据坡度设置颜色
          let color
          if (slopeData.slope < 5) {
            color = Cesium.Color.GREEN
          } else if (slopeData.slope < 15) {
            color = Cesium.Color.YELLOW
          } else if (slopeData.slope < 30) {
            color = Cesium.Color.ORANGE
          } else {
            color = Cesium.Color.RED
          }

          const entity = this.viewer.entities.add({
            position: Cesium.Cartesian3.fromDegrees(lon, lat, slopeData.center),
            point: {
              pixelSize: 10,
              color: color.withAlpha(0.7)
            }
          })

          entities.push(entity)
        }
      }
    }

    return entities
  }
}

地形数据格式 ​

1. Quantized-Mesh 格式 ​

Cesium 使用的高效地形格式，支持：

  • 高度数据压缩
  • 法线向量
  • 水体掩码
  • 元数据

2. HeightMap 格式 ​

基于图像的高程数据：

  • PNG/JPEG 高程图
  • 灰度值表示高程
  • 简单但精度有限

应用场景 ​

  1. 地形可视化 - 山地、峡谷、高原的真实展示
  2. 路径规划 - 考虑地形起伏的路线设计
  3. 视域分析 - 观察点可视范围计算
  4. 地质勘探 - 地形剖面和坡度分析
  5. 灾害评估 - 洪水淹没、滑坡风险分析

性能优化建议 ​

  1. 合理选择采样级别 - 根据需求平衡精度和性能
  2. 批量采样 - 一次采样多个点而不是逐个采样
  3. 缓存采样结果 - 避免重复采样相同位置
  4. 使用适当的地形夸张 - 过大的夸张会影响性能
  5. LOD 管理 - 远处使用低精度地形
