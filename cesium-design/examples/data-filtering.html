数据过滤与筛选示例 ​

这是一个演示如何在 Cesium 中实现数据过滤和筛选功能的综合示例。

功能说明 ​

  • 多条件数据筛选
  • 实时过滤结果更新
  • 范围滑块过滤
  • 分类筛选和搜索
  • 过滤历史记录
  • 数据统计和分析

核心 API ​

  • Cesium.Entity.show - 控制实体显示/隐藏
  • Cesium.DataSource.entities - 实体集合操作
  • Cesium.Property.getValue() - 获取属性值
  • Cesium.CallbackProperty - 动态属性回调

// 数据过滤管理器
class DataFilterManager {
  constructor() {
    this.filters = new Map()
    this.entities = []
  }

  setEntities(entities) {
    this.entities = entities
  }

  addFilter(name, filterFunction) {
    this.filters.set(name, filterFunction)
    return this.applyFilters()
  }

  removeFilter(name) {
    this.filters.delete(name)
    return this.applyFilters()
  }

  applyFilters() {
    const filteredEntities = this.entities.filter(entity => {
      if (filterMode.value === 'AND') {
        // 所有过滤器都必须通过
        for (let [name, filterFunc] of this.filters) {
          if (!filterFunc(entity)) {
            return false
          }
        }
        return true
      } else {
        // 任一过滤器通过即可
        if (this.filters.size === 0) return true

        for (let [name, filterFunc] of this.filters) {
          if (filterFunc(entity)) {
            return true
          }
        }
        return false
      }
    })

    this.updateEntityVisibility(filteredEntities)
    return filteredEntities
  }

  updateEntityVisibility(filteredEntities) {
    const filteredSet = new Set(filteredEntities)

    this.entities.forEach(entity => {
      entity.show = filteredSet.has(entity)
    })
  }

  getFilterStats() {
    const visible = this.entities.filter(entity => entity.show).length
    return {
      total: this.entities.length,
      visible: visible,
      hidden: this.entities.length - visible
    }
  }
}

const loadCesium = () => {
  return new Promise((resolve, reject) => {
    if (window.Cesium) {
      resolve(window.Cesium)
      return
    }

    const script = document.createElement('script')
    script.src = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Cesium.js'
    script.onload = () => {
      const link = document.createElement('link')
      link.rel = 'stylesheet'
      link.href = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Widgets/widgets.css'
      link.onload = () => resolve(window.Cesium)
      link.onerror = () => reject(new Error('加载 Cesium CSS 失败'))
      document.head.appendChild(link)
    }
    script.onerror = () => reject(new Error('加载 Cesium JS 失败'))
    document.head.appendChild(script)
  })
}

    case 'all':
      filterManager.removeFilter('quick')
      break
    case 'capital':
      filterManager.addFilter('quick', entity =>
        entity.properties.category.getValue() === 'capital'
      )
      break
    case 'major':
      filterManager.addFilter('quick', entity =>
        entity.properties.category.getValue() === 'major'
      )
      break
    case 'large':
      filterManager.addFilter('quick', entity =>
        entity.properties.population.getValue() > 15000000
      )
      break
  }

  updateResults()
  saveFilterState(`快速过滤: ${getQuickFilterLabel(filterType)}`)
}

        return similarity >= 0.6

    } else {
      filterManager.addFilter('search', entity =>
        entity.name.toLowerCase().includes(searchLower)
      )
    }
  }

  updateResults()
  if (searchText.value.trim()) {
    saveFilterState(`文本搜索: ${searchText.value}`)
  }
}

    const gdp = entity.properties.gdp.getValue()

  }

  updateResults()
  saveFilterState(`分类过滤: ${selectedCategories.value.join(', ')}`)
}

// 更新结果
const updateResults = () => {
  visibleEntities.value = allEntities.filter(entity => entity.show)
  Object.assign(filterStats, filterManager.getFilterStats())
}

// 保存过滤状态
const saveFilterState = (description) => {
  const state = {
    activeQuickFilter: activeQuickFilter.value,
    searchText: searchText.value,
    selectedCategories: [...selectedCategories.value],
    populationRange: { ...populationRange },
    gdpRange: { ...gdpRange },
    filterMode: filterMode.value
  }

  // 移除当前位置之后的历史记录
  filterHistory.value = filterHistory.value.slice(0, currentHistoryIndex.value + 1)

  // 添加新状态
  filterHistory.value.push({
    state: JSON.parse(JSON.stringify(state)),
    description: description,
    timestamp: new Date().toISOString(),
    isCurrent: true
  })

  // 更新当前索引
  if (filterHistory.value.length > 1) {
    filterHistory.value[currentHistoryIndex.value].isCurrent = false
  }
  currentHistoryIndex.value = filterHistory.value.length - 1

  // 限制历史记录大小
  if (filterHistory.value.length > 20) {
    filterHistory.value.shift()
    currentHistoryIndex.value--
  }
}

// 恢复过滤状态
const restoreFilterState = (index) => {
  if (index >= 0 && index < filterHistory.value.length) {
    const historyItem = filterHistory.value[index]
    const state = historyItem.state

    // 更新当前状态标记
    filterHistory.value.forEach((item, i) => {
      item.isCurrent = i === index
    })
    currentHistoryIndex.value = index

    // 恢复状态
    activeQuickFilter.value = state.activeQuickFilter
    searchText.value = state.searchText
    selectedCategories.value = state.selectedCategories

const calculateSimilarity = (str1, str2) => {
  const longer = str1.length > str2.length ? str1 : str2
  const shorter = str1.length > str2.length ? str2 : str1

  if (longer.length === 0) return 1.0

  const editDistance = levenshteinDistance(longer, shorter)
  return (longer.length - editDistance) / longer.length
}

// 计算编辑距离
const levenshteinDistance = (str1, str2) => {
  const matrix = []

  for (let i = 0; i <= str2.length; i++) {
    matrix[i] = [i]
  }

  for (let j = 0; j <= str1.length; j++) {
    matrix[0][j] = j
  }

  for (let i = 1; i <= str2.length; i++) {
    for (let j = 1; j <= str1.length; j++) {
      if (str2.charAt(i - 1) === str1.charAt(j - 1)) {
        matrix[i][j] = matrix[i - 1][j - 1]
      } else {
        matrix[i][j] = Math.min(
          matrix[i - 1][j - 1] + 1,
          matrix[i][j - 1] + 1,
          matrix[i - 1][j] + 1
        )
      }
    }
  }

  return matrix[str2.length][str1.length]
}

// 数据过滤管理器
class DataFilterManager {
  constructor(viewer) {
    this.viewer = viewer
    this.filters = new Map()
    this.originalEntities = []
    this.filteredEntities = []
  }
  
  // 添加过滤器
  addFilter(name, filterFunction) {
    this.filters.set(name, filterFunction)
    this.applyFilters()
  }
  
  // 移除过滤器
  removeFilter(name) {
    this.filters.delete(name)
    this.applyFilters()
  }
  
  // 应用所有过滤器
  applyFilters() {
    const entities = this.getAllEntities()
    
    this.filteredEntities = entities.filter(entity => {
      // 所有过滤器都必须通过
      for (let [name, filterFunc] of this.filters) {
        if (!filterFunc(entity)) {
          return false
        }
      }
      return true
    })
    
    // 更新实体显示状态
    this.updateEntityVisibility()
  }
  
  // 更新实体显示状态
  updateEntityVisibility() {
    const entities = this.getAllEntities()
    const filteredSet = new Set(this.filteredEntities)
    
    entities.forEach(entity => {
      entity.show = filteredSet.has(entity)
    })
  }
  
  // 获取所有实体
  getAllEntities() {
    if (this.originalEntities.length === 0) {
      this.viewer.dataSources.get(0).entities.values.forEach(entity => {
        this.originalEntities.push(entity)
      })
    }
    return this.originalEntities
  }
  
  // 获取过滤结果统计
  getFilterStats() {
    return {
      total: this.originalEntities.length,
      filtered: this.filteredEntities.length,
      hidden: this.originalEntities.length - this.filteredEntities.length
    }
  }
}

// 创建复合过滤器
const createCompoundFilter = (conditions) => {
  return (entity) => {
    const properties = entity.properties
    if (!properties) return false
    
    return conditions.every(condition => {
      const { property, operator, value } = condition
      const entityValue = properties[property]?.getValue()
      
      if (entityValue === undefined) return false
      
      switch (operator) {
        case 'equals':
          return entityValue === value
        case 'not_equals':
          return entityValue !== value
        case 'greater_than':
          return entityValue > value
        case 'less_than':
          return entityValue < value
        case 'greater_equal':
          return entityValue >= value
        case 'less_equal':
          return entityValue <= value
        case 'contains':
          return entityValue.toString().toLowerCase().includes(value.toLowerCase())
        case 'starts_with':
          return entityValue.toString().toLowerCase().startsWith(value.toLowerCase())
        case 'ends_with':
          return entityValue.toString().toLowerCase().endsWith(value.toLowerCase())
        case 'in_range':
          return entityValue >= value.min && entityValue <= value.max
        default:
          return true
      }
    })
  }
}

// 创建地理空间过滤器
const createSpatialFilter = (viewer, filterType, parameters) => {
  return (entity) => {
    const position = entity.position?.getValue()
    if (!position) return false
    
    const cartographic = Cesium.Cartographic.fromCartesian(position)
    const longitude = Cesium.Math.toDegrees(cartographic.longitude)
    const latitude = Cesium.Math.toDegrees(cartographic.latitude)
    
    switch (filterType) {
      case 'bbox':
        // 边界框过滤
        const { west, south, east, north } = parameters
        return longitude >= west && longitude <= east && 
               latitude >= south && latitude <= north
      
      case 'circle':
        // 圆形区域过滤
        const { centerLon, centerLat, radius } = parameters
        const centerCartesian = Cesium.Cartesian3.fromDegrees(centerLon, centerLat)
        const distance = Cesium.Cartesian3.distance(position, centerCartesian)
        return distance <= radius
      
      case 'polygon':
        // 多边形区域过滤（简化版点在多边形内判断）
        const { points } = parameters
        return isPointInPolygon({ longitude, latitude }, points)
      
      default:
        return true
    }
  }
}

// 点在多边形内判断（射线法）
const isPointInPolygon = (point, polygon) => {
  let inside = false
  const { longitude: x, latitude: y } = point
  
  for (let i = 0, j = polygon.length - 1; i < polygon.length; j = i++) {
    const xi = polygon[i].longitude
    const yi = polygon[i].latitude
    const xj = polygon[j].longitude
    const yj = polygon[j].latitude
    
    if (((yi > y) !== (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi)) {
      inside = !inside
    }
  }
  
  return inside
}

// 过滤器组合管理器
class FilterCombinationManager {
  constructor(filterManager) {
    this.filterManager = filterManager
    this.filterGroups = new Map()
    this.combinationMode = 'AND' // 'AND' | 'OR'
  }
  
  // 创建过滤器组
  createFilterGroup(groupName, filters, mode = 'AND') {
    this.filterGroups.set(groupName, { filters, mode })
    this.updateCombinedFilter()
  }
  
  // 更新组合过滤器
  updateCombinedFilter() {
    const combinedFilter = (entity) => {
      const groupResults = []
      
      for (let [groupName, group] of this.filterGroups) {
        const { filters, mode } = group
        
        if (mode === 'AND') {
          // 组内所有过滤器都必须通过
          const groupResult = filters.every(filter => filter(entity))
          groupResults.push(groupResult)
        } else {
          // 组内任一过滤器通过即可
          const groupResult = filters.some(filter => filter(entity))
          groupResults.push(groupResult)
        }
      }
      
      // 组间组合
      if (this.combinationMode === 'AND') {
        return groupResults.every(result => result)
      } else {
        return groupResults.some(result => result)
      }
    }
    
    this.filterManager.addFilter('combined', combinedFilter)
  }
  
  // 设置组合模式
  setCombinationMode(mode) {
    this.combinationMode = mode
    this.updateCombinedFilter()
  }
}

// 过滤历史管理器
class FilterHistoryManager {
  constructor() {
    this.history = []
    this.currentIndex = -1
    this.maxHistorySize = 50
  }
  
  // 保存过滤状态
  saveState(filterState, description) {
    // 移除当前位置之后的历史记录
    this.history = this.history.slice(0, this.currentIndex + 1)
    
    // 添加新状态
    this.history.push({
      state: JSON.parse(JSON.stringify(filterState)),
      description: description,
      timestamp: new Date().toISOString()
    })
    
    // 限制历史记录大小
    if (this.history.length > this.maxHistorySize) {
      this.history.shift()
    } else {
      this.currentIndex++
    }
  }
  
  // 撤销操作
  undo() {
    if (this.currentIndex > 0) {
      this.currentIndex--
      return this.history[this.currentIndex]
    }
    return null
  }
  
  // 重做操作
  redo() {
    if (this.currentIndex < this.history.length - 1) {
      this.currentIndex++
      return this.history[this.currentIndex]
    }
    return null
  }
  
  // 获取历史记录列表
  getHistory() {
    return this.history.map((item, index) => ({
      ...item,
      isCurrent: index === this.currentIndex
    }))
  }
  
  // 跳转到指定历史状态
  goToState(index) {
    if (index >= 0 && index < this.history.length) {
      this.currentIndex = index
      return this.history[index]
    }
    return null

// 过滤器性能优化
class OptimizedFilterManager extends DataFilterManager {
  constructor(viewer) {
    super(viewer)
    this.filterCache = new Map()
    this.lastFilterHash = null
  }
  
  // 缓存过滤结果
  applyFilters() {
    const filterHash = this.getFilterHash()
    
    // 如果过滤条件没有变化，使用缓存结果
    if (filterHash === this.lastFilterHash && this.filterCache.has(filterHash)) {
      this.filteredEntities = this.filterCache.get(filterHash)
      this.updateEntityVisibility()
      return
    }
    
    // 执行过滤
    const entities = this.getAllEntities()
    this.filteredEntities = entities.filter(entity => {
      for (let [name, filterFunc] of this.filters) {
        if (!filterFunc(entity)) {
          return false
        }
      }
      return true
    })
    
    // 缓存结果
    this.filterCache.set(filterHash, [...this.filteredEntities])
    this.lastFilterHash = filterHash
    
    // 限制缓存大小
    if (this.filterCache.size > 10) {
      const firstKey = this.filterCache.keys().next().value
      this.filterCache.delete(firstKey)
    }
    
    this.updateEntityVisibility()
  }
  
  // 生成过滤器哈希
  getFilterHash() {
    const filterKeys = Array.from(this.filters.keys()).sort()
    return filterKeys.join('|')
  }
  
  // 批量更新实体显示状态
  updateEntityVisibility() {
    // 暂时禁用渲染
    this.viewer.scene.requestRenderMode = false
    
    const entities = this.getAllEntities()
    const filteredSet = new Set(this.filteredEntities)
    
    entities.forEach(entity => {
      entity.show = filteredSet.has(entity)
    })
    
    // 重新启用渲染
    this.viewer.scene.requestRenderMode = true
    this.viewer.scene.requestRender()
  }
}

// 模糊搜索过滤器
const createFuzzySearchFilter = (searchText, threshold = 0.6) => {
  return (entity) => {
    const entityName = entity.name || ''
    const similarity = calculateSimilarity(searchText.toLowerCase(), entityName.toLowerCase())
    return similarity >= threshold
  }
}

// 计算字符串相似度（简化版）
const calculateSimilarity = (str1, str2) => {
  const longer = str1.length > str2.length ? str1 : str2
  const shorter = str1.length > str2.length ? str2 : str1
  
  if (longer.length === 0) return 1.0
  
  const editDistance = levenshteinDistance(longer, shorter)
  return (longer.length - editDistance) / longer.length
}

// 计算编辑距离
const levenshteinDistance = (str1, str2) => {
  const matrix = []
  
  for (let i = 0; i <= str2.length; i++) {
    matrix[i] = [i]
  }
  
  for (let j = 0; j <= str1.length; j++) {
    matrix[0][j] = j
  }
  
  for (let i = 1; i <= str2.length; i++) {
    for (let j = 1; j <= str1.length; j++) {
      if (str2.charAt(i - 1) === str1.charAt(j - 1)) {
        matrix[i][j] = matrix[i - 1][j - 1]
      } else {
        matrix[i][j] = Math.min(
          matrix[i - 1][j - 1] + 1,
          matrix[i][j - 1] + 1,
          matrix[i - 1][j] + 1
        )
      }
    }
  }
  
  return matrix[str2.length][str1.length]
}

// 智能推荐过滤器
class SmartFilterRecommender {
  constructor(filterManager) {
    this.filterManager = filterManager
    this.userPreferences = new Map()
    this.filterUsageStats = new Map()
  }
  
  // 记录用户过滤偏好
  recordFilterUsage(filterName, filterParams) {
    const key = `${filterName}_${JSON.stringify(filterParams)}`
    const count = this.filterUsageStats.get(key) || 0
    this.filterUsageStats.set(key, count + 1)
  }
  
  // 获取推荐过滤器
  getRecommendedFilters(currentContext) {
    const recommendations = []
    
    // 基于使用频率推荐
    const sortedFilters = Array.from(this.filterUsageStats.entries())
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
    
    sortedFilters.forEach(([filterKey, count]) => {
      recommendations.push({
        type: 'frequent',
        filterKey: filterKey,
        score: count,
        description: `常用过滤器 (使用 ${count} 次)`
      })
    })
    
    // 基于上下文推荐
    if (currentContext.selectedEntities && currentContext.selectedEntities.length > 0) {
      const selectedEntity = currentContext.selectedEntities[0]
      const properties = selectedEntity.properties
      
      if (properties) {
        recommendations.push({
          type: 'contextual',
          filterKey: 'similar_entities',
          score: 0.8,
          description: '查找相似实体'
        })
      }
    }
    
    return recommendations.sort((a, b) => b.score - a.score)
  }
}
