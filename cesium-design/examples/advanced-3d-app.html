高级 3D 应用示例 ​

这是一个综合性的高级 3D 应用示例，展示了 Cesium 的多种高级功能的组合使用。

功能说明 ​

  • 地下空间可视化（管线、地质结构）
  • 模型动画系统（路径动画、关键帧动画）
  • 多层次场景管理
  • 时间轴控制
  • 高级交互功能
  • 场景切换与过渡

核心 API ​

  • Cesium.ClippingPlane - 裁剪平面
  • Cesium.ClippingPlaneCollection - 裁剪平面集合
  • Cesium.SampledPositionProperty - 采样位置属性
  • Cesium.VelocityOrientationProperty - 速度方向属性
  • Cesium.PathGraphics - 路径图形

              光照效果

              阴影效果

              雾效果

          <h4>统计信息</h4>

              <span class="stat-value">{{ sceneNames[currentScene] }}</span>
            </div>

              <span class="stat-label">实体数量:</span>
              <span class="stat-value">{{ entityCount }}</span>

              <span class="stat-label">FPS:</span>

<script setup>
import { ref, onMounted, onUnmounted, computed, getCurrentInstance } from 'vue'

const { proxy } = getCurrentInstance()

const cesiumContainer = ref(null)
const loading = ref(true)
const error = ref('')

let viewer = null
let clippingPlanes = null
let pathAnimation = null

// 全屏切换
const toggleFullscreen = () => {
  const element = proxy.$el
  if (!document.fullscreenElement) {
    element.requestFullscreen()
  } else {
    document.exitFullscreen()
  }
}

let animationInterval = null

// 场景状态
const currentScene = ref('surface')
const sceneNames = {
  surface: '地表场景',
  underground: '地下场景',
  animation: '动画场景'
}

// 地下可视化状态
const undergroundMode = ref(false)
const clippingDepth = ref(20)

// 动画状态
const hasAnimation = ref(false)
const isPlaying = ref(false)
const animationSpeed = ref(1.0)
const animationDuration = ref(60)
const currentTime = ref(0)

// 图层状态
const layers = ref([
  { name: '地表', visible: true, opacity: 1.0 },
  { name: '建筑', visible: true, opacity: 1.0 },
  { name: '地下设施', visible: false, opacity: 0.8 }
])

// 环境状态
const lighting = ref(false)
const shadows = ref(false)
const fog = ref(false)

// 统计信息
const entityCount = ref(0)
const fps = ref(60)

// 加载 Cesium
const loadCesium = () => {
  return new Promise((resolve, reject) => {
    if (window.Cesium) {
      resolve(window.Cesium)
      return
    }

    const script = document.createElement('script')
    script.src = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Cesium.js'
    script.onload = () => {
      const link = document.createElement('link')
      link.rel = 'stylesheet'
      link.href = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Widgets/widgets.css'
      link.onload = () => resolve(window.Cesium)
      link.onerror = () => reject(new Error('加载 Cesium CSS 失败'))
      document.head.appendChild(link)
    }
    script.onerror = () => reject(new Error('加载 Cesium JS 失败'))
    document.head.appendChild(script)
  })
}

// 初始化 Cesium
const initCesium = async () => {
  try {
    loading.value = true
    const Cesium = await loadCesium()

    Cesium.Ion.defaultAccessToken = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJqdGkiOiI5N2UyMjcwOS00MDY1LTQxYjEtYjZjMy00YTU0ZTg1YmJjMGEiLCJpZCI6MjU5MCwic2NvcGVzIjpbImFzciIsImdjIl0sImlhdCI6MTU0Mjg2MTUzMX0.Hvl3oBH8t1pJOOkOXJJ0M0pHcJJ8Gx8Hb3Ks3k7Ks3k'

    viewer = new Cesium.Viewer(cesiumContainer.value, {
      animation: false,
      timeline: false,
      baseLayerPicker: false,
      geocoder: false,
      homeButton: true,
      sceneModePicker: false,
      navigationHelpButton: false,
      fullscreenButton: false
    })

    // 设置初始视角
    viewer.camera.setView({
      destination: Cesium.Cartesian3.fromDegrees(116.4074, 39.9042, 5000),
      orientation: {
        heading: Cesium.Math.toRadians(0),
        pitch: Cesium.Math.toRadians(-45),
        roll: 0.0
      }
    })

    // 开始性能监控
    startPerformanceMonitoring()

    loading.value = false
  } catch (err) {
    error.value = err.message
    loading.value = false
  }
}

// 切换场景
const switchScene = async () => {
  if (!viewer) return

  const Cesium = window.Cesium

  // 清除所有实体
  viewer.entities.removeAll()
  entityCount.value = 0

  // 重置状态
  if (undergroundMode.value) {
    toggleUndergroundMode()
  }
  if (hasAnimation.value) {
    stopAnimation()
  }

  // 根据场景设置相机
  switch (currentScene.value) {
    case 'surface':
      await viewer.camera.flyTo({
        destination: Cesium.Cartesian3.fromDegrees(116.4074, 39.9042, 5000),
        orientation: {
          heading: Cesium.Math.toRadians(0),
          pitch: Cesium.Math.toRadians(-45),
          roll: 0.0
        },
        duration: 2
      })
      break

    case 'underground':
      await viewer.camera.flyTo({
        destination: Cesium.Cartesian3.fromDegrees(116.4074, 39.9042, 2000),
        orientation: {
          heading: Cesium.Math.toRadians(0),
          pitch: Cesium.Math.toRadians(-60),
          roll: 0.0
        },
        duration: 2
      })
      break

    case 'animation':
      await viewer.camera.flyTo({
        destination: Cesium.Cartesian3.fromDegrees(116.4074, 39.9042, 10000),
        orientation: {
          heading: Cesium.Math.toRadians(0),
          pitch: Cesium.Math.toRadians(-30),
          roll: 0.0
        },
        duration: 2
      })
      break
  }
}

// 切换地下模式
const toggleUndergroundMode = () => {
  if (!viewer) return

  const Cesium = window.Cesium

  if (undergroundMode.value) {
    // 启用地下模式
    viewer.scene.globe.translucency.enabled = true
    viewer.scene.globe.translucency.frontFaceAlpha = 0.3

    // 创建裁剪平面
    clippingPlanes = new Cesium.ClippingPlaneCollection({
      planes: [
        new Cesium.ClippingPlane(new Cesium.Cartesian3(0.0, 0.0, 1.0), -clippingDepth.value)
      ],
      edgeWidth: 1.0,
      edgeColor: Cesium.Color.WHITE
    })

    viewer.scene.globe.clippingPlanes = clippingPlanes
  } else {
    // 禁用地下模式
    viewer.scene.globe.translucency.enabled = false
    viewer.scene.globe.clippingPlanes = undefined
    clippingPlanes = null
  }
}

// 更新裁剪深度
const updateClippingDepth = () => {
  if (clippingPlanes) {
    const plane = clippingPlanes.get(0)
    plane.distance = -clippingDepth.value
  }
}

// 创建管线
const createPipeline = () => {
  if (!viewer) return

  const Cesium = window.Cesium

  // 创建管线路径
  const positions = [
    Cesium.Cartesian3.fromDegrees(116.40, 39.90, -20),
    Cesium.Cartesian3.fromDegrees(116.41, 39.90, -20),
    Cesium.Cartesian3.fromDegrees(116.41, 39.91, -20),
    Cesium.Cartesian3.fromDegrees(116.42, 39.91, -20)
  ]

  // 创建管道形状
  const shape = []
  const radius = 0.5
  const segments = 16

  for (let i = 0; i < segments; i++) {
    const angle = (i / segments) * 2 * Math.PI
    shape.push(new Cesium.Cartesian2(radius * Math.cos(angle), radius * Math.sin(angle)))
  }

  viewer.entities.add({
    name: '地下管线',
    polylineVolume: {
      positions: positions,
      shape: shape,
      material: Cesium.Color.BLUE.withAlpha(0.8),
      outline: true,
      outlineColor: Cesium.Color.WHITE
    }
  })

  entityCount.value = viewer.entities.values.length
}

// 创建地质层
const createGeologicalLayers = () => {
  if (!viewer) return

  const Cesium = window.Cesium

  const layers = [
    { name: '表土层', depth: -5, color: Cesium.Color.BROWN },
    { name: '砂土层', depth: -15, color: Cesium.Color.YELLOW },
    { name: '粘土层', depth: -30, color: Cesium.Color.ORANGE },
    { name: '岩石层', depth: -50, color: Cesium.Color.GRAY }
  ]

  layers.forEach(layer => {
    viewer.entities.add({
      name: layer.name,
      rectangle: {
        coordinates: Cesium.Rectangle.fromDegrees(116.39, 39.89, 116.43, 39.93),
        height: layer.depth,
        extrudedHeight: layer.depth - 10,
        material: layer.color.withAlpha(0.6),
        outline: true,
        outlineColor: Cesium.Color.BLACK
      }
    })
  })

  entityCount.value = viewer.entities.values.length
}

// 清除地下实体
const clearUnderground = () => {
  if (!viewer) return
  viewer.entities.removeAll()
  entityCount.value = 0
}

// 创建路径动画
const createPathAnimation = () => {
  if (!viewer) return

  const Cesium = window.Cesium

  // 定义路径点
  const pathPositions = [
    { longitude: 116.38, latitude: 39.88, height: 1000 },
    { longitude: 116.40, latitude: 39.90, height: 1500 },
    { longitude: 116.42, latitude: 39.92, height: 1200 },
    { longitude: 116.44, latitude: 39.94, height: 1000 }
  ]

  const startTime = Cesium.JulianDate.now()
  const stopTime = Cesium.JulianDate.addSeconds(startTime, animationDuration.value, new Cesium.JulianDate())

  // 创建采样位置属性
  const positionProperty = new Cesium.SampledPositionProperty()

  pathPositions.forEach((pos, index) => {
    const time = Cesium.JulianDate.addSeconds(
      startTime,
      (animationDuration.value / pathPositions.length) * index,
      new Cesium.JulianDate()
    )

    const position = Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.height)
    positionProperty.addSample(time, position)
  })

  // 创建飞机实体
  const entity = viewer.entities.add({
    availability: new Cesium.TimeIntervalCollection([
      new Cesium.TimeInterval({ start: startTime, stop: stopTime })
    ]),
    position: positionProperty,
    orientation: new Cesium.VelocityOrientationProperty(positionProperty),
    model: {
      uri: 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/SampleData/models/CesiumAir/Cesium_Air.glb',
      scale: 2.0,
      minimumPixelSize: 64
    },
    path: {
      resolution: 1,
      material: new Cesium.PolylineGlowMaterialProperty({
        glowPower: 0.1,
        color: Cesium.Color.YELLOW
      }),
      width: 3,
      leadTime: 0,
      trailTime: animationDuration.value
    }
  })

  // 设置时钟
  viewer.clock.startTime = startTime.clone()
  viewer.clock.stopTime = stopTime.clone()
  viewer.clock.currentTime = startTime.clone()
  viewer.clock.clockRange = Cesium.ClockRange.LOOP_STOP
  viewer.clock.multiplier = animationSpeed.value

  pathAnimation = { entity, startTime, stopTime }
  hasAnimation.value = true
  entityCount.value = viewer.entities.values.length

  // 跟踪动画
  viewer.trackedEntity = entity
}

// 播放动画
const playAnimation = () => {
  if (!viewer || !pathAnimation) return

  if (isPlaying.value) {
    viewer.clock.shouldAnimate = false
    isPlaying.value = false
  } else {
    viewer.clock.shouldAnimate = true
    isPlaying.value = true

    // 开始更新当前时间
    animationInterval = setInterval(() => {
      if (viewer && pathAnimation) {
        const elapsed = window.Cesium.JulianDate.secondsDifference(
          viewer.clock.currentTime,
          pathAnimation.startTime
        )
        currentTime.value = elapsed
      }
    }, 100)
  }
}

// 停止动画
const stopAnimation = () => {
  if (!viewer || !pathAnimation) return

  viewer.clock.shouldAnimate = false
  viewer.clock.currentTime = pathAnimation.startTime.clone()
  viewer.trackedEntity = undefined
  isPlaying.value = false
  currentTime.value = 0

  if (animationInterval) {
    clearInterval(animationInterval)
    animationInterval = null
  }
}

// 更新动画速度
const updateAnimationSpeed = () => {
  if (viewer) {
    viewer.clock.multiplier = animationSpeed.value
  }
}

// 切换图层
const toggleLayer = (layer) => {
  // 简化实现：这里只是演示
  console.log(`图层 ${layer.name} 可见性: ${layer.visible}`)
}

// 更新图层透明度
const updateLayerOpacity = (layer) => {
  console.log(`图层 ${layer.name} 透明度: ${layer.opacity}`)
}

// 切换光照
const toggleLighting = () => {
  if (viewer) {
    viewer.scene.globe.enableLighting = lighting.value
  }
}

// 切换阴影
const toggleShadows = () => {
  if (viewer) {
    viewer.shadows = shadows.value
  }
}

// 切换雾效果
const toggleFog = () => {
  if (viewer) {
    viewer.scene.fog.enabled = fog.value
  }
}

// 开始性能监控
const startPerformanceMonitoring = () => {
  let frameCount = 0
  let lastTime = performance.now()

  const updateFPS = () => {
    frameCount++
    const currentTime = performance.now()

    if (currentTime - lastTime >= 1000) {
      fps.value = Math.round((frameCount * 1000) / (currentTime - lastTime))
      frameCount = 0
      lastTime = currentTime
    }

    requestAnimationFrame(updateFPS)
  }

  updateFPS()
}

// 获取 FPS 样式类
const getFPSClass = (fpsValue) => {
  if (fpsValue >= 50) return 'good'
  if (fpsValue >= 30) return 'medium'
  return 'poor'
}

// 清理资源
const cleanup = () => {
  if (animationInterval) {
    clearInterval(animationInterval)
  }

  if (viewer) {
    viewer.destroy()
    viewer = null
  }
}

onMounted(() => {
  initCesium()
})

onUnmounted(() => {
  cleanup()
})
</script>

// 地下空间可视化管理器
class UndergroundVisualizationManager {
  constructor(viewer) {
    this.viewer = viewer
    this.clippingPlanes = null
    this.undergroundEntities = []
  }

  // 启用地下模式
  enableUndergroundMode() {
    // 设置地球透明度
    this.viewer.scene.globe.translucency.enabled = true
    this.viewer.scene.globe.translucency.frontFaceAlpha = 0.3
    this.viewer.scene.globe.translucency.backFaceAlpha = 0.3

    // 创建裁剪平面
    const clippingPlanes = new Cesium.ClippingPlaneCollection({
      planes: [
        new Cesium.ClippingPlane(
          new Cesium.Cartesian3(0.0, 0.0, 1.0),
          0.0
        )
      ],
      edgeWidth: 1.0,
      edgeColor: Cesium.Color.WHITE
    })

    this.viewer.scene.globe.clippingPlanes = clippingPlanes
    this.clippingPlanes = clippingPlanes
  }

  // 禁用地下模式
  disableUndergroundMode() {
    this.viewer.scene.globe.translucency.enabled = false
    this.viewer.scene.globe.clippingPlanes = undefined
    this.clippingPlanes = null
  }

  // 设置裁剪深度
  setClippingDepth(depth) {
    if (this.clippingPlanes) {
      const plane = this.clippingPlanes.get(0)
      plane.distance = -depth
    }
  }

  // 创建地下管线
  createUndergroundPipeline(positions, options = {}) {
    const {
      diameter = 1.0,
      color = Cesium.Color.BLUE,
      depth = -10
    } = options

    // 将位置调整到地下
    const undergroundPositions = positions.map(pos => {
      const cartographic = Cesium.Cartographic.fromDegrees(
        pos.longitude,
        pos.latitude,
        depth
      )
      return Cesium.Cartesian3.fromRadians(
        cartographic.longitude,
        cartographic.latitude,
        cartographic.height
      )
    })

    const pipeline = this.viewer.entities.add({
      name: '地下管线',
      polylineVolume: {
        positions: undergroundPositions,
        shape: this.computePipeShape(diameter),
        material: color.withAlpha(0.8),
        outline: true,
        outlineColor: Cesium.Color.WHITE,
        outlineWidth: 2
      }
    })

    this.undergroundEntities.push(pipeline)
    return pipeline
  }

  // 计算管道形状
  computePipeShape(diameter) {
    const radius = diameter / 2
    const positions = []
    const segments = 32

    for (let i = 0; i < segments; i++) {
      const angle = (i / segments) * 2 * Math.PI
      positions.push(
        new Cesium.Cartesian2(
          radius * Math.cos(angle),
          radius * Math.sin(angle)
        )
      )
    }

    return positions
  }

  // 创建地质层
  createGeologicalLayer(bounds, depth, color, name) {
    const { west, south, east, north } = bounds

    const layer = this.viewer.entities.add({
      name: name,
      rectangle: {
        coordinates: Cesium.Rectangle.fromDegrees(west, south, east, north),
        height: depth,
        extrudedHeight: depth - 5,
        material: color.withAlpha(0.6),
        outline: true,
        outlineColor: Cesium.Color.BLACK
      }
    })

    this.undergroundEntities.push(layer)
    return layer
  }

  // 清除地下实体
  clearUndergroundEntities() {
    this.undergroundEntities.forEach(entity => {
      this.viewer.entities.remove(entity)
    })
    this.undergroundEntities = []
  }
}

// 路径动画管理器
class PathAnimationManager {
  constructor(viewer) {
    this.viewer = viewer
    this.animations = new Map()
  }

  // 创建路径动画
  createPathAnimation(modelUrl, pathPositions, options = {}) {
    const {
      duration = 60, // 秒
      startTime = Cesium.JulianDate.now(),
      scale = 1.0,
      showPath = true
    } = options

    const stopTime = Cesium.JulianDate.addSeconds(
      startTime,
      duration,
      new Cesium.JulianDate()
    )

    // 创建采样位置属性
    const positionProperty = new Cesium.SampledPositionProperty()

    pathPositions.forEach((pos, index) => {
      const time = Cesium.JulianDate.addSeconds(
        startTime,
        (duration / pathPositions.length) * index,
        new Cesium.JulianDate()
      )

      const position = Cesium.Cartesian3.fromDegrees(
        pos.longitude,
        pos.latitude,
        pos.height || 0
      )

      positionProperty.addSample(time, position)
    })

    // 设置插值算法
    positionProperty.setInterpolationOptions({
      interpolationDegree: 5,
      interpolationAlgorithm: Cesium.LagrangePolynomialApproximation
    })

    // 创建实体
    const entity = this.viewer.entities.add({
      availability: new Cesium.TimeIntervalCollection([
        new Cesium.TimeInterval({
          start: startTime,
          stop: stopTime
        })
      ]),
      position: positionProperty,
      orientation: new Cesium.VelocityOrientationProperty(positionProperty),
      model: {
        uri: modelUrl,
        scale: scale,
        minimumPixelSize: 64
      },
      path: showPath ? {
        resolution: 1,
        material: new Cesium.PolylineGlowMaterialProperty({
          glowPower: 0.1,
          color: Cesium.Color.YELLOW
        }),
        width: 3,
        leadTime: 0,
        trailTime: duration
      } : undefined
    })

    // 设置时钟
    this.viewer.clock.startTime = startTime.clone()
    this.viewer.clock.stopTime = stopTime.clone()
    this.viewer.clock.currentTime = startTime.clone()
    this.viewer.clock.clockRange = Cesium.ClockRange.LOOP_STOP
    this.viewer.clock.multiplier = 1

    const animationId = `path_${Date.now()}`
    this.animations.set(animationId, {
      entity,
      startTime,
      stopTime,
      duration
    })

    return {
      id: animationId,
      entity,
      startTime,
      stopTime
    }
  }

  // 播放动画
  playAnimation(animationId) {
    const animation = this.animations.get(animationId)
    if (animation) {
      this.viewer.clock.currentTime = animation.startTime.clone()
      this.viewer.clock.shouldAnimate = true
    }
  }

  // 暂停动画
  pauseAnimation() {
    this.viewer.clock.shouldAnimate = false
  }

  // 停止动画
  stopAnimation(animationId) {
    const animation = this.animations.get(animationId)
    if (animation) {
      this.viewer.clock.shouldAnimate = false
      this.viewer.clock.currentTime = animation.startTime.clone()
    }
  }

  // 设置动画速度
  setAnimationSpeed(multiplier) {
    this.viewer.clock.multiplier = multiplier
  }

  // 删除动画
  removeAnimation(animationId) {
    const animation = this.animations.get(animationId)
    if (animation) {
      this.viewer.entities.remove(animation.entity)
      this.animations.delete(animationId)
    }
  }
}

// 关键帧动画管理器
class KeyframeAnimationManager {
  constructor(viewer) {
    this.viewer = viewer
    this.animations = []
  }

  // 创建关键帧动画
  createKeyframeAnimation(entity, keyframes, options = {}) {
    const {
      duration = 5000,
      loop = false,
      easing = 'linear'
    } = options

    const animation = {
      entity,
      keyframes,
      duration,
      loop,
      easing,
      startTime: Date.now(),
      isPlaying: false
    }

    this.animations.push(animation)
    return animation
  }

  // 播放关键帧动画
  playKeyframeAnimation(animation) {
    animation.isPlaying = true
    animation.startTime = Date.now()

    const animate = () => {
      if (!animation.isPlaying) return

      const elapsed = Date.now() - animation.startTime
      let progress = elapsed / animation.duration

      if (progress >= 1) {
        if (animation.loop) {
          animation.startTime = Date.now()
          progress = 0
        } else {
          animation.isPlaying = false
          progress = 1
        }
      }

      // 应用缓动函数
      const easedProgress = this.applyEasing(progress, animation.easing)

      // 插值关键帧
      this.interpolateKeyframes(animation.entity, animation.keyframes, easedProgress)

      if (animation.isPlaying) {
        requestAnimationFrame(animate)
      }
    }

    animate()
  }

  // 应用缓动函数
  applyEasing(t, easing) {
    switch (easing) {
      case 'easeIn':
        return t * t
      case 'easeOut':
        return t * (2 - t)
      case 'easeInOut':
        return t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t
      case 'linear':
      default:
        return t
    }
  }

  // 插值关键帧
  interpolateKeyframes(entity, keyframes, progress) {
    // 找到当前进度对应的关键帧区间
    let startFrame = null
    let endFrame = null

    for (let i = 0; i < keyframes.length - 1; i++) {
      if (progress >= keyframes[i].time && progress <= keyframes[i + 1].time) {
        startFrame = keyframes[i]
        endFrame = keyframes[i + 1]
        break
      }
    }

    if (!startFrame || !endFrame) return

    // 计算区间内的进度
    const frameProgress = (progress - startFrame.time) / (endFrame.time - startFrame.time)

    // 插值位置
    if (startFrame.position && endFrame.position) {
      const startPos = Cesium.Cartesian3.fromDegrees(
        startFrame.position.longitude,
        startFrame.position.latitude,
        startFrame.position.height || 0
      )
      const endPos = Cesium.Cartesian3.fromDegrees(
        endFrame.position.longitude,
        endFrame.position.latitude,
        endFrame.position.height || 0
      )

      entity.position = Cesium.Cartesian3.lerp(
        startPos,
        endPos,
        frameProgress,
        new Cesium.Cartesian3()
      )
    }

    // 插值缩放
    if (startFrame.scale !== undefined && endFrame.scale !== undefined) {
      const scale = startFrame.scale + (endFrame.scale - startFrame.scale) * frameProgress
      if (entity.model) {
        entity.model.scale = scale
      }
    }

    // 插值颜色
    if (startFrame.color && endFrame.color) {
      const startColor = Cesium.Color.fromCssColorString(startFrame.color)
      const endColor = Cesium.Color.fromCssColorString(endFrame.color)
      const color = Cesium.Color.lerp(startColor, endColor, frameProgress, new Cesium.Color())

      if (entity.model) {
        entity.model.color = color
      }
    }
  }

  // 停止关键帧动画
  stopKeyframeAnimation(animation) {
    animation.isPlaying = false
  }
}

// 场景管理器
class SceneManager {
  constructor(viewer) {
    this.viewer = viewer
    this.scenes = new Map()
    this.currentScene = null
  }

  // 注册场景
  registerScene(name, sceneConfig) {
    this.scenes.set(name, {
      name,
      config: sceneConfig,
      entities: [],
      camera: sceneConfig.camera,
      environment: sceneConfig.environment
    })
  }

  // 切换场景
  async switchScene(sceneName, options = {}) {
    const {
      transition = true,
      duration = 2000
    } = options

    const scene = this.scenes.get(sceneName)
    if (!scene) {
      console.warn(`场景 ${sceneName} 不存在`)
      return
    }

    // 清除当前场景
    if (this.currentScene) {
      this.clearScene(this.currentScene)
    }

    // 设置环境
    if (scene.environment) {
      this.applyEnvironment(scene.environment)
    }

    // 切换相机
    if (scene.camera) {
      if (transition) {
        await this.viewer.camera.flyTo({
          destination: Cesium.Cartesian3.fromDegrees(
            scene.camera.longitude,
            scene.camera.latitude,
            scene.camera.height
          ),
          orientation: {
            heading: Cesium.Math.toRadians(scene.camera.heading || 0),
            pitch: Cesium.Math.toRadians(scene.camera.pitch || -90),
            roll: 0
          },
          duration: duration / 1000
        })
      } else {
        this.viewer.camera.setView({
          destination: Cesium.Cartesian3.fromDegrees(
            scene.camera.longitude,
            scene.camera.latitude,
            scene.camera.height
          ),
          orientation: {
            heading: Cesium.Math.toRadians(scene.camera.heading || 0),
            pitch: Cesium.Math.toRadians(scene.camera.pitch || -90),
            roll: 0
          }
        })
      }
    }

    // 加载场景内容
    if (scene.config.onLoad) {
      await scene.config.onLoad(this.viewer, scene)
    }

    this.currentScene = scene
  }

  // 应用环境设置
  applyEnvironment(environment) {
    if (environment.lighting !== undefined) {
      this.viewer.scene.globe.enableLighting = environment.lighting
    }

    if (environment.shadows !== undefined) {
      this.viewer.shadows = environment.shadows
    }

    if (environment.fog !== undefined) {
      this.viewer.scene.fog.enabled = environment.fog
    }

    if (environment.skyAtmosphere !== undefined) {
      this.viewer.scene.skyAtmosphere.show = environment.skyAtmosphere
    }
  }

  // 清除场景
  clearScene(scene) {
    scene.entities.forEach(entity => {
      this.viewer.entities.remove(entity)
    })
    scene.entities = []
  }

  // 获取当前场景
  getCurrentScene() {
    return this.currentScene
  }
}

// 多层次可视化管理器
class MultiLayerVisualizationManager {
  constructor(viewer) {
    this.viewer = viewer
    this.layers = new Map()
  }

  // 创建图层
  createLayer(name, options = {}) {
    const {
      visible = true,
      opacity = 1.0,
      zIndex = 0
    } = options

    const layer = {
      name,
      visible,
      opacity,
      zIndex,
      entities: []
    }

    this.layers.set(name, layer)
    return layer
  }

  // 添加实体到图层
  addEntityToLayer(layerName, entity) {
    const layer = this.layers.get(layerName)
    if (layer) {
      layer.entities.push(entity)
      entity.show = layer.visible

      // 应用透明度
      if (entity.model) {
        entity.model.color = Cesium.Color.WHITE.withAlpha(layer.opacity)
      }
    }
  }

  // 设置图层可见性
  setLayerVisibility(layerName, visible) {
    const layer = this.layers.get(layerName)
    if (layer) {
      layer.visible = visible
      layer.entities.forEach(entity => {
        entity.show = visible
      })
    }
  }

  // 设置图层透明度
  setLayerOpacity(layerName, opacity) {
    const layer = this.layers.get(layerName)
    if (layer) {
      layer.opacity = opacity
      layer.entities.forEach(entity => {
        if (entity.model) {
          const currentColor = entity.model.color.getValue()
          entity.model.color = currentColor.withAlpha(opacity)
        }
      })
    }
  }

  // 获取图层
  getLayer(layerName) {
    return this.layers.get(layerName)
  }

  // 获取所有图层
  getAllLayers() {
    return Array.from(this.layers.values())
  }

  // 删除图层
  removeLayer(layerName) {
    const layer = this.layers.get(layerName)
    if (layer) {
      layer.entities.forEach(entity => {
        this.viewer.entities.remove(entity)
      })
      this.layers.delete(layerName)
    }
  }
}
