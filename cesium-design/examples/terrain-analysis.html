地形分析工具 ​

本示例展示如何在 Cesium 中实现地形分析功能，包括坡度分析、坡向分析、视域分析、剖面分析等。

核心功能 ​

1. 坡度分析 ​

  • 坡度计算 - 计算地形表面的坡度
  • 坡度着色 - 根据坡度值进行颜色渲染
  • 坡度分级 - 按坡度范围分类显示

2. 坡向分析 ​

  • 坡向计算 - 计算地形表面的朝向
  • 坡向着色 - 根据朝向进行颜色渲染
  • 八方位显示 - 东、南、西、北等方位

3. 视域分析 ​

  • 可视区域 - 计算观察点的可视范围
  • 通视分析 - 判断两点之间是否可见
  • 视线剖面 - 显示视线路径的高程剖面

4. 剖面分析 ​

  • 地形剖面 - 沿指定路径的地形剖面
  • 高程曲线 - 绘制高程变化曲线
  • 坡度曲线 - 绘制坡度变化曲线

5. 等高线生成 ​

  • 等高线提取 - 从地形数据提取等高线
  • 等高线渲染 - 可视化显示等高线
  • 高程标注 - 标注等高线高程值

坡度计算 ​

javascript
// 计算坡度（使用有限差分法）
async function calculateSlope(position, sampleDistance = 10) {
  const cartographic = Cesium.Cartographic.fromCartesian(position);
  
  // 采样周围4个点
  const north = new Cesium.Cartographic(
    cartographic.longitude,
    cartographic.latitude + Cesium.Math.toRadians(sampleDistance / 111320),
    0
  );
  const south = new Cesium.Cartographic(
    cartographic.longitude,
    cartographic.latitude - Cesium.Math.toRadians(sampleDistance / 111320),
    0
  );
  const east = new Cesium.Cartographic(
    cartographic.longitude + Cesium.Math.toRadians(sampleDistance / (111320 * Math.cos(cartographic.latitude))),
    cartographic.latitude,
    0
  );
  const west = new Cesium.Cartographic(
    cartographic.longitude - Cesium.Math.toRadians(sampleDistance / (111320 * Math.cos(cartographic.latitude))),
    cartographic.latitude,
    0
  );
  
  // 查询高程
  const positions = [cartographic, north, south, east, west];
  const updatedPositions = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    positions
  );
  
  // 计算梯度
  const dz_dx = (updatedPositions[3].height - updatedPositions[4].height) / (2 * sampleDistance);
  const dz_dy = (updatedPositions[1].height - updatedPositions[2].height) / (2 * sampleDistance);
  
  // 计算坡度（度）
  const slope = Math.atan(Math.sqrt(dz_dx * dz_dx + dz_dy * dz_dy));
  return Cesium.Math.toDegrees(slope);
}

// 坡度着色
function getSlopeColor(slope) {
  if (slope < 5) return Cesium.Color.GREEN;
  if (slope < 15) return Cesium.Color.YELLOW;
  if (slope < 25) return Cesium.Color.ORANGE;
  if (slope < 35) return Cesium.Color.RED;
  return Cesium.Color.DARKRED;
}

坡向计算 ​

// 计算坡向（方位角）
async function calculateAspect(position, sampleDistance = 10) {
  const cartographic = Cesium.Cartographic.fromCartesian(position);
  
  // 采样周围4个点
  const north = new Cesium.Cartographic(
    cartographic.longitude,
    cartographic.latitude + Cesium.Math.toRadians(sampleDistance / 111320),
    0
  );
  const south = new Cesium.Cartographic(
    cartographic.longitude,
    cartographic.latitude - Cesium.Math.toRadians(sampleDistance / 111320),
    0
  );
  const east = new Cesium.Cartographic(
    cartographic.longitude + Cesium.Math.toRadians(sampleDistance / (111320 * Math.cos(cartographic.latitude))),
    cartographic.latitude,
    0
  );
  const west = new Cesium.Cartographic(
    cartographic.longitude - Cesium.Math.toRadians(sampleDistance / (111320 * Math.cos(cartographic.latitude))),
    cartographic.latitude,
    0
  );
  
  // 查询高程
  const positions = [north, south, east, west];
  const updatedPositions = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    positions
  );
  
  // 计算梯度
  const dz_dx = (updatedPositions[2].height - updatedPositions[3].height) / (2 * sampleDistance);
  const dz_dy = (updatedPositions[0].height - updatedPositions[1].height) / (2 * sampleDistance);
  
  // 计算坡向（度，0=北，90=东，180=南，270=西）
  let aspect = Math.atan2(dz_dx, dz_dy);
  aspect = Cesium.Math.toDegrees(aspect);
  aspect = (aspect + 360) % 360;
  
  return aspect;
}

// 坡向着色
function getAspectColor(aspect) {
  // 北: 蓝色, 东: 黄色, 南: 红色, 西: 绿色
  const hue = aspect;
  return Cesium.Color.fromHsl(hue / 360, 0.8, 0.5);
}

视域分析 ​

javascript
// 视域分析
async function viewshedAnalysis(observerPosition, radius, viewAngle) {
  const observerCart = Cesium.Cartographic.fromCartesian(observerPosition);
  
  // 生成采样点网格
  const gridSize = 50;
  const samplePositions = [];
  
  for (let i = 0; i < gridSize; i++) {
    for (let j = 0; j < gridSize; j++) {
      const angle = (i / gridSize) * 2 * Math.PI;
      const distance = (j / gridSize) * radius;
      
      const lon = observerCart.longitude + (distance * Math.sin(angle)) / (111320 * Math.cos(observerCart.latitude));
      const lat = observerCart.latitude + (distance * Math.cos(angle)) / 111320;
      
      samplePositions.push(new Cesium.Cartographic(lon, lat, 0));
    }
  }
  
  // 查询地形高程
  const updatedPositions = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    samplePositions
  );
  
  // 计算可见性
  const visiblePositions = [];
  const invisiblePositions = [];
  
  for (const targetCart of updatedPositions) {
    const isVisible = await checkVisibility(observerCart, targetCart);
    if (isVisible) {
      visiblePositions.push(Cesium.Cartographic.toCartesian(targetCart));
    } else {
      invisiblePositions.push(Cesium.Cartographic.toCartesian(targetCart));
    }
  }
  
  return { visible: visiblePositions, invisible: invisiblePositions };
}

// 检查两点之间的可见性
async function checkVisibility(observer, target) {
  const numSamples = 100;
  const samplePositions = [];
  
  // 在两点之间采样
  for (let i = 0; i <= numSamples; i++) {
    const t = i / numSamples;
    const lon = observer.longitude + t * (target.longitude - observer.longitude);
    const lat = observer.latitude + t * (target.latitude - observer.latitude);
    samplePositions.push(new Cesium.Cartographic(lon, lat, 0));
  }
  
  // 查询地形高程
  const updatedPositions = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    samplePositions
  );
  
  // 检查视线是否被遮挡
  const observerHeight = observer.height;
  const targetHeight = target.height;
  
  for (let i = 1; i < updatedPositions.length - 1; i++) {
    const t = i / numSamples;
    const expectedHeight = observerHeight + t * (targetHeight - observerHeight);
    const actualHeight = updatedPositions[i].height;
    
    if (actualHeight > expectedHeight) {
      return false; // 被遮挡
    }
  }
  
  return true; // 可见
}

// 地形剖面分析
async function terrainProfile(startPosition, endPosition, numSamples = 100) {
  const startCart = Cesium.Cartographic.fromCartesian(startPosition);
  const endCart = Cesium.Cartographic.fromCartesian(endPosition);
  
  // 在路径上采样
  const samplePositions = [];
  for (let i = 0; i <= numSamples; i++) {
    const t = i / numSamples;
    const lon = startCart.longitude + t * (endCart.longitude - startCart.longitude);
    const lat = startCart.latitude + t * (endCart.latitude - startCart.latitude);
    samplePositions.push(new Cesium.Cartographic(lon, lat, 0));
  }
  
  // 查询地形高程
  const updatedPositions = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    samplePositions
  );
  
  // 计算距离和高程数据
  const profileData = [];
  let cumulativeDistance = 0;
  
  for (let i = 0; i < updatedPositions.length; i++) {
    if (i > 0) {
      const pos1 = Cesium.Cartographic.toCartesian(updatedPositions[i - 1]);
      const pos2 = Cesium.Cartographic.toCartesian(updatedPositions[i]);
      cumulativeDistance += Cesium.Cartesian3.distance(pos1, pos2);
    }
    
    profileData.push({
      distance: cumulativeDistance,
      elevation: updatedPositions[i].height,
      position: updatedPositions[i]
    });
  }
  
  return profileData;
}

// 生成等高线
async function generateContours(bounds, interval = 100) {
  // 采样地形数据
  const gridSize = 100;
  const elevationGrid = [];
  
  for (let i = 0; i <= gridSize; i++) {
    elevationGrid[i] = [];
    for (let j = 0; j <= gridSize; j++) {
      const lon = bounds.west + (i / gridSize) * (bounds.east - bounds.west);
      const lat = bounds.south + (j / gridSize) * (bounds.north - bounds.south);
      const cartographic = new Cesium.Cartographic(lon, lat, 0);
      
      const updated = await Cesium.sampleTerrainMostDetailed(
        viewer.terrainProvider,
        [cartographic]
      );
      
      elevationGrid[i][j] = updated[0].height;
    }
  }
  
  // 使用 Marching Squares 算法提取等高线
  const contours = marchingSquares(elevationGrid, interval);
  
  return contours;
}

// Marching Squares 算法（简化版）

          grid[i][j + 1]
        ];
