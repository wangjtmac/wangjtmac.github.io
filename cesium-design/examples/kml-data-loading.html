KML 数据加载示例 ​

这是一个演示如何使用 Cesium 加载和处理 KML 数据的综合示例。

功能说明 ​

  • 加载不同类型的 KML 数据（点标记、路径、区域）
  • 演示 KML 样式的解析和显示
  • 支持 KML 网络链接和时间动画
  • 提供 KML 数据的交互和信息展示
  • 支持 KMZ 压缩文件格式

核心 API ​

  • Cesium.KmlDataSource - KML 数据源
  • viewer.dataSources - 数据源管理
  • dataSource.entities - 实体集合
  • KmlDataSource.load() - 加载 KML 数据

// 加载示例地标
const loadSamplePlaces = async () => {
  try {
    const dataUrl = 'data:text/xml;charset=utf-8,' + encodeURIComponent(samplePlacesKml)
    const dataSource = await window.Cesium.KmlDataSource.load(dataUrl, {
      camera: viewer.scene.camera,
      canvas: viewer.scene.canvas
    })

    dataSource.name = '示例地标'
    viewer.dataSources.add(dataSource)

    kmlSources.value.push({
      id: ++kmlSourceCounter,
      name: '示例地标',
      dataSource: dataSource,
      visible: true,
      count: dataSource.entities.values.length
    })

    viewer.flyTo(dataSource)

  } catch (err) {
    console.error('加载示例地标失败:', err)
  }
}

// 从字符串加载 KML
const loadKmlFromString = async (kmlString, name) => {
  try {
    const dataUrl = 'data:text/xml;charset=utf-8,' + encodeURIComponent(kmlString)
    const dataSource = await window.Cesium.KmlDataSource.load(dataUrl, {
      camera: viewer.scene.camera,
      canvas: viewer.scene.canvas
    })

    dataSource.name = name
    viewer.dataSources.add(dataSource)

    kmlSources.value.push({
      id: ++kmlSourceCounter,
      name: name,
      dataSource: dataSource,
      visible: true,
      count: dataSource.entities.values.length
    })

    viewer.flyTo(dataSource)

  } catch (err) {
    console.error('加载 KML 失败:', err)
  }
}

// 切换 KML 数据源可见性
const toggleKmlSource = (sourceId) => {
  const source = kmlSources.value.find(s => s.id === sourceId)
  if (source) {
    source.visible = !source.visible
    source.dataSource.show = source.visible
  }
}

// 删除 KML 数据源
const removeKmlSource = (sourceId) => {
  const index = kmlSources.value.findIndex(s => s.id === sourceId)
  if (index !== -1) {
    const source = kmlSources.value[index]
    viewer.dataSources.remove(source.dataSource)
    kmlSources.value.splice(index, 1)
  }
}

// 清除所有 KML
const clearAllKml = () => {
  viewer.dataSources.removeAll()
  kmlSources.value = []
  selectedKmlEntity.value = null
  hasTimeData.value = false
}

// 时间控制
const playAnimation = () => {
  viewer.clock.shouldAnimate = true
}

const pauseAnimation = () => {
  viewer.clock.shouldAnimate = false
}

const resetAnimation = () => {
  viewer.clock.currentTime = viewer.clock.startTime
}

const updateTimeMultiplier = () => {
  viewer.clock.multiplier = timeMultiplier.value * 60
}

// 更新当前时间显示
const updateCurrentTime = () => {
  if (hasTimeData.value) {
    setInterval(() => {
      const current = viewer.clock.currentTime
      currentTime.value = window.Cesium.JulianDate.toIso8601(current)
    }, 1000)
  }
}

// 格式化时间
const formatTime = (timeString) => {
  if (!timeString) return ''
  return new Date(timeString).toLocaleString()
}

// 格式化位置
const formatPosition = (position) => {
  if (!position) return ''
  const cartographic = window.Cesium.Cartographic.fromCartesian(position)
  const lon = window.Cesium.Math.toDegrees(cartographic.longitude).toFixed(4)
  const lat = window.Cesium.Math.toDegrees(cartographic.latitude).toFixed(4)
  return `${lon}°, ${lat}°`
}

// 设置点击事件
const setupEventHandlers = () => {
  handler = new window.Cesium.ScreenSpaceEventHandler(viewer.scene.canvas)

  handler.setInputAction((click) => {
    const pickedObject = viewer.scene.pick(click.position)

    if (pickedObject && pickedObject.id) {
      const entity = pickedObject.id

      selectedKmlEntity.value = {
        name: entity.name || '未命名',
        description: entity.description || '',
        position: entity.position ? entity.position.getValue() : null
      }

      viewer.selectedEntity = entity
    } else {
      selectedKmlEntity.value = null
      viewer.selectedEntity = undefined
    }
  }, window.Cesium.ScreenSpaceEventType.LEFT_CLICK)
}

// 加载 Cesium
const loadCesium = () => {
  return new Promise((resolve, reject) => {
    if (window.Cesium) {
      resolve(window.Cesium)
      return
    }

    const script = document.createElement('script')
    script.src = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Cesium.js'
    script.onload = () => {
      const link = document.createElement('link')
      link.rel = 'stylesheet'
      link.href = 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/Widgets/widgets.css'
      link.onload = () => resolve(window.Cesium)
      link.onerror = () => reject(new Error('加载 Cesium CSS 失败'))
      document.head.appendChild(link)
    }
    script.onerror = () => reject(new Error('加载 Cesium JS 失败'))
    document.head.appendChild(script)
  })
}

// 初始化 Cesium
const initCesium = async () => {
  try {
    loading.value = true
    const Cesium = await loadCesium()

    Cesium.Ion.defaultAccessToken = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJqdGkiOiI5N2UyMjcwOS00MDY1LTQxYjEtYjZjMy00YTU0ZTg1YmJjMGEiLCJpZCI6MjU5MCwic2NvcGVzIjpbImFzciIsImdjIl0sImlhdCI6MTU0Mjg2MTUzMX0.Hvl3oBH8t1pJOOkOXJJ0M0pHcJJ8Gx8Hb3Ks3k7Ks3k'

    viewer = new Cesium.Viewer(cesiumContainer.value, {
      animation: true,
      timeline: true,
      baseLayerPicker: false,
      geocoder: false,
      homeButton: true,
      sceneModePicker: false,
      navigationHelpButton: false,
      fullscreenButton: false
    })

    // 设置初始视角（中国）
    viewer.camera.setView({
      destination: Cesium.Cartesian3.fromDegrees(104.0, 35.0, 5000000)
    })

    // 添加影像图层
    viewer.imageryLayers.addImageryProvider(
      new Cesium.UrlTemplateImageryProvider({
        url: "https://webst04.is.autonavi.com/appmaptile?style=6&x={x}&y={y}&z={z}"
      })
    )

    // 设置事件处理
    setupEventHandlers()

    loading.value = false
  } catch (err) {
    error.value = err.message
    loading.value = false
  }
}

// 清理资源
const cleanup = () => {
  if (handler) {
    handler.destroy()
    handler = null
  }

  if (viewer) {
    viewer.destroy()
    viewer = null
  }
}

onMounted(() => {
  initCesium()
})

onUnmounted(() => {
  cleanup()
})
</script>

// 简化 KML 几何体
const simplifyKml = (kmlString) => {
  // 移除不必要的精度
  return kmlString.replace(
    /(\d+\.\d{6})\d+/g, 
    '$1'
  )

// 压缩 KML 为 KMZ
const createKmz = async (kmlString, images = []) => {
  const zip = new JSZip()
  zip.file('doc.kml', kmlString)
  
  // 添加图片资源
  images.forEach(image => {
    zip.file(image.name, image.data)
  })
  
  return await zip.generateAsync({ type: 'blob' })
}

2. 动态 KML 生成 ​

javascript
const generateKml = (features) => {
  let kml = `<?xml version="1.0" encoding="UTF-8"?>
<kml xmlns="http://www.opengis.net/kml/2.2">
  <Document>
    <name>动态生成的 KML</name>`
  
  features.forEach(feature => {
    kml += `
    <Placemark>
      <name>${feature.name}</name>
      <description>${feature.description}</description>
      <Point>
        <coordinates>${feature.lon},${feature.lat},${feature.alt}</coordinates>
      </Point>
    </Placemark>`
  })
  
  kml += `
  </Document>
</kml>`
  
  return kml
}

3. KML 数据解析 ​

javascript
const parseKmlData = (dataSource) => {
  const entities = dataSource.entities.values
  const parsedData = []
  
  entities.forEach(entity => {
    const data = {
      name: entity.name,
      description: entity.description,
      position: null,
      properties: {}
    }
    
    // 提取位置信息
    if (entity.position) {
      const cartographic = Cesium.Cartographic.fromCartesian(
        entity.position.getValue()
      )
      data.position = {
        longitude: Cesium.Math.toDegrees(cartographic.longitude),
        latitude: Cesium.Math.toDegrees(cartographic.latitude),
        height: cartographic.height
      }
    }
    
    // 提取扩展数据
    if (entity.kml && entity.kml.extendedData) {
      const extendedData = entity.kml.extendedData
      Object.keys(extendedData).forEach(key => {
        data.properties[key] = extendedData[key]
      })
    }
    
    parsedData.push(data)
  })
  
  return parsedData
}

1. 大文件处理 ​

javascript
// 分块加载大型 KML 文件
const loadLargeKml = async (url) => {
  try {
    const response = await fetch(url)
    const text = await response.text()
    
    // 解析 KML 并分批加载
    const parser = new DOMParser()
    const kmlDoc = parser.parseFromString(text, 'text/xml')
    const placemarks = kmlDoc.getElementsByTagName('Placemark')
    
    const batchSize = 100
    for (let i = 0; i < placemarks.length; i += batchSize) {
      const batch = Array.from(placemarks).slice(i, i + batchSize)
      await loadKmlBatch(batch)
      
      // 给浏览器一些时间处理其他任务
      await new Promise(resolve => setTimeout(resolve, 10))
    }
  } catch (error) {
    console.error('加载 KML 失败:', error)
  }
}
