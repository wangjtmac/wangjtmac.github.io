3D Tiles 展示示例 ​

这是一个演示如何在 Cesium 中加载和管理 3D Tiles 数据集的综合示例。

功能说明 ​

  • 支持多种 3D Tiles 数据集加载
  • LOD（细节层次）自动管理
  • 3D Tiles 样式定制
  • 性能监控和优化
  • 点云数据可视化
  • 建筑群批量展示

核心 API ​

  • Cesium.Cesium3DTileset - 3D Tiles 数据集
  • Cesium.Cesium3DTileStyle - 3D Tiles 样式
  • Cesium.Cesium3DTileFeature - 3D Tiles 要素
  • Cesium.ScreenSpaceEventType - 屏幕空间事件

import { ref, onMounted, onUnmounted, reactive, computed, getCurrentInstance } from 'vue'

const { proxy } = getCurrentInstance()

const cesiumContainer = ref(null)
const loading = ref(true)
const error = ref('')

let viewer = null
let performanceMonitor = null
let interactionManager = null

// 全屏切换
const toggleFullscreen = () => {
  const element = proxy.$el
  if (!document.fullscreenElement) {
    element.requestFullscreen()
  } else {
    document.exitFullscreen()
  }
}

let tilesetCounter = 0

// Tileset 状态
const loadedTilesets = ref([])
const selectedTileset = ref(null)
const selectedStyle = ref('')
const customColor = ref('#ffffff')
const opacity = ref(100)
const screenSpaceError = ref(16)
const maxMemoryUsage = ref(512)

// 调试选项
const debugOptions = reactive({
  showBoundingVolume: false,
  wireframe: false,
  showStatistics: false
})

// 交互状态
const featureSelectionEnabled = ref(false)
const measurementEnabled = ref(false)
const selectedFeature = ref(null)
const featureProperties = ref({})

// 统计信息
const statistics = ref(null)
const currentFPS = ref(60)
const renderTime = ref(0)

// 示例 Tileset 配置
const tilesetConfigs = {
  building: {
    name: '建筑群',
    type: 'building',
    url: 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/SampleData/Cesium3DTiles/Batched/BatchedColors/tileset.json'
  },
  pointCloud: {
    name: '点云数据',
    type: 'pointcloud',
    url: 'https://cesium.com/downloads/cesiumjs/releases/1.136/Build/Cesium/SampleData/Cesium3DTiles/PointCloud/PointCloudRGB/tileset.json'
  },
  photogrammetry: {
    name: '摄影测量',
    type: 'photogrammetry',

  }
}

// 样式定义
const styleDefinitions = {
  heightColor: {
    color: {
      conditions: [
        ['${Height} >= 83', 'rgba(45, 0, 75, 0.5)'],
        ['${Height} >= 80', 'rgba(102, 37, 6, 0.5)'],

// 性能监控器
class PerformanceMonitor {
  constructor(viewer) {
    this.viewer = viewer
    this.frameCount = 0
    this.lastTime = performance.now()
    this.fps = 60
    this.renderTime = 0
  }

  update() {
    const currentTime = performance.now()
    this.frameCount++

    if (currentTime - this.lastTime >= 1000) {
      this.fps = Math.round((this.frameCount * 1000) / (currentTime - this.lastTime))
      this.frameCount = 0
      this.lastTime = currentTime
    }

    // 估算渲染时间
    this.renderTime = Math.round(1000 / (this.fps || 1))

    return {
      fps: this.fps,
      renderTime: this.renderTime
    }
  }
}

// 交互管理器
class InteractionManager {
  constructor(viewer) {
    this.viewer = viewer
    this.handler = new window.Cesium.ScreenSpaceEventHandler(viewer.scene.canvas)
    this.selectedFeature = null
    this.originalColor = null
    this.setupEventHandlers()
  }

  setupEventHandlers() {
    this.handler.setInputAction((click) => {
      this.handleClick(click.position)
    }, window.Cesium.ScreenSpaceEventType.LEFT_CLICK)

    this.handler.setInputAction((movement) => {
      this.handleMouseMove(movement.endPosition)
    }, window.Cesium.ScreenSpaceEventType.MOUSE_MOVE)
  }

  handleClick(position) {
    const pickedFeature = this.viewer.scene.pick(position)

    if (this.selectedFeature) {
      this.clearSelection()
    }

    if (pickedFeature instanceof window.Cesium.Cesium3DTileFeature) {
      this.selectedFeature = pickedFeature
      this.highlightFeature(pickedFeature, window.Cesium.Color.YELLOW)
      this.showFeatureInfo(pickedFeature)
    }
  }

  handleMouseMove(position) {
    const pickedFeature = this.viewer.scene.pick(position)

    if (pickedFeature instanceof window.Cesium.Cesium3DTileFeature) {
      this.viewer.canvas.style.cursor = 'pointer'
    } else {
      this.viewer.canvas.style.cursor = 'default'
    }
  }

  highlightFeature(feature, color) {
    if (!this.originalColor) {
      this.originalColor = window.Cesium.Color.clone(feature.color)
    }
    feature.color = color
  }

  clearSelection() {
    if (this.selectedFeature && this.originalColor) {
      this.selectedFeature.color = this.originalColor
      this.selectedFeature = null
      this.originalColor = null
    }
  }

  showFeatureInfo(feature) {
    const properties = {}
    const propertyNames = feature.getPropertyNames()

    propertyNames.forEach(name => {
      properties[name] = feature.getProperty(name)
    })

    selectedFeature.value = feature
    featureProperties.value = properties
  }

  destroy() {
    if (this.handler) {
      this.handler.destroy()
    }
  }
}

// 加载 Cesium
const loadCesium = () => {
  return new Promise((resolve, reject) => {
    // 检查是否已加载
    if (window.Cesium) {
      resolve(window.Cesium)
      return
    }

    // 动态加载 Cesium CSS
    const link = document.createElement('link')
    link.rel = 'stylesheet'
    link.href = 'https://cdn.staticfile.net/cesium/1.113.0/Widgets/widgets.css'
    document.head.appendChild(link)

    // 动态加载 Cesium JS
    const script = document.createElement('script')
    script.src = 'https://cdn.staticfile.net/cesium/1.113.0/Cesium.js'
    script.onload = () => resolve(window.Cesium)
    script.onerror = () => reject(new Error('Cesium 加载失败'))
    document.head.appendChild(script)
  })
}

// 初始化 Cesium
const initCesium = async () => {
  try {
    loading.value = true
    const Cesium = await loadCesium()

    viewer = new Cesium.Viewer(cesiumContainer.value, {
      animation: false,
      timeline: false,
      baseLayerPicker: false,
      geocoder: false,
      homeButton: true,
      sceneModePicker: false,
      navigationHelpButton: false,
      fullscreenButton: false,
      imageryProvider: false // 禁用默认影像层
    })

    // 手动添加高德地图影像图层
    viewer.imageryLayers.addImageryProvider(
      new Cesium.UrlTemplateImageryProvider({
        url: "https://webst01.is.autonavi.com/appmaptile?style=6&x={x}&y={y}&z={z}",
        minimumLevel: 0,
        maximumLevel: 18
      })
    )

    // 设置初始视角
    viewer.camera.setView({
      destination: Cesium.Cartesian3.fromDegrees(-75.612, 40.042, 300)
    })

    // 创建性能监控器
    performanceMonitor = new PerformanceMonitor(viewer)

    // 创建交互管理器
    interactionManager = new InteractionManager(viewer)

    // 开始性能监控
    startPerformanceMonitoring()

    loading.value = false
  } catch (err) {
    error.value = err.message
    loading.value = false
  }
}

// 加载 Tileset
const loadTileset = async (config) => {
  try {
    const Cesium = window.Cesium

    const tileset = await Cesium.Cesium3DTileset.fromUrl(config.url, {
      maximumScreenSpaceError: screenSpaceError.value,
      maximumMemoryUsage: maxMemoryUsage.value
    })

    viewer.scene.primitives.add(tileset)

    // 创建 tileset 数据对象
    const tilesetData = {
      id: ++tilesetCounter,
      name: config.name,
      type: config.type,
      tileset: tileset,
      visible: true,
      ready: false
    }

    loadedTilesets.value.push(tilesetData)

    // 监听就绪事件
    tileset.readyPromise.then(() => {
      tilesetData.ready = true
      viewer.zoomTo(tileset)
      updateStatistics()
    })

    // 监听瓦片加载事件
    tileset.tileLoad.addEventListener(() => {
      updateStatistics()
    })

    tileset.tileUnload.addEventListener(() => {
      updateStatistics()
    })

    return tilesetData
  } catch (err) {
    console.error('Tileset 加载失败:', err)
    throw err
  }
}

// 加载建筑群 Tileset
const loadBuildingTileset = () => {
  loadTileset(tilesetConfigs.building)
}

// 加载点云 Tileset
const loadPointCloudTileset = () => {
  loadTileset(tilesetConfigs.pointCloud)
}

// 加载摄影测量 Tileset
const loadPhotogrammetryTileset = () => {
  loadTileset(tilesetConfigs.photogrammetry)
}

// 清除所有 Tilesets
const clearAllTilesets = () => {
  loadedTilesets.value.forEach(tilesetData => {
    viewer.scene.primitives.remove(tilesetData.tileset)
  })
  loadedTilesets.value = []
  selectedTileset.value = null
  selectedFeature.value = null
  statistics.value = null
}

// 选择 Tileset
const selectTileset = (tilesetData) => {
  selectedTileset.value = tilesetData
  viewer.zoomTo(tilesetData.tileset)
  updateStatistics()
}

// 切换 Tileset 可见性
const toggleTilesetVisibility = (tilesetData) => {
  tilesetData.visible = !tilesetData.visible
  tilesetData.tileset.show = tilesetData.visible
}

// 移除 Tileset
const removeTileset = (tilesetData) => {
  const index = loadedTilesets.value.indexOf(tilesetData)
  if (index !== -1) {
    viewer.scene.primitives.remove(tilesetData.tileset)
    loadedTilesets.value.splice(index, 1)

    if (selectedTileset.value === tilesetData) {
      selectedTileset.value = null
      statistics.value = null
    }
  }
}

// 应用样式
const applyStyle = () => {
  if (!selectedTileset.value || !selectedStyle.value) return

  const Cesium = window.Cesium
  const styleDefinition = styleDefinitions[selectedStyle.value]

  if (styleDefinition) {
    selectedTileset.value.tileset.style = new Cesium.Cesium3DTileStyle(styleDefinition)
  }
}

// 应用自定义颜色
const applyCustomColor = () => {
  if (!selectedTileset.value) return

  const Cesium = window.Cesium
  const color = Cesium.Color.fromCssColorString(customColor.value)

  selectedTileset.value.tileset.style = new Cesium.Cesium3DTileStyle({
    color: `color("${customColor.value}")`
  })
}

// 应用透明度
const applyOpacity = () => {
  if (!selectedTileset.value) return

  const Cesium = window.Cesium
  const alpha = opacity.value / 100

  selectedTileset.value.tileset.style = new Cesium.Cesium3DTileStyle({
    color: `rgba(255, 255, 255, ${alpha})`
  })
}

// 更新屏幕空间误差
const updateScreenSpaceError = () => {
  if (!selectedTileset.value) return

  selectedTileset.value.tileset.maximumScreenSpaceError = screenSpaceError.value
}

// 更新最大内存使用
const updateMaxMemoryUsage = () => {
  if (!selectedTileset.value) return

  selectedTileset.value.tileset.maximumMemoryUsage = maxMemoryUsage.value
}

// 更新调试选项
const updateDebugOptions = () => {
  if (!selectedTileset.value) return

  const tileset = selectedTileset.value.tileset
  tileset.debugShowBoundingVolume = debugOptions.showBoundingVolume
  tileset.debugWireframe = debugOptions.wireframe
  tileset.debugShowRenderingStatistics = debugOptions.showStatistics
}

// 启用要素选择
const enableFeatureSelection = () => {
  featureSelectionEnabled.value = !featureSelectionEnabled.value
  measurementEnabled.value = false
}

// 启用测量
const enableMeasurement = () => {
  measurementEnabled.value = !measurementEnabled.value
  featureSelectionEnabled.value = false
}

// 重置相机
const resetCamera = () => {
  if (selectedTileset.value) {
    viewer.zoomTo(selectedTileset.value.tileset)
  } else {
    viewer.camera.setView({
      destination: window.Cesium.Cartesian3.fromDegrees(-75.612, 40.042, 300)
    })
  }
}

// 关闭要素信息
const closeFeatureInfo = () => {
  selectedFeature.value = null
  featureProperties.value = {}

  if (interactionManager) {
    interactionManager.clearSelection()
  }
}

// 更新统计信息
const updateStatistics = () => {
  if (!selectedTileset.value) return

  const tileset = selectedTileset.value.tileset

  statistics.value = {
    tilesLoaded: tileset.tilesLoaded || 0,
    tilesLoading: 0, // 简化处理
    memoryUsage: tileset.totalMemoryUsageInBytes || 0,
    geometricError: tileset.geometricError || 0
  }
}

// 开始性能监控
const startPerformanceMonitoring = () => {
  const updatePerformance = () => {
    if (performanceMonitor) {
      const perf = performanceMonitor.update()
      currentFPS.value = perf.fps
      renderTime.value = perf.renderTime
    }
    requestAnimationFrame(updatePerformance)
  }

  updatePerformance()
}

// 格式化内存使用量
const formatMemoryUsage = (bytes) => {
  if (bytes === 0) return '0 B'

  const k = 1024
  const sizes = ['B', 'KB', 'MB', 'GB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))

  return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i]
}

// 格式化值
const formatValue = (value) => {
  if (typeof value === 'number') {
    return value.toLocaleString()
  }
  return value?.toString() || 'N/A'
}

// 获取 FPS 样式类
const getFPSClass = (fps) => {
  if (fps >= 50) return 'good'
  if (fps >= 30) return 'medium'
  return 'poor'
}

const cleanup = () => {
  if (interactionManager) {
    interactionManager.destroy()
  }

  if (viewer) {
    viewer.destroy()
    viewer = null
  }
}

onMounted(() => {

    const conditions = []
    
    for (let i = 0; i < colorStops.length - 1; i++) {
      const currentStop = colorStops[i]
      const nextStop = colorStops[i + 1]
      
      const value = minValue + (maxValue - minValue) * currentStop.position
      conditions.push([
        `\${${property}} >= ${value}`,
        currentStop.color
      ])
    }
    
    // 添加默认条件
    conditions.push(['true', colorStops[colorStops.length - 1].color])

    const styleDefinition = {
      color: { conditions }
    }

    this.styles.set(name, styleDefinition)
    return styleDefinition
  }

// 3D Tiles 性能监控器
class Cesium3DTilePerformanceMonitor {
  constructor(viewer) {
    this.viewer = viewer
    this.tilesets = new Set()
    this.statistics = {
      tilesLoaded: 0,
      tilesLoading: 0,
      tilesFailed: 0,
      memoryUsage: 0,
      geometricError: 0,
      screenSpaceError: 0
    }
    this.callbacks = new Set()
  }

  // 添加监控的 tileset
  addTileset(tileset) {
    this.tilesets.add(tileset)
    this.setupTilesetEvents(tileset)
  }

  // 移除监控的 tileset
  removeTileset(tileset) {
    this.tilesets.delete(tileset)
  }

  // 设置 tileset 事件监听
  setupTilesetEvents(tileset) {
    // 瓦片加载完成事件
    tileset.tileLoad.addEventListener((tile) => {
      this.statistics.tilesLoaded++
      this.notifyCallbacks('tileLoaded', { tile, tileset })
    })

    // 瓦片加载失败事件
    tileset.tileFailed.addEventListener((tile) => {
      this.statistics.tilesFailed++
      this.notifyCallbacks('tileFailed', { tile, tileset })
    })

    // 瓦片卸载事件
    tileset.tileUnload.addEventListener((tile) => {
      this.statistics.tilesLoaded--
      this.notifyCallbacks('tileUnloaded', { tile, tileset })
    })

    // 所有瓦片加载完成事件
    tileset.allTilesLoaded.addEventListener(() => {
      this.notifyCallbacks('allTilesLoaded', { tileset })
    })

    // 初始瓦片加载完成事件
    tileset.initialTilesLoaded.addEventListener(() => {
      this.notifyCallbacks('initialTilesLoaded', { tileset })
    })

    // 加载进度事件
    tileset.loadProgress.addEventListener((numberOfPendingRequests, numberOfTilesProcessing) => {
      this.statistics.tilesLoading = numberOfPendingRequests + numberOfTilesProcessing
      this.notifyCallbacks('loadProgress', {
        pending: numberOfPendingRequests,
        processing: numberOfTilesProcessing,
        tileset
      })
    })
  }

  // 更新统计信息
  updateStatistics() {
    let totalMemoryUsage = 0
    let totalGeometricError = 0
    let totalScreenSpaceError = 0

    this.tilesets.forEach(tileset => {
      totalMemoryUsage += tileset.totalMemoryUsageInBytes || 0
      totalGeometricError += tileset.geometricError || 0
      totalScreenSpaceError += tileset.maximumScreenSpaceError || 0
    })

    this.statistics.memoryUsage = totalMemoryUsage
    this.statistics.geometricError = totalGeometricError
    this.statistics.screenSpaceError = totalScreenSpaceError

    return this.statistics
  }

  // 获取详细统计信息
  getDetailedStatistics() {
    const stats = {
      global: this.updateStatistics(),
      tilesets: []
    }

    this.tilesets.forEach(tileset => {
      const tilesetStats = {
        url: tileset.url,
        ready: tileset.ready,
        tilesLoaded: tileset.tilesLoaded,
        memoryUsage: tileset.totalMemoryUsageInBytes,
        geometricError: tileset.geometricError,
        maximumScreenSpaceError: tileset.maximumScreenSpaceError,
        boundingSphere: tileset.boundingSphere,
        properties: tileset.properties
      }

      stats.tilesets.push(tilesetStats)
    })

    return stats
  }

  // 添加回调函数
  addCallback(callback) {
    this.callbacks.add(callback)
  }

  // 移除回调函数
  removeCallback(callback) {
    this.callbacks.delete(callback)
  }

  // 通知回调函数
  notifyCallbacks(eventType, data) {
    this.callbacks.forEach(callback => {
      try {
        callback(eventType, data)
      } catch (error) {
        console.error('性能监控回调执行失败:', error)
      }
    })
  }

  // 开始性能监控
  startMonitoring(interval = 1000) {
    this.monitoringInterval = setInterval(() => {
      const stats = this.updateStatistics()
      this.notifyCallbacks('statisticsUpdate', stats)
    }, interval)
  }

  // 停止性能监控
  stopMonitoring() {
    if (this.monitoringInterval) {
      clearInterval(this.monitoringInterval)
      this.monitoringInterval = null
    }
  }
}

// 3D Tiles 交互管理器
class Cesium3DTileInteractionManager {
  constructor(viewer) {
    this.viewer = viewer
    this.handler = new Cesium.ScreenSpaceEventHandler(viewer.scene.canvas)
    this.selectedFeature = null
    this.highlightedFeature = null
    this.originalColor = null
    this.setupEventHandlers()
  }

  // 设置事件处理器
  setupEventHandlers() {
    // 左键点击事件
    this.handler.setInputAction((click) => {
      this.handleClick(click.position)
    }, Cesium.ScreenSpaceEventType.LEFT_CLICK)

    // 鼠标移动事件
    this.handler.setInputAction((movement) => {
      this.handleMouseMove(movement.endPosition)
    }, Cesium.ScreenSpaceEventType.MOUSE_MOVE)
  }

  // 处理点击事件
  handleClick(position) {
    const pickedFeature = this.viewer.scene.pick(position)

    // 清除之前的选择
    if (this.selectedFeature) {
      this.clearSelection()
    }

    if (pickedFeature instanceof Cesium.Cesium3DTileFeature) {
      this.selectedFeature = pickedFeature
      this.highlightFeature(pickedFeature, Cesium.Color.YELLOW)
      this.showFeatureInfo(pickedFeature)
    }
  }

  // 处理鼠标移动事件
  handleMouseMove(position) {
    const pickedFeature = this.viewer.scene.pick(position)

    // 清除之前的高亮
    if (this.highlightedFeature && this.highlightedFeature !== this.selectedFeature) {
      this.clearHighlight()
    }

    if (pickedFeature instanceof Cesium.Cesium3DTileFeature && 
        pickedFeature !== this.selectedFeature) {
      this.highlightedFeature = pickedFeature
      this.highlightFeature(pickedFeature, Cesium.Color.CYAN)
      this.viewer.canvas.style.cursor = 'pointer'
    } else {
      this.viewer.canvas.style.cursor = 'default'
    }
  }

  // 高亮要素
  highlightFeature(feature, color) {
    if (!this.originalColor) {
      this.originalColor = Cesium.Color.clone(feature.color)
    }
    feature.color = color
  }

  // 清除选择
  clearSelection() {
    if (this.selectedFeature && this.originalColor) {
      this.selectedFeature.color = this.originalColor
      this.selectedFeature = null
      this.originalColor = null
    }
  }

  // 清除高亮
  clearHighlight() {
    if (this.highlightedFeature && this.originalColor) {
      this.highlightedFeature.color = this.originalColor
      this.highlightedFeature = null
    }
  }

  // 显示要素信息
  showFeatureInfo(feature) {
    const properties = feature.getPropertyNames()
    let infoHtml = '<h3>要素信息</h3><table>'

    properties.forEach(propertyName => {
      const value = feature.getProperty(propertyName)
      infoHtml += `<tr><td><strong>${propertyName}:</strong></td><td>${value}</td></tr>`
    })

    infoHtml += '</table>'

    // 创建信息窗口
    this.showInfoWindow(infoHtml)
  }

  // 显示信息窗口
  showInfoWindow(content) {
    // 移除现有的信息窗口
    const existingWindow = document.getElementById('feature-info-window')
    if (existingWindow) {
      existingWindow.remove()
    }

    // 创建新的信息窗口
    const infoWindow = document.createElement('div')
    infoWindow.id = 'feature-info-window'
    infoWindow.innerHTML = content
    infoWindow.style.cssText = `
      position: absolute;
      top: 10px;
      right: 10px;
      background: rgba(42, 42, 42, 0.9);
      color: white;
      padding: 15px;
      border-radius: 8px;
      max-width: 300px;
      z-index: 1000;
      font-size: 12px;
    `

    // 添加关闭按钮
    const closeButton = document.createElement('button')
    closeButton.innerHTML = '×'
    closeButton.style.cssText = `
      position: absolute;
      top: 5px;
      right: 8px;
      background: none;
      border: none;
      color: white;
      font-size: 18px;
      cursor: pointer;
    `
    closeButton.onclick = () => infoWindow.remove()

    infoWindow.appendChild(closeButton)
    document.body.appendChild(infoWindow)
  }

  // 批量选择要素
  selectFeaturesByProperty(tileset, propertyName, value) {
    const selectedFeatures = []

    // 遍历所有要素
    const features = tileset.root.children
    this.traverseFeatures(features, (feature) => {
      if (feature.getProperty(propertyName) === value) {
        this.highlightFeature(feature, Cesium.Color.ORANGE)
        selectedFeatures.push(feature)
      }
    })

    return selectedFeatures
  }

  // 遍历要素
  traverseFeatures(features, callback) {
    features.forEach(feature => {
      if (feature instanceof Cesium.Cesium3DTileFeature) {
        callback(feature)
      }
      if (feature.children) {
        this.traverseFeatures(feature.children, callback)
      }
    })
  }

  // 清理资源
  destroy() {
    if (this.handler) {
      this.handler.destroy()
    }
  }
}

// 3D Tiles 数据分析器
class Cesium3DTileAnalyzer {
  constructor(viewer) {
    this.viewer = viewer
    this.analysisResults = new Map()
  }

  // 分析 tileset 属性分布
  analyzePropertyDistribution(tileset, propertyName) {
    const distribution = {
      propertyName: propertyName,
      values: [],
      statistics: {
        min: Infinity,
        max: -Infinity,
        mean: 0,
        count: 0
      }
    }

    let sum = 0
    let count = 0

    // 遍历所有要素
    this.traverseTileset(tileset, (feature) => {
      const value = feature.getProperty(propertyName)
      if (value !== undefined && value !== null) {
        distribution.values.push(value)
        
        if (typeof value === 'number') {
          distribution.statistics.min = Math.min(distribution.statistics.min, value)
          distribution.statistics.max = Math.max(distribution.statistics.max, value)
          sum += value
          count++
        }
      }
    })

    if (count > 0) {
      distribution.statistics.mean = sum / count
      distribution.statistics.count = count
    }

    this.analysisResults.set(`distribution_${propertyName}`, distribution)
    return distribution
  }

  // 空间分析
  analyzeSpatialDistribution(tileset) {
    const spatialAnalysis = {
      boundingBox: null,
      center: null,
      density: 0,
      clusters: []
    }

    const positions = []
    let featureCount = 0

    this.traverseTileset(tileset, (feature) => {
      // 获取要素位置（简化处理）
      const boundingVolume = feature.content?.boundingVolume
      if (boundingVolume) {
        positions.push(boundingVolume.center)
        featureCount++
      }
    })

    if (positions.length > 0) {
      // 计算边界框
      spatialAnalysis.boundingBox = Cesium.BoundingSphere.fromPoints(positions)
      spatialAnalysis.center = spatialAnalysis.boundingBox.center
      
      // 计算密度
      const volume = (4/3) * Math.PI * Math.pow(spatialAnalysis.boundingBox.radius, 3)
      spatialAnalysis.density = featureCount / volume
    }

    this.analysisResults.set('spatial_distribution', spatialAnalysis)
    return spatialAnalysis
  }

  // 质量分析
  analyzeDataQuality(tileset) {
    const qualityAnalysis = {
      completeness: 0,
      consistency: 0,
      accuracy: 0,
      issues: []
    }

    let totalFeatures = 0
    let featuresWithAllProperties = 0
    const propertyNames = new Set()

    // 第一遍遍历：收集所有属性名
    this.traverseTileset(tileset, (feature) => {
      totalFeatures++
      const names = feature.getPropertyNames()
      names.forEach(name => propertyNames.add(name))
    })

    // 第二遍遍历：检查完整性
    this.traverseTileset(tileset, (feature) => {
      const featureProperties = feature.getPropertyNames()
      let hasAllProperties = true

      propertyNames.forEach(propertyName => {
        if (!featureProperties.includes(propertyName)) {
          hasAllProperties = false
          qualityAnalysis.issues.push({
            type: 'missing_property',
            feature: feature,
            property: propertyName
          })
        }
      })

      if (hasAllProperties) {
        featuresWithAllProperties++
      }
    })

    // 计算完整性
    qualityAnalysis.completeness = totalFeatures > 0 ? 
      featuresWithAllProperties / totalFeatures : 0

    this.analysisResults.set('data_quality', qualityAnalysis)
    return qualityAnalysis
  }

  // 遍历 tileset
  traverseTileset(tileset, callback) {
    if (!tileset.ready) {
      console.warn('Tileset 尚未准备就绪')
      return
    }

    const traverse = (tile) => {
      if (tile.content && tile.content.featuresLength > 0) {
        for (let i = 0; i < tile.content.featuresLength; i++) {
          const feature = tile.content.getFeature(i)
          callback(feature)
        }
      }

      if (tile.children) {
        tile.children.forEach(child => traverse(child))
      }
    }

    traverse(tileset.root)
  }

  // 生成分析报告
  generateAnalysisReport() {
    const report = {
      timestamp: new Date().toISOString(),
      analyses: {}
    }

    this.analysisResults.forEach((result, key) => {
      report.analyses[key] = result
    })

    return report
  }

  // 导出分析结果
  exportAnalysisResults(format = 'json') {
    const report = this.generateAnalysisReport()

    switch (format) {
      case 'json':
        return JSON.stringify(report, null, 2)
      case 'csv':
        return this.convertToCSV(report)
      default:
        return report
    }
  }

  // 转换为 CSV 格式
  convertToCSV(report) {
    let csv = 'Analysis Type,Property,Value\n'

    Object.entries(report.analyses).forEach(([analysisType, data]) => {
      if (data.statistics) {
        Object.entries(data.statistics).forEach(([key, value]) => {
          csv += `${analysisType},${key},${value}\n`
        })
      }
    })

    return csv
  }
}

// 动态 LOD 控制器
class DynamicLODController {
  constructor(viewer) {
    this.viewer = viewer
    this.tilesets = new Map()
    this.performanceTarget = 60 // 目标帧率
    this.currentFPS = 60
  }

  // 添加 tileset 到 LOD 控制
  addTileset(name, tileset, lodConfig) {
    this.tilesets.set(name, {
      tileset: tileset,
      config: lodConfig,
      originalSSE: tileset.maximumScreenSpaceError
    })
  }

  // 更新 LOD 设置
  updateLOD() {
    this.currentFPS = this.viewer.scene.frameState.fps || 60

    if (this.currentFPS < this.performanceTarget) {
      // 性能不足，降低质量
      this.decreaseQuality()
    } else if (this.currentFPS > this.performanceTarget + 10) {
      // 性能充足，提高质量
      this.increaseQuality()
    }
  }

  // 降低质量
  decreaseQuality() {
    this.tilesets.forEach((tilesetData, name) => {
      const { tileset, config } = tilesetData
      const currentSSE = tileset.maximumScreenSpaceError
      const newSSE = Math.min(currentSSE * 1.2, config.maxSSE || 64)
      
      tileset.maximumScreenSpaceError = newSSE
    })
  }

  // 提高质量
  increaseQuality() {
    this.tilesets.forEach((tilesetData, name) => {
      const { tileset, config } = tilesetData
      const currentSSE = tileset.maximumScreenSpaceError
      const newSSE = Math.max(currentSSE * 0.9, config.minSSE || 1)
      
      tileset.maximumScreenSpaceError = newSSE
    })
  }

  // 开始自动 LOD 控制
  startAutoLOD(interval = 1000) {
    this.lodInterval = setInterval(() => {
      this.updateLOD()
    }, interval)
  }

  // 停止自动 LOD 控制
  stopAutoLOD() {
    if (this.lodInterval) {
      clearInterval(this.lodInterval)
    }
  }
}

// 3D Tiles 缓存管理器
class Cesium3DTileCacheManager {
  constructor() {
    this.cache = new Map()
    this.maxCacheSize = 1024 * 1024 * 1024 // 1GB
    this.currentCacheSize = 0
  }

  // 添加到缓存
  addToCache(url, tilesetData) {
    const dataSize = this.estimateDataSize(tilesetData)
    
    // 检查缓存空间
    if (this.currentCacheSize + dataSize > this.maxCacheSize) {
      this.evictLRU(dataSize)
    }

    this.cache.set(url, {
      data: tilesetData,
      size: dataSize,
      lastAccessed: Date.now()
    })

    this.currentCacheSize += dataSize
  }

  // 从缓存获取
  getFromCache(url) {
    const cacheEntry = this.cache.get(url)
    if (cacheEntry) {
      cacheEntry.lastAccessed = Date.now()
      return cacheEntry.data
    }
    return null
  }

  // LRU 淘汰策略
  evictLRU(requiredSpace) {
    const entries = Array.from(this.cache.entries())
    entries.sort((a, b) => a[1].lastAccessed - b[1].lastAccessed)

    let freedSpace = 0
    for (const [url, entry] of entries) {
      this.cache.delete(url)
      this.currentCacheSize -= entry.size
      freedSpace += entry.size

      if (freedSpace >= requiredSpace) {
        break
      }
    }
  }

  // 估算数据大小
  estimateDataSize(data) {
    // 简化的大小估算
    return JSON.stringify(data).length * 2
  }

  // 清空缓存
  clearCache() {
    this.cache.clear()
    this.currentCacheSize = 0
  }

  // 获取缓存统计
  getCacheStats() {
    return {
      entryCount: this.cache.size,
      totalSize: this.currentCacheSize,
      maxSize: this.maxCacheSize,
      utilization: this.currentCacheSize / this.maxCacheSize
    }
  }
}
