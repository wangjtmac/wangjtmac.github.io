本示例展示如何在 Cesium 中实现各种测量工具，包括距离测量、面积测量、高度测量、角度测量等功能。

核心功能 ​

1. 距离测量 ​

  • 直线距离 - 测量两点之间的直线距离
  • 折线距离 - 测量多点连线的总长度
  • 贴地距离 - 考虑地形起伏的实际距离
  • 空间距离 - 三维空间中的距离

2. 面积测量 ​

  • 平面面积 - 投影到水平面的面积
  • 表面积 - 考虑地形起伏的实际表面积
  • 多边形面积 - 任意多边形的面积计算

3. 高度测量 ​

  • 点高程 - 查询指定点的地形高程
  • 相对高度 - 两点之间的高度差
  • 垂直距离 - 点到地面的垂直距离

4. 角度测量 ​

  • 方位角 - 两点连线与正北方向的夹角
  • 俯仰角 - 两点连线与水平面的夹角
  • 三点角度 - 三个点形成的角度

5. 体积测量 ​

  • 挖方体积 - 地形挖方的体积计算
  • 填方体积 - 地形填方的体积计算

距离计算 ​

// 1. 直线距离（不考虑地形）
function calculateStraightDistance(point1, point2) {
  return Cesium.Cartesian3.distance(point1, point2);
}

// 2. 贴地距离（考虑地形）
async function calculateTerrainDistance(positions) {
  // 在路径上采样更多点
  const sampledPositions = [];
  for (let i = 0; i < positions.length - 1; i++) {
    const start = positions[i];
    const end = positions[i + 1];
    const numSamples = 100;
    
    for (let j = 0; j < numSamples; j++) {
      const t = j / numSamples;
      const position = Cesium.Cartesian3.lerp(start, end, t, new Cesium.Cartesian3());
      sampledPositions.push(position);
    }
  }
  
  // 查询地形高程
  const cartographics = sampledPositions.map(pos => 
    Cesium.Cartographic.fromCartesian(pos)
  );
  const updatedCartographics = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    cartographics
  );
  
  // 计算总距离
  let totalDistance = 0;
  for (let i = 0; i < updatedCartographics.length - 1; i++) {
    const pos1 = Cesium.Cartographic.toCartesian(updatedCartographics[i]);
    const pos2 = Cesium.Cartographic.toCartesian(updatedCartographics[i + 1]);
    totalDistance += Cesium.Cartesian3.distance(pos1, pos2);
  }
  
  return totalDistance;
}

// 3. 大地线距离（考虑地球曲率）
function calculateGeodesicDistance(point1, point2) {
  const ellipsoid = viewer.scene.globe.ellipsoid;
  const cart1 = ellipsoid.cartesianToCartographic(point1);
  const cart2 = ellipsoid.cartesianToCartographic(point2);
  
  const geodesic = new Cesium.EllipsoidGeodesic(cart1, cart2);
  return geodesic.surfaceDistance;
}

// 1. 平面面积（投影到水平面）
function calculateProjectedArea(positions) {
  const cartographics = positions.map(pos => 
    Cesium.Cartographic.fromCartesian(pos)
  );
  
  // 使用 Shoelace 公式计算面积
  let area = 0;
  for (let i = 0; i < cartographics.length; i++) {
    const j = (i + 1) % cartographics.length;
    area += cartographics[i].longitude * cartographics[j].latitude;
    area -= cartographics[j].longitude * cartographics[i].latitude;
  }
  area = Math.abs(area) / 2;
  
  // 转换为平方米
  const ellipsoid = viewer.scene.globe.ellipsoid;
  const avgLat = cartographics.reduce((sum, c) => sum + c.latitude, 0) / cartographics.length;
  const metersPerDegree = ellipsoid.maximumRadius * Math.PI / 180;
  const metersPerDegreeLat = metersPerDegree;
  const metersPerDegreeLon = metersPerDegree * Math.cos(avgLat);
  
  return area * metersPerDegreeLat * metersPerDegreeLon;
}

高程查询 ​

javascript
// 查询单点高程
async function queryElevation(position) {
  const cartographic = Cesium.Cartographic.fromCartesian(position);
  const updatedCartographic = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    [cartographic]
  );
  
  return updatedCartographic[0].height;
}

// 批量查询高程
async function queryElevations(positions) {
  const cartographics = positions.map(pos => 
    Cesium.Cartographic.fromCartesian(pos)
  );
  const updatedCartographics = await Cesium.sampleTerrainMostDetailed(
    viewer.terrainProvider,
    cartographics
  );
  
  return updatedCartographics.map(c => c.height);
}

角度计算 ​

javascript
// 计算方位角（0-360度，正北为0）
function calculateAzimuth(point1, point2) {
  const cart1 = Cesium.Cartographic.fromCartesian(point1);
  const cart2 = Cesium.Cartographic.fromCartesian(point2);
  
  const dLon = cart2.longitude - cart1.longitude;
  const y = Math.sin(dLon) * Math.cos(cart2.latitude);
  const x = Math.cos(cart1.latitude) * Math.sin(cart2.latitude) -
            Math.sin(cart1.latitude) * Math.cos(cart2.latitude) * Math.cos(dLon);
  
  let azimuth = Math.atan2(y, x);
  azimuth = Cesium.Math.toDegrees(azimuth);
  azimuth = (azimuth + 360) % 360;
  
  return azimuth;
}

// 计算俯仰角
function calculatePitch(point1, point2) {
  const distance = Cesium.Cartesian3.distance(point1, point2);
  const cart1 = Cesium.Cartographic.fromCartesian(point1);
  const cart2 = Cesium.Cartographic.fromCartesian(point2);
  const heightDiff = cart2.height - cart1.height;
  
  const pitch = Math.atan2(heightDiff, distance);
  return Cesium.Math.toDegrees(pitch);
}

// 计算三点角度
function calculateAngle(point1, vertex, point2) {
  const v1 = Cesium.Cartesian3.subtract(point1, vertex, new Cesium.Cartesian3());
  const v2 = Cesium.Cartesian3.subtract(point2, vertex, new Cesium.Cartesian3());
  
  Cesium.Cartesian3.normalize(v1, v1);
  Cesium.Cartesian3.normalize(v2, v2);
  
  const dot = Cesium.Cartesian3.dot(v1, v2);
  const angle = Math.acos(Cesium.Math.clamp(dot, -1, 1));
  
  return Cesium.Math.toDegrees(angle);
}

// 创建测量工具的鼠标交互
class MeasurementHandler {
  constructor(viewer, measurementType) {
    this.viewer = viewer;
    this.measurementType = measurementType;
    this.positions = [];
    this.handler = new Cesium.ScreenSpaceEventHandler(viewer.scene.canvas);
    
    this.setupHandlers();
  }
  
  setupHandlers() {
    // 左键点击添加点
    this.handler.setInputAction((click) => {
      const position = this.viewer.scene.pickPosition(click.position);
      if (Cesium.defined(position)) {
        this.positions.push(position);
        this.updateMeasurement();
      }
    }, Cesium.ScreenSpaceEventType.LEFT_CLICK);
    
    // 右键点击完成测量
    this.handler.setInputAction(() => {
      this.finishMeasurement();
    }, Cesium.ScreenSpaceEventType.RIGHT_CLICK);
    
    // 鼠标移动显示预览
    this.handler.setInputAction((movement) => {
      const position = this.viewer.scene.pickPosition(movement.endPosition);
      if (Cesium.defined(position) && this.positions.length > 0) {
        this.updatePreview(position);
      }
    }, Cesium.ScreenSpaceEventType.MOUSE_MOVE);
  }

  }

  }
  
  finishMeasurement() {
    // 完成测量
    this.handler.destroy();
  }
}
