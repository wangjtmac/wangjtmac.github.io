<template>
  <div class="cesium-container-wrapper">
    <div class="control-panel">
      <h3>坐标转换工具</h3>

      <!-- 经纬度转笛卡尔坐标 -->

        <h4>经纬度 → 笛卡尔坐标</h4>

        <div class="input-group">
          <label>纬度：</label>
          <input type="number" v-model.number="degrees.latitude" placeholder="39.9042">
        </div>
        <div class="input-group">
          <label>高度：</label>
          <input type="number" v-model.number="degrees.height" placeholder="0">
        </div>
        <button @click="degreesToCartesian">转换</button>
        <div class="result">
          <p>X: {{ cartesian.x.toFixed(2) }}</p>
          <p>Y: {{ cartesian.y.toFixed(2) }}</p>
          <p>Z: {{ cartesian.z.toFixed(2) }}</p>
        </div>
      </div>

      <!-- 笛卡尔坐标转经纬度 -->
      <div class="conversion-section">
        <h4>笛卡尔坐标 → 经纬度</h4>
        <div class="input-group">
          <label>X：</label>
          <input type="number" v-model.number="cartesianInput.x" placeholder="-2178244.0">
        </div>
        <div class="input-group">
          <label>Y：</label>
          <input type="number" v-model.number="cartesianInput.y" placeholder="4356834.0">
        </div>
        <div class="input-group">
          <label>Z：</label>
          <input type="number" v-model.number="cartesianInput.z" placeholder="4077985.0">
        </div>
        <button @click="cartesianToDegrees">转换</button>
        <div class="result">
          <p>经度: {{ degreesResult.longitude.toFixed(6) }}°</p>
          <p>纬度: {{ degreesResult.latitude.toFixed(6) }}°</p>
          <p>高度: {{ degreesResult.height.toFixed(2) }} 米</p>
        </div>
      </div>

      <!-- 坐标可视化 -->
      <div class="visualization-section">
        <h4>坐标可视化</h4>
        <button @click="addPointFromDegrees">添加点（经纬度）</button>
        <button @click="addPointFromCartesian">添加点（笛卡尔）</button>
        <button @click="clearPoints">清除所有点</button>
      </div>
    </div>

    <div ref="cesiumContainer" class="cesium-container">
      <div v-if="loading" class="loading-overlay">
        <div class="loading-spinner">加载 Cesium 中...</div>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue'

const cesiumContainer = ref(null)
const loading = ref(true)

// 坐标数据
const degrees = ref({ longitude: 116.4074, latitude: 39.9042, height: 0 })
const cartesian = ref({ x: 0, y: 0, z: 0 })
const cartesianInput = ref({ x: -2178244.0, y: 4356834.0, z: 4077985.0 })
const degreesResult = ref({ longitude: 0, latitude: 0, height: 0 })

let viewer = null
let points = []

/**
 * 组件挂载时初始化 Cesium 查看器
 * 功能：动态加载 Cesium 资源并初始化地球视图
 * 参数：无
 * 返回值：无
 */
onMounted(async () => {
  if (!cesiumContainer.value) return

  try {
    // 动态加载 Cesium
    await loadCesium()
    Cesium.Ion.defaultAccessToken = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJqdGkiOiIzMDRkZjcyNS03YmI5LTQ2ODQtOTI2NC1jMzYyMDRhZTYxMGIiLCJpZCI6Mzg1NDA5LCJpYXQiOjE3Njk4MjQ4NDh9.DF_GKt9FLM4O6VEG_Bb_DNOPV01YnhyhVAFAAakt9Bs'
    // 初始化 Cesium 查看器
    viewer = new Cesium.Viewer(cesiumContainer.value, {
      animation: false,
      baseLayerPicker: false,
      fullscreenButton: true,
      geocoder: false,
      homeButton: true,
      infoBox: false,
      sceneModePicker: true,
      selectionIndicator: false,
      timeline: false,
      navigationHelpButton: false,
      scene3DOnly: true,
      requestRenderMode: true,
      fullscreenElement: cesiumContainer.value
    })

    // 设置相机位置到北京
    viewer.camera.setView({
      destination: Cesium.Cartesian3.fromDegrees(116.4, 39.9, 1000000.0)
    })

    // 手动添加一个可靠的影像图层
    viewer.imageryLayers.addImageryProvider(
      new Cesium.UrlTemplateImageryProvider({
        url: "https://webst04.is.autonavi.com/appmaptile?style=6&x={x}&y={y}&z={z}"
      })
    );

    // 修复全屏按钮功能
    // fixFullscreenButton()

    loading.value = false
  } catch (error) {
    console.error('Cesium 初始化失败:', error)
    loading.value = false
  }
})

onUnmounted(() => {
  if (viewer) {
    viewer.destroy()
  }
})

/**
 * 经纬度转笛卡尔坐标
 * 功能：将经纬度坐标转换为笛卡尔坐标
 * 参数：无
 * 返回值：无
 */
function degreesToCartesian() {
  if (!window.Cesium) return

  const { longitude, latitude, height } = degrees.value

  // 使用 Cesium 提供的转换函数
  const cartesian3 = Cesium.Cartesian3.fromDegrees(longitude, latitude, height)

  cartesian.value = {
    x: cartesian3.x,
    y: cartesian3.y,
    z: cartesian3.z
  }
}

/**
 * 笛卡尔坐标转经纬度
 * 功能：将笛卡尔坐标转换为经纬度坐标
 * 参数：无
 * 返回值：无
 */
function cartesianToDegrees() {
  if (!window.Cesium) return

  const { x, y, z } = cartesianInput.value
  const cartesian3 = new Cesium.Cartesian3(x, y, z)

  // 将笛卡尔坐标转换为地理坐标（弧度）
  const cartographic = Cesium.Cartographic.fromCartesian(cartesian3)

  // 将弧度转换为角度
  degreesResult.value = {
    longitude: Cesium.Math.toDegrees(cartographic.longitude),
    latitude: Cesium.Math.toDegrees(cartographic.latitude),
    height: cartographic.height
  }
}

/**
 * 从经纬度添加可视化点
 * 功能：在地球上添加一个可视化点标记
 * 参数：无
 * 返回值：无
 */
function addPointFromDegrees() {
  if (!viewer || !window.Cesium) return

  const { longitude, latitude, height } = degrees.value

  // 创建点实体
  const point = viewer.entities.add({
    position: Cesium.Cartesian3.fromDegrees(longitude, latitude, height),
    point: {
      pixelSize: 10,
      color: Cesium.Color.YELLOW,
      outlineColor: Cesium.Color.BLACK,
      outlineWidth: 2,
      heightReference: Cesium.HeightReference.CLAMP_TO_GROUND
    },
    label: {
      text: `(${longitude.toFixed(4)}°, ${latitude.toFixed(4)}°)`,
      font: '14pt monospace',
      fillColor: Cesium.Color.BLACK,
      outlineColor: Cesium.Color.WHITE,
      outlineWidth: 2,
      pixelOffset: new Cesium.Cartesian2(0, -30),
      heightReference: Cesium.HeightReference.CLAMP_TO_GROUND
    }
  })

  points.push(point)

  // 飞向该点
  viewer.flyTo(point)
}

/**
 * 从笛卡尔坐标添加可视化点
 * 功能：在地球上添加一个可视化点标记
 * 参数：无
 * 返回值：无
 */
function addPointFromCartesian() {
  if (!viewer || !window.Cesium) return

  const { x, y, z } = cartesianInput.value
  const position = new Cesium.Cartesian3(x, y, z)

  // 将笛卡尔坐标转换为经纬度用于标签显示
  const cartographic = Cesium.Cartographic.fromCartesian(position)
  const longitude = Cesium.Math.toDegrees(cartographic.longitude)
  const latitude = Cesium.Math.toDegrees(cartographic.latitude)

  // 创建点实体
  const point = viewer.entities.add({
    position: position,
    point: {
      pixelSize: 10,
      color: Cesium.Color.RED,
      outlineColor: Cesium.Color.BLACK,
      outlineWidth: 2,
      heightReference: Cesium.HeightReference.CLAMP_TO_GROUND
    },
    label: {
      text: `(${longitude.toFixed(4)}°, ${latitude.toFixed(4)}°)`,
      font: '14pt monospace',
      fillColor: Cesium.Color.BLACK,
      outlineColor: Cesium.Color.WHITE,
      outlineWidth: 2,
      pixelOffset: new Cesium.Cartesian2(0, -30),
      heightReference: Cesium.HeightReference.CLAMP_TO_GROUND
    }
  })

  points.push(point)

  // 飞向该点
  viewer.flyTo(point)
}

/**
 * 清除所有可视化点
 * 功能：移除所有添加到地球上的点标记
 * 参数：无
 * 返回值：无
 */
function clearPoints() {
  if (!viewer) return

  points.forEach(point => {
    viewer.entities.remove(point)
  })

  points = []
}

/**
 * 动态加载 Cesium 资源
 * 功能：异步加载 Cesium JS 和 CSS 文件
 * 参数：无
 * 返回值：Promise 对象，加载成功时 resolve
 */
async function loadCesium() {
  return new Promise((resolve, reject) => {
    if (window.Cesium) {
      resolve()
      return
    }

    const link = document.createElement('link')
    link.rel = 'stylesheet'
    link.href = 'https://cdn.staticfile.net/cesium/1.113.0/Widgets/widgets.css'
    document.head.appendChild(link)

    const script = document.createElement('script')
    script.src = 'https://cdn.staticfile.net/cesium/1.113.0/Cesium.js'
    script.onload = () => resolve()
    script.onerror = () => reject(new Error('Cesium 加载失败'))
    document.head.appendChild(script)
  })
}

/**
 * 修复全屏按钮功能
 * 功能：修复 Cesium 默认全屏按钮的跨浏览器兼容性问题
 * 参数：无
 * 返回值：无
 */
function fixFullscreenButton() {
  if (!viewer) return

  const fullscreenButton = viewer.cesiumWidget._toolbar._elements.find(
    element => element && element._tooltip && element._tooltip.content === '全屏'
  )

  if (fullscreenButton) {
    fullscreenButton._element.onclick = () => {
      const container = cesiumContainer.value
      if (!container) return

      if (!document.fullscreenElement) {
        if (container.requestFullscreen) {
          container.requestFullscreen()
        } else if (container.webkitRequestFullscreen) {
          container.webkitRequestFullscreen()
        } else if (container.msRequestFullscreen) {
          container.msRequestFullscreen()
        }
      } else {
        if (document.exitFullscreen) {
          document.exitFullscreen()
        } else if (document.webkitExitFullscreen) {
          document.webkitExitFullscreen()
        } else if (document.msExitFullscreen) {
          document.msExitFullscreen()
        }
      }
    }
  }
}
</script>

说明 ​

这个示例展示了如何在 Vue 组件中实现 Cesium 坐标转换功能。主要功能包括：

  • 地理坐标（经纬度）与笛卡尔坐标的相互转换
  • 弧度和角度之间的转换
  • 在 Cesium 视图中可视化坐标点
  • 动态加载 Cesium CDN 资源
  • 完整的错误处理和加载状态

技术要点 ​

  1. 坐标转换：使用 Cesium 提供的 Cartesian3.fromDegrees() 和 Cartographic.fromCartesian() 方法
  2. 单位转换：使用 Cesium.Math.toDegrees() 和 Cesium.Math.toRadians() 进行角度和弧度转换
  3. 实体管理：使用 viewer.entities 管理可视化点
  4. 动态加载：使用 Promise 异步加载 Cesium JS 和 CSS 文件
  5. 组件生命周期：在 onMounted 中初始化，onUnmounted 中清理资源

核心 API ​

  • Cesium.Cartesian3 - 笛卡尔坐标表示
  • Cesium.Cartographic - 地理坐标表示（弧度）
  • Cesium.Cartesian3.fromDegrees() - 经纬度转笛卡尔坐标
  • Cesium.Cartographic.fromCartesian() - 笛卡尔坐标转地理坐标
  • Cesium.Math.toDegrees() - 弧度转角度
  • Cesium.Math.toRadians() - 角度转弧度

注意事项 ​

  • 示例需要网络连接才能加载 Cesium CDN 资源
  • 坐标转换精度受浮点数运算限制
  • 可视化点会自动吸附到地形表面（使用 CLAMP_TO_GROUND）
  • 全屏功能需要浏览器支持 Fullscreen API
  • 组件会自动清理资源，避免内存泄漏
