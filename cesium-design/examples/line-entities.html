线实体示例 ​

这是一个演示如何使用 Cesium Entity API 创建和自定义线实体的示例。

功能说明 ​

  • 实现不同样式的线实体创建
  • 演示线实体的各种属性设置
  • 实现线实体的动态更新
  • 提供交互式控制线实体样式

核心 API ​

  • viewer.entities.add() - 添加实体到场景
  • Cesium.PolylineGraphics - 线图形类
  • Cesium.Color - 颜色类
  • Cesium.Cartesian3 - 笛卡尔坐标类

<script setup>
import { ref, onMounted, onUnmounted, computed, getCurrentInstance } from "vue";
import carIcon from "../public/icons/car.svg"
const { proxy } = getCurrentInstance()


const cesiumContainer = ref(null);
const loading = ref(true);
const error = ref('');
const currentType = ref("simple");
const lineWidth = ref(3);
const lineColor = ref("#ff0000");
const isAnimating = ref(false);

// 线实体列表
let viewer = null;
let lines = ref([]);
// 存储每条线的完整坐标数组
let lineFullPositions = ref([]);
let animationInterval = null;
let animationPhase = 0;
// 汽车实体
let carEntity = null;

// 全屏切换
const toggleFullscreen = () => {
  const element = proxy.$el
  if (!document.fullscreenElement) {
    element.requestFullscreen()
  } else {
    document.exitFullscreen()
  }
}

/**
 * 组件挂载时初始化 Cesium 查看器
 * 功能：动态加载 Cesium 资源并初始化地球视图
 * 参数：无
 * 返回值：无
 */
onMounted(async () => {
  if (!cesiumContainer.value) return;

  try {
    // 动态加载 Cesium
    await loadCesium();

    // 初始化 Cesium 查看器
    viewer = new Cesium.Viewer(cesiumContainer.value, {
      animation: false,
      baseLayerPicker: false,
      fullscreenButton: false,
      geocoder: false,
      homeButton: true,
      infoBox: false,
      sceneModePicker: true,
      selectionIndicator: false,
      timeline: false,
      navigationHelpButton: false,
      scene3DOnly: true,
      requestRenderMode: true,
      imageryProvider: false // 禁用默认影像层
    });

    // 手动添加高德地图影像图层
    viewer.imageryLayers.addImageryProvider(
      new Cesium.UrlTemplateImageryProvider({
        url: "https://webst01.is.autonavi.com/appmaptile?style=6&x={x}&y={y}&z={z}",
        minimumLevel: 0,
        maximumLevel: 18
      })
    );

    // 设置相机位置到中国区域
    viewer.camera.setView({
      destination: Cesium.Cartesian3.fromDegrees(104.0, 35.0, 5000000.0),
    });

    // 创建简单线实体
    createSimpleLine();

    // 初始化完成后，将镜头移动到简单线的起点
    // 等待一小段时间确保线已创建
    setTimeout(() => {
      if (lineFullPositions.value.length > 0 && lineFullPositions.value[0].length > 0) {
        const startPosition = lineFullPositions.value[0][0];
        resetCameraToStart(startPosition);
      }
    }, 500);

    loading.value = false;
  } catch (error) {
    error.value = error.message || 'Cesium 初始化失败';
    loading.value = false;
  }
});

onUnmounted(() => {
  if (viewer) {
    // 清除汽车实体
    if (carEntity) {
      viewer.entities.remove(carEntity);
      carEntity = null;
    }
    viewer.destroy();
  }

  if (animationInterval) {
    clearInterval(animationInterval);
  }
});

/**
 * 计算当前线实体数量
 * 功能：实时计算当前场景中的线实体数量
 * 参数：无
 * 返回值：线实体数量
 */
const lineCount = computed(() => {
  return lines.value.length;
});

/**
 * 获取当前类型名称
 * 功能：根据当前类型返回对应的中文名称
 * 参数：无
 * 返回值：类型名称
 */
const currentTypeName = computed(() => {
  const typeMap = {
    simple: "简单线",
    dashed: "虚线",
    glowing: "发光线",
    multi: "多段线",
  };
  return typeMap[currentType.value] || "未知";
});

/**
 * 创建简单线实体
 * 功能：创建一条简单的实线
 * 参数：无
 * 返回值：无
 */
function createSimpleLine() {
  if (!viewer || !window.Cesium) return;

  currentType.value = "simple";

  // 清除现有线实体
  clearAllLines();

  // 创建从北京到上海的线
  const positions = [
    Cesium.Cartesian3.fromDegrees(116.4074, 39.9042, 0), // 北京
    Cesium.Cartesian3.fromDegrees(121.4737, 31.2304, 0), // 上海
  ];

  // 存储完整的坐标数组用于动画
  lineFullPositions.value.push(positions);

  // 初始只显示起点
  const initialPositions = [positions[0], positions[0]];

  const line = viewer.entities.add({
    name: "北京-上海线",
    polyline: {
      positions: initialPositions,
      width: lineWidth.value,
      material: Cesium.Color.fromCssColorString(lineColor.value),
      clampToGround: true,
    },
  });

  lines.value.push(line);

  // 飞向线实体
  viewer.flyTo(line);

  // 如果有汽车实体，将其移动到新线的起点
  if (carEntity && carEntity.main) {
    carEntity.main.position = positions[0];
    // 重置镜头到起点
    resetCameraToStart(positions[0]);
  }
}

/**
 * 更新线条样式
 * 功能：更新所有线实体的样式
 * 参数：无
 * 返回值：无
 */
function updateLineStyle() {
  if (!viewer || lines.value.length === 0) return;

  lines.value.forEach((line) => {
    if (line.polyline) {
      line.polyline.width = lineWidth.value;

      // 根据当前类型更新材质
      switch (currentType.value) {
        case "simple":
          line.polyline.material = Cesium.Color.fromCssColorString(
            lineColor.value
          );
          break;
        case "dashed":
          line.polyline.material = new Cesium.PolylineDashMaterialProperty({
            color: Cesium.Color.fromCssColorString(lineColor.value),
            dashLength: 16.0,
            gapColor: Cesium.Color.WHITE,
          });
          break;
        case "glowing":
          line.polyline.material = new Cesium.PolylineGlowMaterialProperty({
            glowPower: 0.2,
            color: Cesium.Color.fromCssColorString(lineColor.value),
          });
          break;
        case "multi":
          line.polyline.material = Cesium.Color.fromCssColorString(
            lineColor.value
          );
          break;
      }
    }
  });
}

/**
 * 切换动画状态
 * 功能：开始或停止线实体动画
 * 参数：无
 * 返回值：无
 */
function toggleAnimation() {
  if (!viewer || lines.value.length === 0) return;

  isAnimating.value = !isAnimating.value;

  if (isAnimating.value) {
    startAnimation();
  } else {
    stopAnimation();
  }
}

/**
 * 开始动画
 * 功能：开始线实体动画效果
 * 参数：无
 * 返回值：无
 */
async function startAnimation() {
  if (animationInterval) {
    clearInterval(animationInterval);
  }

  animationPhase = 0;

  // 如果有完整坐标数组，创建汽车实体
  if (lineFullPositions.value.length > 0 && lineFullPositions.value[0].length > 0) {
    const startPosition = lineFullPositions.value[0][0];
    if (startPosition) {
      await createCarEntity(startPosition);
    }
  }

  // 为每条线创建动态坐标更新函数
  lines.value.forEach((line, index) => {
    if (line.polyline && lineFullPositions.value[index]) {
      const fullPositions = lineFullPositions.value[index];

      // 使用Cesium的CallbackProperty来动态生成坐标
      line.polyline.positions = new Cesium.CallbackProperty(() => {
        // 安全检查：确保坐标数组有效
        if (!fullPositions || fullPositions.length <= 1) {
          return fullPositions || [];
        }

        // 如果动画完成（进度达到100%），返回完整坐标
        if (animationPhase >= 1.0) {
          return fullPositions;
        }

        // 注意：这里移除了 !isAnimating.value 的检查
        // 这样即使动画停止，也会保持当前的绘制状态，而不是显示完整线条

        // 如果只有两个点，使用插值生成中间点
        if (fullPositions.length === 2) {
          const start = fullPositions[0];
          const end = fullPositions[1];

          // 安全检查：确保两个点都存在
          if (!start || !end) {
            return fullPositions;
          }

          // 使用Cesium的线性插值生成中间点
          const currentPosition = Cesium.Cartesian3.lerp(
            start,
            end,
            animationPhase,
            new Cesium.Cartesian3()
          );

          // 返回包含起点和当前中间点的数组
          return [start, currentPosition];
        }

        // 对于多个点的线，计算当前应该在哪个线段上
        // 将整条线分成 (n-1) 个线段，每个线段占总进度的 1/(n-1)
        const segmentCount = fullPositions.length - 1;
        const totalProgress = animationPhase * segmentCount;

        // 当前在第几个线段（从0开始）
        const currentSegmentIndex = Math.floor(totalProgress);

        // 在当前线段内的进度（0-1）
        const segmentProgress = totalProgress - currentSegmentIndex;

        // 如果动画完成，返回完整数组
        if (currentSegmentIndex >= segmentCount) {
          return fullPositions;
        }

        // 如果还在第一个点
        if (currentSegmentIndex < 0 || totalProgress <= 0) {
          return [fullPositions[0], fullPositions[0]];
        }

        // 获取当前线段的起点和终点
        const segmentStart = fullPositions[currentSegmentIndex];
        const segmentEnd = fullPositions[currentSegmentIndex + 1];

        // 安全检查
        if (!segmentStart || !segmentEnd) {
          return fullPositions.slice(0, currentSegmentIndex + 1);
        }

        // 在当前线段上插值计算当前位置
        const currentPosition = Cesium.Cartesian3.lerp(
          segmentStart,
          segmentEnd,
          segmentProgress,
          new Cesium.Cartesian3()
        );

        // 返回已完成的线段 + 当前正在绘制的线段
        const result = fullPositions.slice(0, currentSegmentIndex + 1);
        result.push(currentPosition);

        return result;
      }, false);
    }
  });

  // 更新动画相位的定时器
  animationInterval = setInterval(() => {
    if (!viewer || lines.value.length === 0 || lineFullPositions.value.length === 0) return;

    // 动画相位控制在0到1之间，大幅降低速度从0.002改为0.0005
    // 镜头拉近后需要非常慢的速度才能看清楚细节
    animationPhase += 0.0005;
    if (animationPhase >= 1.0) {
      animationPhase = 1.0;
      // 动画完成后停止动画
      stopAnimation();
      return;
    }

    // 更新汽车位置
    updateCarPosition();

    // 强制Cesium重新渲染场景
    viewer.scene.requestRender();
  }, 30);
}

/**
 * 创建汽车实体
 * 功能：创建一个表示汽车的实体，用于在线上行驶动画
 * 参数：起始位置坐标
 * 返回值：无
 */
async function createCarEntity(startPosition) {
  if (!viewer || !window.Cesium || !startPosition) return;

  // 如果已经有汽车实体，先清除
  if (carEntity) {
    carEntity.parts.forEach(part => {
      viewer.entities.remove(part);
    });
    carEntity = null;
  }

  // 创建汽车实体容器
  carEntity = {
    main: null,
    parts: []
  };

  const elevatedPosition = Cesium.Cartesian3.add(
    startPosition,
    new Cesium.Cartesian3(0, 0, 0),
    new Cesium.Cartesian3()
  );

  // 尝试加载本地 SVG 文件，如果失败则使用备用图标
  let carImageUrl = carIcon;
  
  // 测试图片是否可以加载
  try {
    const response = await fetch(carImageUrl);
    if (!response.ok) {
      console.warn('Failed to load car.svg, using fallback icon');
      // 使用 Cesium 内置的图标作为备用
      carImageUrl = viewer.entities.add({
        position: elevatedPosition,
        point: {
          pixelSize: 20,
          color: Cesium.Color.RED,
          outlineColor: Cesium.Color.WHITE,
          outlineWidth: 2,
        }
      });
      carEntity.main = carImageUrl;
      carEntity.parts.push(carImageUrl);
      return;
    }
  } catch (error) {
    console.error('Error checking car.svg:', error);
    // 使用点作为备用
    const fallbackPoint = viewer.entities.add({
      name: "Car",
      position: elevatedPosition,
      point: {
        pixelSize: 20,
        color: Cesium.Color.RED,
        outlineColor: Cesium.Color.WHITE,
        outlineWidth: 2,
      }
    });
    carEntity.main = fallbackPoint;
    carEntity.parts.push(fallbackPoint);
    return;
  }

  // 使用 billboard 显示 SVG 汽车
  const carBillboard = viewer.entities.add({
    name: "Car",
    position: elevatedPosition,
    billboard: {
      image: carImageUrl,
      width: 80,  // 增大尺寸以提高清晰度
      height: 80,
      heightReference: Cesium.HeightReference.CLAMP_TO_GROUND,
      verticalOrigin: Cesium.VerticalOrigin.CENTER,
      horizontalOrigin: Cesium.HorizontalOrigin.CENTER,
      scaleByDistance: new Cesium.NearFarScalar(1000, 2.5, 1000000, 1.0),
      rotation: 0,
      alignedAxis: Cesium.Cartesian3.UNIT_Z,
      disableDepthTestDistance: Number.POSITIVE_INFINITY, // 始终显示在最前面
      pixelOffset: new Cesium.Cartesian2(0, 0),
    },
  });
  
  carEntity.main = carBillboard;
  carEntity.parts.push(carBillboard);
}

/**
 * 更新汽车所有部件的位置
 * 功能：根据主车身位置更新汽车所有部件的位置，并计算旋转角度，同时更新镜头跟随
 * 参数：新的车身位置，下一个目标位置（可选）
 * 返回值：无
 */
function updateCarPartsPosition(newPosition, nextPosition = null) {
  if (!carEntity || !newPosition || !window.Cesium) return;

  // 更新汽车 billboard 的位置
  if (carEntity.main) {
    carEntity.main.position = newPosition;
    
    // 如果有下一个位置，计算旋转角度
    if (nextPosition) {
      // 计算从当前位置到下一个位置的方向向量
      const direction = Cesium.Cartesian3.subtract(
        nextPosition,
        newPosition,
        new Cesium.Cartesian3()
      );
      
      // 将方向向量归一化
      Cesium.Cartesian3.normalize(direction, direction);
      
      // 获取当前位置的地表法向量
      const normal = Cesium.Cartesian3.normalize(
        newPosition,
        new Cesium.Cartesian3()
      );
      
      // 计算东向量
      const east = Cesium.Cartesian3.cross(
        Cesium.Cartesian3.UNIT_Z,
        normal,
        new Cesium.Cartesian3()
      );
      Cesium.Cartesian3.normalize(east, east);
      
      // 计算北向量
      const north = Cesium.Cartesian3.cross(
        normal,
        east,
        new Cesium.Cartesian3()
      );
      Cesium.Cartesian3.normalize(north, north);
      
      // 计算方向向量在东向和北向上的投影
      const eastComponent = Cesium.Cartesian3.dot(direction, east);
      const northComponent = Cesium.Cartesian3.dot(direction, north);
      
      // 计算旋转角度（弧度）
      // Math.atan2(east, north) 给出从正北顺时针的角度
      let heading = Math.atan2(eastComponent, northComponent);
      
      // Cesium billboard rotation:
      // - 0 弧度 = 图片默认方向（车头朝上/北）
      // - 正值 = 逆时针旋转
      // - 负值 = 顺时针旋转
      // 
      // heading 是从正北顺时针测量的角度
      // billboard.rotation 需要逆时针角度，所以取负值
      carEntity.main.billboard.rotation = -heading;
      
      // 调试输出 - 显示当前位置和方向信息
      const currentCart = Cesium.Cartographic.fromCartesian(newPosition);
      const nextCart = Cesium.Cartographic.fromCartesian(nextPosition);
      console.log('Car Position:', 
        'Lon:', Cesium.Math.toDegrees(currentCart.longitude).toFixed(2),
        'Lat:', Cesium.Math.toDegrees(currentCart.latitude).toFixed(2),
        '→ Next:', 
        'Lon:', Cesium.Math.toDegrees(nextCart.longitude).toFixed(2),
        'Lat:', Cesium.Math.toDegrees(nextCart.latitude).toFixed(2)
      );
      console.log('Direction Vector - East:', eastComponent.toFixed(3), 'North:', northComponent.toFixed(3));
      console.log('Heading:', Cesium.Math.toDegrees(heading).toFixed(1), '° (from North, clockwise)');
      console.log('Billboard Rotation:', Cesium.Math.toDegrees(-heading).toFixed(1), '° (counter-clockwise)');
    }
    
    // 镜头跟随汽车，保持汽车在屏幕中心
    if (viewer && isAnimating.value) {
      // 将笛卡尔坐标转换为地理坐标
      const cartographic = Cesium.Cartographic.fromCartesian(newPosition);
      const longitude = Cesium.Math.toDegrees(cartographic.longitude);
      const latitude = Cesium.Math.toDegrees(cartographic.latitude);
      
      // 计算镜头位置：在汽车正上方一定高度
      const cameraHeight = 50000; // 降低到50km高度，可以看到道路细节
      const cameraPosition = Cesium.Cartesian3.fromDegrees(
        longitude, 
        latitude, 
        cameraHeight
      );
      
      // 直接设置镜头位置，无过渡动画，实时跟随
      viewer.camera.setView({
        destination: cameraPosition,
        orientation: {
          heading: 0,
          pitch: Cesium.Math.toRadians(-90), // 垂直俯视
          roll: 0
        }
      });
    }
  }
}

/**
 * 更新汽车位置
 * 功能：根据当前动画相位更新汽车的位置和方向
 * 参数：无
 * 返回值：无
 */
function updateCarPosition() {
  if (!viewer || !window.Cesium || !carEntity || lineFullPositions.value.length === 0) return;

  // 获取当前线的完整坐标数组
  const fullPositions = lineFullPositions.value[0];
  if (!fullPositions || fullPositions.length <= 1) return;

  let carPosition;
  let nextPosition = null;

  // 如果只有两个点，直接使用插值
  if (fullPositions.length === 2) {
    const start = fullPositions[0];
    const end = fullPositions[1];

    // 安全检查
    if (!start || !end) return;

    // 使用线性插值计算汽车位置
    carPosition = Cesium.Cartesian3.lerp(
      start,
      end,
      animationPhase,
      new Cesium.Cartesian3()
    );
    
    // 下一个位置就是终点
    nextPosition = end;
  } else {
    // 对于多个点的线，使用与动画相同的逻辑
    const segmentCount = fullPositions.length - 1;
    const totalProgress = animationPhase * segmentCount;

    // 当前在第几个线段
    const currentSegmentIndex = Math.floor(totalProgress);

    // 在当前线段内的进度
    const segmentProgress = totalProgress - currentSegmentIndex;

    // 如果动画完成，使用最后一个点
    if (currentSegmentIndex >= segmentCount) {
      carPosition = fullPositions[fullPositions.length - 1];
      // 动画完成时，使用倒数第二个点作为方向参考
      if (fullPositions.length >= 2) {
        nextPosition = fullPositions[fullPositions.length - 1];
        const prevPosition = fullPositions[fullPositions.length - 2];
        // 保持最后的方向
        updateCarPartsPosition(carPosition, nextPosition);
        return;
      }
    } else if (currentSegmentIndex < 0 || totalProgress <= 0) {
      // 如果还在起点
      carPosition = fullPositions[0];
      nextPosition = fullPositions[1];
    } else {
      // 获取当前线段的起点和终点
      const segmentStart = fullPositions[currentSegmentIndex];
      const segmentEnd = fullPositions[currentSegmentIndex + 1];

      // 安全检查
      if (!segmentStart || !segmentEnd) return;

      // 在当前线段上插值计算汽车位置
      carPosition = Cesium.Cartesian3.lerp(
        segmentStart,
        segmentEnd,
        segmentProgress,
        new Cesium.Cartesian3()
      );
      
      // 下一个位置就是当前线段的终点
      nextPosition = segmentEnd;
    }
  }

  // 更新汽车位置和方向
  if (carPosition && nextPosition) {
    updateCarPartsPosition(carPosition, nextPosition);
  }
}

/**

/**
 * 重置镜头到起点
 * 功能：将镜头移动到线的起点位置
 * 参数：起点位置坐标
 * 返回值：无
 */
function resetCameraToStart(startPosition) {
  if (!viewer || !window.Cesium || !startPosition) return;

  // 将笛卡尔坐标转换为地理坐标
  const cartographic = Cesium.Cartographic.fromCartesian(startPosition);
  const longitude = Cesium.Math.toDegrees(cartographic.longitude);
  const latitude = Cesium.Math.toDegrees(cartographic.latitude);
  
  // 计算镜头位置：在起点正上方
  const cameraHeight = 50000; // 50km 高度，可以看到道路
  const cameraPosition = Cesium.Cartesian3.fromDegrees(
    longitude, 
    latitude, 
    cameraHeight
  );
  
  // 平滑飞向起点
  viewer.camera.flyTo({
    destination: cameraPosition,
    orientation: {
      heading: 0,
      pitch: Cesium.Math.toRadians(-90), // 垂直俯视
      roll: 0
    },
    duration: 1.5 // 1.5秒的飞行时间
  });
}

/**
 * 创建示例线实体
 * 功能：在初始化时创建一些示例线实体
 * 参数：无
 * 返回值：无
 */
function createExampleLines() {
  if (!viewer || !window.Cesium) return;

  // 创建一条示例线（北京到天津）
  const positions = [
    Cesium.Cartesian3.fromDegrees(116.4074, 39.9042, 0), // 北京
    Cesium.Cartesian3.fromDegrees(117.1902, 39.1079, 0), // 天津
  ];

  const exampleLine = viewer.entities.add({
    name: "示例线",
    polyline: {
      positions: positions,
      width: 3,
      material: Cesium.Color.BLUE,
      clampToGround: true,
    },
  });

  lines.value.push(exampleLine);
}

/**
 * 动态加载 Cesium 资源
 * 功能：异步加载 Cesium JS 和 CSS 文件
 * 参数：无
 * 返回值：Promise 对象，加载成功时 resolve
 */
async function loadCesium() {
  return new Promise((resolve, reject) => {
    // 检查是否已加载
    if (window.Cesium) {
      resolve();
      return;
    }

    // 动态加载 Cesium CSS
    const link = document.createElement("link");
    link.rel = "stylesheet";
    link.href = "https://cdn.staticfile.net/cesium/1.113.0/Widgets/widgets.css";
    document.head.appendChild(link);

    // 动态加载 Cesium JS
    const script = document.createElement("script");
    script.src = "https://cdn.staticfile.net/cesium/1.113.0/Cesium.js";
    script.onload = () => resolve();
    script.onerror = () => reject(new Error("Cesium 加载失败"));
    document.head.appendChild(script);
  });
}
</script>
