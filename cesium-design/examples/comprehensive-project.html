本示例是一个综合性的 Cesium 应用项目，整合了前面学习的所有知识点，展示如何构建一个完整的智慧城市可视化平台。

项目概述 ​

功能模块 ​

  1. 基础地图 - 影像图层、地形数据、坐标系统
  2. 3D 建筑 - 城市建筑模型、3D Tiles
  3. 实时数据 - 车辆轨迹、传感器数据
  4. 数据可视化 - GeoJSON、热力图、统计图表
  5. 分析工具 - 测量、地形分析、视域分析
  6. 视觉效果 - 光照、阴影、粒子效果
  7. 交互功能 - 点击查询、信息面板、搜索定位

技术栈 ​

  • 前端框架: Vue 3
  • 3D 引擎: CesiumJS
  • 数据格式: GeoJSON, 3D Tiles, glTF
  • 状态管理: Pinia (可选)
  • UI 组件: Element Plus (可选)

1. Viewer 管理器 ​

javascript
// managers/ViewerManager.js
export class ViewerManager {
  constructor(container, options = {}) {
    this.viewer = new Cesium.Viewer(container, {
      terrainProvider: Cesium.createWorldTerrain(),
      imageryProvider: new Cesium.IonImageryProvider({ assetId: 2 }),
      shadows: true,
      shouldAnimate: true,
      ...options
    });
    
    this.initializeViewer();
  }
  
  initializeViewer() {
    // 配置光照
    this.viewer.scene.globe.enableLighting = true;
    this.viewer.scene.light = new Cesium.SunLight();
    
    // 配置阴影
    this.viewer.shadowMap.size = 2048;
    this.viewer.shadowMap.softShadows = true;
    
    // 设置初始视角
    this.flyToCity();
  }
  
  flyToCity() {
    this.viewer.camera.flyTo({
      destination: Cesium.Cartesian3.fromDegrees(116.4, 39.9, 5000),
      orientation: {
        heading: Cesium.Math.toRadians(0),
        pitch: Cesium.Math.toRadians(-45),
        roll: 0
      },
      duration: 2
    });
  }
  
  destroy() {
    this.viewer.destroy();
  }
}

2. 实体管理器 ​

javascript
// managers/EntityManager.js
export class EntityManager {
  constructor(viewer) {
    this.viewer = viewer;
    this.entities = new Map();
    this.collections = new Map();
  }
  
  // 添加建筑
  addBuilding(id, data) {
    const entity = this.viewer.entities.add({
      id: id,
      position: Cesium.Cartesian3.fromDegrees(data.lon, data.lat, 0),
      polygon: {
        hierarchy: Cesium.Cartesian3.fromDegreesArray(data.coordinates),
        extrudedHeight: data.height,
        material: Cesium.Color.fromCssColorString(data.color).withAlpha(0.8),
        outline: true,
        outlineColor: Cesium.Color.BLACK,
        shadows: Cesium.ShadowMode.ENABLED
      },
      properties: data.properties
    });
    
    this.entities.set(id, entity);
    return entity;
  }
  
  // 添加车辆
  addVehicle(id, data) {
    const position = new Cesium.SampledPositionProperty();
    
    data.trajectory.forEach(point => {
      const time = Cesium.JulianDate.fromIso8601(point.time);
      const pos = Cesium.Cartesian3.fromDegrees(point.lon, point.lat, point.alt);
      position.addSample(time, pos);
    });
    
    const entity = this.viewer.entities.add({
      id: id,
      position: position,
      orientation: new Cesium.VelocityOrientationProperty(position),
      model: {
        uri: 'models/vehicle.glb',
        minimumPixelSize: 64,
        maximumScale: 200
      },
      path: {
        show: true,
        leadTime: 0,
        trailTime: 300,
        width: 3,
        material: new Cesium.PolylineGlowMaterialProperty({
          glowPower: 0.3,
          color: Cesium.Color.YELLOW
        })
      },
      properties: data.properties
    });
    
    this.entities.set(id, entity);
    return entity;
  }
  
  // 添加传感器
  addSensor(id, data) {
    const entity = this.viewer.entities.add({
      id: id,
      position: Cesium.Cartesian3.fromDegrees(data.lon, data.lat, data.alt),
      point: {
        pixelSize: 10,
        color: this.getSensorColor(data.status),
        outlineColor: Cesium.Color.WHITE,
        outlineWidth: 2
      },
      label: {
        text: data.name,
        font: '12px sans-serif',
        fillColor: Cesium.Color.WHITE,
        pixelOffset: new Cesium.Cartesian2(0, -20),
        showBackground: true,
        backgroundColor: new Cesium.Color(0, 0, 0, 0.7)
      },
      properties: data.properties
    });
    
    this.entities.set(id, entity);
    return entity;
  }
  
  getSensorColor(status) {
    switch (status) {
      case 'normal': return Cesium.Color.GREEN;
      case 'warning': return Cesium.Color.YELLOW;
      case 'error': return Cesium.Color.RED;
      default: return Cesium.Color.GRAY;
    }
  }
  
  // 移除实体
  removeEntity(id) {
    const entity = this.entities.get(id);
    if (entity) {
      this.viewer.entities.remove(entity);
      this.entities.delete(id);
    }
  }
  
  // 清空所有实体
  clear() {
    this.viewer.entities.removeAll();
    this.entities.clear();
  }
}

3. 数据管理器 ​

// managers/DataManager.js
export class DataManager {
  constructor() {
    this.cache = new Map();
  }
  
  // 加载 GeoJSON 数据
  async loadGeoJSON(url) {
    if (this.cache.has(url)) {
      return this.cache.get(url);
    }
    
    const response = await fetch(url);
    const data = await response.json();
    this.cache.set(url, data);
    return data;
  }
  
  // 加载 3D Tiles
  async load3DTiles(viewer, url) {
    const tileset = viewer.scene.primitives.add(
      new Cesium.Cesium3DTileset({
        url: url,
        maximumScreenSpaceError: 16,
        maximumMemoryUsage: 512
      })
    );
    
    await tileset.readyPromise;
    return tileset;
  }
  
  // 实时数据更新
  startRealtimeUpdate(callback, interval = 1000) {
    return setInterval(() => {
      this.fetchRealtimeData().then(callback);
    }, interval);
  }
  
  async fetchRealtimeData() {
    // 模拟实时数据
    return {
      vehicles: this.generateVehicleData(),
      sensors: this.generateSensorData()
    };
  }
  
  generateVehicleData() {
    // 生成随机车辆数据
    return Array.from({ length: 10 }, (_, i) => ({
      id: `vehicle_${i}`,
      lon: 116.4 + Math.random() * 0.1,
      lat: 39.9 + Math.random() * 0.1,
      speed: Math.random() * 60,
      direction: Math.random() * 360
    }));
  }
  
  generateSensorData() {
    // 生成随机传感器数据
    return Array.from({ length: 20 }, (_, i) => ({
      id: `sensor_${i}`,
      lon: 116.4 + Math.random() * 0.1,
      lat: 39.9 + Math.random() * 0.1,
      value: Math.random() * 100,
      status: ['normal', 'warning', 'error'][Math.floor(Math.random() * 3)]
    }));
  }
}

// managers/ToolManager.js
export class ToolManager {
  constructor(viewer) {
    this.viewer = viewer;
    this.currentTool = null;
    this.handler = null;
  }
  
  // 激活测量工具
  activateMeasureTool(type) {
    this.deactivateCurrentTool();
    
    this.handler = new Cesium.ScreenSpaceEventHandler(this.viewer.scene.canvas);
    this.currentTool = type;
    
    switch (type) {
      case 'distance':
        this.setupDistanceMeasure();
        break;
      case 'area':
        this.setupAreaMeasure();
        break;
      case 'height':
        this.setupHeightMeasure();
        break;
    }
  }
  
  setupDistanceMeasure() {
    const positions = [];
    
    this.handler.setInputAction((click) => {
      const position = this.viewer.scene.pickPosition(click.position);
      if (Cesium.defined(position)) {
        positions.push(position);
        
        if (positions.length >= 2) {
          const distance = Cesium.Cartesian3.distance(
            positions[positions.length - 2],
            positions[positions.length - 1]
          );
          
          console.log('Distance:', distance.toFixed(2), 'm');
        }
      }
    }, Cesium.ScreenSpaceEventType.LEFT_CLICK);
  }
  
  setupAreaMeasure() {
    // 实现面积测量
  }
  
  setupHeightMeasure() {
    // 实现高度测量
  }
  
  // 停用当前工具
  deactivateCurrentTool() {
    if (this.handler) {
      this.handler.destroy();
      this.handler = null;
    }
    this.currentTool = null;
  }
}

5. Vue 组件 ​

vue
<!-- components/CesiumViewer.vue -->
<template>
  <div class="cesium-viewer-container">
    <div ref="cesiumContainer" class="cesium-container"></div>
    
    <ControlPanel 
      @layer-change="handleLayerChange"
      @tool-select="handleToolSelect"
    />
    
    <InfoPanel 
      :selected-entity="selectedEntity"
      @close="selectedEntity = null"
    />
  </div>
</template>

<script>
import { ViewerManager } from '../managers/ViewerManager';
import { EntityManager } from '../managers/EntityManager';
import { DataManager } from '../managers/DataManager';
import { ToolManager } from '../managers/ToolManager';
import ControlPanel from './ControlPanel.vue';
import InfoPanel from './InfoPanel.vue';

export default {
  name: 'CesiumViewer',
  components: {
    ControlPanel,
    InfoPanel
  },
  data() {
    return {
      viewerManager: null,
      entityManager: null,
      dataManager: null,
      toolManager: null,
      selectedEntity: null
    };
  },
  mounted() {
    this.initializeApp();
  },
  beforeUnmount() {
    this.cleanup();
  },
  methods: {
    async initializeApp() {
      // 初始化管理器
      this.viewerManager = new ViewerManager(this.$refs.cesiumContainer);
      this.entityManager = new EntityManager(this.viewerManager.viewer);
      this.dataManager = new DataManager();
      this.toolManager = new ToolManager(this.viewerManager.viewer);
      
      // 加载数据
      await this.loadInitialData();
      
      // 设置事件监听
      this.setupEventListeners();
      
      // 启动实时数据更新
      this.startRealtimeUpdates();
    },
    
    async loadInitialData() {
      // 加载建筑数据
      const buildings = await this.dataManager.loadGeoJSON('/data/buildings.json');
      buildings.features.forEach(feature => {
        this.entityManager.addBuilding(feature.id, feature.properties);
      });
      
      // 加载 3D Tiles
      await this.dataManager.load3DTiles(
        this.viewerManager.viewer,
        '/data/tileset.json'
      );
    },
    
    setupEventListeners() {
      const viewer = this.viewerManager.viewer;
      
      // 点击事件
      const handler = new Cesium.ScreenSpaceEventHandler(viewer.scene.canvas);
      handler.setInputAction((click) => {
        const pickedObject = viewer.scene.pick(click.position);
        if (Cesium.defined(pickedObject) && pickedObject.id) {
          this.selectedEntity = pickedObject.id;
        }
      }, Cesium.ScreenSpaceEventType.LEFT_CLICK);
    },
    
    startRealtimeUpdates() {
      this.dataManager.startRealtimeUpdate((data) => {
        // 更新车辆位置
        data.vehicles.forEach(vehicle => {
          // 更新或创建车辆实体
        });
        
        // 更新传感器状态
        data.sensors.forEach(sensor => {
          // 更新传感器显示
        });
      }, 1000);
    },
    
    handleLayerChange(layer) {
      // 处理图层切换
    },
    
    handleToolSelect(tool) {
      this.toolManager.activateMeasureTool(tool);
    },
    
    cleanup() {
      if (this.viewerManager) {
        this.viewerManager.destroy();
      }
    }
  }
};
</script>
