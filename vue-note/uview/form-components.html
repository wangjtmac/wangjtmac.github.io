uView UI 表单组件 ​

表单组件是用户交互的重要部分，用于收集、验证和提交用户输入的数据。uView UI提供了丰富的表单组件，包括输入框、单选框、复选框、开关、滑块、日期选择器等。本章将详细介绍这些表单组件的使用方法和最佳实践。

输入框组件 (u-input) ​

输入框是最基础的表单元素，用于接收用户的文本输入。

基本用法 ​

vue
<template>
  <view>
    <!-- 基础输入框 -->
    <u-input
      v-model="value1"
      placeholder="请输入内容"
      border="bottom"
    ></u-input>
    
    <!-- 带图标的输入框 -->
    <u-input
      v-model="value2"
      placeholder="请输入用户名"
      prefixIcon="account"
      prefixIconStyle="font-size: 22px;color: #909399"
    ></u-input>
    
    <!-- 密码输入框 -->
    <u-input
      v-model="value3"
      type="password"
      placeholder="请输入密码"
      suffixIcon="eye"
      :password="showPassword"
      @clickSuffixIcon="showPassword = !showPassword"
    ></u-input>
    
    <!-- 禁用状态 -->
    <u-input
      v-model="value4"
      disabled
      placeholder="禁用状态"
    ></u-input>
  </view>
</template>

<script>
export default {
  data() {
    return {
      value1: '',
      value2: '',
      value3: '',
      value4: '这是禁用状态',
      showPassword: true
    }
  }
}
</script>

输入框验证 ​

输入框可以结合uView UI的表单验证功能使用：

vue
<template>
  <view>
    <u-form :model="form" ref="uForm">
      <u-form-item label="用户名" prop="username">
        <u-input
          v-model="form.username"
          placeholder="请输入用户名"
        ></u-input>
      </u-form-item>
      
      <u-form-item label="手机号" prop="phone">
        <u-input
          v-model="form.phone"
          placeholder="请输入手机号"
          type="number"
        ></u-input>
      </u-form-item>
    </u-form>
    
    <u-button text="提交" type="primary" @click="submit"></u-button>
  </view>
</template>

<script>
export default {
  data() {
    return {
      form: {
        username: '',
        phone: ''
      },
      rules: {
        username: [
          { required: true, message: '请输入用户名', trigger: 'blur' },
          { min: 3, max: 15, message: '用户名长度在3-15个字符之间', trigger: 'blur' }
        ],
        phone: [
          { required: true, message: '请输入手机号', trigger: 'blur' },
          { pattern: /^1[3-9]\d{9}$/, message: '请输入正确的手机号', trigger: 'blur' }
        ]
      }
    }
  },
  onReady() {
    this.$refs.uForm.setRules(this.rules);
  },
  methods: {
    submit() {
      this.$refs.uForm.validate(valid => {
        if (valid) {
          console.log('表单验证通过');
          uni.showToast({
            title: '表单验证通过',
            icon: 'none'
          });
        } else {
          console.log('表单验证失败');
        }
      });
    }
  }
}
</script>

日期选择器组件 (u-datetime-picker) ​

日期选择器用于选择日期或时间。

基本用法 ​

vue
<template>
  <view>
    <!-- 日期选择器 -->
    <u-datetime-picker
      v-model="show1"
      :value="date1"
      mode="date"
      @confirm="confirm1"
      @cancel="cancel1"
    ></u-datetime-picker>
    <u-button text="选择日期" @click="show1 = true"></u-button>
    
    <!-- 时间选择器 -->
    <u-datetime-picker
      v-model="show2"
      :value="time"
      mode="time"
      @confirm="confirm2"
      @cancel="cancel2"
    ></u-datetime-picker>
    <u-button text="选择时间" @click="show2 = true"></u-button>
    
    <!-- 日期时间选择器 -->
    <u-datetime-picker
      v-model="show3"
      :value="datetime"
      mode="datetime"
      @confirm="confirm3"
      @cancel="cancel3"
    ></u-datetime-picker>
    <u-button text="选择日期时间" @click="show3 = true"></u-button>
  </view>
</template>

<script>
export default {
  data() {
    return {
      show1: false,
      date1: Number(new Date()),
      show2: false,
      time: Number(new Date()),
      show3: false,
      datetime: Number(new Date())
    }
  },
  methods: {
    confirm1(e) {
      this.date1 = e.value;
      console.log('选择的日期：', this.$u.timeFormat(this.date1, 'yyyy-mm-dd'));
    },
    cancel1() {
      console.log('取消选择日期');
    },
    confirm2(e) {
      this.time = e.value;
      console.log('选择的时间：', this.$u.timeFormat(this.time, 'hh:MM'));
    },
    cancel2() {
      console.log('取消选择时间');
    },
    confirm3(e) {
      this.datetime = e.value;
      console.log('选择的日期时间：', this.$u.timeFormat(this.datetime, 'yyyy-mm-dd hh:MM'));
    },
    cancel3() {
      console.log('取消选择日期时间');
    }
  }
}
</script>

表单组件 (u-form, u-form-item) ​

表单组件用于将多个表单元素组合在一起，并提供统一的验证和提交功能。

基本用法 ​

vue
<template>
  <view>
    <u-form :model="form" ref="uForm">
      <u-form-item label="姓名" prop="name">
        <u-input v-model="form.name" placeholder="请输入姓名"></u-input>
      </u-form-item>
      
      <u-form-item label="性别" prop="gender">
        <u-radio-group v-model="form.gender">
          <u-radio name="1">男</u-radio>
          <u-radio name="2">女</u-radio>
        </u-radio-group>
      </u-form-item>
      
      <u-form-item label="爱好" prop="hobby">
        <u-checkbox-group v-model="form.hobby">
          <u-checkbox name="1">阅读</u-checkbox>
          <u-checkbox name="2">游戏</u-checkbox>
          <u-checkbox name="3">音乐</u-checkbox>
        </u-checkbox-group>
      </u-form-item>
      
      <u-form-item label="生日" prop="birthday">
        <u-input 
          v-model="form.birthday" 
          placeholder="请选择生日" 
          disabled 
          @click="showDatePicker = true"
        ></u-input>
      </u-form-item>
      
      <u-form-item label="是否学生" prop="isStudent">
        <u-switch v-model="form.isStudent"></u-switch>
      </u-form-item>
    </u-form>
    
    <u-button text="提交" type="primary" @click="submit"></u-button>
    
    <u-datetime-picker
      v-model="showDatePicker"
      :value="birthday"
      mode="date"
      @confirm="confirmBirthday"
    ></u-datetime-picker>
  </view>
</template>

<script>
export default {
  data() {
    return {
      form: {
        name: '',
        gender: '1',
        hobby: [],
        birthday: '',
        isStudent: false
      },
      rules: {
        name: [
          { required: true, message: '请输入姓名', trigger: 'blur' },
          { min: 2, max: 20, message: '姓名长度在2-20个字符之间', trigger: 'blur' }
        ],
        gender: [
          { required: true, message: '请选择性别', trigger: 'change' }
        ],
        hobby: [
          { required: true, message: '请选择爱好', trigger: 'change' },
          { min: 1, message: '至少选择一个爱好', trigger: 'change' }
        ],
        birthday: [
          { required: true, message: '请选择生日', trigger: 'change' }
        ]
      },
      showDatePicker: false,
      birthday: Number(new Date())
    }
  },
  onReady() {
    this.$refs.uForm.setRules(this.rules);
  },
  methods: {
    submit() {
      this.$refs.uForm.validate(valid => {
        if (valid) {
          console.log('表单验证通过');
          console.log('表单数据：', this.form);
          uni.showToast({
            title: '表单验证通过',
            icon: 'none'
          });
        } else {
          console.log('表单验证失败');
        }
      });
    },
    confirmBirthday(e) {
      this.birthday = e.value;
      this.form.birthday = this.$u.timeFormat(this.birthday, 'yyyy-mm-dd');
    }
  }
}
</script>

表单验证规则 ​

uView UI的表单验证支持多种规则：

js
rules: {
  // 必填验证
  name: [
    { required: true, message: '请输入姓名', trigger: 'blur' }
  ],
  
  // 长度验证
  password: [
    { required: true, message: '请输入密码', trigger: 'blur' },
    { min: 6, max: 20, message: '密码长度在6-20个字符之间', trigger: 'blur' }
  ],
  
  // 正则验证
  phone: [
    { required: true, message: '请输入手机号', trigger: 'blur' },
    { pattern: /^1[3-9]\d{9}$/, message: '请输入正确的手机号', trigger: 'blur' }
  ],
  
  // 自定义验证函数
  code: [
    { required: true, message: '请输入验证码', trigger: 'blur' },
    { validator: (rule, value, callback) => {
      if (value.length !== 6) {
        callback(new Error('验证码长度应为6位'));
      } else {
        callback();
      }
    }, trigger: 'blur' }
  ]
}
