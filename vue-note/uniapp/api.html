UniApp API 使用 ​

UniApp 提供了丰富的 API，可以帮助开发者快速实现各种功能，包括网络请求、数据缓存、设备能力调用以及界面交互等。这些 API 都是跨平台的，可以在不同平台上一致地使用。

网络请求 ​

UniApp 提供了统一的网络请求 API，可以在不同平台上发起 HTTP 请求。

uni.request ​

uni.request 是最基本的网络请求 API，用于发起 HTTP 请求。

js
uni.request({
  url: 'https://api.example.com/data',
  method: 'GET',
  data: {
    id: 1,
    name: 'test'
  },
  header: {
    'content-type': 'application/json',
    'Authorization': 'Bearer token'
  },
  success: (res) => {
    console.log('请求成功', res.data);
  },
  fail: (err) => {
    console.error('请求失败', err);
  },
  complete: () => {
    console.log('请求完成');
  }
});

使用 Promise 风格 ​

UniApp 的 API 支持 Promise 风格调用，可以使用 async/await 简化异步操作。

js
async function fetchData() {
  try {
    const res = await uni.request({
      url: 'https://api.example.com/data',
      method: 'GET',
      data: { id: 1 }
    });
    console.log('请求成功', res.data);
    return res.data;
  } catch (err) {
    console.error('请求失败', err);
    throw err;
  }
}

请求拦截器 ​

可以使用 uni.addInterceptor 添加请求拦截器，对请求进行统一处理。

js
// 添加请求拦截器
uni.addInterceptor('request', {
  invoke(args) {
    // 请求前处理
    console.log('请求拦截器', args);
    // 添加公共参数
    args.header = {
      ...args.header,
      'Authorization': 'Bearer ' + getToken()
    };
    // 添加公共域名
    if (!args.url.startsWith('http')) {
      args.url = 'https://api.example.com' + args.url;
    }
    return args;
  },
  success(res) {
    // 请求成功后处理
    console.log('请求成功拦截器', res);
    // 统一处理响应
    if (res.statusCode === 200) {
      if (res.data.code === 0) {
        return res.data.data;
      } else {
        uni.showToast({
          title: res.data.message || '请求失败',
          icon: 'none'
        });
        return Promise.reject(res.data);
      }
    } else {
      uni.showToast({
        title: '网络错误：' + res.statusCode,
        icon: 'none'
      });
      return Promise.reject(res);
    }
  },
  fail(err) {
    // 请求失败后处理
    console.error('请求失败拦截器', err);
    uni.showToast({
      title: '网络请求失败',
      icon: 'none'
    });
    return Promise.reject(err);
  },
  complete(res) {
    // 请求完成后处理
    console.log('请求完成拦截器', res);
  }
});

上传文件 ​

使用 uni.uploadFile 上传文件：

js
uni.uploadFile({
  url: 'https://api.example.com/upload',
  filePath: tempFilePath,
  name: 'file',
  formData: {
    'user': 'test'
  },
  success: (res) => {
    console.log('上传成功', res.data);
  },
  fail: (err) => {
    console.error('上传失败', err);
  }
});

下载文件 ​

使用 uni.downloadFile 下载文件：

js
uni.downloadFile({
  url: 'https://example.com/file.pdf',
  success: (res) => {
    if (res.statusCode === 200) {
      console.log('下载成功', res.tempFilePath);
      // 保存文件到本地
      uni.saveFile({
        tempFilePath: res.tempFilePath,
        success: (saveRes) => {
          console.log('文件保存成功', saveRes.savedFilePath);
        }
      });
    }
  },
  fail: (err) => {
    console.error('下载失败', err);
  }
});

数据缓存 ​

UniApp 提供了统一的数据缓存 API，可以在不同平台上存储和读取数据。

存储数据 ​

使用 uni.setStorage 或 uni.setStorageSync 存储数据：

js
// 异步存储
uni.setStorage({
  key: 'userInfo',
  data: {
    name: 'Tom',
    age: 18
  },
  success: () => {
    console.log('存储成功');
  }
});

// 同步存储
try {
  uni.setStorageSync('token', 'abc123');
  console.log('存储成功');
} catch (e) {
  console.error('存储失败', e);
}

读取数据 ​

使用 uni.getStorage 或 uni.getStorageSync 读取数据：

js
// 异步读取
uni.getStorage({
  key: 'userInfo',
  success: (res) => {
    console.log('读取成功', res.data);
  },
  fail: (err) => {
    console.error('读取失败', err);
  }
});

// 同步读取
try {
  const token = uni.getStorageSync('token');
  if (token) {
    console.log('读取成功', token);
  }
} catch (e) {
  console.error('读取失败', e);
}

删除数据 ​

使用 uni.removeStorage 或 uni.removeStorageSync 删除数据：

js
// 异步删除
uni.removeStorage({
  key: 'userInfo',
  success: () => {
    console.log('删除成功');
  }
});

// 同步删除
try {
  uni.removeStorageSync('token');
  console.log('删除成功');
} catch (e) {
  console.error('删除失败', e);
}

清空数据 ​

使用 uni.clearStorage 或 uni.clearStorageSync 清空所有数据：

js
// 异步清空
uni.clearStorage();

// 同步清空
try {
  uni.clearStorageSync();
  console.log('清空成功');
} catch (e) {
  console.error('清空失败', e);
}

获取存储信息 ​

使用 uni.getStorageInfo 或 uni.getStorageInfoSync 获取存储信息：

js
// 异步获取
uni.getStorageInfo({
  success: (res) => {
    console.log('存储信息', res.keys, res.currentSize, res.limitSize);
  }
});

// 同步获取
try {
  const res = uni.getStorageInfoSync();
  console.log('存储信息', res.keys, res.currentSize, res.limitSize);
} catch (e) {
  console.error('获取失败', e);
}

摄像头和相册 ​

使用 uni.chooseImage 从相册选择图片或使用相机拍照：

js
uni.chooseImage({
  count: 9, // 最多可选择的图片张数，默认为9
  sizeType: ['original', 'compressed'], // 可选择原图或压缩图
  sourceType: ['album', 'camera'], // 可选择性开放访问相册、相机
  success: (res) => {
    console.log('图片临时路径', res.tempFilePaths);
    console.log('图片临时文件对象', res.tempFiles);
  }
});

使用 uni.chooseVideo 从相册选择视频或使用相机录制视频：

js
uni.chooseVideo({
  sourceType: ['album', 'camera'], // 可选择性开放访问相册、相机
  maxDuration: 60, // 拍摄视频最长拍摄时间，单位秒
  camera: 'back', // 默认拉起后置摄像头
  success: (res) => {
    console.log('视频临时路径', res.tempFilePath);
    console.log('视频时长', res.duration);
    console.log('视频大小', res.size);
    console.log('视频高度', res.height);
    console.log('视频宽度', res.width);
  }
});

蓝牙 ​

使用 uni.openBluetoothAdapter 初始化蓝牙模块：

js
uni.openBluetoothAdapter({
  success: (res) => {
    console.log('初始化蓝牙成功');
    // 开始搜索蓝牙设备
    uni.startBluetoothDevicesDiscovery({
      success: (res) => {
        console.log('开始搜索蓝牙设备');
      }
    });
    // 监听蓝牙设备发现事件
    uni.onBluetoothDeviceFound((res) => {
      console.log('发现新设备', res.devices);
    });
  },
  fail: (err) => {
    console.error('初始化蓝牙失败', err);
  }
});

界面交互 ​

UniApp 提供了丰富的界面交互 API，可以实现各种交互效果。

消息提示框 ​

使用 uni.showToast 显示消息提示框：

js
uni.showToast({
  title: '操作成功',
  icon: 'success', // 可选值：success、loading、none
  duration: 2000, // 提示的延迟时间，单位毫秒
  mask: false, // 是否显示透明蒙层，防止触摸穿透
  success: () => {
    console.log('显示成功');
  }
});

关闭消息提示框：

js
uni.hideToast();

加载提示框 ​

使用 uni.showLoading 显示加载提示框：

js
uni.showLoading({
  title: '加载中...',
  mask: true // 是否显示透明蒙层，防止触摸穿透
});

关闭加载提示框：

js
uni.hideLoading();

模态对话框 ​

使用 uni.showModal 显示模态对话框：

js
uni.showModal({
  title: '提示',
  content: '确定要删除吗？',
  showCancel: true, // 是否显示取消按钮
  cancelText: '取消', // 取消按钮的文字
  cancelColor: '#999', // 取消按钮的文字颜色
  confirmText: '确定', // 确认按钮的文字
  confirmColor: '#007aff', // 确认按钮的文字颜色
  success: (res) => {
    if (res.confirm) {
      console.log('用户点击确定');
    } else if (res.cancel) {
      console.log('用户点击取消');
    }
  }
});

使用 uni.showActionSheet 显示操作菜单：

下拉刷新 ​

开启下拉刷新：

js
// 在 pages.json 中配置
{
  "pages": [
    {
      "path": "pages/index/index",
      "style": {
        "enablePullDownRefresh": true,
        "backgroundTextStyle": "dark"
      }
    }
  ]
}

监听下拉刷新事件：

js
export default {
  onPullDownRefresh() {
    console.log('触发下拉刷新');
    // 刷新数据
    this.refreshData().then(() => {
      // 停止下拉刷新
      uni.stopPullDownRefresh();
    });
  },
  methods: {
    refreshData() {
      return new Promise((resolve) => {
        setTimeout(() => {
          resolve();
        }, 2000);
      });
    }
  }
}

手动触发下拉刷新：

js
uni.startPullDownRefresh();

停止下拉刷新：

js
uni.stopPullDownRefresh();

动画 ​

创建动画实例：

js
const animation = uni.createAnimation({
  duration: 1000, // 动画持续时间
  timingFunction: 'ease', // 动画效果
  delay: 0 // 动画延迟时间
});

定义动画：

js
// 顺序执行
animation.scale(2).rotate(45).step();
// 接着执行
animation.scale(1).rotate(0).step();

this.animationData = animation.export();

在模板中使用动画：

html
<view :animation="animationData" class="animated-view"></view>
