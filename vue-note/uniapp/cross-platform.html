UniApp 跨端兼容 ​

UniApp 的核心优势之一是跨平台能力，可以让一套代码运行在多个平台上。但是，不同平台之间存在差异，需要采取一些措施来处理这些差异，确保应用在各个平台上都能正常运行和良好展示。

条件编译 ​

条件编译是 UniApp 中处理跨平台差异的主要方式，可以根据不同平台编写特定的代码。

条件编译语法 ​

UniApp 支持在 Vue 文件的 template、script、style 标签中以及 js、css 文件中使用条件编译。

条件编译的语法是：以 #ifdef 或 #ifndef 加平台标识开头，以 #endif 结尾。

  • #ifdef：当符合平台标识时，编译到对应平台
  • #ifndef：当不符合平台标识时，编译到对应平台

平台标识 ​

UniApp 支持的平台标识如下：

条件编译示例 ​

在 template 中使用条件编译 ​

html
<template>
  <view>
    <!-- #ifdef APP-PLUS -->
    <view>仅在 App 平台显示</view>
    <!-- #endif -->
    
    <!-- #ifdef H5 -->
    <view>仅在 H5 平台显示</view>
    <!-- #endif -->
    
    <!-- #ifdef MP-WEIXIN -->
    <view>仅在微信小程序显示</view>
    <!-- #endif -->
    
    <!-- #ifndef MP-ALIPAY -->
    <view>在除了支付宝小程序的其他平台显示</view>
    <!-- #endif -->
    
    <!-- 多平台共用一段代码 -->
    <!-- #ifdef H5 || MP-WEIXIN || APP-PLUS -->
    <view>在 H5、微信小程序、App 平台显示</view>
    <!-- #endif -->
  </view>
</template>

在 script 中使用条件编译 ​

js
<script>
export default {
  data() {
    return {
      // 所有平台通用数据
      title: 'UniApp'
    }
  },
  methods: {
    // #ifdef APP-PLUS
    // 仅在 App 平台执行的方法
    appMethod() {
      console.log('App 平台特有方法');
    },
    // #endif
    
    // #ifdef H5
    // 仅在 H5 平台执行的方法
    h5Method() {
      console.log('H5 平台特有方法');
    },
    // #endif
    
    // 根据不同平台返回不同的值
    getPlatformValue() {
      // #ifdef APP-PLUS
      return 'App 平台';
      // #endif
      
      // #ifdef H5
      return 'H5 平台';
      // #endif
      
      // #ifdef MP-WEIXIN
      return '微信小程序平台';
      // #endif
      
      // 默认返回值
      return '其他平台';
    }
  },
  // #ifdef APP-PLUS
  // 仅在 App 平台执行的生命周期
  onBackPress() {
    console.log('监听返回按钮事件');
    return false;
  },
  // #endif
  
  // #ifdef MP-WEIXIN
  // 仅在微信小程序执行的生命周期
  onShareAppMessage() {
    return {
      title: '分享标题',
      path: '/pages/index/index'
    };
  }
  // #endif
}
</script>

运行期判断平台 ​

除了编译期的条件编译，UniApp 还提供了运行期判断平台的方式，可以使用 uni.getSystemInfoSync().platform 来判断当前运行的平台。

js
const systemInfo = uni.getSystemInfoSync();
const platform = systemInfo.platform; // ios、android、windows、mac、devtools

if (platform === 'ios') {
  // iOS 平台特有代码
} else if (platform === 'android') {
  // Android 平台特有代码
}

API 差异 ​

不同平台的 API 支持情况不同，需要进行兼容处理。

API 兼容性检查 ​

在使用某些 API 前，可以先检查当前平台是否支持该 API：

js
// 检查是否支持某个 API
if (uni.canIUse('getLocation')) {
  uni.getLocation({
    success: (res) => {
      console.log('当前位置', res);
    }
  });
} else {
  console.log('当前平台不支持获取位置');
}

API 差异化处理 ​

js
// 分享功能
function share() {
  // #ifdef APP-PLUS
  // App 平台使用原生分享
  uni.share({
    provider: 'weixin',
    scene: 'WXSceneSession',
    type: 1,
    summary: '分享内容',
    success: (res) => {
      console.log('分享成功');
    }
  });
  // #endif
  
  // #ifdef MP-WEIXIN
  // 微信小程序使用自带的分享
  // 通过页面的 onShareAppMessage 方法处理
  // #endif
  
  // #ifdef H5
  // H5 平台使用 Web Share API 或自定义分享弹窗
  if (navigator.share) {
    navigator.share({
      title: '分享标题',
      text: '分享内容',
      url: window.location.href
    });
  } else {
    // 显示自定义分享弹窗
    this.showSharePopup = true;
  }
  // #endif
}

组件兼容性处理 ​

html
<template>
  <view>
    <!-- 所有平台通用的按钮 -->
    <button type="primary">通用按钮</button>
    
    <!-- #ifdef APP-PLUS -->
    <!-- App 平台特有的按钮 -->

    <!-- #ifdef MP-WEIXIN -->
    <!-- 微信小程序特有的按钮 -->
    <button type="primary" open-type="contact">联系客服</button>
    <!-- #endif -->
    
    <!-- 根据平台使用不同的组件实现相同的功能 -->
    <!-- #ifdef APP-PLUS || H5 -->
    <picker mode="date" :value="date" @change="bindDateChange">
      <view class="picker">当前选择: {{ date }}</view>
    </picker>
    <!-- #endif -->
    
    <!-- #ifdef MP-ALIPAY -->
    <!-- 支付宝小程序的日期选择器写法可能有差异 -->
    <picker mode="date" :value="date" @change="bindDateChange">
      <view class="picker">当前选择: {{ date }}</view>
    </picker>
    <!-- #endif -->
  </view>
</template>

<script>
export default {
  data() {
    return {
      date: '2023-01-01'
    }
  },
  methods: {
    bindDateChange(e) {
      this.date = e.detail.value;
    }
  }
}
</script>

适配方案 ​

屏幕适配 ​

UniApp 提供了一套统一的屏幕适配方案，可以适配不同尺寸的屏幕。

rpx 响应式单位 ​

UniApp 提供了 rpx 作为响应式单位，可以根据屏幕宽度自动缩放。在 750rpx 宽度的屏幕下，1rpx = 1物理像素。

css
.container {
  width: 750rpx; /* 屏幕宽度 */
  height: 100rpx;
  padding: 20rpx;
  font-size: 28rpx;
}

媒体查询 ​

可以使用媒体查询适配不同尺寸的屏幕：

css
/* 所有屏幕通用样式 */
.content {
  padding: 20rpx;
}

/* 小屏幕手机 */
@media screen and (max-width: 375px) {
  .content {
    padding: 10rpx;
    font-size: 24rpx;
  }
}

/* 大屏幕手机 */
@media screen and (min-width: 768px) {
  .content {
    padding: 30rpx;
    font-size: 32rpx;
  }
}

功能降级处理 ​

当某些平台不支持特定功能时，需要提供降级方案：

js
// 生物识别认证
function biometricAuthentication() {
  // #ifdef APP-PLUS
  // App 平台使用指纹或面容识别
  const deviceInfo = uni.getSystemInfoSync();
  if (deviceInfo.provider === 'apple') {
    // iOS 设备
    uni.startSoterAuthentication({
      requestAuthModes: ['fingerPrint', 'facial'],
      challenge: '123456',
      authContent: '请验证指纹或面容',
      success: () => {
        console.log('认证成功');
      },
      fail: () => {
        // 降级到密码认证
        this.showPasswordAuth();
      }
    });
  } else {
    // Android 设备
    plus.fingerprint.authenticate(() => {
      console.log('指纹认证成功');
    }, () => {
      // 降级到密码认证
      this.showPasswordAuth();
    }, {
      message: '请验证指纹'
    });
  }
  // #endif
  
  // #ifdef H5 || MP
  // H5 和小程序平台直接使用密码认证
  this.showPasswordAuth();
  // #endif
}

// 密码认证（降级方案）
function showPasswordAuth() {
  uni.showModal({
    title: '密码验证',
    content: '请输入密码进行验证',
    editable: true,
    success: (res) => {
      if (res.confirm && res.content) {
        // 验证密码
        this.verifyPassword(res.content);
      }
    }
  });
}

封装平台差异 ​

将平台差异封装成统一的接口，便于调用：

js
// platform.js

// 分享功能封装
export function shareContent(options) {
  const { title, summary, imageUrl, href } = options;
  
  // #ifdef APP-PLUS
  uni.share({
    provider: 'weixin',
    scene: 'WXSceneSession',
    type: 1,
    title,
    summary,
    imageUrl,
    href,
    success: options.success,
    fail: options.fail
  });
  // #endif
  
  // #ifdef H5
  if (navigator.share) {
    navigator.share({
      title,
      text: summary,
      url: href
    }).then(options.success).catch(options.fail);
  } else {
    // 显示自定义分享弹窗
    showSharePopup(options);
  }
  // #endif
  
  // #ifdef MP-WEIXIN
  // 微信小程序通过 onShareAppMessage 处理，这里可以缓存分享参数
  getApp().globalData.shareInfo = {
    title,
    imageUrl,
    path: href
  };
  // #endif
}

// 支付功能封装
export function doPay(options) {
  const { orderInfo, success, fail } = options;
  
  // #ifdef APP-PLUS
  uni.requestPayment({
    provider: 'alipay', // 或 'wxpay'
    orderInfo,
    success,
    fail
  });
  // #endif
  
  // #ifdef MP-WEIXIN
  uni.requestPayment({
    timeStamp: orderInfo.timeStamp,
    nonceStr: orderInfo.nonceStr,
    package: orderInfo.package,
    signType: orderInfo.signType,
    paySign: orderInfo.paySign,
    success,
    fail
  });
  // #endif
  
  // #ifdef H5
  // H5 平台可能需要跳转到支付页面
  window.location.href = orderInfo.h5PayUrl;
  // #endif
}

总结 ​
