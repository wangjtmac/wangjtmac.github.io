UniApp 网络请求封装 ​

网络请求基础 ​

UniApp 提供了跨平台的网络请求 API，可以在各端发起 HTTP 请求，获取服务器数据。基础的网络请求使用 uni.request() 方法：

js
uni.request({
  url: 'https://api.example.com/data',
  method: 'GET',
  data: {
    param1: 'value1',
    param2: 'value2'
  },
  header: {
    'content-type': 'application/json'
  },
  success: (res) => {
    console.log(res.data);
  },
  fail: (err) => {
    console.error(err);
  },
  complete: () => {
    console.log('请求完成');
  }
});

虽然基础 API 可以满足简单的网络请求需求，但在实际项目中，我们通常需要对网络请求进行封装，以实现：

  1. 统一的请求配置和错误处理
  2. 请求和响应拦截
  3. 身份验证和 token 管理
  4. 请求重试和取消
  5. 缓存管理
  6. 更优雅的 Promise 接口

封装网络请求类 ​

下面是一个基于 Promise 的网络请求封装示例：

js
// utils/request.js

/**
 * 网络请求工具类
 */
class Request {
  constructor(options = {}) {
    // 默认配置
    this.config = {
      baseUrl: options.baseUrl || '',
      header: options.header || {},
      timeout: options.timeout || 60000,
      dataType: options.dataType || 'json',
      responseType: options.responseType || 'text'
    }
    
    // 拦截器
    this.interceptors = {
      request: (config) => {
        return config;
      },
      response: (response) => {
        return response;
      }
    }
  }
  
  /**
   * 设置请求拦截器
   * @param {Function} callback - 请求拦截器回调函数
   */
  setRequestInterceptor(callback) {
    if (typeof callback === 'function') {
      this.interceptors.request = callback;
    }
  }
  
  /**
   * 设置响应拦截器
   * @param {Function} callback - 响应拦截器回调函数
   */
  setResponseInterceptor(callback) {
    if (typeof callback === 'function') {
      this.interceptors.response = callback;
    }
  }
  
  /**
   * 发起请求
   * @param {Object} options - 请求配置
   * @returns {Promise} - 返回Promise对象
   */
  request(options = {}) {
    // 合并配置
    const config = {
      ...this.config,
      ...options,
      header: { ...this.config.header, ...options.header }
    };
    
    // 请求拦截
    const interceptedConfig = this.interceptors.request(config);
    
    // 发起请求
    return new Promise((resolve, reject) => {
      uni.request({
        url: interceptedConfig.baseUrl + interceptedConfig.url,
        method: interceptedConfig.method || 'GET',
        data: interceptedConfig.data,
        header: interceptedConfig.header,
        timeout: interceptedConfig.timeout,
        dataType: interceptedConfig.dataType,
        responseType: interceptedConfig.responseType,
        success: (res) => {
          // 响应拦截
          const interceptedRes = this.interceptors.response(res);
          resolve(interceptedRes);
        },
        fail: (err) => {
          reject(err);
        }
      });
    });
  }
  
  /**
   * GET请求
   * @param {String} url - 请求地址
   * @param {Object} data - 请求参数
   * @param {Object} options - 其他配置
   * @returns {Promise} - 返回Promise对象
   */
  get(url, data = {}, options = {}) {
    return this.request({
      url,
      data,
      method: 'GET',
      ...options
    });
  }
  
  /**
   * POST请求
   * @param {String} url - 请求地址
   * @param {Object} data - 请求参数
   * @param {Object} options - 其他配置
   * @returns {Promise} - 返回Promise对象
   */
  post(url, data = {}, options = {}) {
    return this.request({
      url,
      data,
      method: 'POST',
      ...options
    });
  }
  
  /**
   * PUT请求
   * @param {String} url - 请求地址
   * @param {Object} data - 请求参数
   * @param {Object} options - 其他配置
   * @returns {Promise} - 返回Promise对象
   */
  put(url, data = {}, options = {}) {
    return this.request({
      url,
      data,
      method: 'PUT',
      ...options
    });
  }
  
  /**
   * DELETE请求
   * @param {String} url - 请求地址
   * @param {Object} data - 请求参数
   * @param {Object} options - 其他配置
   * @returns {Promise} - 返回Promise对象
   */
  delete(url, data = {}, options = {}) {
    return this.request({
      url,
      data,
      method: 'DELETE',
      ...options
    });
  }
}

export default Request;

使用封装的请求类 ​

创建请求实例并配置拦截器：

js
// api/index.js
import Request from '@/utils/request';
import store from '@/store';

// 创建请求实例
const http = new Request({
  baseUrl: 'https://api.example.com',
  timeout: 10000,
  header: {
    'content-type': 'application/json'
  }
});

// 请求拦截器
http.setRequestInterceptor((config) => {
  // 获取 token
  const token = uni.getStorageSync('token') || store.state.user.token;
  
  // 添加 token 到请求头
  if (token) {
    config.header['Authorization'] = `Bearer ${token}`;
  }
  
  // 添加时间戳防止缓存
  if (config.method === 'GET') {
    config.data = {
      ...config.data,
      _t: Date.now()
    };
  }
  
  // 显示加载提示
  uni.showLoading({
    title: '加载中...',
    mask: true
  });
  
  return config;
});

// 响应拦截器
http.setResponseInterceptor((response) => {
  // 隐藏加载提示
  uni.hideLoading();
  
  // 请求成功，但业务状态可能有错误
  if (response.statusCode === 200) {
    // 假设后端返回格式为 { code: 0, data: {}, message: '' }
    const { code, data, message } = response.data;
    
    if (code === 0) {
      // 业务成功
      return data;
    } else {
      // 业务失败
      uni.showToast({
        title: message || '请求失败',
        icon: 'none'
      });
      return Promise.reject(message);
    }
  } else if (response.statusCode === 401) {
    // 未授权，需要登录
    uni.showToast({
      title: '请先登录',
      icon: 'none'
    });
    
    // 清除 token
    uni.removeStorageSync('token');
    store.commit('user/CLEAR_USER');
    
    // 跳转到登录页
    setTimeout(() => {
      uni.navigateTo({
        url: '/pages/login/login'
      });
    }, 1500);
    
    return Promise.reject('未授权');
  } else {
    // 其他错误
    uni.showToast({
      title: `请求失败(${response.statusCode})`,
      icon: 'none'
    });
    return Promise.reject(`请求失败(${response.statusCode})`);
  }
});

export default http;

请求取消 ​

js
// utils/request.js (扩展)

class Request {
  constructor(options = {}) {
    // 原有配置...
    
    // 存储请求任务，用于取消
    this.taskMap = new Map();
  }
  
  /**
   * 发起请求
   * @param {Object} options - 请求配置
   * @returns {Promise} - 返回Promise对象
   */
  request(options = {}) {
    // 合并配置...
    
    // 请求拦截...
    
    // 生成请求标识
    const requestId = options.requestId || `${options.url}_${Date.now()}`;
    
    // 发起请求
    return new Promise((resolve, reject) => {
      const task = uni.request({
        // 请求配置...
        
        complete: () => {
          // 请求完成后，从 taskMap 中移除
          this.taskMap.delete(requestId);
        }
      });
      
      // 将请求任务存储到 taskMap
      this.taskMap.set(requestId, task);
    });
  }
  
  /**
   * 取消请求
   * @param {String} requestId - 请求标识
   */
  cancel(requestId) {
    const task = this.taskMap.get(requestId);
    if (task) {
      task.abort();
      this.taskMap.delete(requestId);
    }
  }
  
  /**
   * 取消所有请求
   */
  cancelAll() {
    for (const [requestId, task] of this.taskMap.entries()) {
      task.abort();
      this.taskMap.delete(requestId);
    }
  }
}

请求重试 ​

js
// utils/request.js (扩展)

class Request {
  constructor(options = {}) {
    // 原有配置...
    
    // 重试配置
    this.retryConfig = {
      count: options.retryCount || 3,
      delay: options.retryDelay || 1000
    };
  }
  
  /**
   * 发起请求
   * @param {Object} options - 请求配置
   * @param {Number} retryCount - 当前重试次数
   * @returns {Promise} - 返回Promise对象
   */
  request(options = {}, retryCount = 0) {
    // 合并配置...
    
    // 请求拦截...
    
    // 发起请求
    return new Promise((resolve, reject) => {
      uni.request({
        // 请求配置...
        
        fail: (err) => {
          // 判断是否需要重试
          if (
            options.retry !== false && 
            retryCount < this.retryConfig.count && 
            this.shouldRetry(err)
          ) {
            setTimeout(() => {
              // 重试请求
              this.request(options, retryCount + 1)
                .then(resolve)
                .catch(reject);
            }, this.retryConfig.delay);
          } else {
            reject(err);
          }
        }
      });
    });
  }
  
  /**
   * 判断是否应该重试
   * @param {Object} error - 错误信息
   * @returns {Boolean} - 是否重试
   */
  shouldRetry(error) {
    // 网络错误、超时等情况下重试
    return error.errMsg && (
      error.errMsg.includes('timeout') || 
      error.errMsg.includes('network')
    );
  }
}

请求缓存 ​

js
// utils/request.js (扩展)

class Request {
  constructor(options = {}) {
    // 原有配置...
    
    // 缓存配置
    this.cacheConfig = {
      enable: options.enableCache || false,
      maxAge: options.cacheMaxAge || 60000, // 默认缓存1分钟
      exclude: options.cacheExclude || [] // 排除缓存的接口
    };
    
    // 缓存存储
    this.cache = new Map();
  }
  
  /**
   * 发起请求
   * @param {Object} options - 请求配置
   * @returns {Promise} - 返回Promise对象
   */
  request(options = {}) {
    // 合并配置...
    
    // 请求拦截...
    
    // 处理缓存
    if (
      this.cacheConfig.enable && 
      options.method === 'GET' && 
      options.cache !== false &&
      !this.cacheConfig.exclude.includes(options.url)
    ) {
      // 生成缓存 key
      const cacheKey = this.generateCacheKey(options);
      
      // 获取缓存
      const cache = this.cache.get(cacheKey);
      
      // 如果缓存存在且未过期，直接返回缓存数据
      if (cache && cache.expires > Date.now()) {
        return Promise.resolve(cache.data);
      }
      
      // 发起请求
      return new Promise((resolve, reject) => {
        uni.request({
          // 请求配置...
          
          success: (res) => {
            // 响应拦截
            const interceptedRes = this.interceptors.response(res);
            
            // 设置缓存
            const cacheMaxAge = options.cacheMaxAge || this.cacheConfig.maxAge;
            this.cache.set(cacheKey, {
              data: interceptedRes,
              expires: Date.now() + cacheMaxAge
            });
            
            resolve(interceptedRes);
          },
          fail: (err) => {
            reject(err);
          }
        });
      });
    }
    
    // 不使用缓存，直接发起请求
    return new Promise((resolve, reject) => {
      // 原有请求逻辑...
    });
  }
  
  /**
   * 生成缓存 key
   * @param {Object} options - 请求配置
   * @returns {String} - 缓存 key
   */
  generateCacheKey(options) {
    const { url, data } = options;
    return `${url}_${JSON.stringify(data || {})}`;
  }
  
  /**
   * 清除指定 URL 的缓存
   * @param {String} url - 请求地址
   * @param {Object} data - 请求参数
   */
  clearCache(url, data = {}) {
    const cacheKey = this.generateCacheKey({ url, data });
    this.cache.delete(cacheKey);
  }
  
  /**
   * 清除所有缓存
   */
  clearAllCache() {
    this.cache.clear();
  }
}

实践心得 ​

  1. 统一错误处理：

    • 在响应拦截器中统一处理错误，避免在每个业务代码中重复处理
    • 根据错误类型（网络错误、业务错误、授权错误等）采取不同的处理策略

  2. 请求管理：

    • 对于可能重复发起的请求，考虑实现请求防抖或节流
    • 页面离开时取消未完成的请求，避免内存泄漏

  3. 缓存策略：

    • 对于不常变化的数据（如配置信息、字典数据等），可以使用缓存减少请求次数
    • 设置合理的缓存过期时间，平衡实时性和性能

  4. 跨平台适配：

    • 注意不同平台的网络请求限制和特性，如小程序的域名白名单
    • 使用条件编译处理平台差异

  5. 性能优化：

    • 避免频繁的短时间内请求同一接口
    • 合理使用并发请求，避免一次性发起过多请求
    • 考虑实现请求队列，控制并发数量

电商应用的商品模块 ​

js
// api/modules/product.js
import http from '@/api/index';

export default {
  /**
   * 获取商品列表
   * @param {Object} params - 查询参数
   * @returns {Promise} - 返回Promise对象
   */
  getProductList(params) {
    return http.get('/products', params, {
      // 启用缓存，缓存时间30秒
      cache: true,
      cacheMaxAge: 30000
    });
  },
  
  /**
   * 获取商品详情
   * @param {String|Number} id - 商品ID
   * @returns {Promise} - 返回Promise对象
   */
  getProductDetail(id) {
    return http.get(`/products/${id}`, {}, {
      // 不使用缓存，确保获取最新数据
      cache: false
    });
  },
  
  /**
   * 搜索商品
   * @param {Object} params - 搜索参数
   * @returns {Promise} - 返回Promise对象
   */
  searchProducts(params) {
    // 生成请求ID，用于取消请求
    const requestId = `search_${Date.now()}`;
    
    return http.get('/products/search', params, {
      requestId,
      // 禁用重试
      retry: false
    });
  },
  
  /**
   * 取消搜索请求
   */
  cancelSearch() {
    // 取消所有包含 search_ 前缀的请求
    http.cancelAll((requestId) => requestId.startsWith('search_'));
  }
};

在页面中使用 ​

<template>
  <view class="search-page">
    <view class="search-bar">
      <input

        <view 
          v-for="(item, index) in products" 
          :key="item.id" 
          class="product-item"
          @tap="viewProduct(item.id)"
        >
          <image :src="item.image" mode="aspectFill" />
          <view class="product-info">
            <text class="product-name">{{ item.name }}</text>
            <text class="product-price">¥{{ item.price }}</text>
          </view>
        </view>
      </view>
    </view>
  </view>

<script>
import api from '@/api/index';
import debounce from '@/utils/debounce';

export default {
  data() {
    return {
      keyword: '',
      products: [],
      loading: false,
      error: ''
    };
  },
  created() {
    // 创建防抖搜索函数
    this.debouncedSearch = debounce(this.searchProducts, 500);
  },
  onUnload() {
    // 页面卸载时取消搜索请求
    api.product.cancelSearch();
  },
  methods: {
    onKeywordChange() {
      if (this.keyword.trim()) {
        this.debouncedSearch();
      } else {
        this.products = [];
      }
    },
    async searchProducts() {
      // 取消之前的搜索请求
      api.product.cancelSearch();
      
      this.loading = true;
      this.error = '';
      
      try {
        const data = await api.product.searchProducts({
          keyword: this.keyword,
          page: 1,
          size: 20
        });
        
        this.products = data.list || [];
      } catch (err) {
        // 如果是请求被取消，不显示错误
        if (err.errMsg && err.errMsg.includes('abort')) {
          return;
        }
        
        console.error('搜索失败', err);
        this.error = '搜索失败，请重试';
      } finally {
        this.loading = false;
      }
    },
    clearSearch() {
      this.keyword = '';
      this.products = [];
      api.product.cancelSearch();
    },
    viewProduct(id) {
      uni.navigateTo({
        url: `/pages/product/detail?id=${id}`
      });
    }
  }
};
</script>

总结 ​

在 UniApp 中，合理封装网络请求模块可以大幅提高开发效率和代码质量。通过封装请求类、拦截器、API 模块等，我们可以实现统一的请求管理、错误处理、身份验证等功能，使代码更加清晰、可维护。

高级功能如请求取消、重试、缓存等，可以根据项目需求选择性实现，进一步提升应用性能和用户体验。在实际开发中，应当根据项目规模和需求，选择适当的封装方式和功能，避免过度设计。
