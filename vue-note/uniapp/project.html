UniApp实战项目 ​

本章节将带领大家完成一个完整的UniApp实战项目，从项目设计、架构搭建到功能实现，最终打包发布。通过这个实战项目，我们将综合运用前面所学的所有知识点。

项目概述 ​

我们将开发一个"多平台社区资讯App"，具有以下功能：

  • 用户注册与登录
  • 资讯浏览与搜索
  • 社区互动（点赞、评论、分享）
  • 个人中心
  • 设置与偏好

该项目将同时适配微信小程序、H5、iOS和Android平台。

技术栈选择 ​

  • 前端框架：Vue3 + UniApp
  • 状态管理：Pinia
  • UI组件库：uView
  • 网络请求：uni-request（自定义封装）
  • 数据存储：本地存储 + 云端数据库

项目准备工作 ​

核心功能实现 ​

1. 网络请求封装 ​

在utils目录下创建request.js文件，封装网络请求：

js
// utils/request.js
import { useUserStore } from '@/store/user';

// 基础URL
const BASE_URL = 'https://api.example.com';

// 请求拦截器
const requestInterceptor = (config) => {
  const userStore = useUserStore();
  
  // 添加token到请求头
  if (userStore.token) {
    config.header = {
      ...config.header,
      'Authorization': `Bearer ${userStore.token}`
    };
  }
  
  return config;
};

// 响应拦截器
const responseInterceptor = (response) => {
  if (response.statusCode === 200) {
    // 请求成功
    if (response.data.code === 0) {
      return response.data.data;
    } else {
      // 业务错误
      uni.showToast({
        title: response.data.message || '请求失败',
        icon: 'none'
      });
      return Promise.reject(response.data);
    }
  } else if (response.statusCode === 401) {
    // 未授权，跳转登录
    const userStore = useUserStore();
    userStore.logout();
    uni.navigateTo({
      url: '/pages/login/login'
    });
    return Promise.reject(response);
  } else {
    // 其他错误
    uni.showToast({
      title: '网络异常，请稍后再试',
      icon: 'none'
    });
    return Promise.reject(response);
  }
};

// 请求方法
const request = (options) => {
  options.url = BASE_URL + options.url;
  options = requestInterceptor(options);
  
  return new Promise((resolve, reject) => {
    uni.request({
      ...options,
      success: (res) => {
        try {
          const result = responseInterceptor(res);
          resolve(result);
        } catch (error) {
          reject(error);
        }
      },
      fail: (err) => {
        uni.showToast({
          title: '网络异常，请稍后再试',
          icon: 'none'
        });
        reject(err);
      }
    });
  });
};

// 导出请求方法
export default {
  get: (url, data = {}, options = {}) => {
    return request({
      url,
      data,
      method: 'GET',
      ...options
    });
  },
  post: (url, data = {}, options = {}) => {
    return request({
      url,
      data,
      method: 'POST',
      ...options
    });
  },
  put: (url, data = {}, options = {}) => {
    return request({
      url,
      data,
      method: 'PUT',
      ...options
    });
  },
  delete: (url, data = {}, options = {}) => {
    return request({
      url,
      data,
      method: 'DELETE',
      ...options
    });
  }
};

创建用户状态管理模块：

js
// store/user.js
import { defineStore } from 'pinia';

export const useUserStore = defineStore('user', {
  state: () => ({
    token: uni.getStorageSync('token') || '',
    userInfo: uni.getStorageSync('userInfo') || null
  }),
  
  getters: {
    isLoggedIn: (state) => !!state.token
  },
  
  actions: {
    // 登录
    async login(username, password) {
      try {
        // 模拟登录API调用
        const res = await uni.request({
          url: 'https://api.example.com/login',
          method: 'POST',
          data: { username, password }
        });
        
        if (res.data.code === 0) {
          this.token = res.data.data.token;
          this.userInfo = res.data.data.userInfo;
          
          // 存储到本地
          uni.setStorageSync('token', this.token);
          uni.setStorageSync('userInfo', this.userInfo);
          
          return true;
        }
        return false;
      } catch (error) {
        console.error('登录失败', error);
        return false;
      }
    },
    
    // 登出
    logout() {
      this.token = '';
      this.userInfo = null;
      
      // 清除本地存储
      uni.removeStorageSync('token');
      uni.removeStorageSync('userInfo');
      
      // 跳转到登录页
      uni.reLaunch({
        url: '/pages/login/login'
      });
    },
    
    // 更新用户信息
    updateUserInfo(userInfo) {
      this.userInfo = { ...this.userInfo, ...userInfo };
      uni.setStorageSync('userInfo', this.userInfo);
    }
  }
});

3. 页面实现 ​

登录页面实现 ​

vue
<!-- pages/login/login.vue -->
<template>
  <view class="login-container">
    <view class="logo">
      <image src="/static/images/logo.png" mode="aspectFit"></image>
    </view>
    
    <view class="form-item">
      <input 
        type="text" 
        v-model="form.username" 
        placeholder="请输入用户名" 
        class="input"
      />
    </view>
    
    <view class="form-item">
      <input 
        type="password" 
        v-model="form.password" 
        placeholder="请输入密码" 
        class="input"
      />
    </view>
    
    <view class="btn-area">
      <button 
        type="primary" 
        :loading="loading" 
        @click="handleLogin"
      >登录</button>
      
      <view class="register-link">
        <text @click="goToRegister">没有账号？立即注册</text>
      </view>
    </view>
  </view>
</template>

<script setup>
import { ref, reactive } from 'vue';
import { useUserStore } from '@/store/user';
import { onLoad } from '@dcloudio/uni-app';

// 表单数据
const form = reactive({
  username: '',
  password: ''
});

// 加载状态
const loading = ref(false);

// 用户状态管理
const userStore = useUserStore();

// 登录处理
const handleLogin = async () => {
  // 表单验证
  if (!form.username || !form.password) {
    uni.showToast({
      title: '请输入用户名和密码',
      icon: 'none'
    });
    return;
  }
  
  // 设置加载状态
  loading.value = true;
  
  try {
    // 调用登录接口
    const success = await userStore.login(form.username, form.password);
    
    if (success) {
      uni.showToast({
        title: '登录成功',
        icon: 'success'
      });
      
      // 跳转到首页
      uni.switchTab({
        url: '/pages/index/index'
      });
    } else {
      uni.showToast({
        title: '用户名或密码错误',
        icon: 'none'
      });
    }
  } catch (error) {
    console.error('登录失败', error);
    uni.showToast({
      title: '登录失败，请稍后再试',
      icon: 'none'
    });
  } finally {
    loading.value = false;
  }
};

// 跳转到注册页
const goToRegister = () => {
  uni.navigateTo({
    url: '/pages/register/register'
  });
};

// 页面加载
onLoad(() => {
  // 如果已登录，跳转到首页
  if (userStore.isLoggedIn) {
    uni.switchTab({
      url: '/pages/index/index'
    });
  }
});
</script>

首页实现 ​

vue
<!-- pages/index/index.vue -->
<template>
  <view class="container">
    <!-- 搜索框 -->
    <view class="search-box">
      <view class="search-input" @click="goToSearch">
        <text class="iconfont icon-search"></text>
        <text>搜索文章、话题</text>
      </view>
    </view>
    
    <!-- 轮播图 -->
    <swiper 
      class="banner" 
      indicator-dots 
      autoplay 
      circular
      :interval="3000"
      :duration="500"
    >
      <swiper-item v-for="(item, index) in bannerList" :key="index">
        <image :src="item.image" mode="aspectFill" @click="handleBannerClick(item)"></image>
      </swiper-item>
    </swiper>
    
    <!-- 分类导航 -->
    <view class="category-nav">
      <view 
        class="category-item" 
        v-for="(item, index) in categoryList" 
        :key="index"
        @click="handleCategoryClick(item)"
      >
        <image :src="item.icon" mode="aspectFit"></image>
        <text>{{ item.name }}</text>
      </view>
    </view>
    
    <!-- 文章列表 -->
    <view class="article-list">
      <view class="list-header">
        <text class="title">推荐文章</text>
        <text class="more" @click="goToMore">查看更多</text>
      </view>
      
      <view 
        class="article-item" 
        v-for="(item, index) in articleList" 
        :key="index"
        @click="goToDetail(item.id)"
      >
        <view class="article-info">
          <view class="article-title">{{ item.title }}</view>
          <view class="article-desc">{{ item.description }}</view>
          <view class="article-meta">
            <text>{{ item.author }}</text>
            <text>{{ item.publishTime }}</text>
            <text>{{ item.viewCount }}阅读</text>
          </view>
        </view>
        <view class="article-image">
          <image :src="item.coverImage" mode="aspectFill"></image>
        </view>
      </view>
      
      <!-- 加载更多 -->
      <view class="load-more" v-if="hasMore">
        <text v-if="loading">加载中...</text>
        <text v-else @click="loadMore">加载更多</text>
      </view>
      <view class="no-more" v-else>
        <text>没有更多数据了</text>
      </view>
    </view>
  </view>
</template>

<script setup>
import { ref, onMounted } from 'vue';
import { onPullDownRefresh, onReachBottom } from '@dcloudio/uni-app';
import request from '@/utils/request';

// 轮播图数据
const bannerList = ref([]);
// 分类数据
const categoryList = ref([]);
// 文章列表
const articleList = ref([]);
// 是否有更多数据
const hasMore = ref(true);
// 加载状态
const loading = ref(false);
// 当前页码
const page = ref(1);
// 每页数量
const pageSize = ref(10);

// 获取轮播图数据
const fetchBanners = async () => {
  try {
    const res = await request.get('/api/banners');
    bannerList.value = res.data || [];
  } catch (error) {
    console.error('获取轮播图失败', error);
  }
};

// 获取分类数据
const fetchCategories = async () => {
  try {
    const res = await request.get('/api/categories');
    categoryList.value = res.data || [];
  } catch (error) {
    console.error('获取分类失败', error);
  }
};

// 获取文章列表
const fetchArticles = async (reset = false) => {
  if (loading.value) return;
  
  loading.value = true;
  
  try {
    const res = await request.get('/api/articles', {
      page: reset ? 1 : page.value,
      pageSize: pageSize.value
    });
    
    if (reset) {
      articleList.value = res.data || [];
      page.value = 1;
    } else {
      articleList.value = [...articleList.value, ...(res.data || [])];
      page.value += 1;
    }
    
    // 判断是否还有更多数据
    hasMore.value = res.data && res.data.length === pageSize.value;
  } catch (error) {
    console.error('获取文章列表失败', error);
  } finally {
    loading.value = false;
    // 停止下拉刷新
    uni.stopPullDownRefresh();
  }
};

// 加载更多
const loadMore = () => {
  if (!hasMore.value || loading.value) return;
  fetchArticles();
};

// 跳转到搜索页
const goToSearch = () => {
  uni.navigateTo({
    url: '/pages/search/search'
  });
};

// 跳转到更多文章
const goToMore = () => {
  uni.navigateTo({
    url: '/pages/article/list'
  });
};

// 跳转到文章详情
const goToDetail = (id) => {
  uni.navigateTo({
    url: `/pages/article/detail?id=${id}`
  });
};

// 处理轮播图点击
const handleBannerClick = (item) => {
  if (item.link) {
    uni.navigateTo({
      url: item.link
    });
  }
};

// 处理分类点击
const handleCategoryClick = (item) => {
  uni.navigateTo({
    url: `/pages/category/list?id=${item.id}&name=${item.name}`
  });
};

// 页面加载
onMounted(() => {
  fetchBanners();
  fetchCategories();
  fetchArticles();
});

// 下拉刷新
onPullDownRefresh(() => {
  fetchArticles(true);
});

// 上拉加载
onReachBottom(() => {
  loadMore();
});
</script>

跨平台适配 ​

在实战项目中，我们需要处理不同平台的差异，确保应用在各个平台上都能正常运行。以下是几个关键的适配点：

1. 条件编译 ​

使用条件编译处理平台差异：

vue
<!-- 平台特定代码 -->
<template>
  <view class="container">
    <!-- #ifdef MP-WEIXIN -->
    <button open-type="share">微信分享</button>
    <!-- #endif -->
    
    <!-- #ifdef APP-PLUS -->
    <button @click="shareToApp">APP分享</button>
    <!-- #endif -->
    
    <!-- #ifdef H5 -->
    <button @click="shareToWeb">网页分享</button>
    <!-- #endif -->
  </view>

<script setup>
// #ifdef APP-PLUS
const shareToApp = () => {
  // APP分享逻辑
  uni.share({
    provider: 'weixin',
    scene: 'WXSceneSession',
    type: 0,
    title: '分享标题',
    summary: '分享内容',
    imageUrl: '分享图片',
    href: '分享链接',
    success: function() {
      console.log('分享成功');
    },
    fail: function() {
      console.log('分享失败');
    }
  });
};
// #endif

// utils/platform.js

// 分享功能封装
export const shareContent = (options) => {
  // #ifdef MP-WEIXIN
  return new Promise((resolve, reject) => {
    // 微信小程序通过页面的onShareAppMessage处理
    resolve({
      title: options.title,
      path: options.path,
      imageUrl: options.imageUrl
    });
  });
  // #endif
  
  // #ifdef APP-PLUS
  return new Promise((resolve, reject) => {
    uni.share({
      provider: options.provider || 'weixin',
      scene: options.scene || 'WXSceneSession',
      type: options.type || 0,
      title: options.title,
      summary: options.summary,
      imageUrl: options.imageUrl,
      href: options.href,
      success: function() {
        resolve();
      },
      fail: function(err) {
        reject(err);
      }
    });
  });
  // #endif
  
  // #ifdef H5
  return new Promise((resolve, reject) => {
    if (navigator.share) {
      navigator.share({
        title: options.title,
        text: options.summary,
        url: options.href,
      }).then(resolve).catch(reject);
    } else {
      // 复制链接
      const input = document.createElement('input');
      input.setAttribute('value', options.href);
      document.body.appendChild(input);
      input.select();
      document.execCommand('copy');
      document.body.removeChild(input);
      uni.showToast({
        title: '链接已复制，请手动分享',
        icon: 'none'
      });
      resolve();
    }
  });
  // #endif
};

// 支付功能封装
export const doPay = (options) => {
  return new Promise((resolve, reject) => {
    // #ifdef MP-WEIXIN
    uni.requestPayment({
      timeStamp: options.timeStamp,
      nonceStr: options.nonceStr,
      package: options.package,
      signType: options.signType,
      paySign: options.paySign,
      success: function(res) {
        resolve(res);
      },
      fail: function(err) {
        reject(err);
      }
    });
    // #endif
    
    // #ifdef APP-PLUS
    uni.requestPayment({
      provider: 'wxpay', // 或者 alipay
      orderInfo: options.orderInfo, // 订单数据
      success: function(res) {
        resolve(res);
      },
      fail: function(err) {
        reject(err);
      }
    });
    // #endif
    
    // #ifdef H5
    if (options.payType === 'wxpay') {
      // 微信H5支付，通常是跳转到微信支付页面
      window.location.href = options.mwebUrl;
      resolve();
    } else if (options.payType === 'alipay') {
      // 支付宝H5支付，通常是表单提交
      const div = document.createElement('div');
      div.innerHTML = options.formData;
      document.body.appendChild(div);
      document.forms[0].submit();
      resolve();
    } else {
      reject(new Error('不支持的支付类型'));
    }
    // #endif
  });
};
