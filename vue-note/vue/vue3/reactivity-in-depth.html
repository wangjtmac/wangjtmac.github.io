响应式原理深入 ​

Vue 3 的响应式系统是基于 ES6 的 Proxy 实现的，相比 Vue 2 的 Object.defineProperty 有了显著的改进。本文将深入探讨 Vue 3 响应式系统的实现原理。

响应式系统概览 ​

核心概念 ​

  1. Proxy: 拦截对象操作
  2. Reflect: 反射操作
  3. Track: 依赖收集
  4. Trigger: 依赖触发
  5. Effect: 副作用函数

基本流程 ​

数据变化 → Proxy 拦截 → 触发更新 → 重新渲染
    ↑                              ↓
依赖收集 ← 访问数据 ← 副作用函数执行

Proxy vs Object.defineProperty ​

Vue 2 的限制 ​

javascript
// Vue 2 的响应式限制
const data = { count: 0 }

// ❌ 无法检测新增属性
data.newProp = 'new value'

// ❌ 无法检测数组索引变化
data.list[0] = 'new item'

// ❌ 无法检测数组长度变化
data.list.length = 0

Vue 3 的改进 ​

javascript
// Vue 3 使用 Proxy 解决了这些问题
import { reactive } from 'vue'

const data = reactive({ count: 0 })

// ✅ 可以检测新增属性
data.newProp = 'new value'

// ✅ 可以检测数组变化
data.list = []
data.list[0] = 'new item'
data.list.length = 0

响应式 API 实现原理 ​

reactive 实现 ​

javascript
// 简化版 reactive 实现
const reactiveMap = new WeakMap()

function reactive(target) {
  // 如果已经是响应式对象，直接返回
  if (reactiveMap.has(target)) {
    return reactiveMap.get(target)
  }
  
  // 创建 Proxy
  const proxy = new Proxy(target, {
    get(target, key, receiver) {
      // 依赖收集
      track(target, key)
      
      const result = Reflect.get(target, key, receiver)
      
      // 如果是对象，递归创建响应式
      if (isObject(result)) {
        return reactive(result)
      }
      
      return result
    },
    
    set(target, key, value, receiver) {
      const oldValue = target[key]
      const result = Reflect.set(target, key, value, receiver)
      
      // 如果值发生变化，触发更新
      if (oldValue !== value) {
        trigger(target, key)
      }
      
      return result
    },
    
    deleteProperty(target, key) {
      const hadKey = hasOwn(target, key)
      const result = Reflect.deleteProperty(target, key)
      
      if (result && hadKey) {
        trigger(target, key)
      }
      
      return result
    }
  })
  
  reactiveMap.set(target, proxy)
  return proxy
}

ref 实现 ​

javascript
// 简化版 ref 实现
class RefImpl {
  constructor(value) {
    this._value = convert(value)
  }
  
  get value() {
    // 依赖收集
    track(this, 'value')
    return this._value
  }
  
  set value(newValue) {
    if (newValue !== this._value) {
      this._value = convert(newValue)
      // 触发更新
      trigger(this, 'value')
    }
  }
}

function ref(value) {
  return new RefImpl(value)
}

function convert(value) {
  return isObject(value) ? reactive(value) : value
}

computed 实现 ​

javascript
// 简化版 computed 实现
class ComputedRefImpl {
  constructor(getter, setter) {
    this._getter = getter
    this._setter = setter
    this._dirty = true
    this._value = undefined
    
    // 创建 effect
    this.effect = new ReactiveEffect(getter, () => {
      if (!this._dirty) {
        this._dirty = true
        // 触发依赖此计算属性的更新
        trigger(this, 'value')
      }
    })
  }
  
  get value() {
    // 依赖收集
    track(this, 'value')
    
    if (this._dirty) {
      this._value = this.effect.run()
      this._dirty = false
    }
    
    return this._value
  }
  
  set value(newValue) {
    this._setter(newValue)
  }
}

function computed(getterOrOptions) {
  let getter, setter
  
  if (typeof getterOrOptions === 'function') {
    getter = getterOrOptions
    setter = () => console.warn('Computed property is readonly')
  } else {
    getter = getterOrOptions.get
    setter = getterOrOptions.set
  }
  
  return new ComputedRefImpl(getter, setter)
}

依赖收集和触发 ​

依赖收集 (Track) ​

javascript
// 当前活跃的 effect
let activeEffect = null

// 依赖映射表
const targetMap = new WeakMap()

function track(target, key) {
  if (!activeEffect) return
  
  let depsMap = targetMap.get(target)
  if (!depsMap) {
    targetMap.set(target, (depsMap = new Map()))
  }
  
  let dep = depsMap.get(key)
  if (!dep) {
    depsMap.set(key, (dep = new Set()))
  }
  
  // 建立双向依赖关系
  dep.add(activeEffect)
  activeEffect.deps.push(dep)
}

依赖触发 (Trigger) ​

javascript
function trigger(target, key) {
  const depsMap = targetMap.get(target)
  if (!depsMap) return
  
  const dep = depsMap.get(key)
  if (!dep) return
  
  // 执行所有相关的 effect
  dep.forEach(effect => {
    if (effect !== activeEffect) {
      if (effect.scheduler) {
        effect.scheduler()
      } else {
        effect.run()
      }
    }
  })
}

Effect 实现 ​

javascript
class ReactiveEffect {
  constructor(fn, scheduler = null) {
    this.fn = fn
    this.scheduler = scheduler
    this.deps = []
    this.active = true
  }
  
  run() {
    if (!this.active) {
      return this.fn()
    }
    
    try {
      // 清理之前的依赖
      this.cleanup()
      
      // 设置当前活跃的 effect
      activeEffect = this
      
      // 执行函数，触发依赖收集
      return this.fn()
    } finally {
      activeEffect = null
    }
  }
  
  cleanup() {
    this.deps.forEach(dep => {
      dep.delete(this)
    })
    this.deps.length = 0
  }
  
  stop() {
    if (this.active) {
      this.cleanup()
      this.active = false
    }
  }
}

function effect(fn, options = {}) {
  const _effect = new ReactiveEffect(fn, options.scheduler)
  
  // 立即执行一次
  _effect.run()
  
  // 返回 runner 函数
  const runner = _effect.run.bind(_effect)
  runner.effect = _effect
  
  return runner
}

响应式系统的优化 ​

1. 浅层响应式 ​

javascript
function shallowReactive(target) {
  return new Proxy(target, {
    get(target, key, receiver) {
      track(target, key)
      return Reflect.get(target, key, receiver)
    },
    
    set(target, key, value, receiver) {
      const result = Reflect.set(target, key, value, receiver)
      trigger(target, key)
      return result
    }
    
    // 不递归处理嵌套对象
  })
}

2. 只读响应式 ​

javascript
function readonly(target) {
  return new Proxy(target, {
    get(target, key, receiver) {
      // 只收集依赖，不允许修改
      track(target, key)
      
      const result = Reflect.get(target, key, receiver)
      return isObject(result) ? readonly(result) : result
    },
    
    set() {
      console.warn('Cannot set readonly property')
      return true
    },
    
    deleteProperty() {
      console.warn('Cannot delete readonly property')
      return true
    }
  })
}

3. 标记原始对象 ​

javascript
const ReactiveFlags = {
  IS_REACTIVE: '__v_isReactive',
  IS_READONLY: '__v_isReadonly',
  RAW: '__v_raw'
}

function markRaw(value) {
  Object.defineProperty(value, ReactiveFlags.SKIP, {
    value: true,
    configurable: true
  })
  return value
}

性能优化策略 ​

1. 批量更新 ​

javascript
// 调度器实现批量更新
const queue = []
let isFlushing = false

function queueJob(job) {
  if (!queue.includes(job)) {
    queue.push(job)
  }
  
  if (!isFlushing) {
    isFlushing = true
    Promise.resolve().then(flushJobs)
  }
}

function flushJobs() {
  try {
    queue.forEach(job => job())
  } finally {
    queue.length = 0
    isFlushing = false
  }
}

2. 停止追踪 ​

javascript
function stop(runner) {
  runner.effect.stop()
}

// 使用示例
const runner = effect(() => {
  console.log(state.count)
})

// 停止响应式追踪
stop(runner)

3. 暂停/恢复追踪 ​

javascript
let shouldTrack = true

function pauseTracking() {
  shouldTrack = false
}

function resumeTracking() {
  shouldTrack = true
}

function track(target, key) {
  if (!activeEffect || !shouldTrack) return

function getDependencies(target) {
  const depsMap = targetMap.get(target)
  if (!depsMap) return {}
  
  const result = {}
  depsMap.forEach((dep, key) => {
    result[key] = Array.from(dep).map(effect => effect.fn.toString())
  })
  
  return result
}

最佳实践 ​

1. 合理选择响应式 API ​

javascript
// ✅ 对象使用 reactive
const state = reactive({
  user: { name: 'John' },
  list: []
})

// ✅ 基本类型使用 ref
const count = ref(0)
const message = ref('Hello')

// ✅ 大对象使用 shallowReactive
const largeObject = shallowReactive({
  data: new Array(10000).fill(0)
})

2. 避免不必要的响应式 ​

javascript
// ❌ 不需要响应式的数据
const config = reactive({
  apiUrl: 'https://api.example.com',
  version: '1.0.0'
})

// ✅ 使用 markRaw 或普通对象
const config = markRaw({
  apiUrl: 'https://api.example.com',
  version: '1.0.0'
})

3. 正确使用 toRefs ​

javascript
// ✅ 解构响应式对象
const state = reactive({ count: 0, name: 'Vue' })
const { count, name } = toRefs(state)

// 现在 count 和 name 都是 ref
console.log(count.value) // 0
count.value++ // 触发更新

总结 ​

Vue 3 的响应式系统特点：

  1. Proxy 基础: 解决了 Vue 2 的诸多限制
  2. 精确追踪: 只追踪实际访问的属性
  3. 性能优化: 批量更新、懒计算等优化策略
  4. 类型安全: 更好的 TypeScript 支持
  5. 调试友好: 提供丰富的调试工具

理解响应式原理有助于：

  • 写出更高效的代码
  • 避免常见的响应式陷阱
  • 更好地调试应用问题
  • 优化应用性能
