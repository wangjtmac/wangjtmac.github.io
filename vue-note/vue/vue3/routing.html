路由 ​

Vue Router 是 Vue.js 的官方路由管理器，为单页应用提供了强大的路由功能。本文将介绍 Vue Router 4 在 Vue 3 中的使用。

基础路由 ​

安装和设置 ​

bash
npm install vue-router@4
javascript
// router/index.js
import { createRouter, createWebHistory } from 'vue-router'
import Home from '../views/Home.vue'
import About from '../views/About.vue'

const routes = [
  {
    path: '/',
    name: 'Home',
    component: Home
  },
  {
    path: '/about',
    name: 'About',
    component: About
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

export default router
javascript
// main.js
import { createApp } from 'vue'
import App from './App.vue'
import router from './router'

const app = createApp(App)
app.use(router)
app.mount('#app')

动态路由 ​

路径参数 ​

javascript
// router/index.js
const routes = [
  {
    path: '/user/:id',
    name: 'User',
    component: () => import('../views/User.vue')
  },
  {
    path: '/user/:id/post/:postId',
    name: 'UserPost',
    component: () => import('../views/UserPost.vue')
  }
]
vue
<!-- User.vue -->
<template>
  <div>
    <h2>用户 {{ $route.params.id }}</h2>
    <p>用户信息...</p>
    
    <router-link :to="`/user/${$route.params.id}/post/1`">
      查看文章
    </router-link>
  </div>
</template>

<script setup>
import { useRoute, useRouter } from 'vue-router'
import { watch } from 'vue'

const route = useRoute()
const router = useRouter()

// 监听路由参数变化
watch(
  () => route.params.id,
  (newId, oldId) => {
    console.log(`用户ID从 ${oldId} 变为 ${newId}`)
    // 重新获取用户数据
    fetchUser(newId)
  }
)

function fetchUser(id) {
  console.log(`获取用户 ${id} 的数据`)
}

// 编程式导航
function goToProfile() {
  router.push(`/user/${route.params.id}/profile`)
}
</script>

查询参数 ​

vue
<template>
  <div>
    <h2>搜索结果</h2>
    <p>关键词: {{ $route.query.q }}</p>
    <p>页码: {{ $route.query.page || 1 }}</p>
    
    <div>
      <input v-model="searchQuery" @keyup.enter="search" />
      <button @click="search">搜索</button>
    </div>
    
    <div>
      <button @click="prevPage" :disabled="currentPage <= 1">上一页</button>
      <span>第 {{ currentPage }} 页</span>
      <button @click="nextPage">下一页</button>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'
import { useRoute, useRouter } from 'vue-router'

const route = useRoute()
const router = useRouter()

const searchQuery = ref(route.query.q || '')

const currentPage = computed(() => 
  parseInt(route.query.page) || 1
)

function search() {
  router.push({
    path: '/search',
    query: {
      q: searchQuery.value,
      page: 1
    }
  })
}

function prevPage() {
  router.push({
    query: {
      ...route.query,
      page: currentPage.value - 1
    }
  })
}

function nextPage() {
  router.push({
    query: {
      ...route.query,
      page: currentPage.value + 1
    }
  })
}
</script>

路由守卫 ​

全局守卫 ​

javascript
// router/index.js
const router = createRouter({
  history: createWebHistory(),
  routes
})

// 全局前置守卫
router.beforeEach((to, from, next) => {
  console.log('导航到:', to.path)
  
  // 检查认证
  if (to.meta.requiresAuth && !isAuthenticated()) {
    next('/login')
  } else {
    next()
  }
})

// 全局解析守卫
router.beforeResolve((to, from, next) => {
  console.log('路由解析完成')
  next()
})

// 全局后置钩子
router.afterEach((to, from) => {
  console.log('导航完成')
  // 更新页面标题
  document.title = to.meta.title || 'Vue App'
})

function isAuthenticated() {
  return !!localStorage.getItem('token')
}

组件内守卫 ​

vue
<template>
  <div>
    <h2>编辑文章</h2>
    <form>
      <textarea v-model="content" placeholder="文章内容"></textarea>
      <button @click="save">保存</button>
    </form>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { onBeforeRouteLeave, onBeforeRouteUpdate } from 'vue-router'

const content = ref('')
const hasUnsavedChanges = ref(false)

// 监听内容变化
watch(content, () => {
  hasUnsavedChanges.value = true
})

// 路由离开守卫
onBeforeRouteLeave((to, from, next) => {
  if (hasUnsavedChanges.value) {
    const answer = window.confirm('有未保存的更改，确定要离开吗？')
    if (answer) {
      next()
    } else {
      next(false)
    }
  } else {
    next()
  }
})

// 路由更新守卫
onBeforeRouteUpdate((to, from, next) => {
  console.log('路由参数更新')
  next()
})

function save() {
  // 保存逻辑
  hasUnsavedChanges.value = false
}
</script>

路由过渡 ​

基础过渡 ​

vue
<template>
  <div>
    <router-view v-slot="{ Component }">
      <transition name="fade" mode="out-in">
        <component :is="Component" />
      </transition>
    </router-view>
  </div>
</template>

<style>
.fade-enter-active,
.fade-leave-active {
  transition: opacity 0.3s ease;
}

.fade-enter-from,
.fade-leave-to {
  opacity: 0;
}
</style>

动态过渡 ​

vue
<template>
  <div>
    <router-view v-slot="{ Component, route }">
      <transition :name="getTransitionName(route)" mode="out-in">
        <component :is="Component" :key="route.path" />
      </transition>
    </router-view>
  </div>
</template>

<script setup>
function getTransitionName(route) {
  // 根据路由深度决定过渡效果
  const depth = route.path.split('/').length
  return depth > 2 ? 'slide-left' : 'slide-right'
}
</script>

<style>
.slide-left-enter-active,
.slide-left-leave-active,
.slide-right-enter-active,
.slide-right-leave-active {
  transition: transform 0.3s ease;
}

.slide-left-enter-from {
  transform: translateX(100%);
}

.slide-left-leave-to {
  transform: translateX(-100%);
}

.slide-right-enter-from {
  transform: translateX(-100%);
}

.slide-right-leave-to {
  transform: translateX(100%);
}
</style>

滚动行为 ​

javascript
const router = createRouter({
  history: createWebHistory(),
  routes,
  scrollBehavior(to, from, savedPosition) {
    // 如果有保存的位置（浏览器前进/后退）
    if (savedPosition) {
      return savedPosition
    }
    
    // 如果有锚点
    if (to.hash) {
      return {
        el: to.hash,
        behavior: 'smooth'
      }
    }
    
    // 默认滚动到顶部
    return { top: 0 }
  }
})

最佳实践 ​

  1. 路由结构: 保持路由结构清晰和一致
  2. 懒加载: 使用路由懒加载优化性能
  3. 守卫使用: 合理使用路由守卫进行权限控制
  4. 元信息: 充分利用路由元信息
  5. 错误处理: 提供完善的错误处理机制
  6. 用户体验: 使用过渡动画提升用户体验

总结 ​

Vue Router 4 提供了：

  1. 强大的路由功能: 支持各种路由模式和配置
  2. 组合式 API: 更好的 TypeScript 支持
  3. 性能优化: 懒加载和代码分割
  4. 灵活的守卫系统: 完善的权限控制
  5. 良好的用户体验: 过渡动画和滚动控制

掌握 Vue Router 是构建单页应用的关键技能。
