自定义指令 ​

Vue 3 允许你创建自定义指令来封装对 DOM 的直接操作。自定义指令提供了一种重用涉及普通元素的底层 DOM 访问的逻辑的方法。

指令基础 ​

指令钩子函数 ​

Vue 3 中的指令钩子函数：

javascript
const myDirective = {
  // 在绑定元素的 attribute 前
  // 或事件监听器应用前调用
  created(el, binding, vnode, prevVnode) {
    // 下面会介绍各个参数的细节
  },
  
  // 在元素被插入到 DOM 前调用
  beforeMount(el, binding, vnode, prevVnode) {},
  
  // 在绑定元素的父组件
  // 及他自己的所有子节点都挂载完成后调用
  mounted(el, binding, vnode, prevVnode) {},
  
  // 绑定元素的父组件更新前调用
  beforeUpdate(el, binding, vnode, prevVnode) {},
  
  // 在绑定元素的父组件
  // 及他自己的所有子节点都更新后调用
  updated(el, binding, vnode, prevVnode) {},
  
  // 绑定元素的父组件卸载前调用
  beforeUnmount(el, binding, vnode, prevVnode) {},
  
  // 绑定元素的父组件卸载后调用
  unmounted(el, binding, vnode, prevVnode) {}
}

钩子参数 ​

  • el：指令绑定到的元素
  • binding：包含指令信息的对象
  • vnode：代表绑定元素的底层 VNode
  • prevVnode：代表之前的渲染中指令所绑定元素的 VNode

全局注册指令 ​

javascript
// main.js
import { createApp } from 'vue'

const app = createApp({})

// 注册全局自定义指令 v-focus
app.directive('focus', {
  mounted(el) {
    el.focus()
  }
})

局部注册指令 ​

选项式 API ​

vue
<script>
export default {
  directives: {
    focus: {
      mounted(el) {
        el.focus()
      }
    }
  }
}
</script>

组合式 API ​

vue
<script setup>
// 在 <script setup> 中，任何以 v 开头的驼峰式命名的变量
// 都可以被用作一个自定义指令
const vFocus = {
  mounted: (el) => el.focus()
}
</script>

<template>
  <input v-focus />
</template>

实用指令示例 ​

1. 防抖指令 ​

javascript
const debounce = {
  mounted(el, binding) {
    let timer
    el.addEventListener('click', () => {
      if (timer) {
        clearTimeout(timer)
      }
      timer = setTimeout(() => {
        binding.value()
      }, binding.arg || 1000)
    })
  }
}

// 使用
app.directive('debounce', debounce)
vue
<template>
  <!-- 1000ms 防抖 -->
  <button v-debounce="handleClick">点击我</button>
  
  <!-- 500ms 防抖 -->
  <button v-debounce:500="handleClick">点击我</button>
</template>

2. 权限控制指令 ​

javascript
const permission = {
  mounted(el, binding) {
    const { value } = binding
    const userPermissions = getUserPermissions() // 获取用户权限
    
    if (!userPermissions.includes(value)) {
      el.style.display = 'none'
      // 或者移除元素
      // el.parentNode?.removeChild(el)
    }
  },
  
  updated(el, binding) {
    const { value } = binding
    const userPermissions = getUserPermissions()
    
    if (!userPermissions.includes(value)) {
      el.style.display = 'none'
    } else {
      el.style.display = ''
    }
  }
}

app.directive('permission', permission)
vue
<template>
  <button v-permission="'admin'">管理员按钮</button>
  <div v-permission="'user'">用户内容</div>
</template>

4. 点击外部指令 ​

javascript
const clickOutside = {
  mounted(el, binding) {
    el._clickOutsideHandler = (event) => {
      if (!(el === event.target || el.contains(event.target))) {
        binding.value(event)
      }
    }
    document.addEventListener('click', el._clickOutsideHandler)
  },
  
  unmounted(el) {
    document.removeEventListener('click', el._clickOutsideHandler)
    delete el._clickOutsideHandler
  }
}

app.directive('click-outside', clickOutside)
vue
<template>
  <div v-click-outside="closeModal" class="modal">
    <p>点击外部关闭模态框</p>
  </div>
</template>

<script setup>
const closeModal = () => {
  console.log('点击了外部区域')
}
</script>

5. 复制到剪贴板指令 ​

javascript
const copy = {
  mounted(el, binding) {
    el._copyHandler = async () => {
      try {
        await navigator.clipboard.writeText(binding.value)
        console.log('复制成功')
        
        // 可以添加视觉反馈
        el.classList.add('copied')
        setTimeout(() => {
          el.classList.remove('copied')
        }, 1000)
      } catch (err) {
        console.error('复制失败:', err)
      }
    }
    
    el.addEventListener('click', el._copyHandler)
  },
  
  updated(el, binding) {
    // 更新要复制的内容
    el._copyValue = binding.value
  },
  
  unmounted(el) {
    el.removeEventListener('click', el._copyHandler)
    delete el._copyHandler
  }
}

app.directive('copy', copy)
vue
<template>
  <button v-copy="textToCopy">复制文本</button>
</template>

<style>
.copied {
  background-color: #4caf50 !important;
  color: white !important;
}
</style>

指令参数和修饰符 ​

动态参数 ​

vue
<template>
  <div v-pin:[direction]="200">固定定位</div>
</template>

<script setup>
import { ref } from 'vue'

const direction = ref('top')

const vPin = {
  mounted(el, binding) {
    el.style.position = 'fixed'
    const s = binding.arg || 'top'
    el.style[s] = binding.value + 'px'
  },
  
  updated(el, binding) {
    const s = binding.arg || 'top'
    el.style[s] = binding.value + 'px'
  }
}
</script>

修饰符 ​

javascript
const highlight = {
  mounted(el, binding) {
    const { modifiers, value } = binding
    
    if (modifiers.bold) {
      el.style.fontWeight = 'bold'
    }
    
    if (modifiers.italic) {
      el.style.fontStyle = 'italic'
    }
    
    if (modifiers.underline) {
      el.style.textDecoration = 'underline'
    }
    
    el.style.color = value || 'red'
  }
}

app.directive('highlight', highlight)
vue
<template>
  <p v-highlight.bold.italic="'blue'">蓝色粗体斜体文本</p>
  <p v-highlight.underline="'green'">绿色下划线文本</p>
</template>

指令的简化形式 ​

如果指令只需要在 mounted 和 updated 时触发相同的行为，可以简化：

javascript
// 简化形式
app.directive('color', (el, binding) => {
  el.style.color = binding.value
})

// 等同于
app.directive('color', {
  mounted(el, binding) {
    el.style.color = binding.value
  },
  updated(el, binding) {
    el.style.color = binding.value
  }
})

TypeScript 支持 ​

类型定义 ​

typescript
import type { Directive } from 'vue'

interface DebounceBinding {
  value: () => void
  arg?: string
}

const debounce: Directive<HTMLElement, () => void> = {
  mounted(el, binding) {
    let timer: number
    el.addEventListener('click', () => {
      if (timer) {
        clearTimeout(timer)
      }
      timer = setTimeout(() => {
        binding.value()
      }, parseInt(binding.arg || '1000'))
    })
  }
}

最佳实践 ​

1. 命名约定 ​

javascript
// ✅ 好的命名
app.directive('focus', focusDirective)
app.directive('click-outside', clickOutsideDirective)
app.directive('lazy-load', lazyLoadDirective)

// ❌ 避免的命名
app.directive('f', focusDirective) // 太简短
app.directive('myDirective', myDirective) // 不够具体

2. 清理资源 ​

javascript
const myDirective = {
  mounted(el, binding) {
    // 添加事件监听器
    el._handler = () => { /* ... */ }
    el.addEventListener('click', el._handler)
  },
  
  unmounted(el) {
    // 清理事件监听器
    if (el._handler) {
      el.removeEventListener('click', el._handler)
      delete el._handler
    }
  }
}

3. 错误处理 ​

javascript
const safeDirective = {
  mounted(el, binding) {
    try {
      // 指令逻辑
      if (typeof binding.value !== 'function') {
        console.warn('指令期望一个函数')
        return
      }
      
      // 执行指令逻辑
    } catch (error) {
      console.error('指令执行错误:', error)
    }
  }
}

总结 ​

自定义指令是 Vue 3 中强大的功能：

  1. 封装 DOM 操作：将复杂的 DOM 操作封装成可复用的指令
  2. 生命周期钩子：提供完整的生命周期管理
  3. 参数和修饰符：支持动态参数和修饰符
  4. 全局和局部注册：灵活的注册方式
  5. TypeScript 支持：完整的类型支持

合理使用自定义指令可以提高代码的复用性和可维护性。
