Vant 源码解析 ​

Vant 是有赞前端团队开源的轻量、可靠的移动端 Vue 组件库。本文将深入分析 Vant 的源码结构和核心实现。

源码结构 ​

Vant 采用 monorepo 架构：

packages/
├── vant/           # 主包
├── vant-cli/       # 构建工具
├── vant-icons/     # 图标库
├── vant-use/       # 组合式函数
└── vant-compat/    # 兼容层

src/
├── action-sheet/   # 动作面板
├── button/         # 按钮
├── cell/           # 单元格
├── dialog/         # 弹窗
├── field/          # 输入框
├── list/           # 列表
├── popup/          # 弹出层
├── toast/          # 轻提示
├── utils/          # 工具函数
└── style/          # 样式文件

<template>
  <button
    :type="nativeType"
    :class="classes"
    :style="style"
    :disabled="disabled"
    @click="onClick"
  >
    <div class="van-button__content">
      <van-loading
        v-if="loading"
        :size="loadingSize"
        :type="loadingType"
        :color="loadingColor"
        class="van-button__loading"
      />
      <van-icon
        v-if="icon && !loading"
        :name="icon"
        :class-prefix="iconPrefix"
        class="van-button__icon"
      />
      <span v-if="$slots.default" class="van-button__text">
        <slot />
      </span>
    </div>
  </button>
</template>

<script>
import { computed, defineComponent } from 'vue'
import { createNamespace, addUnit, BORDER_SURROUND } from '../utils'
import { Loading } from '../loading'
import { Icon } from '../icon'

const [name, bem] = createNamespace('button')

export default defineComponent({
  name,

  components: {
    VanLoading: Loading,
    VanIcon: Icon,
  },

  props: {
    tag: {
      type: String,
      default: 'button',
    },
    text: String,
    icon: String,
    type: {
      type: String,
      default: 'default',
    },
    size: {
      type: String,
      default: 'normal',
    },
    color: String,
    block: Boolean,
    plain: Boolean,
    round: Boolean,
    square: Boolean,
    loading: Boolean,
    hairline: Boolean,
    disabled: Boolean,
    iconPrefix: String,
    nativeType: {
      type: String,
      default: 'button',
    },
    loadingSize: String,
    loadingType: String,
    loadingText: String,
  },

  emits: ['click'],

  setup(props, { emit, slots }) {
    const style = computed(() => {
      const { color, plain } = props
      if (color) {
        const style = {
          color: plain ? color : 'white',
          background: plain ? 'transparent' : color,
        }

        if (!plain) {
          style.borderColor = color
        }

        return style
      }
    })

    const classes = computed(() => {
      const { type, size, block, round, plain, square, loading, disabled, hairline } = props

      return [
        bem([
          type,
          size,
          {
            block,
            round,
            plain,
            square,
            loading,
            disabled,
            hairline,
          },
        ]),
        { [BORDER_SURROUND]: hairline },
      ]
    })

    const loadingColor = computed(() => {
      if (props.plain) {
        return props.color
      }
      if (props.type === 'default') {
        return '#c9c9c9'
      }
    })

    const onClick = (event) => {
      if (!props.loading && !props.disabled) {
        emit('click', event)
      }
    }

    return {
      bem,
      style,
      classes,
      onClick,
      loadingColor,
    }
  },
})
</script>

createNamespace ​

typescript
// src/utils/create.ts
export type BEM = (
  element?: string,
  modifier?: string | Record<string, any>
) => string

export function createBEM(name: string): BEM {
  return (element, modifier) => {
    let cls = name

    if (element) {
      cls += `__${element}`
    }

    if (modifier) {
      if (typeof modifier === 'string') {
        cls += `--${modifier}`
      } else {
        Object.keys(modifier).forEach((key) => {
          if (modifier[key]) {
            cls += `--${key}`
          }
        })
      }
    }

    return cls
  }
}

export function createNamespace(name: string) {
  const prefixedName = `van-${name}`
  return [prefixedName, createBEM(prefixedName)] as const
}

单位处理 ​

// src/utils/format.ts
import { isDef, isNumeric } from './validate'

export function addUnit(value?: string | number): string | undefined {
  if (isDef(value)) {
    return isNumeric(value) ? `${value}px` : String(value)
  }
  return undefined
}

export function getSizeStyle(
  originSize?: string | number
): {
  width: string
  height: string
} | undefined {
  if (isDef(originSize)) {
    const size = addUnit(originSize)
    return {
      width: size!,
      height: size!,
    }
  }
}

export function getZIndexStyle(zIndex?: string | number) {
  const style: Record<string, any> = {}
  if (zIndex !== undefined) {
    style.zIndex = +zIndex
  }
  return style
}

<template>
  <van-overlay
    v-if="overlay"
    :show="show"
    :class="overlayClass"
    :style="overlayStyle"
    :duration="duration"
    :custom-style="overlayStyle"

    @after-leave="onClosed"
  >
    <div

      :class="classes"
      :style="style"

    >
      <slot />
    </div>
  </van-transition>
</template>

<script>
import { computed, watch, defineComponent } from 'vue'
import {
  isDef,
  addUnit,
  isObject,
  numericProp,
  unknownProp,
  makeStringProp,
  createNamespace,
  HAPTICS_FEEDBACK,
} from '../utils'
import { useLockScroll } from '../composables/use-lock-scroll'
import { useLazyRender } from '../composables/use-lazy-render'
import { Overlay } from '../overlay'
import { Transition } from '../transition'

const [name, bem] = createNamespace('popup')

const popupSharedProps = {
  show: Boolean,
  zIndex: numericProp,
  overlay: {
    type: Boolean,
    default: true,
  },
  duration: numericProp,
  teleport: [String, Object],
  lockScroll: {
    type: Boolean,
    default: true,
  },
  lazyRender: {
    type: Boolean,
    default: true,
  },
  beforeClose: Function,
  overlayClass: unknownProp,
  overlayStyle: Object,
  destroyOnClose: Boolean,
  safeAreaInsetTop: Boolean,
  safeAreaInsetBottom: Boolean,
}

const popupSharedPropKeys = Object.keys(popupSharedProps)

export { popupSharedProps, popupSharedPropKeys }

export default defineComponent({
  name,

  inheritAttrs: false,

  props: {
    ...popupSharedProps,
    round: Boolean,
    position: makeStringProp('center'),
    closeIcon: makeStringProp('cross'),
    closeable: Boolean,
    transition: String,
    iconPrefix: String,
    closeOnPopstate: Boolean,
    closeIconPosition: makeStringProp('top-right'),
    safeAreaInsetTop: Boolean,
    safeAreaInsetBottom: Boolean,
  },

  emits: [
    'open',
    'close',
    'opened',
    'closed',
    'keydown',
    'update:show',
    'clickOverlay',
    'clickCloseIcon',
  ],

  setup(props, { emit, attrs, slots }) {
    let opened = false
    let shouldReopen = false
    const zIndex = ref()
    const popupRef = ref()

    const lazyRender = useLazyRender(() => props.show || !props.lazyRender)

    const style = computed(() => {
      const style = {
        zIndex: zIndex.value,
      }

      if (isDef(props.duration)) {
        const key = props.position === 'center' ? 'animationDuration' : 'transitionDuration'
        style[key] = `${props.duration}s`
      }

      return style
    })

    const open = () => {
      if (!opened) {
        opened = true

        if (props.zIndex !== undefined) {
          zIndex.value = +props.zIndex
        }

        emit('open')
      }
    }

    const close = () => {
      if (opened) {
        callInterceptor(props.beforeClose, {
          args: [action],
          done() {
            opened = false
            emit('close')
            emit('update:show', false)
          },
        })
      }
    }

    const onClickOverlay = (event) => {
      emit('clickOverlay', event)

      if (props.closeOnClickOverlay) {
        close()
      }
    }

    const renderOverlay = () => {
      if (props.overlay) {
        return (
          <Overlay
            v-slots={{ default: slots['overlay-content'] }}
            show={props.show}
            class={props.overlayClass}
            zIndex={zIndex.value}
            duration={props.duration}
            customStyle={props.overlayStyle}
            role={props.closeOnClickOverlay ? 'button' : undefined}
            tabindex={props.closeOnClickOverlay ? 0 : undefined}
            onClick={onClickOverlay}
          />
        )
      }
    }

    const onClickCloseIcon = (event) => {
      emit('clickCloseIcon', event)
      close()
    }

    const renderCloseIcon = () => {
      if (props.closeable) {
        return (
          <Icon
            role="button"
            tabindex={0}
            name={props.closeIcon}
            class={[
              bem('close-icon', props.closeIconPosition),
              HAPTICS_FEEDBACK,
            ]}
            classPrefix={props.iconPrefix}
            onClick={onClickCloseIcon}
          />
        )
      }
    }

    let timer
    const onOpened = () => {
      if (timer) clearTimeout(timer)
      timer = setTimeout(() => {
        emit('opened')
      })
    }

    const onClosed = () => emit('closed')

    const onKeydown = (event) => emit('keydown', event)

    const renderPopup = lazyRender(() => {
      const { round, position, safeAreaInsetTop, safeAreaInsetBottom } = props

      return withDirectives(
        <div
          ref={popupRef}
          class={[
            bem({
              round,
              [position]: position !== 'center',
            }),
            {
              'van-safe-area-top': safeAreaInsetTop,
              'van-safe-area-bottom': safeAreaInsetBottom,
            },
          ]}
          style={style.value}
          onKeydown={onKeydown}
          {...attrs}
        >
          {slots.default?.()}
          {renderCloseIcon()}
        </div>,
        [[vShow, props.show]]
      )
    })

    const renderTransition = () => {
      const { position, transition, transitionAppear } = props
      const name = position === 'center' ? 'van-fade' : `van-popup-slide-${position}`

      return (
        <Transition
          v-slots={{ default: renderPopup }}
          name={transition || name}
          appear={transitionAppear}
          customStyle={style.value}
          onAfterEnter={onOpened}
          onAfterLeave={onClosed}
        />
      )
    }

    watch(
      () => props.show,
      (show) => {
        if (show && !opened) {
          open()

          if (attrs.tabindex === 0) {
            nextTick(() => {
              popupRef.value?.focus()
            })
          }
        }
        if (!show && opened) {
          opened = false
          emit('close')
        }
      }
    )

    useLockScroll(popupRef, () => props.show && props.lockScroll)

    useEventListener('popstate', () => {
      if (props.closeOnPopstate) {
        close()
        shouldReopen = false
      }
    })

    onMounted(() => {
      if (props.show) {
        open()
      }
    })

    onActivated(() => {
      if (shouldReopen) {
        emit('update:show', true)
        shouldReopen = false
      }
    })

    onDeactivated(() => {
      if (props.show && props.teleport) {
        close()
        shouldReopen = true
      }
    })

    provide(POPUP_TOGGLE_KEY, () => props.show)

    return () => {
      if (props.teleport) {
        return (
          <Teleport to={props.teleport}>
            {renderOverlay()}
            {renderTransition()}
          </Teleport>
        )
      }

      return (
        <>
          {renderOverlay()}
          {renderTransition()}
        </>
      )
    }
  },
})
</script>

useLockScroll ​

typescript
// src/composables/use-lock-scroll.ts
import { watch, onUnmounted } from 'vue'
import { getScrollTop, setScrollTop } from '../utils'

let totalLockCount = 0

const BODY_LOCK_CLASS = 'van-overflow-hidden'

export function useLockScroll(
  rootRef: Ref<HTMLElement | undefined>,
  shouldLock: () => boolean
) {
  const touch = useTouch()

  const onTouchMove = (event: TouchEvent) => {
    touch.move(event)

    const direction = touch.deltaY.value > 0 ? '10' : '01'
    const el = getScrollParent(
      event.target as Element,
      rootRef.value
    ) as HTMLElement
    const { scrollHeight, offsetHeight, scrollTop } = el
    let status = '11'

    if (scrollTop === 0) {
      status = offsetHeight >= scrollHeight ? '00' : '01'
    } else if (scrollTop + offsetHeight >= scrollHeight) {
      status = '10'
    }

    if (
      status !== '11' &&
      touch.isVertical() &&
      !(parseInt(status, 2) & parseInt(direction, 2))
    ) {
      preventDefault(event, true)
    }
  }

  const lock = () => {
    document.addEventListener('touchstart', touch.start)
    document.addEventListener('touchmove', onTouchMove, { passive: false })

    if (!totalLockCount) {
      document.body.classList.add(BODY_LOCK_CLASS)
    }

    totalLockCount++
  }

  const unlock = () => {
    if (totalLockCount) {
      document.removeEventListener('touchstart', touch.start)
      document.removeEventListener('touchmove', onTouchMove)

      totalLockCount--

      if (!totalLockCount) {
        document.body.classList.remove(BODY_LOCK_CLASS)
      }
    }
  }

  const init = () => shouldLock() && lock()

  const destroy = () => shouldLock() && unlock()

  onMounted(init)
  onActivated(init)
  onDeactivated(destroy)
  onUnmounted(destroy)

  watch(shouldLock, (value) => {
    value ? lock() : unlock()
  })
}

useTouch ​

typescript
// src/composables/use-touch.ts
import { ref } from 'vue'

const MIN_DISTANCE = 10

function getDirection(x: number, y: number) {
  if (x > y) {
    return 'horizontal'
  }
  if (y > x) {
    return 'vertical'
  }
  return ''
}

export function useTouch() {
  const startX = ref(0)
  const startY = ref(0)
  const deltaX = ref(0)
  const deltaY = ref(0)
  const offsetX = ref(0)
  const offsetY = ref(0)
  const direction = ref('')

  const isVertical = () => direction.value === 'vertical'
  const isHorizontal = () => direction.value === 'horizontal'

  const reset = () => {
    deltaX.value = 0
    deltaY.value = 0
    offsetX.value = 0
    offsetY.value = 0
    direction.value = ''
  }

  const start = ((event: TouchEvent) => {
    reset()
    startX.value = event.touches[0].clientX
    startY.value = event.touches[0].clientY
  }) as EventListener

  const move = ((event: TouchEvent) => {
    const touch = event.touches[0]
    deltaX.value = (touch.clientX < 0 ? 0 : touch.clientX) - startX.value
    deltaY.value = touch.clientY - startY.value
    offsetX.value = Math.abs(deltaX.value)
    offsetY.value = Math.abs(deltaY.value)

    const LOCK_DIRECTION_DISTANCE = 10
    if (
      !direction.value ||
      (offsetX.value < LOCK_DIRECTION_DISTANCE &&
        offsetY.value < LOCK_DIRECTION_DISTANCE)
    ) {
      direction.value = getDirection(offsetX.value, offsetY.value)
    }
  }) as EventListener

  return {
    move,
    start,
    reset,
    startX,
    startY,
    deltaX,
    deltaY,
    offsetX,
    offsetY,
    direction,
    isVertical,
    isHorizontal,
  }
}

总结 ​

Vant 的源码特点：

  1. 移动端优化: 专为移动端设计的组件库
  2. TypeScript 支持: 完整的 TypeScript 类型定义
  3. 组合式函数: 丰富的 Hooks 系统
  4. 主题定制: 基于 CSS 变量的主题系统
  5. 构建工具: 专门的 CLI 工具链

Vant 为移动端 Vue 应用提供了完整的组件库解决方案，其设计理念值得学习。
