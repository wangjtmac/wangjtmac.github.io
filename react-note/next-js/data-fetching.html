数据获取 ​

Next.js 提供了多种数据获取方法，适用于不同的渲染策略和使用场景。

Pages Router 数据获取 ​

getStaticProps (静态生成) ​

在构建时获取数据，适用于数据不经常变化的页面。

jsx
// pages/posts/index.js
export default function Posts({ posts }) {
  return (
    <div>
      <h1>文章列表</h1>
      {posts.map(post => (
        <div key={post.id}>
          <h2>{post.title}</h2>
          <p>{post.excerpt}</p>
        </div>
      ))}
    </div>
  )
}

export async function getStaticProps() {
  // 在构建时运行
  const res = await fetch('https://api.example.com/posts')
  const posts = await res.json()

  return {
    props: {
      posts,
    },
    // 可选：启用 ISR
    revalidate: 60, // 60 秒后重新生成
  }
}

getServerSideProps (服务器端渲染) ​

在每次请求时获取数据，适用于需要实时数据的页面。

jsx
// pages/dashboard.js
export default function Dashboard({ user, stats }) {
  return (
    <div>
      <h1>欢迎, {user.name}</h1>
      <div>
        <p>今日访问: {stats.todayVisits}</p>
        <p>总访问: {stats.totalVisits}</p>
      </div>
    </div>
  )
}

export async function getServerSideProps(context) {
  // 在每次请求时运行
  const { req, res, query } = context
  
  // 获取用户信息
  const userRes = await fetch(`https://api.example.com/user/${query.id}`, {
    headers: {
      Cookie: req.headers.cookie,
    },
  })
  const user = await userRes.json()

  // 获取统计数据
  const statsRes = await fetch('https://api.example.com/stats')
  const stats = await statsRes.json()

  return {
    props: {
      user,
      stats,
    },
  }
}

getStaticPaths (动态路由预生成) ​

为动态路由指定要预渲染的路径。

jsx
// pages/posts/[id].js
export default function Post({ post }) {
  return (
    <div>
      <h1>{post.title}</h1>
      <p>{post.content}</p>
    </div>
  )
}

export async function getStaticPaths() {
  // 获取所有文章 ID
  const res = await fetch('https://api.example.com/posts')
  const posts = await res.json()

  // 生成路径
  const paths = posts.map(post => ({
    params: { id: post.id.toString() }
  }))

  return {
    paths,
    fallback: false, // 或 true、'blocking'
  }
}

export async function getStaticProps({ params }) {
  // 获取特定文章
  const res = await fetch(`https://api.example.com/posts/${params.id}`)
  const post = await res.json()

  return {
    props: {
      post,
    },
  }
}

App Router 数据获取 ​

服务器组件 ​

在 App Router 中，组件默认是服务器组件，可以直接使用 async/await。

jsx
// app/posts/page.js
async function getPosts() {
  const res = await fetch('https://api.example.com/posts', {
    next: { revalidate: 60 } // ISR
  })
  
  if (!res.ok) {
    throw new Error('Failed to fetch posts')
  }
  
  return res.json()
}

export default async function PostsPage() {
  const posts = await getPosts()

  return (
    <div>
      <h1>文章列表</h1>
      {posts.map(post => (
        <div key={post.id}>
          <h2>{post.title}</h2>
          <p>{post.excerpt}</p>
        </div>
      ))}
    </div>
  )
}

并行数据获取 ​

jsx
// app/dashboard/page.js
async function getUser() {
  const res = await fetch('https://api.example.com/user')
  return res.json()
}

async function getStats() {
  const res = await fetch('https://api.example.com/stats')
  return res.json()
}

export default async function Dashboard() {
  // 并行获取数据
  const [user, stats] = await Promise.all([
    getUser(),
    getStats()
  ])

  return (
    <div>
      <h1>欢迎, {user.name}</h1>
      <div>
        <p>今日访问: {stats.todayVisits}</p>
        <p>总访问: {stats.totalVisits}</p>
      </div>
    </div>
  )
}

流式渲染 ​

使用 Suspense 实现流式渲染：

jsx
// app/posts/page.js
import { Suspense } from 'react'

async function PostList() {
  const posts = await fetch('https://api.example.com/posts')
  const data = await posts.json()
  
  return (
    <div>
      {data.map(post => (
        <div key={post.id}>{post.title}</div>
      ))}
    </div>
  )
}

export default function PostsPage() {
  return (
    <div>
      <h1>文章列表</h1>
      <Suspense fallback={<div>加载中...</div>}>
        <PostList />
      </Suspense>
    </div>
  )
}

客户端数据获取 ​

SWR ​

jsx
import useSWR from 'swr'

const fetcher = (url) => fetch(url).then(res => res.json())

export default function Profile() {
  const { data, error, isLoading } = useSWR('/api/user', fetcher)

  if (error) return <div>加载失败</div>
  if (isLoading) return <div>加载中...</div>

  return <div>你好 {data.name}!</div>
}

React Query ​

jsx
import { useQuery } from '@tanstack/react-query'

function fetchPosts() {
  return fetch('/api/posts').then(res => res.json())
}

export default function Posts() {
  const { data, isLoading, error } = useQuery({
    queryKey: ['posts'],
    queryFn: fetchPosts
  })

  if (isLoading) return <div>加载中...</div>
  if (error) return <div>出错了: {error.message}</div>

  return (
    <div>
      {data.map(post => (
        <div key={post.id}>{post.title}</div>
      ))}
    </div>
  )
}

缓存和重新验证 ​

fetch 选项 ​

jsx
// 缓存 60 秒
fetch('https://api.example.com/data', {
  next: { revalidate: 60 }
})

// 不缓存
fetch('https://api.example.com/data', {
  cache: 'no-store'
})

// 强制缓存
fetch('https://api.example.com/data', {
  cache: 'force-cache'
})

标签重新验证 ​

jsx
// 添加标签
fetch('https://api.example.com/posts', {
  next: { tags: ['posts'] }
})

// 重新验证标签
import { revalidateTag } from 'next/cache'

export async function POST() {
  // 更新数据后重新验证
  revalidateTag('posts')
}

错误边界 ​

jsx

export default function Error({ error, reset }) {

    <div>
      <h2>出错了!</h2>
      <p>{error.message}</p>
      <button onClick={reset}>重试</button>
    </div>
  )
}

加载状态 ​

jsx
// app/posts/loading.js
export default function Loading() {
  return <div>加载中...</div>
}

最佳实践 ​

  1. 选择合适的方法：根据数据更新频率选择获取方法
  2. 缓存策略：合理使用缓存提高性能
  3. 错误处理：始终处理错误情况
  4. 加载状态：提供良好的用户体验
  5. 并行获取：尽可能并行获取数据
