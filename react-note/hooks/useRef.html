useRef Hook ​

什么是 useRef？ ​

useRef 是 React 提供的一个 Hook，用于创建一个可变的引用对象，该对象在组件的整个生命周期内保持不变。useRef 返回的对象有一个 .current 属性，初始值为传入的参数，可以随时修改这个属性的值。

useRef 有两个主要用途：

  1. 访问 DOM 元素：通过 ref 属性将引用附加到 React 元素上
  2. 存储可变值：类似于类组件中的实例变量，可以存储任何可变值

与 state 不同，修改 ref.current 的值不会导致组件重新渲染。

基本语法 ​

jsx
const refContainer = useRef(initialValue);

参数 ​

  • initialValue：ref 对象的 .current 属性的初始值，可以是任何类型的值

返回值 ​

返回一个可变的 ref 对象，其 .current 属性被初始化为传入的参数。返回的对象在组件的整个生命周期内保持不变。

基本用法 ​

创建 ref ​

jsx
import React, { useRef } from 'react';

function MyComponent() {
  // 创建一个 ref，初始值为 null
  const myRef = useRef(null);
  
  // 使用 TypeScript 时可以指定类型
  const inputRef = useRef<HTMLInputElement>(null);
  
  // 也可以存储任何值
  const countRef = useRef(0);
  
  // ...
}

访问 DOM 元素 ​

jsx
import React, { useRef } from 'react';

function TextInputWithFocusButton() {
  // 创建一个 ref 对象
  const inputRef = useRef(null);
  
  // 使用 ref 访问 DOM 元素
  const focusInput = () => {
    inputRef.current.focus();
  };
  
  return (
    <>
      <input ref={inputRef} type="text" />
      <button onClick={focusInput}>聚焦输入框</button>
    </>
  );
}

存储可变值 ​

jsx
import React, { useState, useRef, useEffect } from 'react';

function Counter() {
  const [count, setCount] = useState(0);
  
  // 使用 ref 存储渲染之间的值
  const prevCountRef = useRef();
  const renderCountRef = useRef(0);
  
  useEffect(() => {
    // 存储当前 count 值，在下次渲染时可以作为上一次的值使用
    prevCountRef.current = count;
    // 增加渲染计数
    renderCountRef.current += 1;
  }, [count]);
  
  return (
    <>
      <p>当前计数: {count}</p>
      <p>上一次计数: {prevCountRef.current}</p>
      <p>组件渲染次数: {renderCountRef.current}</p>
      <button onClick={() => setCount(count + 1)}>增加计数</button>
    </>
  );
}

常见应用场景 ​

1. 管理焦点、文本选择或媒体播放 ​

jsx
function MediaPlayer() {
  const videoRef = useRef(null);
  
  const playVideo = () => {
    if (videoRef.current) {
      videoRef.current.play();
    }
  };
  
  const pauseVideo = () => {
    if (videoRef.current) {
      videoRef.current.pause();
    }
  };
  
  return (
    <div>
      <video ref={videoRef} src="/my-video.mp4" />
      <button onClick={playVideo}>播放</button>
      <button onClick={pauseVideo}>暂停</button>
    </div>
  );
}

2. 管理定时器和间隔器 ​

jsx
function Timer() {
  const [seconds, setSeconds] = useState(0);
  const timerRef = useRef(null);
  
  const startTimer = () => {
    if (timerRef.current !== null) return;
    
    timerRef.current = setInterval(() => {
      setSeconds(prev => prev + 1);
    }, 1000);
  };
  
  const stopTimer = () => {
    clearInterval(timerRef.current);
    timerRef.current = null;
  };
  
  // 组件卸载时清除定时器
  useEffect(() => {
    return () => {
      if (timerRef.current) clearInterval(timerRef.current);
    };
  }, []);
  
  return (
    <div>
      <p>计时器: {seconds} 秒</p>
      <button onClick={startTimer}>开始</button>
      <button onClick={stopTimer}>停止</button>
    </div>
  );
}

3. 保存上一个值 ​

jsx
function usePrevious(value) {
  const ref = useRef();
  
  useEffect(() => {
    ref.current = value;
  }, [value]);
  
  return ref.current;
}

function MyComponent() {
  const [count, setCount] = useState(0);
  const prevCount = usePrevious(count);
  
  return (
    <div>
      <p>当前值: {count}, 上一个值: {prevCount}</p>
      <button onClick={() => setCount(count + 1)}>增加</button>
    </div>
  );
}

4. 避免不必要的重新渲染 ​

jsx
function ExpensiveComponent({ data }) {
  // 使用 ref 存储数据，避免不必要的重新渲染
  const dataRef = useRef(data);
  
  // 只有当数据实际变化时才更新 ref
  if (JSON.stringify(dataRef.current) !== JSON.stringify(data)) {
    dataRef.current = data;
  }
  
  // 使用 ref 中的数据进行渲染
  return <div>{/* 使用 dataRef.current 渲染内容 */}</div>;
}

实际应用示例 ​

表单处理 ​

jsx
import React, { useRef, useState } from 'react';

function FormWithRef() {
  const [submitted, setSubmitted] = useState(false);
  const nameRef = useRef(null);
  const emailRef = useRef(null);
  const messageRef = useRef(null);
  
  const handleSubmit = (e) => {
    e.preventDefault();
    
    // 获取表单值
    const formData = {
      name: nameRef.current.value,
      email: emailRef.current.value,
      message: messageRef.current.value
    };
    
    // 验证表单
    if (!formData.name || !formData.email || !formData.message) {
      alert('请填写所有字段');
      return;
    }
    
    // 提交表单
    console.log('提交的数据:', formData);
    setSubmitted(true);
    
    // 清空表单
    nameRef.current.value = '';
    emailRef.current.value = '';
    messageRef.current.value = '';
  };
  
  return (
    <div>
      {submitted && <div>表单已提交！</div>}
      
      <form onSubmit={handleSubmit}>
        <div>
          <label>姓名:</label>
          <input ref={nameRef} type="text" />
        </div>
        
        <div>
          <label>邮箱:</label>
          <input ref={emailRef} type="email" />
        </div>
        
        <div>
          <label>留言:</label>
          <textarea ref={messageRef}></textarea>
        </div>
        
        <button type="submit">提交</button>
      </form>
    </div>
  );
}

实现图片预览 ​

jsx
import React, { useRef, useState } from 'react';

function ImageUploader() {
  const [preview, setPreview] = useState(null);
  const fileInputRef = useRef(null);
  
  const handleFileChange = (e) => {
    const file = e.target.files[0];
    if (!file) return;
    
    // 验证文件类型
    if (!file.type.startsWith('image/')) {
      alert('请选择图片文件');
      return;
    }
    
    // 创建预览URL
    const reader = new FileReader();
    reader.onload = () => {
      setPreview(reader.result);
    };
    reader.readAsDataURL(file);
  };
  
  const handleUploadClick = () => {
    // 触发文件选择对话框
    fileInputRef.current.click();
  };
  
  return (
    <div>
      <input
        type="file"
        ref={fileInputRef}
        onChange={handleFileChange}
        style={{ display: 'none' }}
        accept="image/*"
      />
      
      <button onClick={handleUploadClick}>
        选择图片
      </button>
      
      {preview && (
        <div>
          <h3>图片预览</h3>
          <img 
            src={preview} 
            alt="预览" 
            style={{ maxWidth: '300px', maxHeight: '300px' }} 
          />
        </div>
      )}
    </div>
  );
}

useRef vs useState ​

特性useRefuseState
触发重新渲染❌ 不会✅ 会
更新时机同步异步批量更新
用途存储不需要在 UI 中反映的值存储需要在 UI 中反映的值
访问方式ref.current通过状态变量直接访问
更新方式直接修改 ref.current调用 setter 函数

常见陷阱和解决方案 ​

1. 初始化时访问 DOM 元素 ​

jsx
// ❌ 错误方式：在组件渲染时直接访问 ref
function WrongComponent() {
  const inputRef = useRef(null);
  
  // 这会导致错误，因为在首次渲染时 inputRef.current 为 null
  inputRef.current.focus();
  
  return <input ref={inputRef} />;
}

// ✅ 正确方式：在 useEffect 中访问 ref
function CorrectComponent() {
  const inputRef = useRef(null);
  
  useEffect(() => {
    // 组件挂载后 inputRef.current 才会有值
    inputRef.current.focus();
  }, []);
  
  return <input ref={inputRef} />;
}

2. 在条件渲染中使用 ref ​

jsx
// ❌ 错误方式：可能导致 ref 为 null
function WrongComponent({ showInput }) {
  const inputRef = useRef(null);
  
  const focusInput = () => {
    // 如果 showInput 为 false，这里会出错
    inputRef.current.focus();
  };
  
  return (
    <>
      {showInput && <input ref={inputRef} />}
      <button onClick={focusInput}>聚焦</button>
    </>
  );
}

// ✅ 正确方式：检查 ref 是否存在
function CorrectComponent({ showInput }) {
  const inputRef = useRef(null);
  
  const focusInput = () => {
    // 先检查 ref 是否存在
    if (inputRef.current) {
      inputRef.current.focus();
    }
  };
  
  return (
    <>
      {showInput && <input ref={inputRef} />}
      <button onClick={focusInput}>聚焦</button>
    </>
  );
}

最佳实践 ​

  1. 访问 DOM 元素：使用 useRef 而不是直接使用 DOM API（如 document.getElementById）

  2. 存储不需要触发重新渲染的值：如果某个值的变化不需要反映在 UI 上，使用 useRef 而不是 useState

  3. 清理副作用：使用 useRef 存储定时器 ID、动画帧 ID 等，并在组件卸载时清理它们

  4. 类型安全：使用 TypeScript 时，为 ref 提供正确的类型，例如：

    tsx
    const inputRef = useRef<HTMLInputElement>(null);

  5. 避免过度使用：不要使用 useRef 来规避 React 的数据流原则，只在真正需要时使用

总结 ​

useRef 是一个强大的 Hook，可以帮助我们:

  • 访问和操作 DOM 元素
  • 存储在组件生命周期内持久化的可变值
  • 管理副作用，如定时器和动画
  • 在不触发重新渲染的情况下存储和更新值

理解 useRef 的特性和用法，可以帮助我们编写更高效、更灵活的 React 组件。
