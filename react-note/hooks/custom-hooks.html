自定义 Hooks ​

什么是自定义 Hooks？ ​

自定义 Hooks 是一种在 React 组件之间重用状态逻辑的机制。它们是利用 React 的 Hooks 特性创建的函数，可以提取组件逻辑到可重用的函数中，使代码更加简洁和模块化。

自定义 Hooks 是以 "use" 开头的 JavaScript 函数，可以调用其他 Hooks，并遵循 Hooks 的规则。它们不仅可以共享状态逻辑，还可以共享副作用、订阅、定时器等功能。

创建自定义 Hook 的规则 ​

  1. 自定义 Hook 必须以 "use" 开头，这是一个约定，让 React 能够识别它是一个 Hook
  2. 自定义 Hook 可以调用其他 Hook，包括内置 Hook 和其他自定义 Hook
  3. 自定义 Hook 必须遵循 Hook 的规则，即只能在组件顶层或其他 Hook 中调用
  4. 自定义 Hook 应该返回有用的值或函数，供组件使用

创建自定义 Hook ​

创建自定义 Hook 的步骤：

  1. 创建一个以 "use" 开头的函数
  2. 在函数内部使用其他 React Hooks
  3. 返回需要在组件中使用的值或函数

基本示例：useWindowSize ​

jsx
import { useState, useEffect } from 'react';

function useWindowSize() {
  // 初始化状态为当前窗口大小
  const [windowSize, setWindowSize] = useState({
    width: window.innerWidth,
    height: window.innerHeight
  });
  
  // 添加事件监听器并在组件卸载时移除
  useEffect(() => {
    // 处理窗口大小变化的函数
    function handleResize() {
      setWindowSize({
        width: window.innerWidth,
        height: window.innerHeight
      });
    }
    
    // 添加事件监听器
    window.addEventListener('resize', handleResize);
    
    // 清理函数，移除事件监听器
    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, []); // 空依赖数组，只在组件挂载和卸载时执行
  
  return windowSize;
}

使用自定义 Hook ​

jsx
function WindowSizeComponent() {
  const windowSize = useWindowSize();
  
  return (
    <div>
      <p>窗口宽度: {windowSize.width}px</p>
      <p>窗口高度: {windowSize.height}px</p>
    </div>
  );
}

常用自定义 Hooks 示例 ​

1. useLocalStorage ​

将状态同步到 localStorage，在页面刷新后保持状态。

jsx
function useLocalStorage(key, initialValue) {
  // 创建状态，初始值从localStorage获取，如果没有则使用传入的初始值
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(error);
      return initialValue;
    }
  });
  
  // 创建一个更新函数，同时更新状态和localStorage
  const setValue = (value) => {
    try {
      // 允许值是一个函数，类似于setState
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);
      window.localStorage.setItem(key, JSON.stringify(valueToStore));
    } catch (error) {
      console.error(error);
    }
  };
  
  return [storedValue, setValue];
}

// 使用示例
function ThemeToggler() {
  const [theme, setTheme] = useLocalStorage('theme', 'light');
  
  return (
    <div>
      <p>当前主题: {theme}</p>
      <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
        切换主题
      </button>
    </div>
  );
}

2. useFetch ​

处理 API 请求，包括加载状态和错误处理。

jsx
function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  
  useEffect(() => {
    let isMounted = true;
    const controller = new AbortController();
    
    async function fetchData() {
      try {
        setLoading(true);
        const response = await fetch(url, { signal: controller.signal });
        
        if (!response.ok) {
          throw new Error(`HTTP error! Status: ${response.status}`);
        }
        
        const result = await response.json();
        
        if (isMounted) {
          setData(result);
          setError(null);
        }
      } catch (error) {
        if (isMounted && error.name !== 'AbortError') {
          setError(error);
          setData(null);
        }
      } finally {
        if (isMounted) {
          setLoading(false);
        }
      }
    }
    
    fetchData();
    
    // 清理函数
    return () => {
      isMounted = false;
      controller.abort();
    };
  }, [url]);
  
  return { data, loading, error };
}

// 使用示例
function PostViewer() {
  const [postId, setPostId] = useState(1);
  const { data, loading, error } = useFetch(`https://jsonplaceholder.typicode.com/posts/${postId}`);
  
  if (loading) return <div>加载中...</div>;
  if (error) return <div>加载失败: {error.message}</div>;
  
  return (
    <div>
      <h1>{data.title}</h1>
      <p>{data.body}</p>
      <button onClick={() => setPostId(postId + 1)}>下一篇</button>
    </div>
  );
}

3. useForm ​

处理表单状态和验证。

jsx
function useForm(initialValues, validate) {
  const [values, setValues] = useState(initialValues);
  const [errors, setErrors] = useState({});
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  const handleChange = (e) => {
    const { name, value } = e.target;
    setValues({
      ...values,
      [name]: value
    });
  };
  
  const handleSubmit = (e) => {
    e.preventDefault();
    const validationErrors = validate(values);
    setErrors(validationErrors);
    if (Object.keys(validationErrors).length === 0) {
      setIsSubmitting(true);
    }
  };
  
  useEffect(() => {
    if (isSubmitting) {
      // 这里可以添加提交后的逻辑，比如API调用
      setIsSubmitting(false);
    }
  }, [isSubmitting]);
  
  const resetForm = () => {
    setValues(initialValues);
    setErrors({});
  };
  
  return {
    values,
    errors,
    handleChange,
    handleSubmit,
    resetForm,
    isSubmitting
  };
}

// 使用示例
function SignupForm() {
  const validate = (values) => {
    const errors = {};
    if (!values.email) {
      errors.email = '请输入邮箱';
    } else if (!/\S+@\S+\.\S+/.test(values.email)) {
      errors.email = '邮箱格式不正确';
    }
    if (!values.password) {
      errors.password = '请输入密码';
    } else if (values.password.length < 6) {
      errors.password = '密码至少需要6个字符';
    }
    return errors;
  };
  
  const { values, errors, handleChange, handleSubmit } = useForm(
    { email: '', password: '' },
    validate
  );
  
  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>邮箱:</label>
        <input
          type="text"
          name="email"
          value={values.email}
          onChange={handleChange}
        />
        {errors.email && <p>{errors.email}</p>}
      </div>
      <div>
        <label>密码:</label>
        <input
          type="password"
          name="password"
          value={values.password}
          onChange={handleChange}
        />
        {errors.password && <p>{errors.password}</p>}
      </div>
      <button type="submit">注册</button>
    </form>
  );
}

4. useDebounce ​

对值进行防抖处理，避免频繁更新引起的性能问题。

jsx
function useDebounce(value, delay) {
  const [debouncedValue, setDebouncedValue] = useState(value);
  
  useEffect(() => {
    // 设置定时器
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);
    
    // 清理函数，清除定时器
    return () => {
      clearTimeout(timer);
    };
  }, [value, delay]);
  
  return debouncedValue;
}

// 使用示例
function SearchComponent() {
  const [searchTerm, setSearchTerm] = useState('');
  const debouncedSearchTerm = useDebounce(searchTerm, 500);
  const [results, setResults] = useState([]);
  
  useEffect(() => {
    if (debouncedSearchTerm) {
      // 执行搜索，例如API调用
      searchAPI(debouncedSearchTerm).then(data => {
        setResults(data);
      });
    }
  }, [debouncedSearchTerm]);
  
  return (
    <div>
      <input
        type="text"
        value={searchTerm}
        onChange={(e) => setSearchTerm(e.target.value)}
        placeholder="搜索..."
      />
      <ul>
        {results.map(item => (
          <li key={item.id}>{item.name}</li>
        ))}
      </ul>
    </div>
  );
}

5. useMediaQuery ​

响应式设计中检测媒体查询的匹配情况。

jsx
function useMediaQuery(query) {
  const [matches, setMatches] = useState(false);

  useEffect(() => {
    const mediaQuery = window.matchMedia(query);
    
    // 设置初始值
    setMatches(mediaQuery.matches);
    
    // 创建事件监听器
    const handler = (event) => {
      setMatches(event.matches);
    };
    
    // 添加事件监听器
    mediaQuery.addEventListener('change', handler);
    
    // 清理函数
    return () => {
      mediaQuery.removeEventListener('change', handler);
    };
  }, [query]);

  return matches;
}

// 使用示例
function ResponsiveComponent() {
  const isMobile = useMediaQuery('(max-width: 768px)');
  const isTablet = useMediaQuery('(min-width: 769px) and (max-width: 1024px)');
  const isDesktop = useMediaQuery('(min-width: 1025px)');
  
  return (
    <div>
      <h1>响应式组件</h1>
      {isMobile && <p>移动端视图</p>}
      {isTablet && <p>平板视图</p>}
      {isDesktop && <p>桌面视图</p>}
    </div>
  );
}

6. useOnClickOutside ​

检测点击元素外部的事件。

jsx
function useOnClickOutside(ref, handler) {
  useEffect(() => {
    const listener = (event) => {
      // 如果点击的是元素内部，不做任何处理
      if (!ref.current || ref.current.contains(event.target)) {
        return;
      }
      
      // 否则，调用处理函数
      handler(event);
    };
    
    // 添加事件监听器
    document.addEventListener('mousedown', listener);
    document.addEventListener('touchstart', listener);
    
    // 清理函数
    return () => {
      document.removeEventListener('mousedown', listener);
      document.removeEventListener('touchstart', listener);
    };
  }, [ref, handler]);
}

// 使用示例
function Dropdown() {
  const [isOpen, setIsOpen] = useState(false);
  const dropdownRef = useRef();
  
  // 点击外部时关闭下拉菜单
  useOnClickOutside(dropdownRef, () => {
    if (isOpen) setIsOpen(false);
  });
  
  return (
    <div ref={dropdownRef}>
      <button onClick={() => setIsOpen(!isOpen)}>
        {isOpen ? '关闭' : '打开'} 菜单
      </button>
      
      {isOpen && (
        <ul>
          <li>选项 1</li>
          <li>选项 2</li>
          <li>选项 3</li>
        </ul>
      )}
    </div>
  );
}

自定义 Hooks 的优势 ​

  1. 逻辑复用 - 可以在多个组件之间共享逻辑，避免代码重复
  2. 关注点分离 - 将复杂逻辑从组件中提取出来，使组件更加简洁
  3. 可测试性 - 自定义 Hooks 可以独立测试，提高代码质量
  4. 可组合性 - 可以组合多个自定义 Hooks 构建复杂功能
  5. 封装实现细节 - 将实现细节隐藏在 Hook 内部，提供简洁的 API

自定义 Hooks 的最佳实践 ​

  1. 命名约定 - 始终以 "use" 开头命名自定义 Hooks
  2. 保持简单 - 每个 Hook 专注于单一功能，避免过于复杂
  3. 处理清理工作 - 在 useEffect 中返回清理函数，避免内存泄漏
  4. 设计良好的 API - 提供清晰、一致的参数和返回值
  5. 文档和测试 - 为自定义 Hooks 编写文档和测试
  6. TypeScript 支持 - 使用 TypeScript 为自定义 Hooks 提供类型定义

总结 ​

自定义 Hooks 是 React 中一种强大的代码复用机制，它使我们能够提取组件逻辑到可重用的函数中，同时保持代码的简洁和可维护性。通过遵循一些最佳实践，我们可以创建出高质量、可复用的自定义 Hooks，提高开发效率和代码质量。
