useActionData Hook ​

useActionData 是 React Router 提供的一个 Hook，用于访问当前路由的 action 函数返回的数据。这对于处理表单提交后的响应数据特别有用。

基本用法 ​

jsx
import { useActionData, Form } from 'react-router-dom';

function ContactForm() {
  const actionData = useActionData();
  
  return (
    <Form method="post">
      <input type="text" name="name" />
      {actionData?.errors?.name && (
        <p className="error">{actionData.errors.name}</p>
      )}
      
      <input type="email" name="email" />
      {actionData?.errors?.email && (
        <p className="error">{actionData.errors.email}</p>
      )}
      
      <button type="submit">提交</button>
      
      {actionData?.success && (
        <p className="success">表单提交成功！</p>
      )}
    </Form>
  );
}

语法 ​

jsx
const actionData = useActionData();

返回值 ​

useActionData 返回当前路由的 action 函数返回的数据。如果 action 尚未执行或正在执行中，则返回 undefined。

示例 ​

基本表单验证 ​

首先，定义带有 action 的路由：

jsx
import { createBrowserRouter } from 'react-router-dom';
import ContactForm from './ContactForm';

const router = createBrowserRouter([
  {
    path: "/contact",
    element: <ContactForm />,
    action: async ({ request }) => {
      const formData = await request.formData();
      const name = formData.get("name");
      const email = formData.get("email");
      
      const errors = {};
      
      if (!name) {
        errors.name = "姓名是必填项";
      }
      
      if (!email) {
        errors.email = "邮箱是必填项";
      } else if (!email.includes("@")) {
        errors.email = "请输入有效的邮箱地址";
      }
      
      if (Object.keys(errors).length > 0) {
        return { errors };
      }
      
      // 处理表单提交逻辑
      // ...
      
      return { success: true };
    }
  }
]);

然后，在组件中使用 useActionData 获取 action 返回的数据：

jsx
import { useActionData, Form } from 'react-router-dom';

function ContactForm() {
  const actionData = useActionData();
  
  return (
    <Form method="post">
      <div>
        <label htmlFor="name">姓名：</label>
        <input 
          type="text" 
          id="name" 
          name="name" 
          aria-invalid={actionData?.errors?.name ? true : undefined}
          aria-describedby={actionData?.errors?.name ? "name-error" : undefined}
        />
        {actionData?.errors?.name && (
          <p className="error" id="name-error">{actionData.errors.name}</p>
        )}
      </div>
      
      <div>
        <label htmlFor="email">邮箱：</label>
        <input 
          type="email" 
          id="email" 
          name="email"
          aria-invalid={actionData?.errors?.email ? true : undefined}
          aria-describedby={actionData?.errors?.email ? "email-error" : undefined}
        />
        {actionData?.errors?.email && (
          <p className="error" id="email-error">{actionData.errors.email}</p>
        )}
      </div>
      
      <button type="submit">提交</button>
      
      {actionData?.success && (
        <p className="success">表单提交成功！</p>
      )}
    </Form>
  );
}

处理异步操作状态 ​

结合 useNavigation 和 useActionData 处理表单提交状态：

jsx
import { useActionData, useNavigation, Form } from 'react-router-dom';

function LoginForm() {
  const actionData = useActionData();
  const navigation = useNavigation();
  const isSubmitting = navigation.state === "submitting";
  
  return (
    <Form method="post">
      <div>
        <label htmlFor="username">用户名：</label>
        <input 
          type="text" 
          id="username" 
          name="username"
          disabled={isSubmitting}
        />
        {actionData?.errors?.username && (
          <p className="error">{actionData.errors.username}</p>
        )}
      </div>
      
      <div>
        <label htmlFor="password">密码：</label>
        <input 
          type="password" 
          id="password" 
          name="password"
          disabled={isSubmitting}
        />
        {actionData?.errors?.password && (
          <p className="error">{actionData.errors.password}</p>
        )}
      </div>
      
      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? "登录中..." : "登录"}
      </button>
      
      {actionData?.error && (
        <p className="error">{actionData.error}</p>
      )}
    </Form>
  );
}

处理不同类型的响应 ​

action 可以返回不同类型的数据，根据操作的结果：

jsx
// 路由配置
{
  path: "/products/new",
  element: <NewProduct />,
  action: async ({ request }) => {
    const formData = await request.formData();
    
    try {
      // 尝试创建产品
      const product = await createProduct(formData);
      return { product };
    } catch (error) {
      return { error: error.message };
    }
  }
}

在组件中处理不同类型的响应：

jsx
import { useActionData, Form, useNavigate } from 'react-router-dom';
import { useEffect } from 'react';

function NewProduct() {
  const actionData = useActionData();
  const navigate = useNavigate();
  
  // 如果产品创建成功，重定向到产品详情页面
  useEffect(() => {
    if (actionData?.product) {
      navigate(`/products/${actionData.product.id}`);
    }
  }, [actionData, navigate]);
  
  return (
    <Form method="post">
      <h1>新建产品</h1>
      
      {actionData?.error && (
        <p className="error">{actionData.error}</p>
      )}
      
      <div>
        <label htmlFor="name">产品名称：</label>
        <input type="text" id="name" name="name" />
      </div>
      
      <div>
        <label htmlFor="price">价格：</label>
        <input type="number" id="price" name="price" min="0" step="0.01" />
      </div>
      
      <div>
        <label htmlFor="description">描述：</label>
        <textarea id="description" name="description" />
      </div>
      
      <button type="submit">创建产品</button>
    </Form>
  );
}

与客户端验证结合 ​

结合客户端验证和服务器验证：

jsx
import { useActionData, Form } from 'react-router-dom';
import { useState } from 'react';

function SignupForm() {
  const actionData = useActionData();
  const [formErrors, setFormErrors] = useState({});
  
  function validateForm(event) {
    const form = event.currentTarget;
    const password = form.elements.password.value;
    const confirmPassword = form.elements.confirmPassword.value;
    
    const errors = {};
    
    if (password !== confirmPassword) {
      errors.confirmPassword = "两次输入的密码不一致";
    }
    
    if (Object.keys(errors).length > 0) {
      event.preventDefault();
      setFormErrors(errors);
    } else {
      setFormErrors({});
    }
  }
  
  return (
    <Form method="post" onSubmit={validateForm}>
      <div>
        <label htmlFor="email">邮箱：</label>
        <input type="email" id="email" name="email" />
        {actionData?.errors?.email && (
          <p className="error">{actionData.errors.email}</p>
        )}
      </div>
      
      <div>
        <label htmlFor="password">密码：</label>
        <input type="password" id="password" name="password" />
        {actionData?.errors?.password && (
          <p className="error">{actionData.errors.password}</p>
        )}
      </div>
      
      <div>
        <label htmlFor="confirmPassword">确认密码：</label>
        <input type="password" id="confirmPassword" name="confirmPassword" />
        {formErrors.confirmPassword && (
          <p className="error">{formErrors.confirmPassword}</p>
        )}
      </div>
      
      <button type="submit">注册</button>
    </Form>
  );
}

注意事项 ​

  1. 初始状态：在首次渲染时，useActionData 返回 undefined，因为 action 尚未执行。

  2. 与 useLoaderData 的区别：useActionData 用于访问 action 函数的返回数据（通常是表单提交后），而 useLoaderData 用于访问 loader 函数的返回数据（通常是页面加载时）。

  3. 数据持久性：action 数据会在导航到新路由时被清除。如果需要在导航之间保持数据，可以考虑使用 React 的状态管理或其他持久化方法。

  4. 错误处理：action 函数可以抛出错误，这些错误可以通过 useRouteError hook 捕获，而不是 useActionData。

  5. 与其他 hooks 结合：useActionData 通常与 useNavigation 结合使用，以提供关于表单提交状态的完整反馈。
