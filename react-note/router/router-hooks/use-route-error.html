useRouteError Hook ​

useRouteError 是 React Router v6.4+ 提供的一个 Hook，用于在错误边界组件中获取路由错误信息。当路由的 loader 或 action 函数抛出错误时，这个 Hook 可以访问到这些错误。

基本用法 ​

jsx
import { useRouteError } from 'react-router-dom';

function ErrorBoundary() {
  const error = useRouteError();
  
  return (
    <div className="error-container">
      <h1>出错了！</h1>
      <p>{error.message || '发生了一个错误'}</p>
      <p>状态码: {error.status || '未知'}</p>
    </div>
  );
}

// 在路由配置中使用错误边界
<Route 
  path="/products/:id" 
  element={<ProductDetail />}
  errorElement={<ErrorBoundary />}
  loader={productLoader}
/>

语法 ​

jsx
const error = useRouteError();

返回值 ​

useRouteError 返回路由过程中抛出的错误对象。这个错误可能来自：

  • loader 函数
  • action 函数
  • 组件渲染过程

详细说明 ​

错误类型 ​

useRouteError 可以返回不同类型的错误：

  1. Response 对象：从 loader 或 action 中抛出的 Response 对象
  2. Error 对象：JavaScript 错误对象
  3. 普通值：任何通过 throw 抛出的值

处理 Response 错误 ​

当 loader 或 action 抛出 Response 对象时：

jsx
import { useRouteError } from 'react-router-dom';

function ErrorBoundary() {
  const error = useRouteError();
  
  if (error instanceof Response) {
    const status = error.status;
    
    if (status === 404) {
      return <h1>页面未找到</h1>;
    }
    
    return (
      <div>
        <h1>{status} 错误</h1>
        <p>{error.statusText}</p>
      </div>
    );
  }
  
  return <h1>发生了一个错误</h1>;
}

处理 JavaScript 错误 ​

当捕获到 JavaScript Error 对象时：

jsx
import { useRouteError } from 'react-router-dom';

function ErrorBoundary() {
  const error = useRouteError();
  
  if (error instanceof Error) {
    return (
      <div>
        <h1>错误</h1>
        <p>{error.message}</p>
        <pre>{error.stack}</pre>
      </div>
    );
  }
  
  return <h1>发生了一个未知错误</h1>;
}

示例 ​

基本错误处理 ​

jsx
import { useRouteError, isRouteErrorResponse } from 'react-router-dom';

function ErrorBoundary() {
  const error = useRouteError();
  
  if (isRouteErrorResponse(error)) {
    // 处理 Response 错误
    return (
      <div>
        <h1>{error.status} {error.statusText}</h1>
        <p>{error.data}</p>
      </div>
    );
  }
  
  if (error instanceof Error) {
    // 处理 JavaScript 错误
    return (
      <div>
        <h1>错误</h1>
        <p>{error.message}</p>
      </div>
    );
  }
  
  // 处理其他类型的错误
  return (
    <div>
      <h1>发生了一个错误</h1>
      <p>{String(error)}</p>
    </div>
  );
}

处理特定状态码 ​

jsx
import { useRouteError, isRouteErrorResponse } from 'react-router-dom';

function ErrorBoundary() {
  const error = useRouteError();
  
  if (isRouteErrorResponse(error)) {
    if (error.status === 404) {
      return (
        <div>
          <h1>404 - 页面未找到</h1>
          <p>抱歉，您访问的页面不存在。</p>
          <Link to="/">返回首页</Link>
        </div>
      );
    }
    
    if (error.status === 401) {
      return (
        <div>
          <h1>401 - 未授权</h1>
          <p>您需要登录才能访问此页面。</p>
          <Link to="/login">登录</Link>
        </div>
      );
    }
    
    if (error.status === 503) {
      return (
        <div>
          <h1>503 - 服务不可用</h1>
          <p>服务器暂时不可用，请稍后再试。</p>
        </div>
      );
    }
    
    // 处理其他状态码
    return (
      <div>
        <h1>{error.status} - {error.statusText}</h1>
        <p>{error.data}</p>
      </div>
    );
  }
  
  // 处理非 Response 错误
  return (
    <div>
      <h1>出错了！</h1>
      <p>{error instanceof Error ? error.message : String(error)}</p>
    </div>
  );
}

从 loader 抛出错误 ​

jsx
// 产品加载器
export async function productLoader({ params }) {
  const { productId } = params;
  
  const response = await fetch(`/api/products/${productId}`);
  
  if (response.status === 404) {
    throw new Response('产品未找到', { status: 404 });
  }
  
  if (!response.ok) {
    throw new Response('加载产品失败', { 
      status: response.status,
      statusText: response.statusText
    });
  }
  
  return response.json();
}

// 错误边界组件
function ProductError() {
  const error = useRouteError();
  
  return (
    <div className="product-error">
      <h1>产品加载失败</h1>
      {isRouteErrorResponse(error) ? (
        <>
          <p>状态: {error.status}</p>
          <p>消息: {error.data}</p>
        </>
      ) : (
        <p>错误: {error instanceof Error ? error.message : String(error)}</p>
      )}
      <Link to="/products">返回产品列表</Link>
    </div>
  );
}

// 路由配置
<Route 
  path="/products/:productId" 
  element={<ProductDetail />}
  loader={productLoader}
  errorElement={<ProductError />}
/>

从 action 抛出错误 ​

jsx
// 表单处理 action
export async function createProductAction({ request }) {
  const formData = await request.formData();
  const product = Object.fromEntries(formData);
  
  try {
    const response = await fetch('/api/products', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(product),
    });
    
    if (response.status === 401) {
      throw new Response('未授权', { status: 401 });
    }
    
    if (!response.ok) {
      throw new Response('创建产品失败', { status: response.status });
    }
    
    return response.json();
  } catch (error) {
    if (error instanceof Response) throw error;
    throw new Response('提交表单时出错', { status: 500 });
  }
}

// 错误边界组件
function FormError() {
  const error = useRouteError();
  const navigate = useNavigate();
  
  return (
    <div className="form-error">
      <h1>表单提交失败</h1>
      {isRouteErrorResponse(error) && error.status === 401 ? (
        <div>
          <p>您需要登录才能执行此操作。</p>
          <button onClick={() => navigate('/login')}>登录</button>
        </div>
      ) : (
        <p>{isRouteErrorResponse(error) ? error.data : String(error)}</p>
      )}
      <button onClick={() => navigate(-1)}>返回</button>
    </div>
  );
}

// 路由配置
<Route 
  path="/products/new" 
  element={<NewProduct />}
  action={createProductAction}
  errorElement={<FormError />}
/>

嵌套错误处理 ​

在嵌套路由中，错误会冒泡到最近的错误边界：

jsx
<Routes>
  <Route 
    path="/" 
    element={<Layout />} 
    errorElement={<RootErrorBoundary />}
  >
    <Route index element={<Home />} />
    <Route 
      path="products" 
      element={<Products />}
    >
      <Route 
        path=":productId" 
        element={<ProductDetail />}
        loader={productLoader}
        errorElement={<ProductError />} // 处理产品加载错误
      />
    </Route>
    <Route 
      path="checkout" 
      element={<Checkout />}
      action={checkoutAction}
      // 没有错误边界，错误会冒泡到根错误边界
    />
  </Route>
</Routes>

注意事项 ​

  1. 错误类型检查：使用 isRouteErrorResponse 辅助函数来检查错误是否是 Response 类型。

  2. 错误边界位置：错误边界的位置决定了用户界面的哪些部分会在错误发生时保留。

  3. 错误数据：从 loader 或 action 抛出的 Response 对象可以包含数据，可以通过 error.data 访问。

  4. 开发环境：在开发环境中，React 会显示额外的错误覆盖层，这是正常的行为。

最佳实践 ​

  1. 提供友好的错误信息：错误信息应该对用户友好，避免技术术语。

  2. 区分错误类型：根据错误类型和状态码提供不同的错误处理逻辑。

  3. 提供恢复选项：在错误页面提供返回按钮、重试按钮或其他恢复选项。

  4. 分层错误处理：为不同级别的路由提供专门的错误边界，以便更精确地处理错误。

  5. 记录错误：考虑将错误记录到日志系统，以便后续分析和修复。

jsx
function ErrorBoundary() {
  const error = useRouteError();
  
  useEffect(() => {
    // 记录错误
    logError(error);
  }, [error]);
  
  // 渲染错误界面
}

相关 API ​

  • errorElement - 路由错误边界属性
  • isRouteErrorResponse - 检查错误是否是路由响应错误
  • useNavigate - 用于在错误处理后导航
  • loader - 路由数据加载函数
  • action - 路由表单处理函数
