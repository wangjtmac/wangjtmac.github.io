通用工具函数 ​

概述 ​

本模块收集了 TypeScript 开发中常用的通用工具函数代码示例，涵盖了类型操作、数据转换、验证、函数式编程等多个方面。

代码模板说明 ​

每个工具函数都提供 TypeScript 和 JavaScript 两个版本，您可以通过 Tab 切换来选择适合的版本：

  • TypeScript 版本：包含完整的类型定义，适合 TypeScript 项目
  • JavaScript 版本：简化版本，适合纯 JavaScript 项目

使用建议 ​

  • 如果您使用 TypeScript，建议使用 TypeScript 版本以获得更好的类型安全
  • 如果您使用 JavaScript，可以直接使用 JavaScript 版本，无需删除类型定义

Tab 切换使用说明 ​

每个函数代码块都提供了 Tab 切换功能，点击上方的 "TypeScript" 或 "JavaScript" 标签即可切换显示对应的代码版本。

函数组合 ​

typescript
/**
 * 函数管道（从左到右执行）
 * @param fns 要组合的函数数组
 * @returns 组合后的函数
 */
function pipe<T>(...fns: Array<(arg: T) => T>): (arg: T) => T {
    return (arg: T) => fns.reduce((acc, fn) => fn(acc), arg);
}

/**
 * 函数组合（从右到左执行）
 * @param fns 要组合的函数数组
 * @returns 组合后的函数
 */
function compose<T>(...fns: Array<(arg: T) => T>): (arg: T) => T {
    return (arg: T) => fns.reduceRight((acc, fn) => fn(acc), arg);
}

/**
 * 柯里化函数
 * @param fn 要柯里化的函数
 * @returns 柯里化后的函数
 */
function curry<T extends any[], R>(fn: (...args: T) => R): (...args: T) => R {
    return function curried(...args: T) {
        if (args.length >= fn.length) {
            return fn(...args);
        } else {
            return (...moreArgs: T) => curried(...args, ...moreArgs);
        }
    };
}

并发请求工具 ​

typescript
/**
 * 批量请求函数生成器
 * @param data 数据数组
 * @param fn 处理每个数据项的函数
 * @returns 返回函数数组，每个函数调用时会执行 fn(item, index)
 */
function getBatchRequestFn<T, R>(data: T[], fn: (item: T, index: number) => R): (() => R)[] {
    const res: (() => R)[] = [];
    data.forEach((item, i) => {
        res.push(() => fn(item, i));
    });
    return res;
}

/**
 * 并发请求控制器
 * @param reqs 请求函数数组
 * @param maxNum 最大并发数，默认为 1
 * @returns Promise，解析为所有请求的结果数组
 */
function concurRequest<T>(reqs: (() => Promise<T>)[], maxNum: number = 1): Promise<T[]> {
    return new Promise(resolve => {
        if (reqs.length === 0) {
            resolve([]);
            return;
        }
        
        let index = 0;
        let count = 0;
        const result: T[] = [];
        
        async function request(): Promise<void> {
            const i = index;
            const req = reqs[i];
            index++;
            
            try {
                const resp = await Promise.resolve(req());
                result[i] = resp;
            } catch (err) {
                result[i] = err as T;
            } finally {
                count++;
                if (count === reqs.length) {
                    resolve(result);
                }
                if (index < reqs.length) {
                    request();
                }
            }
        }
        
        for (let i = 0; i < Math.min(reqs.length, maxNum); i++) {
            request();
        }
    });
}

性能优化函数 ​

typescript
/**
 * 记忆化函数（缓存计算结果）
 * @param fn 要记忆化的函数
 * @returns 记忆化后的函数
 */
function memoize<T extends (...args: any[]) => any>(fn: T): T {
    const cache = new Map<string, ReturnType<T>>();
    
    return ((...args: Parameters<T>) => {
        const key = JSON.stringify(args);
        if (cache.has(key)) {
            return cache.get(key);
        }
        const result = fn(...args);
        cache.set(key, result);
        return result;
    }) as T;
}

/**
 * 防抖函数
 * @param fn 要防抖的函数
 * @param delay 延迟时间（毫秒）
 * @returns 防抖后的函数
 */
function debounce<T extends (...args: any[]) => void>(fn: T, delay: number): T {
    let timeoutId: NodeJS.Timeout;
    
    return ((...args: Parameters<T>) => {
        clearTimeout(timeoutId);
        timeoutId = setTimeout(() => fn(...args), delay);
    }) as T;
}

/**
 * 节流函数
 * @param fn 要节流的函数
 * @param delay 节流间隔（毫秒）
 * @returns 节流后的函数
 */
function throttle<T extends (...args: any[]) => void>(fn: T, delay: number): T {
    let lastCall = 0;
    
    return ((...args: Parameters<T>) => {
        const now = Date.now();
        if (now - lastCall >= delay) {
            lastCall = now;
            fn(...args);
        }
    }) as T;
}

对象处理 ​

typescript
/**
 * 深拷贝对象
 * @param obj 要拷贝的对象
 * @returns 深拷贝后的对象
 */
function deepClone<T>(obj: T): T {
    if (obj === null || typeof obj !== 'object') {
        return obj;
    }
    
    if (obj instanceof Date) {
        return new Date(obj.getTime()) as unknown as T;
    }
    
    if (obj instanceof Array) {
        return obj.map(item => deepClone(item)) as unknown as T;
    }
    
    if (typeof obj === 'object') {
        const cloned = {} as T;
        for (const key in obj) {
            if (obj.hasOwnProperty(key)) {
                cloned[key] = deepClone(obj[key]);
            }
        }
        return cloned;
    }
    
    return obj;
}

/**
 * 深合并对象
 * @param target 目标对象
 * @param sources 源对象
 * @returns 合并后的对象
 */
function deepMerge<T extends Record<string, any>>(target: T, ...sources: Partial<T>[]): T {
    if (!sources.length) return target;
    
    const source = sources.shift();
    if (source === undefined) return target;
    
    if (isObject(target) && isObject(source)) {
        for (const key in source) {
            if (isObject(source[key])) {
                if (!target[key]) Object.assign(target, { [key]: {} });
                deepMerge(target[key], source[key]);
            } else {
                Object.assign(target, { [key]: source[key] });
            }
        }
    }
    
    return deepMerge(target, ...sources);
}

/**
 * 从对象中选取指定属性
 * @param obj 源对象
 * @param keys 要选取的属性键
 * @returns 包含选取属性的新对象
 */
function pick<T extends Record<string, any>, K extends keyof T>(obj: T, keys: K[]): Pick<T, K> {
    const result = {} as Pick<T, K>;
    keys.forEach(key => {
        if (key in obj) {
            result[key] = obj[key];
        }
    });
    return result;
}

/**
 * 从对象中排除指定属性
 * @param obj 源对象
 * @param keys 要排除的属性键
 * @returns 排除指定属性后的新对象
 */
function omit<T extends Record<string, any>, K extends keyof T>(obj: T, keys: K[]): Omit<T, K> {
    const result = { ...obj };
    keys.forEach(key => {
        delete result[key];
    });
    return result;
}

/**
 * 数组去重
 * @param array 要去重的数组
 * @returns 去重后的数组
 */
function unique<T>(array: T[]): T[] {
    return [...new Set(array)];
}

/**
 * 数组分块
 * @param array 要分块的数组
 * @param size 块大小
 * @returns 分块后的二维数组
 */
function chunk<T>(array: T[], size: number): T[][] {
    const chunks: T[][] = [];
    for (let i = 0; i < array.length; i += size) {
        chunks.push(array.slice(i, i + size));
    }
    return chunks;
}

/**
 * 数组扁平化
 * @param array 要扁平化的数组
 * @param depth 扁平化深度
 * @returns 扁平化后的数组
 */
function flatten<T>(array: any[], depth: number = 1): T[] {
    if (depth === 0) return array;
    
    return array.reduce((acc, val) => {
        if (Array.isArray(val)) {
            return acc.concat(flatten(val, depth - 1));
        }
        return acc.concat(val);
    }, []);
}

/**
 * 按属性分组
 * @param array 要分组的数组
 * @param key 分组键
 * @returns 分组后的对象
 */
function groupBy<T extends Record<string, any>, K extends keyof T>(array: T[], key: K): Record<string, T[]> {
    return array.reduce((groups, item) => {
        const groupKey = String(item[key]);
        if (!groups[groupKey]) {
            groups[groupKey] = [];
        }
        groups[groupKey].push(item);
        return groups;
    }, {} as Record<string, T[]>);
}

字符串处理 ​

typescript
/**
 * 转换为驼峰命名
 * @param str 要转换的字符串
 * @returns 驼峰命名字符串
 */
function camelCase(str: string): string {
    return str.replace(/[-_\s]+(.)?/g, (_, c) => c ? c.toUpperCase() : '');
}

/**
 * 转换为蛇形命名
 * @param str 要转换的字符串
 * @returns 蛇形命名字符串
 */
function snakeCase(str: string): string {
    return str.replace(/([A-Z])/g, '_$1').toLowerCase().replace(/^_/, '');
}

/**
 * 转换为短横线命名
 * @param str 要转换的字符串
 * @returns 短横线命名字符串
 */
function kebabCase(str: string): string {
    return str.replace(/([A-Z])/g, '-$1').toLowerCase().replace(/^-/, '');
}

/**
 * 转换为帕斯卡命名
 * @param str 要转换的字符串
 * @returns 帕斯卡命名字符串
 */
function pascalCase(str: string): string {
    return str.replace(/(^|[-_\s]+)(.)/g, (_, __, c) => c.toUpperCase());
}

/**
 * 首字母大写
 * @param str 要转换的字符串
 * @returns 首字母大写的字符串
 */
function capitalize(str: string): string {
    return str.charAt(0).toUpperCase() + str.slice(1);
}

/**
 * 截断字符串
 * @param str 要截断的字符串
 * @param length 最大长度
 * @param suffix 后缀
 * @returns 截断后的字符串
 */
function truncate(str: string, length: number, suffix: string = '...'): string {
    if (str.length <= length) return str;
    return str.slice(0, length - suffix.length) + suffix;
}

console.log(expensiveCalculation(5)); // 25（从缓存中获取）

// 字符串工具使用示例

console.log(snakeCase(str)); // hello_world
console.log(kebabCase(str)); // hello-world
console.log(pascalCase(str)); // HelloWorld
